# Android + Compose 方案二：笔记应用

## 项目介绍

### 项目是什么
这是一个功能完整的笔记应用，使用 Kotlin + Jetpack Compose + Room 数据库构建，支持笔记创建、编辑、分类、搜索、归档等功能。

### 解决什么问题
- 快速记录想法
- 整理和管理笔记
- 分类和标签管理
- 全文搜索

### 核心技术亮点
| 技术 | 说明 |
|------|------|
| Jetpack Compose | 声明式 UI |
| Room | 本地 SQLite 数据库 |
| Hilt | 依赖注入 |
| Coroutines + Flow | 异步编程 |
| Material 3 | 设计系统 |
| Navigation Compose | 导航 |

### 适合场景
- 学习 Room 数据库
- 掌握本地数据持久化
- Compose 综合项目练习

```kotlin
// ===== Note.kt =====

package com.example.notes.data.local.entity

import androidx.room.*
import java.time.LocalDateTime

/**
 * 笔记实体
 * 
 * @Entity 标记为 Room 实体
 * @PrimaryKey 主键
 * @AutoGenerate 自动生成 ID
 */
@Entity(
    tableName = "notes",
    indices = [
        // 索引：加快按分类和状态查询
        Index(value = ["categoryId"]),
        Index(value = ["isArchived"]),
        Index(value = ["createdAt"])
    ]
)
data class Note(
    @PrimaryKey(autoGenerate = true)
    val id: Long = 0,
    
    // ===== 笔记内容 =====
    @ColumnInfo(name = "title")
    val title: String,
    
    @ColumnInfo(name = "content")
    val content: String,
    
    // ===== 格式 =====
    // @ColumnInfo 可以指定列名和其他属性
    @ColumnInfo(name = "color")
    val color: Int = 0xFFFFFFFF.toInt(),  // 笔记颜色（ARGB）
    
    @ColumnInfo(name = "is_pinned")
    val isPinned: Boolean = false,         // 是否置顶
    
    // ===== 分类 =====
    // nullable：笔记可以不属于任何分类
    @ColumnInfo(name = "category_id")
    val categoryId: Long? = null,
    
    // ===== 软删除 =====
    @ColumnInfo(name = "is_archived")
    val isArchived: Boolean = false,
    
    @ColumnInfo(name = "is_deleted")
    val isDeleted: Boolean = false,
    
    // ===== 时间戳 =====
    // TypeConverters：将 LocalDateTime 转换为 Long 存储
    @ColumnInfo(name = "created_at")
    val createdAt: LocalDateTime = LocalDateTime.now(),
    
    @ColumnInfo(name = "updated_at")
    val updatedAt: LocalDateTime = LocalDateTime.now(),
    
    // ===== 其他 =====
    // 提醒时间（可选）
    @ColumnInfo(name = "reminder_at")
    val reminderAt: LocalDateTime? = null
)

/**
 * 分类实体
 */
@Entity(tableName = "categories")
data class Category(
    @PrimaryKey(autoGenerate = true)
    val id: Long = 0,
    
    @ColumnInfo(name = "name")
    val name: String,
    
    @ColumnInfo(name = "color")
    val color: Int = 0xFF2196F3.toInt(),  // 默认蓝色
    
    @ColumnInfo(name = "icon")
    val icon: String? = null,             // 图标名称
    
    @ColumnInfo(name = "sort_order")
    val sortOrder: Int = 0,
    
    @ColumnInfo(name = "created_at")
    val createdAt: LocalDateTime = LocalDateTime.now()
)

/**
 * 笔记和分类的关联（多对多）
 */
@Entity(
    tableName = "note_tags",
    primaryKeys = ["note_id", "tag"],
    indices = [Index(value = ["tag"])]
)
data class NoteTag(
    @ColumnInfo(name = "note_id")
    val noteId: Long,
    
    @ColumnInfo(name = "tag")
    val tag: String
)
```

```kotlin
// ===== NoteDao.kt =====

package com.example.notes.data.local

import androidx.room.*
import com.example.notes.data.local.entity.*
import kotlinx.coroutines.flow.Flow

/**
 * 笔记 DAO（Data Access Object）
 * 
 * Room 会在编译时自动生成实现
 * 
 * Flow：响应式数据流，数据变化时自动通知观察者
 */
@Dao
interface NoteDao {
    
    /**
     * 获取所有笔记（排除已删除的）
     * 
     * Flow 特性：
     * - 当数据库变化时，Flow 会自动发出新数据
     * - Compose 的 collectAsState() 会自动观察 Flow
     */
    @Query("""
        SELECT * FROM notes 
        WHERE is_deleted = 0 
        ORDER BY is_pinned DESC, updated_at DESC
    """)
    fun getAllNotes(): Flow<List<Note>>
    
    /**
     * 获取归档笔记
     */
    @Query("""
        SELECT * FROM notes 
        WHERE is_archived = 1 AND is_deleted = 0
        ORDER BY updated_at DESC
    """)
    fun getArchivedNotes(): Flow<List<Note>>
    
    /**
     * 获取回收站笔记
     */
    @Query("SELECT * FROM notes WHERE is_deleted = 1 ORDER BY updated_at DESC")
    fun getDeletedNotes(): Flow<List<Note>>
    
    /**
     * 获取指定分类的笔记
     */
    @Query("""
        SELECT * FROM notes 
        WHERE category_id = :categoryId AND is_deleted = 0
        ORDER BY is_pinned DESC, updated_at DESC
    """)
    fun getNotesByCategory(categoryId: Long): Flow<List<Note>>
    
    /**
     * 获取置顶笔记
     */
    @Query("""
        SELECT * FROM notes 
        WHERE is_pinned = 1 AND is_deleted = 0
        ORDER BY updated_at DESC
    """)
    fun getPinnedNotes(): Flow<List<Note>>
    
    /**
     * 搜索笔记（标题和内容）
     * 使用 LIKE 进行模糊匹配
     */
    @Query("""
        SELECT * FROM notes 
        WHERE is_deleted = 0
        AND (title LIKE '%' || :query || '%' OR content LIKE '%' || :query || '%')
        ORDER BY is_pinned DESC, updated_at DESC
    """)
    fun searchNotes(query: String): Flow<List<Note>>
    
    /**
     * 获取单个笔记
     */
    @Query("SELECT * FROM notes WHERE id = :id")
    suspend fun getNoteById(id: Long): Note?
    
    /**
     * 获取笔记数量
     */
    @Query("SELECT COUNT(*) FROM notes WHERE is_deleted = 0")
    fun getNoteCount(): Flow<Int>
    
    // ===== 插入 =====
    
    /**
     * 插入笔记
     * @Insert 返回新插入行的 ID
     */
    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insertNote(note: Note): Long
    
    /**
     * 插入分类
     */
    @Insert(onConflict = OnConflictStrategy.REPLACE)
    suspend fun insertCategory(category: Category): Long
    
    // ===== 更新 =====
    
    /**
     * 更新笔记
     */
    @Update
    suspend fun updateNote(note: Note)
    
    /**
     * 更新分类
     */
    @Update
    suspend fun updateCategory(category: Category)
    
    /**
     * 移动到归档
     */
    @Query("UPDATE notes SET is_archived = :isArchived, updated_at = :now WHERE id = :noteId")
    suspend fun setArchived(noteId: Long, isArchived: Boolean, now: LocalDateTime = LocalDateTime.now())
    
    /**
     * 软删除（移入回收站）
     */
    @Query("UPDATE notes SET is_deleted = :isDeleted, updated_at = :now WHERE id = :noteId")
    suspend fun setDeleted(noteId: Long, isDeleted: Boolean, now: LocalDateTime = LocalDateTime.now())
    
    /**
     * 置顶/取消置顶
     */
    @Query("UPDATE notes SET is_pinned = :isPinned, updated_at = :now WHERE id = :noteId")
    suspend fun setPinned(noteId: Long, isPinned: Boolean, now: LocalDateTime = LocalDateTime.now())
    
    // ===== 删除 =====
    
    /**
     * 删除笔记（硬删除）
     */
    @Delete
    suspend fun deleteNote(note: Note)
    
    /**
     * 清空回收站
     */
    @Query("DELETE FROM notes WHERE is_deleted = 1")
    suspend fun emptyTrash()
    
    /**
     * 恢复笔记
     */
    @Query("UPDATE notes SET is_deleted = 0, updated_at = :now WHERE id = :noteId")
    suspend fun restoreNote(noteId: Long, now: LocalDateTime = LocalDateTime.now())
    
    // ===== 分类 DAO =====
    
    @Query("SELECT * FROM categories ORDER BY sort_order ASC")
    fun getAllCategories(): Flow<List<Category>>
    
    @Query("SELECT * FROM categories WHERE id = :id")
    suspend fun getCategoryById(id: Long): Category?
    
    @Query("SELECT COUNT(*) FROM notes WHERE category_id = :categoryId AND is_deleted = 0")
    fun getNoteCountByCategory(categoryId: Long): Flow<Int>
    
    @Delete
    suspend fun deleteCategory(category: Category)
}
```

```kotlin
// ===== NotesDatabase.kt =====

package com.example.notes.data.local

import androidx.room.*
import com.example.notes.data.local.entity.*
import java.time.LocalDateTime

/**
 * Room 数据库
 * 
 * @Database 标记数据库类
 * entities: 数据库包含的表
 * version: 数据库版本（用于迁移）
 */
@Database(
    entities = [
        Note::class,
        Category::class,
        NoteTag::class
    ],
    version = 1,
    exportSchema = false  // 是否导出数据库 schema
)
@TypeConverters(Converters::class)  // 类型转换器
abstract class NotesDatabase : RoomDatabase() {
    
    // DAO 抽象方法（Room 会自动生成实现）
    abstract fun noteDao(): NoteDao
    
    companion object {
        const val DATABASE_NAME = "notes_database"
    }
}

/**
 * 类型转换器
 * 
 * Room 只能存储基本类型和 String
 * 需要将其他类型转换为基本类型
 */
class Converters {
    
    /**
     * LocalDateTime <-> Long
     * 存储为时间戳（毫秒）
     */
    @TypeConverter
    fun fromLocalDateTime(value: LocalDateTime?): Long? {
        return value?.atZone(java.time.ZoneId.systemDefault())?.toInstant()?.toEpochMilli()
    }
    
    @TypeConverter
    fun toLocalDateTime(value: Long?): LocalDateTime? {
        return value?.let {
            java.time.Instant.ofEpochMilli(it)
                .atZone(ZoneId.systemDefault())
                .toLocalDateTime()
        }
    }
}
```

```kotlin
package com.example.notes.data.repository

import com.example.notes.data.local.NoteDao
import com.example.notes.data.local.entity.*
import kotlinx.coroutines.flow.Flow
import java.time.LocalDateTime
import javax.inject.Inject
import javax.inject.Singleton

/**
 * 笔记仓库
 * 
 * 单一数据源原则（Single Source of Truth）：
 * 所有笔记数据都通过这个仓库访问
 */
@Singleton  // 单例：全局只有一个实例
class NotesRepository @Inject constructor(
    private val noteDao: NoteDao
) {
    
    // ===== 笔记操作 =====
    
    /**
     * 获取所有笔记
     */
    fun getAllNotes(): Flow<List<Note>> = noteDao.getAllNotes()
    
    /**
     * 获取归档笔记
     */
    fun getArchivedNotes(): Flow<List<Note>> = noteDao.getArchivedNotes()
    
    /**
     * 获取回收站笔记
     */
    fun getDeletedNotes(): Flow<List<Note>> = noteDao.getDeletedNotes()
    
    /**
     * 获取指定分类的笔记
     */
    fun getNotesByCategory(categoryId: Long): Flow<List<Note>> = 
        noteDao.getNotesByCategory(categoryId)
    
    /**
     * 获取置顶笔记
     */
    fun getPinnedNotes(): Flow<List<Note>> = noteDao.getPinnedNotes()
    
    /**
     * 搜索笔记
     */
    fun searchNotes(query: String): Flow<List<Note>> = noteDao.searchNotes(query)
    
    /**
     * 获取单个笔记
     */
    suspend fun getNoteById(id: Long): Note? = noteDao.getNoteById(id)
    
    /**
     * 创建笔记
     */
    suspend fun createNote(
        title: String,
        content: String,
        categoryId: Long? = null,
        color: Int = 0xFFFFFFFF.toInt()
    ): Long {
        val note = Note(
            title = title,
            content = content,
            categoryId = categoryId,
            color = color,
            createdAt = LocalDateTime.now(),
            updatedAt = LocalDateTime.now()
        )
        return noteDao.insertNote(note)
    }
    
    /**
     * 更新笔记
     */
    suspend fun updateNote(note: Note) {
        noteDao.updateNote(note.copy(updatedAt = LocalDateTime.now()))
    }
    
    /**
     * 删除笔记（软删除）
     */
    suspend fun trashNote(noteId: Long) {
        noteDao.setDeleted(noteId, true)
    }
    
    /**
     * 永久删除笔记
     */
    suspend fun permanentlyDeleteNote(note: Note) {
        noteDao.deleteNote(note)
    }
    
    /**
     * 清空回收站
     */
    suspend fun emptyTrash() {
        noteDao.emptyTrash()
    }
    
    /**
     * 恢复笔记
     */
    suspend fun restoreNote(noteId: Long) {
        noteDao.restoreNote(noteId)
    }
    
    /**
     * 归档笔记
     */
    suspend fun archiveNote(noteId: Long) {
        noteDao.setArchived(noteId, true)
    }
    
    /**
     * 取消归档
     */
    suspend fun unarchiveNote(noteId: Long) {
        noteDao.setArchived(noteId, false)
    }
    
    /**
     * 置顶笔记
     */
    suspend fun pinNote(noteId: Long) {
        noteDao.setPinned(noteId, true)
    }
    
    /**
     * 取消置顶
     */
    suspend fun unpinNote(noteId: Long) {
        noteDao.setPinned(noteId, false)
    }
    
    /**
     * 移动笔记到分类
     */
    suspend fun moveNoteToCategory(noteId: Long, categoryId: Long?) {
        val note = noteDao.getNoteById(noteId) ?: return
        noteDao.updateNote(note.copy(categoryId = categoryId, updatedAt = LocalDateTime.now()))
    }
    
    // ===== 分类操作 =====
    
    /**
     * 获取所有分类
     */
    fun getAllCategories(): Flow<List<Category>> = noteDao.getAllCategories()
    
    /**
     * 创建分类
     */
    suspend fun createCategory(name: String, color: Int = 0xFF2196F3.toInt()): Long {
        val category = Category(
            name = name,
            color = color,
            createdAt = LocalDateTime.now()
        )
        return noteDao.insertCategory(category)
    }
    
    /**
     * 更新分类
     */
    suspend fun updateCategory(category: Category) {
        noteDao.updateCategory(category)
    }
    
    /**
     * 删除分类
     */
    suspend fun deleteCategory(category: Category) {
        // 删除分类前，先将笔记移出该分类
        noteDao.deleteCategory(category)
    }
    
    /**
     * 获取分类下的笔记数量
     */
    fun getNoteCountByCategory(categoryId: Long): Flow<Int> = 
        noteDao.getNoteCountByCategory(categoryId)
}
```

```kotlin
// ===== AppModule.kt =====

package com.example.notes.di

import android.content.Context
import androidx.room.Room
import com.example.notes.data.local.NoteDao
import com.example.notes.data.local.NotesDatabase
import com.example.notes.data.repository.NotesRepository
import dagger.Module
import dagger.Provides
import dagger.hilt.InstallIn
import dagger.hilt.android.qualifiers.ApplicationContext
import dagger.hilt.components.SingletonComponent
import javax.inject.Singleton

@Module
@InstallIn(SingletonComponent::class)
object AppModule {
    
    /**
     * 提供数据库实例
     * 
     * Room 方式构建数据库：
     * - databaseBuilder: 创建数据库构建器
     * - .databaseName: 数据库名称
     * - .fallbackToDestructiveMigration: 降级时删除并重建
     */
    @Provides
    @Singleton
    fun provideDatabase(
        @ApplicationContext context: Context
    ): NotesDatabase {
        return Room.databaseBuilder(
            context,
            NotesDatabase::class.java,
            NotesDatabase.DATABASE_NAME
        )
            .fallbackToDestructiveMigration()  // 版本升级时重建
            .build()
    }
    
    /**
     * 提供 DAO
     */
    @Provides
    @Singleton
    fun provideNoteDao(database: NotesDatabase): NoteDao {
        return database.noteDao()
    }
    
    /**
     * 提供 Repository
     */
    @Provides
    @Singleton
    fun provideNotesRepository(noteDao: NoteDao): NotesRepository {
        return NotesRepository(noteDao)
    }
}
```

```kotlin
// ===== HomeViewModel.kt =====

package com.example.notes.ui.screens.home

import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.example.notes.data.local.entity.Note
import com.example.notes.data.local.entity.Category
import com.example.notes.data.repository.NotesRepository
import dagger.hilt.android.lifecycle.HiltViewModel
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
import javax.inject.Inject

@HiltViewModel
class HomeViewModel @Inject constructor(
    private val repository: NotesRepository
) : ViewModel() {
    
    // ===== 搜索 =====
    private val _searchQuery = MutableStateFlow("")
    val searchQuery: StateFlow<String> = _searchQuery.asStateFlow()
    
    // ===== 选中分类 =====
    private val _selectedCategoryId = MutableStateFlow<Long?>(null)
    val selectedCategoryId: StateFlow<Long?> = _selectedCategoryId.asStateFlow()
    
    // ===== 页面类型 =====
    private val _currentPage = MutableStateFlow(PageType.ALL_NOTES)
    val currentPage: StateFlow<PageType> = _currentPage.asStateFlow()
    
    // ===== 笔记列表 =====
    // combine: 组合多个 Flow，当任一 Flow 变化时重新计算
    val notes: StateFlow<List<Note>> = combine(
        _searchQuery,
        _selectedCategoryId,
        _currentPage
    ) { query, categoryId, page ->
        Triple(query, categoryId, page)
    }.flatMapLatest { (query, categoryId, page) ->
        when (page) {
            PageType.ALL_NOTES -> {
                if (query.isNotBlank()) {
                    repository.searchNotes(query)
                } else if (categoryId != null) {
                    repository.getNotesByCategory(categoryId)
                } else {
                    repository.getAllNotes()
                }
            }
            PageType.ARCHIVED -> repository.getArchivedNotes()
            PageType.TRASH -> repository.getDeletedNotes()
        }
    }.stateIn(
        scope = viewModelScope,
        started = SharingStarted.WhileSubscribed(5000),
        initialValue = emptyList()
    )
    
    // ===== 分类列表 =====
    val categories: StateFlow<List<Category>> = repository.getAllCategories()
        .stateIn(
            scope = viewModelScope,
            started = SharingStarted.WhileSubscribed(5000),
            initialValue = emptyList()
        )
    
    // ===== 操作 =====
    
    /**
     * 搜索
     */
    fun search(query: String) {
        _searchQuery.value = query
    }
    
    /**
     * 选择分类
     */
    fun selectCategory(categoryId: Long?) {
        _selectedCategoryId.value = categoryId
    }
    
    /**
     * 切换页面
     */
    fun switchPage(page: PageType) {
        _currentPage.value = page
        // 切换页面时清除筛选
        _selectedCategoryId.value = null
        _searchQuery.value = ""
    }
    
    /**
     * 创建笔记
     */
    fun createNote(title: String, content: String, categoryId: Long?) {
        viewModelScope.launch {
            repository.createNote(title, content, categoryId)
        }
    }
    
    /**
     * 删除笔记（移到回收站）
     */
    fun trashNote(noteId: Long) {
        viewModelScope.launch {
            repository.trashNote(noteId)
        }
    }
    
    /**
     * 删除笔记（永久）
     */
    fun permanentlyDelete(note: Note) {
        viewModelScope.launch {
            repository.permanentlyDeleteNote(note)
        }
    }
    
    /**
     * 恢复笔记
     */
    fun restoreNote(noteId: Long) {
        viewModelScope.launch {
            repository.restoreNote(noteId)
        }
    }
    
    /**
     * 清空回收站
     */
    fun emptyTrash() {
        viewModelScope.launch {
            repository.emptyTrash()
        }
    }
    
    /**
     * 归档笔记
     */
    fun archiveNote(noteId: Long) {
        viewModelScope.launch {
            repository.archiveNote(noteId)
        }
    }
    
    /**
     * 取消归档
     */
    fun unarchiveNote(noteId: Long) {
        viewModelScope.launch {
            repository.unarchiveNote(noteId)
        }
    }
    
    /**
     * 置顶/取消置顶
     */
    fun togglePin(note: Note) {
        viewModelScope.launch {
            if (note.isPinned) {
                repository.unpinNote(note.id)
            } else {
                repository.pinNote(note.id)
            }
        }
    }
}

/**
 * 页面类型
 */
enum class PageType {
    ALL_NOTES,
    ARCHIVED,
    TRASH
}
```

```kotlin
fun HomeScreen(

) {
    val notes by viewModel.notes.collectAsState()
    val categories by viewModel.categories.collectAsState()
    val searchQuery by viewModel.searchQuery.collectAsState()
    val currentPage by viewModel.currentPage.collectAsState()
    
    var showDeleteDialog by remember { mutableStateOf<Note?>(null) }
    var showEmptyTrashDialog by remember { mutableStateOf(false) }
    
    Scaffold(
        topBar = {
            TopAppBar(
                title = {
                    Text(
                        text = when (currentPage) {
                            PageType.ALL_NOTES -> "我的笔记"
                            PageType.ARCHIVED -> "归档"
                            PageType.TRASH -> "回收站"
                        }
                    )
                },
                actions = {
                    // 分类管理
                    if (currentPage == PageType.ALL_NOTES) {
                        IconButton(onClick = onCategoriesClick) {
                            Icon(Icons.Default.Category, contentDescription = "分类")
                        }
                    }
                    
                    // 清空回收站
                    if (currentPage == PageType.TRASH) {
                        IconButton(onClick = { showEmptyTrashDialog = true }) {
                            Icon(Icons.Default.DeleteForever, contentDescription = "清空")
                        }
                    }
                }
            )
        },
        floatingActionButton = {
            if (currentPage == PageType.ALL_NOTES) {
                FloatingActionButton(onClick = onCreateClick) {
                    Icon(Icons.Default.Add, contentDescription = "创建笔记")
                }
            }
        }
    ) { paddingValues ->
        Column(
            modifier = Modifier
                .fillMaxSize()
                .padding(paddingValues)
        ) {
            // 搜索框
            if (currentPage == PageType.ALL_NOTES) {
                SearchBar(
                    query = searchQuery,
                    onQueryChange = { viewModel.search(it) }
                )
            }
            
            // 分类筛选
            if (currentPage == PageType.ALL_NOTES && categories.isNotEmpty()) {
                CategoryChips(
                    categories = categories,
                    selectedCategoryId = viewModel.selectedCategoryId.value,
                    onCategorySelect = { viewModel.selectCategory(it) }
                )
            }
            
            // 页面切换
            PageTabs(
                currentPage = currentPage,
                onPageChange = { viewModel.switchPage(it) }
            )
            
            // 笔记列表
            if (notes.isEmpty()) {
                EmptyState(currentPage = currentPage)
            } else {
                LazyColumn(
                    modifier = Modifier.fillMaxSize(),
                    contentPadding = PaddingValues(16.dp),
                    verticalArrangement = Arrangement.spacedBy(12.dp)
                ) {
                    items(
                        items = notes,
                        key = { it.id }
                    ) { note ->
                        NoteCard(
                            note = note,
                            onClick = { onNoteClick(note.id) },
                            onDelete = { showDeleteDialog = note },
                            onArchive = { viewModel.archiveNote(note.id) },
                            onRestore = { viewModel.restoreNote(note.id) },
                            onPermanentDelete = { showDeleteDialog = note },
                            onTogglePin = { viewModel.togglePin(note) },
                            isTrashPage = currentPage == PageType.TRASH,
                            isArchivedPage = currentPage == PageType.ARCHIVED
                        )
                    }
                }
            }
        }

        showDeleteDialog?.let { note ->
            AlertDialog(
                onDismissRequest = { showDeleteDialog = null },
                title = { Text(if (currentPage == PageType.TRASH) "永久删除" else "移到回收站") },

                        Text("确定")
                    }

                        Text("取消")
                    }
                }
            )
        }

        if (showEmptyTrashDialog) {
            AlertDialog(
                onDismissRequest = { showEmptyTrashDialog = false },
                title = { Text("清空回收站") },
                text = { Text("确定要清空回收站吗？所有笔记将被永久删除，此操作无法撤销。") },

                    }
```

```kotlin
// ===== EditorViewModel.kt =====

package com.example.notes.ui.screens.editor

import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.example.notes.data.local.entity.Category
import com.example.notes.data.local.entity.Note
import com.example.notes.data.repository.NotesRepository
import dagger.hilt.android.lifecycle.HiltViewModel
import kotlinx.coroutines.flow.*
import kotlinx.coroutines.launch
import javax.inject.Inject

@HiltViewModel
class EditorViewModel @Inject constructor(
    private val repository: NotesRepository
) : ViewModel() {
    
    private val _uiState = MutableStateFlow(EditorUiState())
    val uiState: StateFlow<EditorUiState> = _uiState.asStateFlow()
    
    val categories: StateFlow<List<Category>> = repository.getAllCategories()
        .stateIn(
            scope = viewModelScope,
            started = SharingStarted.WhileSubscribed(5000),
            initialValue = emptyList()
        )
    
    private var currentNoteId: Long? = null
    
    /**
     * 加载笔记
     */
    fun loadNote(noteId: Long) {
        viewModelScope.launch {
            val note = repository.getNoteById(noteId)
            note?.let {
                currentNoteId = it.id
                _uiState.value = EditorUiState(
                    title = it.title,
                    content = it.content,
                    color = it.color,
                    categoryId = it.categoryId
                )
            }
        }
    }
    
    /**
     * 更新标题
     */
    fun updateTitle(title: String) {
        _uiState.update { it.copy(title = title) }
    }
    
    /**
     * 更新内容
     */
    fun updateContent(content: String) {
        _uiState.update { it.copy(content = content) }
    }
    
    /**
     * 更新颜色
     */
    fun updateColor(color: Int) {
        _uiState.update { it.copy(color = color) }
    }
    
    /**
     * 更新分类
     */
    fun updateCategory(categoryId: Long?) {
        _uiState.update { it.copy(categoryId = categoryId) }
    }
    
    /**
     * 保存笔记
     */
    fun saveNote() {
        viewModelScope.launch {
            val state = _uiState.value
            
            // 验证
            if (state.title.isBlank() && state.content.isBlank()) {
                _uiState.update { it.copy(isSaved = true) }
                return@launch
            }
            
            if (currentNoteId != null) {
                // 更新
                val existingNote = repository.getNoteById(currentNoteId!!)
                existingNote?.let {
                    repository.updateNote(
                        it.copy(
                            title = state.title,
                            content = state.content,
                            color = state.color,
                            categoryId = state.categoryId
                        )
                    )
                }
            } else {
                // 新建
                repository.createNote(
                    title = state.title,
                    content = state.content,
                    categoryId = state.categoryId,
                    color = state.color
                )
            }
            
            _uiState.update { it.copy(isSaved = true) }
        }
    }
}

/**
 * 编辑器状态
 */
data class EditorUiState(
    val title: String = "",
    val content: String = "",
    val color: Int = 0xFFFFFFFF.toInt(),
    val categoryId: Long? = null,
    val isSaved: Boolean = false
)
```

```kotlin
    val id: Long = 0,
    val title: String,
    val content: String
)

// 定义 DAO
@Dao
interface NoteDao {
    @Query("SELECT * FROM notes")
    fun getAllNotes(): Flow<List<Note>>  // 返回 Flow 实现响应式
    
    @Insert
    suspend fun insert(note: Note): Long  // suspend 支持协程
    
    @Update
    suspend fun update(note: Note)
    
    @Delete
    suspend fun delete(note: Note)
}

// 创建数据库
@Database(entities = [Note::class], version = 1)
abstract class NotesDatabase : RoomDatabase() {
    abstract fun noteDao(): NoteDao
}
```

### 2. StateFlow vs LiveData

**关键点**：两种响应式状态管理方式。

```kotlin
// StateFlow（推荐）
// - Kotlin 原生
// - 支持协程
// - 可以转换
private val _state = MutableStateFlow(initialValue)
val state: StateFlow<T> = _state.asStateFlow()

// LiveData（传统）
// - Android Jetpack 组件
// - 生命周期感知
// - 需要在主线程更新
private val _liveData = MutableLiveData<T>(initialValue)
val liveData: LiveData<T> = _liveData
```

### 3. Kotlin Flow 操作符

```kotlin
// map：转换数据
notes.map { it.title }

// filter：过滤数据
notes.filter { it.isPinned }

// combine：组合多个 Flow
combine(flow1, flow2) { a, b -> a + b }

// flatMapLatest：切换到新的 Flow
query.flatMapLatest { query ->
    if (query.isBlank()) repository.getAllNotes()
    else repository.searchNotes(query)
}

// stateIn：将 Flow 转换为 StateFlow
.flow.stateIn(
    scope = viewModelScope,
    started = SharingStarted.WhileSubscribed(5000),
    initialValue = emptyList()
)
```

```kotlin
// 基础状态
var name by remember { mutableStateOf("") }

var uiState by remember { mutableStateOf(UiState()) }

val notes by viewModel.notes.collectAsState()
```

### 题目2：Room 数据库迁移？

**标准答案**：

```kotlin
// 方式 1：破坏性迁移（开发用）
Room.databaseBuilder()
    .fallbackToDestructiveMigration()
    .build()

// 方式 2：手动迁移
val MIGRATION_1_2 = object : Migration(1, 2) {
    override fun migrate(database: SupportSQLiteDatabase) {
        database.execSQL("ALTER TABLE notes ADD COLUMN tag TEXT")
    }
}

// 方式 3：自动迁移（KSP）
@Database(
    entities = [Note::class],
    version = 2,
    autoMigrations = [
        AutoMigration(from = 1, to = 2)
    ]
)
```

### 题目3：Kotlin by lazy 和 lateinit 的区别？

**标准答案**：

```kotlin
// by lazy：惰性初始化，只读
val config: Config by lazy {
    loadConfig()  // 首次访问时执行
}

// lateinit：延迟初始化，可变
lateinit var database: Database

fun init() {
    database = createDatabase()  // 稍后初始化
}
```
