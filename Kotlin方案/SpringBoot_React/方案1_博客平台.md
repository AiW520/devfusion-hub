# Spring Boot + React 方案一：博客平台

## 项目介绍

### 项目是什么
这是一个功能完整的博客平台，使用 Kotlin + Spring Boot + React 构建。后端提供 RESTful API，前端使用 React 构建单页应用。

### 解决什么问题
- 内容创作和发布
- 文章分类和标签管理
- 用户评论和互动
- 文章搜索和筛选

### 核心技术亮点
| 技术 | 说明 |
|------|------|
| Spring Boot | 企业级 Web 框架 |
| Spring Data JPA | ORM 框架 |
| Spring Security | 安全框架 |
| React | 前端框架 |
| JWT | 身份认证 |
| Docker | 容器化部署 |

### 适合场景
- 全栈开发学习项目
- 个人博客系统
- 内容管理系统

```kotlin
package com.example.blog.model

import jakarta.persistence.*
import java.time.LocalDateTime

/**
 * 文章实体
 * 
 * @Entity 标记为 JPA 实体
 * @Table 指定数据库表名
 */
@Entity
@Table(name = "articles")
data class Article(
    // ===== 主键 =====
    // @Id 标记主键
    // @GeneratedValue 自动生成主键值
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    val id: Long = 0,
    
    // ===== 基础字段 =====
    // @Column 可选，用于指定列的详细信息
    @Column(nullable = false, length = 200)
    val title: String,
    
    @Column(columnDefinition = "TEXT")
    val content: String,
    
    @Column(length = 500)
    val summary: String? = null,
    
    // ===== 关联关系 =====
    // @ManyToOne: 多篇文章属于一个分类
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "category_id")
    val category: Category? = null,
    
    // @ManyToMany: 文章和标签的多对多关系
    @ManyToMany
    @JoinTable(
        name = "article_tags",
        joinColumns = [JoinColumn(name = "article_id")],
        inverseJoinColumns = [JoinColumn(name = "tag_id")]
    )
    val tags: MutableList<Tag> = mutableListOf(),
    
    // ===== 关联用户 =====
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "author_id")
    val author: User,
    
    // ===== 审计字段 =====
    // @Column 上可以添加 default 表达式
    val createdAt: LocalDateTime = LocalDateTime.now(),
    
    val updatedAt: LocalDateTime = LocalDateTime.now(),
    
    // ===== 状态 =====
    val viewCount: Int = 0,
    
    val likeCount: Int = 0,
    
    @Column(length = 20)
    val status: String = "draft"  // draft, published, archived
) {
    // JPA 要求无参构造函数，但我们可以添加 protected 或 private
    constructor() : this(
        id = 0,
        title = "",
        content = "",
        summary = null,
        category = null,
        tags = mutableListOf(),
        author = User(),
        createdAt = LocalDateTime.now(),
        updatedAt = LocalDateTime.now(),
        viewCount = 0,
        likeCount = 0,
        status = "draft"
    )
}

/**
 * 分类实体
 */
@Entity
@Table(name = "categories")
data class Category(
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    val id: Long = 0,

    val description: String? = null,

    val articles: List<Article> = emptyList()

    constructor() : this(id = 0, name = "", description = null)
}

/**
 * 标签实体

@Entity
@Table(name = "tags")

    @GeneratedValue(strategy = GenerationType.IDENTITY)
    val id: Long = 0,
    
    @Column(nullable = false, unique = true, length = 30)
    val name: String,
    
    @ManyToMany(mappedBy = "tags")
    val articles: MutableList<Article> = mutableListOf()

    constructor() : this(id = 0, name = "")
```

```kotlin
package com.example.blog.repository

import com.example.blog.model.*
import org.springframework.data.domain.*
import org.springframework.data.jpa.repository.*
import org.springframework.data.repository.query.Param

/**
 * 文章仓储接口
 * 
 * Spring Data JPA 会根据方法名自动生成 SQL
 * 这种方式叫做 "方法名查询"（Method Name Query）
 */
interface ArticleRepository : JpaRepository<Article, Long> {
    
    /**
     * 分页查询已发布的文章
     * 方法名规则：findBy + 字段名
     */
    fun findByStatus(status: String, pageable: Pageable): Page<Article>
    
    /**
     * 根据分类查询文章
     */
    fun findByCategoryIdAndStatus(categoryId: Long, status: String, pageable: Pageable): Page<Article>
    
    /**
     * 根据作者查询文章
     */
    fun findByAuthorId(authorId: Long, pageable: Pageable): Page<Article>
    
    /**
     * 搜索文章（标题或内容包含关键词）
     */
    @Query("""
        SELECT a FROM Article a 
        WHERE a.status = 'published'
        AND (a.title LIKE %:keyword% OR a.content LIKE %:keyword%)
    """)
    fun searchByKeyword(@Param("keyword") keyword: String, pageable: Pageable): Page<Article>
    
    /**
     * 查询某标签下的文章
     */
    @Query("""
        SELECT a FROM Article a 
        JOIN a.tags t 
        WHERE t.id = :tagId AND a.status = 'published'
    """)
    fun findByTagId(@Param("tagId") tagId: Long, pageable: Pageable): Page<Article>
    
    /**
     * 统计某作者的文章数量
     */
    fun countByAuthorId(authorId: Long): Long
    
    /**
     * 统计某分类的文章数量
     */
    fun countByCategoryId(categoryId: Long): Long
}
```

### 5. ArticleService.kt

```kotlin
package com.example.blog.service

import com.example.blog.model.*
import com.example.blog.repository.*
import com.example.blog.dto.*
import org.springframework.data.domain.*
import org.springframework.stereotype.Service
import org.springframework.transaction.annotation.Transactional

/**
 * 文章服务
 * 
 * @Service 标记为 Spring 服务组件
 * @Transactional 开启事务支持
 */
@Service
class ArticleService(
    private val articleRepository: ArticleRepository,
    private val categoryRepository: CategoryRepository,
    private val tagRepository: TagRepository
) {
    
    /**
     * 创建文章
     */
    @Transactional
    fun createArticle(
        title: String,
        content: String,
        summary: String?,
        categoryId: Long?,
        tagIds: List<Long>,
        author: User,
        status: String = "draft"
    ): Article {
        // 查找分类
        val category = categoryId?.let { categoryRepository.findById(it).orElse(null) }
        
        // 查找标签
        val tags = tagIds.mapNotNull { tagRepository.findById(it).orElse(null) }
        
        // 创建文章
        val article = Article(
            title = title,
            content = content,
            summary = summary ?: content.take(200),
            category = category,
            tags = tags.toMutableList(),
            author = author,
            status = status,
            createdAt = java.time.LocalDateTime.now(),
            updatedAt = java.time.LocalDateTime.now()
        )
        
        return articleRepository.save(article)
    }
    
    /**
     * 更新文章
     */
    @Transactional
    fun updateArticle(
        id: Long,
        title: String?,
        content: String?,
        summary: String?,
        categoryId: Long?,
        tagIds: List<Long>?,
        status: String?
    ): Article? {
        val article = articleRepository.findById(id).orElse(null) ?: return null
        
        // 更新字段
        title?.let { article::title.set(it) }
        content?.let { article::content.set(it) }
        summary?.let { article::summary.set(it) }
        status?.let { article::status.set(it) }
        article::updatedAt.set(java.time.LocalDateTime.now())
        
        // 更新分类
        categoryId?.let {
            val category = categoryRepository.findById(it).orElse(null)
            article::category.set(category)
        }
        
        // 更新标签
        tagIds?.let {
            val tags = it.mapNotNull { id -> tagRepository.findById(id).orElse(null) }
            article.tags.clear()
            article.tags.addAll(tags)
        }
        
        return articleRepository.save(article)
    }
    
    /**
     * 分页获取已发布文章
     */
    fun getPublishedArticles(page: Int, size: Int): Page<ArticleDto> {
        val pageable = PageRequest.of(page, size, Sort.by("createdAt").descending())
        return articleRepository.findByStatus("published", pageable)
            .map { ArticleDto.fromArticle(it) }
    }
    
    /**
     * 根据 ID 获取文章
     */
    fun getArticleById(id: Long): ArticleDto? {
        return articleRepository.findById(id)
            .filter { it.status == "published" }
            .map { ArticleDto.fromArticle(it) }
            .orElse(null)
    }
    
    /**
     * 根据分类获取文章
     */
    fun getArticlesByCategory(categoryId: Long, page: Int, size: Int): Page<ArticleDto> {
        val pageable = PageRequest.of(page, size, Sort.by("createdAt").descending())
        return articleRepository.findByCategoryIdAndStatus(categoryId, "published", pageable)
            .map { ArticleDto.fromArticle(it) }
    }
    
    /**
     * 搜索文章
     */
    fun searchArticles(keyword: String, page: Int, size: Int): Page<ArticleDto> {
        val pageable = PageRequest.of(page, size, Sort.by("createdAt").descending())
        return articleRepository.searchByKeyword(keyword, pageable)
            .map { ArticleDto.fromArticle(it) }
    }
    
    /**
     * 删除文章
     */
    @Transactional
    fun deleteArticle(id: Long): Boolean {
        return if (articleRepository.existsById(id)) {
            articleRepository.deleteById(id)
            true
        } else {
            false
        }
    }
    
    /**
     * 增加浏览数
     */
    @Transactional
    fun incrementViewCount(id: Long) {
        articleRepository.findById(id).ifPresent {
            it.viewCount.let { count -> articleRepository.updateViewCount(id, count + 1) }
        }
    }
}

/**
 * JPA 更新操作（需要 @Modifying）
 */
interface ArticleRepository : JpaRepository<Article, Long> {
    // ... 其他方法
    
    @Modifying
    @Query("UPDATE Article a SET a.viewCount = :count WHERE a.id = :id")
    fun updateViewCount(@Param("id") id: Long, @Param("count") count: Int)
}
```

```kotlin
package com.example.blog.controller

import com.example.blog.model.*
import com.example.blog.service.*
import com.example.blog.dto.*
import org.springframework.data.domain.*
import org.springframework.http.*
import org.springframework.security.core.annotation.AuthenticationPrincipal
import org.springframework.web.bind.annotation.*

/**
 * 文章控制器
 * 
 * @RestController 组合了 @Controller 和 @ResponseBody
 * 所有方法返回值都会自动序列化为 JSON
 */
@RestController
@RequestMapping("/api/articles")
@CrossOrigin(origins = ["*"])  // 允许跨域
class ArticleController(
    private val articleService: ArticleService
) {
    
    /**
     * 获取文章列表
     * 
     * @GetMapping 等同于 @RequestMapping(method = [RequestMethod.GET])
     */
    @GetMapping
    fun getArticles(
        @RequestParam(defaultValue = "0") page: Int,
        @RequestParam(defaultValue = "10") size: Int,
        @RequestParam(required = false) category: Long?,
        @RequestParam(required = false) keyword: String?
    ): ResponseEntity<Page<ArticleDto>> {
        val articles = when {
            keyword != null -> articleService.searchArticles(keyword, page, size)
            category != null -> articleService.getArticlesByCategory(category, page, size)
            else -> articleService.getPublishedArticles(page, size)
        }
        
        return ResponseEntity.ok(articles)
    }
    
    /**
     * 获取单个文章
     */
    @GetMapping("/{id}")
    fun getArticle(@PathVariable id: Long): ResponseEntity<ArticleDto> {
        val article = articleService.getArticleById(id)
            ?: return ResponseEntity.notFound().build()
        
        // 增加浏览数
        articleService.incrementViewCount(id)
        
        return ResponseEntity.ok(article)
    }
    
    /**
     * 创建文章（需要登录）
     */
    @PostMapping
    fun createArticle(
        @RequestBody request: CreateArticleRequest,
        @AuthenticationPrincipal user: User
    ): ResponseEntity<ArticleDto> {
        val article = articleService.createArticle(
            title = request.title,
            content = request.content,
            summary = request.summary,
            categoryId = request.categoryId,
            tagIds = request.tagIds ?: emptyList(),
            author = user,
            status = request.status ?: "draft"
        )
        
        return ResponseEntity.status(HttpStatusCode.Created)
            .body(ArticleDto.fromArticle(article))
    }
    
    /**
     * 更新文章
     */
    @PutMapping("/{id}")
    fun updateArticle(
        @PathVariable id: Long,
        @RequestBody request: UpdateArticleRequest
    ): ResponseEntity<ArticleDto> {
        val article = articleService.updateArticle(
            id = id,
            title = request.title,
            content = request.content,
            summary = request.summary,
            categoryId = request.categoryId,
            tagIds = request.tagIds,
            status = request.status
        ) ?: return ResponseEntity.notFound().build()
        
        return ResponseEntity.ok(ArticleDto.fromArticle(article))
    }
    
    /**
     * 删除文章
     */
    @DeleteMapping("/{id}")
    fun deleteArticle(@PathVariable id: Long): ResponseEntity<Unit> {
        val deleted = articleService.deleteArticle(id)
        return if (deleted) {
            ResponseEntity.noContent().build()
        } else {
            ResponseEntity.notFound().build()
        }
    }
}
```

```kotlin
package com.example.blog.config

import com.example.blog.security.*
import org.springframework.context.annotation.*
import org.springframework.security.authentication.*
import org.springframework.security.config.annotation.web.builders.HttpSecurity
import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity
import org.springframework.security.config.http.SessionCreationPolicy
import org.springframework.security.crypto.bcrypt.*
import org.springframework.security.crypto.password.PasswordEncoder
import org.springframework.security.web.authentication.UsernamePasswordAuthenticationFilter
import org.springframework.web.cors.*

/**
 * Spring Security 配置
 */
@Configuration
@EnableWebSecurity  // 启用 Spring Security
class SecurityConfig(
    private val jwtTokenProvider: JwtTokenProvider,
    private val jwtAuthenticationFilter: JwtAuthenticationFilter
) {
    
    /**
     * 配置安全规则
     */
    @Bean
    fun securityFilterChain(http: HttpSecurity): HttpSecurity {
        return http
            // ===== CORS 配置 =====
            .cors { it.configurationSource(corsConfigurationSource()) }
            
            // ===== CSRF 禁用 =====
            // 因为使用 JWT，不需要 CSRF 保护
            .csrf { it.disable() }
            
            // ===== 授权规则 =====
            .authorizeHttpRequests { auth ->
                // 公开访问的路径
                auth.requestMatchers(
                    "/api/auth/**",           // 认证相关
                    "/api/articles",           // 文章列表
                    "/api/articles/*",         // 文章详情
                    "/api/categories",         // 分类列表
                    "/api/tags",               // 标签列表
                    "/health"                  // 健康检查
                ).permitAll()
                
                // 需要登录才能访问
                auth.requestMatchers(
                    "/api/articles/create",
                    "/api/articles/*/edit",
                    "/api/my/**"
                ).authenticated()
                
                // 其他请求需要认证
                auth.anyRequest().authenticated()
            }
            
            // ===== JWT 过滤器 =====
            .addFilterBefore(jwtAuthenticationFilter, UsernamePasswordAuthenticationFilter::class.java)
            
            // ===== 会话管理 =====
            // STATELESS: 不创建 HTTP Session，完全依赖 JWT
            .sessionManagement { it.sessionCreationPolicy(SessionCreationPolicy.STATELESS) }
    }
    
    /**
     * 密码编码器
     */
    @Bean
    fun passwordEncoder(): PasswordEncoder {
        // BCryptPasswordEncoder: 使用 BCrypt 算法
        return BCryptPasswordEncoder(12)  // 强度 12
    }
    
    /**
     * CORS 配置
     */
    @Bean
    fun corsConfigurationSource(): CorsConfigurationSource {
        val configuration = CorsConfiguration().apply {
            // 允许的来源
            allowedOrigins = listOf("http://localhost:3000")
            
            // 允许的方法
            allowedMethods = listOf("GET", "POST", "PUT", "DELETE", "OPTIONS")
            
            // 允许的头部
            allowedHeaders = listOf("*")
            
            // 允许携带凭证
            allowCredentials = true
            
            // 预检请求缓存时间
            maxAge = 3600L
        }
        
        return UrlBasedCorsConfigurationSource().apply {
            registerCorsConfiguration("/**", configuration)
        }
    }
}
```

### 8. JwtTokenProvider.kt

```kotlin
package com.example.blog.security

import io.jsonwebtoken.*
import io.jsonwebtoken.security.Keys
import org.springframework.beans.factory.annotation.Value
import org.springframework.security.core.Authentication
import org.springframework.security.core.userdetails.UserDetails
import org.springframework.stereotype.Component
import java.util.*
import javax.crypto.SecretKey

/**
 * JWT Token 提供者
 * 
 * 负责：
 * - 生成 JWT Token
 * - 验证 Token
 * - 从 Token 提取用户信息
 */
@Component
class JwtTokenProvider {
    
    // ===== 配置 =====
    @Value("\${jwt.secret}")
    private lateinit var jwtSecret: String
    
    @Value("\${jwt.expiration}")
    private var jwtExpiration: Long = 86400000  // 24 小时
    
    /**
     * 获取签名密钥
     */
    private fun getSigningKey(): SecretKey {
        return Keys.hmacShaKeyFor(jwtSecret.toByteArray())
    }
    
    /**
     * 生成 Token
     */
    fun generateToken(authentication: Authentication): String {
        val userPrincipal = authentication.principal as UserDetails
        val now = Date()
        val expiryDate = Date(now.time + jwtExpiration)
        
        return Jwts.builder()
            .subject(userPrincipal.username)  // 用户名作为主题
            .issuedAt(now)
            .expiration(expiryDate)
            .signWith(getSigningKey())  // 签名
            .compact()
    }
    
    /**
     * 从 Token 获取用户名
     */
    fun getUsernameFromToken(token: String): String {
        val claims = parseToken(token)
        return claims.payload.subject
    }
    
    /**
     * 验证 Token
     */
    fun validateToken(token: String): Boolean {
        return try {
            parseToken(token)
            true
        } catch (e: Exception) {
            false
        }
    }
    
    /**
     * 解析 Token
     */
    private fun parseToken(token: String): Jws<Claims> {
        return Jwts.parser()
            .verifyWith(getSigningKey())
            .build()
            .parseSignedClaims(token)
    }
}
```

```kotlin
package com.example.blog.security

import jakarta.servlet.FilterChain
import jakarta.servlet.http.*
import org.springframework.security.authentication.UsernamePasswordAuthenticationToken
import org.springframework.security.core.context.*
import org.springframework.security.core.userdetails.*
import org.springframework.security.core.authority.SimpleGrantedAuthority
import org.springframework.security.web.authentication.WebAuthenticationDetailsSource
import org.springframework.stereotype.Component
import org.springframework.web.filter.OncePerRequestFilter

/**
 * JWT 认证过滤器
 * 
 * 继承 OncePerRequestFilter：
 * 确保每个请求只执行一次
 */
@Component
class JwtAuthenticationFilter(
    private val jwtTokenProvider: JwtTokenProvider,
    private val userDetailsService: UserDetailsService  // Spring Security 的用户服务
) : OncePerRequestFilter() {
    
    override fun doFilterInternal(
        request: HttpServletRequest,
        response: HttpServletResponse,
        filterChain: FilterChain
    ) {
        try {
            // ===== 1. 从请求头获取 Token =====
            val jwt = getJwtFromRequest(request)
            
            if (jwt != null && jwtTokenProvider.validateToken(jwt)) {
                // ===== 2. 从 Token 获取用户名 =====
                val username = jwtTokenProvider.getUsernameFromToken(jwt)
                
                // ===== 3. 加载用户详情 =====
                val userDetails = userDetailsService.loadUserByUsername(username)
                
                // ===== 4. 创建认证对象 =====
                val authentication = UsernamePasswordAuthenticationToken(
                    userDetails,
                    null,
                    userDetails.authorities
                )
                
                // ===== 5. 设置认证详情 =====
                authentication.details = WebAuthenticationDetailsSource().buildDetails(request)
                
                // ===== 6. 设置安全上下文 =====
                SecurityContextHolder.getContext().authentication = authentication
            }
        } catch (e: Exception) {
            logger.error("Could not set user authentication in security context", e)
        }
        
        // ===== 7. 继续过滤器链 =====
        filterChain.doFilter(request, response)
    }
    
    /**
     * 从请求中提取 JWT Token
     * 通常在 Authorization 头中：Bearer <token>
     */
    private fun getJwtFromRequest(request: HttpServletRequest): String? {
        val bearerToken = request.getHeader("Authorization")
        
        // 检查是否以 "Bearer " 开头
        if (bearerToken != null && bearerToken.startsWith("Bearer ")) {
            return bearerToken.substring(7)  // 去掉 "Bearer " 前缀
        }
        
        return null
    }
}
```

```typescript
import React, { useState, useEffect } from 'react';
import { Link } from 'react-router-dom';

// 文章列表组件
const ArticleList: React.FC = () => {
  const [articles, setArticles] = useState<Article[]>([]);
  const [loading, setLoading] = useState(true);
  const [page, setPage] = useState(0);
  const [totalPages, setTotalPages] = useState(0);
  const [keyword, setKeyword] = useState('');

  // 获取文章列表
  useEffect(() => {
    fetchArticles();
  }, [page]);

  const fetchArticles = async () => {
    try {
      setLoading(true);
      const response = await articleApi.getArticles({
        page,
        size: 10,
        keyword: keyword || undefined,
      });
      setArticles(response.data.content);
      setTotalPages(response.data.totalPages);
    } catch (error) {
      console.error('Failed to fetch articles:', error);
    } finally {
      setLoading(false);
    }
  };

  // 搜索
  const handleSearch = (e: React.FormEvent) => {
    e.preventDefault();
    setPage(0);
    fetchArticles();
  };

  return (
    <div className="article-list">
      {/* 搜索栏 */}
      <form onSubmit={handleSearch} className="search-form">
        <input
          type="text"
          placeholder="搜索文章..."
          value={keyword}
          onChange={(e) => setKeyword(e.target.value)}
        />
        <button type="submit">搜索</button>
      </form>

      {/* 文章列表 */}
      {loading ? (
        <div className="loading">加载中...</div>
      ) : (
        <div className="articles">
          {articles.map((article) => (
            <article key={article.id} className="article-card">

                <span className="category-tag">{article.category.name}</span>

            </article>
```

### 题目1：Spring Boot 和 Spring Framework 有什么区别？

**标准答案**：

| 特性 | Spring Framework | Spring Boot |
|------|------------------|-------------|
| 配置 | 手动 XML 配置 | 自动配置 |
| 依赖管理 | 手动管理版本 | 起步依赖（Starter） |
| 服务器 | 需要手动配置 | 嵌入式服务器 |
| 监控 | 无 | Spring Actuator |
| 微服务 | 无内置支持 | 原生支持 |

**核心优势**：约定优于配置，自动配置。
