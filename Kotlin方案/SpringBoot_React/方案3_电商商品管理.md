# Spring Boot + React 方案三：电商商品管理系统

## 项目介绍

### 项目是什么
这是一个电商商品管理系统，包含商品管理、分类管理、库存管理、订单管理等功能。使用 Kotlin + Spring Boot + React + PostgreSQL 构建。

### 解决什么问题
- 商品的增删改查
- 商品分类管理
- 库存管理
- 订单处理

### 核心技术亮点
| 技术 | 说明 |
|------|------|
| Spring Boot | 后端框架 |
| Spring Data JPA | ORM 框架 |
| Spring Security | 安全框架 |
| React + Redux | 前端状态管理 |
| 分页和搜索 | 数据查询优化 |

### 适合场景
- 电商后台管理系统开发
- 全栈开发学习
- 企业级应用开发

### 1. Product.kt（商品实体）

```kotlin
package com.example.ecommerce.model

import jakarta.persistence.*
import java.math.BigDecimal
import java.time.LocalDateTime

/**
 * 商品实体
 * 
 * 设计说明：
 * - 商品属于一个分类
 * - 商品有多个图片
 * - 使用 BigDecimal 处理价格（避免浮点精度问题）
 */
@Entity
@Table(name = "products")
data class Product(
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    val id: Long = 0,
    
    // ===== 基本信息 =====
    @Column(nullable = false, length = 200)
    val name: String,
    
    @Column(columnDefinition = "TEXT")
    val description: String? = null,
    
    // ===== 价格相关 =====
    // @Column 精度设置：总长度15位，小数2位
    @Column(precision = 15, scale = 2)
    val price: BigDecimal,
    
    // 原价（用于显示折扣）
    @Column(precision = 15, scale = 2)
    val originalPrice: BigDecimal? = null,
    
    // ===== 库存 =====
    val stock: Int = 0,                    // 库存数量
    val lowStockThreshold: Int = 10,       // 低库存警告阈值
    
    // ===== 分类关联 =====
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "category_id")
    val category: Category? = null,
    
    // ===== 商品图片 =====
    @ElementCollection
    @CollectionTable(name = "product_images", joinColumns = [JoinColumn(name = "product_id")])
    @Column(name = "image_url", length = 500)
    val images: MutableList<String> = mutableListOf(),
    
    // ===== 状态 =====
    @Column(length = 20)
    val status: String = "active",  // active, inactive, deleted
    
    // ===== SEO 相关 =====
    @Column(length = 100)
    val sku: String? = null,        // 商品编码
    @Column(length = 200)
    val brand: String? = null,      // 品牌
    
    // ===== 评分 =====
    val rating: Double = 0.0,     // 平均评分
    val reviewCount: Int = 0,      // 评价数量
    
    // ===== 审计字段 =====
    val createdAt: LocalDateTime = LocalDateTime.now(),
    val updatedAt: LocalDateTime = LocalDateTime.now()
) {
    constructor() : this(
        id = 0, name = "", description = null,
        price = BigDecimal.ZERO, originalPrice = null,
        stock = 0, lowStockThreshold = 10,
        category = null, images = mutableListOf(),
        status = "active", sku = null, brand = null,
        rating = 0.0, reviewCount = 0,
        createdAt = LocalDateTime.now(), updatedAt = LocalDateTime.now()
    )
    
    /**
     * 计算折扣百分比
     */
    fun getDiscountPercentage(): Int? {
        if (originalPrice == null || originalPrice <= BigDecimal.ZERO) return null
        if (price >= originalPrice) return null
        val discount = (originalPrice - price) * BigDecimal(100) / originalPrice
        return discount.toInt()
    }
    
    /**
     * 是否低库存
     */
    fun isLowStock(): Boolean = stock <= lowStockThreshold
    
    /**
     * 是否在售
     */
    fun isActive(): Boolean = status == "active" && stock > 0
}

/**
 * 分类实体
 */
@Entity
@Table(name = "categories")
data class Category(
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    val id: Long = 0,
    
    @Column(nullable = false, length = 50)
    val name: String,
    
    @Column(length = 200)
    val description: String? = null,
    
    @Column(length = 100)
    val icon: String? = null,  // 图标名称或 URL
    
    // 父分类（用于多级分类）
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "parent_id")
    val parent: Category? = null,
    
    // 子分类
    @OneToMany(mappedBy = "parent")
    val children: List<Category> = emptyList(),
    
    // 排序
    val sortOrder: Int = 0,
    
    // 是否启用
    val isActive: Boolean = true,
    
    // 该分类下的商品
    @OneToMany(mappedBy = "category")
    val products: List<Product> = emptyList()
) {
    constructor() : this(
        id = 0, name = "", description = null, icon = null,
        parent = null, children = emptyList(), sortOrder = 0,
        isActive = true, products = emptyList()
    )
}
```

```kotlin
    val discountAmount: BigDecimal = BigDecimal.ZERO,
```

```kotlin
package com.example.ecommerce.repository

import com.example.ecommerce.model.*
import org.springframework.data.domain.*
import org.springframework.data.jpa.repository.*
import org.springframework.data.repository.query.Param
import java.math.BigDecimal

/**
 * 商品仓储接口
 */
interface ProductRepository : JpaRepository<Product, Long> {
    
    /**
     * 分页查询商品
     */
    fun findByStatus(status: String, pageable: Pageable): Page<Product>
    
    /**
     * 根据分类查询商品
     */
    fun findByCategoryIdAndStatus(
        categoryId: Long, 
        status: String, 
        pageable: Pageable
    ): Page<Product>
    
    /**
     * 搜索商品（名称或描述）
     */
    @Query("""
        SELECT p FROM Product p 
        WHERE p.status = 'active'
        AND (p.name LIKE %:keyword% OR p.description LIKE %:keyword%)
        ORDER BY p.createdAt DESC
    """)
    fun searchByKeyword(
        @Param("keyword") keyword: String, 
        pageable: Pageable
    ): Page<Product>
    
    /**
     * 价格区间查询
     */
    @Query("""
        SELECT p FROM Product p 
        WHERE p.status = 'active'
        AND p.price BETWEEN :minPrice AND :maxPrice
    """)
    fun findByPriceRange(
        @Param("minPrice") minPrice: BigDecimal,
        @Param("maxPrice") maxPrice: BigDecimal,
        pageable: Pageable
    ): Page<Product>
    
    /**
     * 查询低库存商品
     */
    @Query("""
        SELECT p FROM Product p 
        WHERE p.stock <= p.lowStockThreshold
        AND p.status = 'active'
        ORDER BY p.stock ASC
    """)
    fun findLowStockProducts(pageable: Pageable): Page<Product>
    
    /**
     * 查询某分类下商品数量
     */
    fun countByCategoryId(categoryId: Long): Long
    
    /**
     * 根据 SKU 查询
     */
    fun findBySku(sku: String): Product?
}
```

### 4. ProductService.kt

```kotlin
package com.example.ecommerce.service

import com.example.ecommerce.model.*
import com.example.ecommerce.repository.*
import com.example.ecommerce.dto.*
import org.springframework.data.domain.*
import org.springframework.stereotype.Service
import org.springframework.transaction.annotation.Transactional
import java.math.BigDecimal

/**
 * 商品服务
 */
@Service
class ProductService(
    private val productRepository: ProductRepository,
    private val categoryRepository: CategoryRepository
) {
    
    /**
     * 创建商品
     */
    @Transactional
    fun createProduct(request: CreateProductRequest): ProductDto {
        val category = request.categoryId?.let {
            categoryRepository.findById(it).orElse(null)
        }
        
        val product = Product(
            name = request.name,
            description = request.description,
            price = request.price,
            originalPrice = request.originalPrice,
            stock = request.stock ?: 0,
            lowStockThreshold = request.lowStockThreshold ?: 10,
            category = category,
            images = request.images?.toMutableList() ?: mutableListOf(),
            sku = request.sku,
            brand = request.brand,
            status = request.status ?: "active",
            createdAt = java.time.LocalDateTime.now(),
            updatedAt = java.time.LocalDateTime.now()
        )
        
        val saved = productRepository.save(product)
        return ProductDto.fromProduct(saved)
    }
    
    /**
     * 更新商品
     */
    @Transactional
    fun updateProduct(id: Long, request: UpdateProductRequest): ProductDto? {
        val product = productRepository.findById(id).orElse(null) ?: return null
        
        // 更新字段
        request.name?.let { product::name.set(it) }
        request.description?.let { product::description.set(it) }
        request.price?.let { product::price.set(it) }
        request.originalPrice?.let { product::originalPrice.set(it) }
        request.stock?.let { product::stock.set(it) }
        request.lowStockThreshold?.let { product::lowStockThreshold.set(it) }
        request.status?.let { product::status.set(it) }
        request.sku?.let { product::sku.set(it) }
        request.brand?.let { product::brand.set(it) }
        product::updatedAt.set(java.time.LocalDateTime.now())
        
        // 更新分类
        request.categoryId?.let {
            val category = categoryRepository.findById(it).orElse(null)
            product::category.set(category)
        }
        
        // 更新图片
        request.images?.let {
            product.images.clear()
            product.images.addAll(it)
        }
        
        val saved = productRepository.save(product)
        return ProductDto.fromProduct(saved)
    }
    
    /**
     * 获取商品详情
     */
    fun getProduct(id: Long): ProductDto? {
        return productRepository.findById(id)
            .map { ProductDto.fromProduct(it) }
            .orElse(null)
    }
    
    /**
     * 分页获取商品列表
     */
    fun getProducts(
        page: Int,
        size: Int,
        categoryId: Long?,
        keyword: String?,
        minPrice: BigDecimal?,
        maxPrice: BigDecimal?,
        sortBy: String?,
        sortOrder: String?
    ): Page<ProductDto> {
        val pageable = createPageable(page, size, sortBy, sortOrder)
        
        val products = when {
            // 分类筛选
            categoryId != null -> 
                productRepository.findByCategoryIdAndStatus(categoryId, "active", pageable)
            
            // 价格区间筛选
            minPrice != null && maxPrice != null ->
                productRepository.findByPriceRange(minPrice, maxPrice, pageable)
            
            // 关键词搜索
            !keyword.isNullOrBlank() ->
                productRepository.searchByKeyword(keyword, pageable)
            
            // 默认：获取所有商品
            else ->
                productRepository.findByStatus("active", pageable)
        }
        
        return products.map { ProductDto.fromProduct(it) }
    }
    
    /**
     * 获取低库存商品
     */
    fun getLowStockProducts(page: Int, size: Int): Page<ProductDto> {
        val pageable = PageRequest.of(page, size)
        return productRepository.findLowStockProducts(pageable)
            .map { ProductDto.fromProduct(it) }
    }
    
    /**
     * 更新库存
     */
    @Transactional
    fun updateStock(id: Long, quantity: Int): Boolean {
        val product = productRepository.findById(id).orElse(null) ?: return false
        product.stock = product.stock + quantity
        if (product.stock < 0) {
            product.stock = 0
        }
        productRepository.save(product)
        return true
    }
    
    /**
     * 删除商品（软删除）
     */
    @Transactional
    fun deleteProduct(id: Long): Boolean {
        val product = productRepository.findById(id).orElse(null) ?: return false
        product.status = "deleted"
        product.updatedAt = java.time.LocalDateTime.now()
        productRepository.save(product)
        return true
    }
    
    // ==================== 私有方法 ====================
    
    private fun createPageable(
        page: Int,
        size: Int,
        sortBy: String?,
        sortOrder: String?
    ): Pageable {
        val actualSortBy = when (sortBy) {
            "price" -> "price"
            "stock" -> "stock"
            "rating" -> "rating"
            "sales" -> "reviewCount"
            else -> "createdAt"
        }
        
        val direction = if (sortOrder?.lowercase() == "asc") 
            Sort.Direction.ASC else Sort.Direction.DESC
        
        return PageRequest.of(page, size, Sort.by(direction, actualSortBy))
    }
}
```

```kotlin
package com.example.ecommerce.controller

import com.example.ecommerce.service.*
import com.example.ecommerce.dto.*
import org.springframework.data.domain.*
import org.springframework.http.*
import org.springframework.web.bind.annotation.*
import java.math.BigDecimal

/**
 * 商品控制器
 */
@RestController
@RequestMapping("/api/products")
@CrossOrigin(origins = ["*"])
class ProductController(
    private val productService: ProductService
) {
    
    /**
     * 创建商品
     */
    @PostMapping
    fun createProduct(@RequestBody request: CreateProductRequest): ResponseEntity<ProductDto> {
        val product = productService.createProduct(request)
        return ResponseEntity.status(HttpStatusCode.Created).body(product)
    }
    
    /**
     * 获取商品列表（管理后台）
     */
    @GetMapping
    fun getProducts(
        @RequestParam(defaultValue = "0") page: Int,
        @RequestParam(defaultValue = "10") size: Int,
        @RequestParam(required = false) categoryId: Long?,
        @RequestParam(required = false) keyword: String?,
        @RequestParam(required = false) minPrice: BigDecimal?,
        @RequestParam(required = false) maxPrice: BigDecimal?,
        @RequestParam(required = false) sortBy: String?,
        @RequestParam(required = false) sortOrder: String?
    ): ResponseEntity<Page<ProductDto>> {
        val products = productService.getProducts(
            page, size, categoryId, keyword, minPrice, maxPrice, sortBy, sortOrder
        )
        return ResponseEntity.ok(products)
    }
    
    /**
     * 获取商品详情
     */
    @GetMapping("/{id}")
    fun getProduct(@PathVariable id: Long): ResponseEntity<ProductDto> {
        val product = productService.getProduct(id)
            ?: return ResponseEntity.notFound().build()
        return ResponseEntity.ok(product)
    }
    
    /**
     * 更新商品
     */
    @PutMapping("/{id}")
    fun updateProduct(
        @PathVariable id: Long,
        @RequestBody request: UpdateProductRequest
    ): ResponseEntity<ProductDto> {
        val product = productService.updateProduct(id, request)
            ?: return ResponseEntity.notFound().build()
        return ResponseEntity.ok(product)
    }
    
    /**
     * 删除商品（软删除）
     */
    @DeleteMapping("/{id}")
    fun deleteProduct(@PathVariable id: Long): ResponseEntity<Unit> {
        val success = productService.deleteProduct(id)
        return if (success) {
            ResponseEntity.noContent().build()
        } else {
            ResponseEntity.notFound().build()
        }
    }
    
    /**
     * 获取低库存商品
     */
    @GetMapping("/low-stock")
    fun getLowStockProducts(
        @RequestParam(defaultValue = "0") page: Int,
        @RequestParam(defaultValue = "10") size: Int
    ): ResponseEntity<Page<ProductDto>> {
        val products = productService.getLowStockProducts(page, size)
        return ResponseEntity.ok(products)
    }
    
    /**
     * 批量更新库存
     */
    @PutMapping("/stock")
    fun updateStock(
        @RequestBody request: UpdateStockRequest
    ): ResponseEntity<Map<String, Boolean>> {
        val success = productService.updateStock(request.productId, request.quantity)
        return ResponseEntity.ok(mapOf("success" to success))
    }
}
```

```typescript
    fetchProducts();
  }, [page, filters]);

  const fetchProducts = async () => {
    try {
      setLoading(true);
      const response = await productApi.getProducts({
        page,
        size: 10,
        ...filters
      });
      
      setProducts(response.data.content);
      setTotalPages(response.data.totalPages);
      setTotalElements(response.data.totalElements);
    } catch (error) {
      console.error('Failed to fetch products:', error);
    } finally {
      setLoading(false);
    }
  };

  // 搜索
  const handleSearch = (e: React.FormEvent) => {
    e.preventDefault();
    setPage(0);
    fetchProducts();
  };

  // 删除商品
  const handleDelete = async (id: number) => {
    if (!confirm('确定要删除这个商品吗？')) return;
    
    try {
      await productApi.delete(id);
      fetchProducts();
    } catch (error) {
      alert('删除失败');
    }
  };

  // 更新库存
  const handleUpdateStock = async (id: number, quantity: number) => {
    try {
      await productApi.updateStock(id, quantity);
      fetchProducts();
    } catch (error) {
      alert('更新失败');
    }
  };

  return (
    <div className="product-management">
      <div className="header">
        <h2>商品管理</h2>
        <button onClick={() => navigate('/products/new')}>
          添加商品
        </button>
      </div>

      {/* 筛选器 */}
      <div className="filters">
        <form onSubmit={handleSearch}>
          <input
            type="text"
            placeholder="搜索商品名称..."
            value={filters.keyword}
            onChange={(e) => setFilters({ ...filters, keyword: e.target.value })}
          />
          <button type="submit">搜索</button>
        </form>
        
        <select
          value={filters.sortBy}
          onChange={(e) => setFilters({ ...filters, sortBy: e.target.value })}
        >
          <option value="createdAt">按时间</option>
          <option value="price">按价格</option>
          <option value="stock">按库存</option>
        </select>
      </div>

      {/* 商品表格 */}
      <table className="product-table">
        <thead>
          <tr>
            <th>ID</th>
            <th>商品名称</th>
            <th>分类</th>
            <th>价格</th>
            <th>库存</th>
            <th>状态</th>
            <th>操作</th>
          </tr>
        </thead>
        <tbody>
          {products.map((product) => (
            <tr key={product.id}>
              <td>{product.id}</td>
              <td>{product.name}</td>
              <td>{product.category?.name || '-'}</td>
              <td>¥{product.price.toFixed(2)}</td>
              <td className={product.isLowStock ? 'low-stock' : ''}>
                {product.stock}
                {product.isLowStock && <span className="warning">⚠️</span>}
              </td>
              <td>
                <span className={`status status-${product.status}`}>
                  {product.status === 'active' ? '在售' : '下架'}
                </span>
              </td>
              <td>
                <button onClick={() => navigate(`/products/${product.id}/edit`)}>
                  编辑
                </button>
                <button onClick={() => handleDelete(product.id)}>
                  删除
                </button>
              </td>
            </tr>
          ))}
        </tbody>
      </table>

      {/* 分页 */}
      <div className="pagination">
        <span>共 {totalElements} 条</span>
        <button disabled={page === 0} onClick={() => setPage(page - 1)}>
          上一页
        </button>
        <span>{page + 1} / {totalPages}</span>
        <button disabled={page >= totalPages - 1} onClick={() => setPage(page + 1)}>
          下一页
        </button>
      </div>
    </div>
  );
};

export default ProductTable;
```

## 代码关键点说明

### 1. BigDecimal 处理价格

**关键点**：价格必须使用 BigDecimal，避免浮点精度问题。

```kotlin
// ❌ 错误：使用 Double
val price: Double = 19.99  // 可能出现精度问题

// ✅ 正确：使用 BigDecimal
val price: BigDecimal = BigDecimal("19.99")

// 创建 BigDecimal 的方式
BigDecimal("19.99")     // 推荐：字符串
BigDecimal.valueOf(19.99)  // 也可以
BigDecimal(19.99)       // ❌ 不推荐：构造函数可能有问题

// 运算
val total = price.multiply(BigDecimal(quantity))
val discount = price.multiply(BigDecimal("0.9"))
```

### 2. JPA 实体状态管理

**关键点**：理解 JPA 实体的状态。

```kotlin
// 实体状态：
// 1. Transient（瞬时）：新建对象，未被持久化
// 2. Managed（托管）：被 EntityManager 管理，修改会自动同步
// 3. Detached（分离）：与 EntityManager 分离，修改不会自动同步
// 4. Removed（删除）：标记为删除

// 修改托管实体
val product = productRepository.findById(1).orElse(null)
product.price = BigDecimal("29.99")  // 自动同步到数据库
productRepository.save(product)      // 可选，但推荐

// 分离实体需要手动合并
val product = Product(...)  // detached
product.price = BigDecimal("39.99")
productRepository.save(product)  // 合并到数据库
```

### 3. 软删除实现

**关键点**：不真正删除数据，只标记状态。

```kotlin
// 软删除
product.status = "deleted"
product.updatedAt = LocalDateTime.now()
productRepository.save(product)

// 查询时过滤已删除
@Query("""
    SELECT p FROM Product p 
    WHERE p.status != 'deleted'
""")
fun findActiveProducts(...): Page<Product>

// 或者使用稀疏列
@Column(nullable = false)
val deleted: Boolean = false

// 视图层过滤
fun findActiveProducts(): List<Product> {
    return products.filter { !it.deleted }
}
```

## 面试要点

### 题目1：Kotlin 的内联函数（inline）是什么？

**标准答案**：

内联函数用于减少函数调用的开销。

```kotlin
// 内联函数
inline fun measureTime(block: () -> Unit) {
    val start = System.currentTimeMillis()
    block()
    println("Elapsed: ${System.currentTimeMillis() - start}ms")
}

// 使用
measureTime {
    // 这里代码会被内联到调用处
    doSomething()
}

// 编译后等效于
val start = System.currentTimeMillis()
doSomething()
println("Elapsed: ${System.currentTimeMillis() - start}ms")
```

**何时使用**：
- 函数参数是 lambda
- lambda 被频繁调用
- 函数体比较简单

### 题目2：Kotlin 的解构声明

**标准答案**：

```kotlin
// 解构声明 - 将对象拆分为多个变量
data class Person(val name: String, val age: Int, val city: String)

val person = Person("John", 30, "Beijing")

// 解构
val (name, age, city) = person
println("$name, $age years old, lives in $city")

// Map 解构
val map = mapOf("a" to 1, "b" to 2)
for ((key, value) in map) {
    println("$key = $value")
}

// 忽略某些值
val (_, _, city2) = person

// 返回多个值
fun getUserInfo(): Triple<String, Int, String> {
    return Triple("John", 30, "Beijing")
}
val (name, age, city) = getUserInfo()
```

### 题目3：Kotlin 的 lateinit 和 by lazy 的区别

**标准答案**：

| 特性 | lateinit | by lazy |
|------|----------|---------|
| 适用对象 | var（可变） | val（不可变） |
| 初始化时机 | 代码块中手动赋值 | 首次访问时自动 |
| 线程安全 | 否 | 是 |
| 可空性 | 必须是明确非空 | 可以推断类型 |
| 失败处理 | UninitializedPropertyAccessException | - |

```kotlin
// lateinit - 稍后初始化
class Service {
    lateinit var database: Database
    
    fun initialize() {
        database = connect()
    }
    
    fun doSomething() {
        // 确保已初始化
        if (::database.isInitialized) {
            database.query()
        }
    }
}

// by lazy - 惰性初始化
class Service {
    // 默认线程安全
    val config: Config by lazy {
        loadConfig()
    }
    
    // 非线程安全版本
    val cache: Map<String, String> by lazy(LazyThreadSafetyMode.NONE) {
        buildCache()
    }
}
```
