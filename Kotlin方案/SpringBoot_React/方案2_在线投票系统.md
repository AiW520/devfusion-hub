# Spring Boot + React 方案二：在线投票系统

## 项目介绍

### 项目是什么
这是一个实时在线投票系统，支持创建投票、参与投票、实时统计、图表展示等功能。使用 Kotlin + Spring Boot + WebSocket + React 构建。

### 解决什么问题
- 收集用户意见和偏好
- 实时统计投票结果
- 多选项投票支持
- 结果可视化展示

### 核心技术亮点
| 技术 | 说明 |
|------|------|
| Spring Boot | 后端框架 |
| WebSocket | 实时通信 |
| SSE | 服务端推送 |
| Redis | 结果缓存 |
| Chart.js | 数据可视化 |

### 适合场景
- 实时应用开发学习
- WebSocket/SSE 使用
- 数据可视化展示

```kotlin
package com.example.voting.model

import jakarta.persistence.*
import java.time.LocalDateTime

/**
 * 投票实体
 * 
 * 设计说明：
 * - 每个投票有多个选项
 * - 支持单选和多选
 * - 可以设置截止时间
 * - 可以限制每个 IP/用户只能投一次
 */
@Entity
@Table(name = "polls")
data class Poll(
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    val id: Long = 0,
    
    // ===== 投票内容 =====
    @Column(nullable = false, length = 200)
    val title: String,
    
    @Column(columnDefinition = "TEXT")
    val description: String? = null,
    
    // ===== 投票设置 =====
    // maxChoices: 最大可选数量（1=单选，>1=多选）
    val maxChoices: Int = 1,
    
    // 是否公开结果
    val showResultsImmediately: Boolean = true,
    
    // 截止时间
    val expiresAt: LocalDateTime? = null,
    
    // 是否启用
    val isActive: Boolean = true,
    
    // ===== 关联关系 =====
    // CascadeType.ALL: 操作投票时，选项也会被级联操作
    // orphanRemoval = true: 删除投票时，选项也会被删除
    @OneToMany(mappedBy = "poll", cascade = [CascadeType.ALL], orphanRemoval = true)
    val options: MutableList<PollOption> = mutableListOf(),
    
    @OneToMany(mappedBy = "poll", cascade = [CascadeType.ALL], orphanRemoval = true)
    val votes: MutableList<Vote> = mutableListOf(),
    
    // ===== 审计字段 =====
    val createdAt: LocalDateTime = LocalDateTime.now(),
    val updatedAt: LocalDateTime = LocalDateTime.now()
) {
    // JPA 构造函数
    constructor() : this(
        id = 0,
        title = "",
        description = null,
        maxChoices = 1,
        showResultsImmediately = true,
        expiresAt = null,
        isActive = true,
        options = mutableListOf(),
        votes = mutableListOf(),
        createdAt = LocalDateTime.now(),
        updatedAt = LocalDateTime.now()
    )
}

/**
 * 投票选项实体
 */
@Entity
@Table(name = "poll_options")
data class PollOption(
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    val id: Long = 0,
    
    @Column(nullable = false, length = 100)
    val text: String,
    
    // 选项在投票中的顺序
    val sortOrder: Int = 0,
    
    // ===== 关联关系 =====
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "poll_id", nullable = false)
    val poll: Poll,
    
    // 投票计数（使用数据库计数器优化查询）
    val voteCount: Int = 0
) {
    constructor() : this(id = 0, text = "", sortOrder = 0, poll = Poll(), voteCount = 0)
}

/**
 * 投票记录实体
 * 
 * 用于记录谁投了什么，防止重复投票
 */
@Entity
@Table(
    name = "votes",
    uniqueConstraints = [UniqueConstraint(columnNames = ["poll_id", "voter_identifier"])]
)
data class Vote(
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    val id: Long = 0,
    
    // 投票者标识（可以是用户ID、IP地址或匿名UUID）
    @Column(name = "voter_identifier", nullable = false, length = 100)
    val voterIdentifier: String,
    
    // 选择的选项 ID（多个用逗号分隔）
    @Column(name = "option_ids", nullable = false, length = 500)
    val optionIds: String,
    
    // ===== 关联关系 =====
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "poll_id", nullable = false)
    val poll: Poll,
    
    // 投票时间
    val createdAt: LocalDateTime = LocalDateTime.now()
) {
    constructor() : this(
        id = 0,
        voterIdentifier = "",
        optionIds = "",
        poll = Poll(),
        createdAt = LocalDateTime.now()
    )
    
    /**
     * 获取选项 ID 列表
     */
    fun getOptionIdList(): List<Long> {
        return optionIds.split(",")
            .filter { it.isNotBlank() }
            .map { it.trim().toLong() }
    }
}
```

### 3. PollService.kt

```kotlin
package com.example.voting.service

import com.example.voting.model.*
import com.example.voting.repository.*
import com.example.voting.dto.*
import org.springframework.stereotype.Service
import org.springframework.transaction.annotation.Transactional
import java.time.LocalDateTime
import java.util.concurrent.ConcurrentHashMap

/**
 * 投票服务
 * 
 * 核心功能：
 * - 创建投票
 * - 投票
 * - 获取投票结果
 * - 实时结果推送
 */
@Service
class PollService(
    private val pollRepository: PollRepository,
    private val voteRepository: VoteRepository,
    private val webSocketService: WebSocketService
) {
    
    /**
     * 创建投票
     */
    @Transactional
    fun createPoll(request: CreatePollRequest): PollDto {
        // 创建投票
        val poll = Poll(
            title = request.title,
            description = request.description,
            maxChoices = request.maxChoices ?: 1,
            showResultsImmediately = request.showResultsImmediately ?: true,
            expiresAt = request.expiresAt,
            createdAt = LocalDateTime.now(),
            updatedAt = LocalDateTime.now()
        )
        
        // 添加选项
        request.options.forEachIndexed { index, text ->
            poll.options.add(
                PollOption(
                    text = text,
                    sortOrder = index,
                    poll = poll
                )
            )
        }
        
        val savedPoll = pollRepository.save(poll)
        return PollDto.fromPoll(savedPoll)
    }
    
    /**
     * 获取投票详情
     */
    fun getPoll(id: Long): PollDto? {
        return pollRepository.findById(id)
            .filter { it.isActive }
            .map { PollDto.fromPoll(it) }
            .orElse(null)
    }
    
    /**
     * 获取投票结果（包含每个选项的票数）
     */
    fun getPollResults(id: Long): PollResultDto? {
        val poll = pollRepository.findById(id).orElse(null) ?: return null
        
        // 计算每个选项的得票数
        val optionResults = poll.options.map { option ->
            // 统计投票记录中包含此选项的数量
            val voteCount = voteRepository.countByPollIdAndOptionId(id, option.id)
            OptionResultDto(
                id = option.id,
                text = option.text,
                voteCount = voteCount.toInt(),
                percentage = calculatePercentage(voteCount, poll.votes.size)
            )
        }.sortedByDescending { it.voteCount }
        
        val totalVotes = poll.votes.size
        
        return PollResultDto(
            id = poll.id,
            title = poll.title,
            description = poll.description,
            totalVotes = totalVotes,
            options = optionResults,
            isActive = poll.isActive && !isExpired(poll),
            expiresAt = poll.expiresAt?.toString(),
            hasVoted = false  // 这个字段需要客户端提供 voterIdentifier 来判断
        )
    }
    
    /**
     * 投票
     */
    @Transactional
    fun vote(pollId: Long, optionIds: List<Long>, voterIdentifier: String): VoteResult {
        val poll = pollRepository.findById(pollId).orElse(null)
            ?: return VoteResult(false, "投票不存在")
        
        // 检查投票是否有效
        if (!poll.isActive) {
            return VoteResult(false, "投票已关闭")
        }
        
        if (isExpired(poll)) {
            return VoteResult(false, "投票已截止")
        }
        
        // 检查选项数量
        if (optionIds.size > poll.maxChoices) {
            return VoteResult(false, "选择的选项超过限制")
        }
        
        // 检查选项是否都属于这个投票
        val validOptionIds = poll.options.map { it.id }.toSet()
        if (!optionIds.all { it in validOptionIds }) {
            return VoteResult(false, "无效的选项")
        }
        
        // 检查是否已经投过票
        if (voteRepository.existsByPollIdAndVoterIdentifier(pollId, voterIdentifier)) {
            return VoteResult(false, "您已经投过票了")
        }
        
        // 创建投票记录
        val vote = Vote(
            poll = poll,
            voterIdentifier = voterIdentifier,
            optionIds = optionIds.joinToString(","),
            createdAt = LocalDateTime.now()
        )
        voteRepository.save(vote)
        
        // 更新每个选项的票数（乐观锁优化）
        optionIds.forEach { optionId ->
            voteRepository.incrementOptionVoteCount(pollId, optionId)
        }
        
        // 获取更新后的结果
        val results = getPollResults(pollId)!!
        
        // 通过 WebSocket 推送更新
        webSocketService.broadcastPollUpdate(pollId, results)
        
        return VoteResult(true, "投票成功", results)
    }
    
    /**
     * 检查用户是否已投票
     */
    fun hasVoted(pollId: Long, voterIdentifier: String): Boolean {
        return voteRepository.existsByPollIdAndVoterIdentifier(pollId, voterIdentifier)
    }
    
    /**
     * 获取活跃的投票列表
     */
    fun getActivePolls(page: Int, size: Int): List<PollDto> {
        val polls = pollRepository.findByIsActiveTrueOrderByCreatedAtDesc(page, size)
        return polls.map { PollDto.fromPoll(it) }
    }
    
    /**
     * 关闭投票
     */
    @Transactional
    fun closePoll(pollId: Long): Boolean {
        val poll = pollRepository.findById(pollId).orElse(null) ?: return false
        poll.isActive = false
        poll.updatedAt = LocalDateTime.now()
        pollRepository.save(poll)
        
        // 推送关闭消息
        webSocketService.broadcastPollClosed(pollId)
        
        return true
    }
    
    // ==================== 私有方法 ====================
    
    /**
     * 计算百分比
     */
    private fun calculatePercentage(count: Long, total: Int): Double {
        if (total == 0) return 0.0
        return (count.toDouble() / total) * 100
    }
    
    /**
     * 检查投票是否过期
     */
    private fun isExpired(poll: Poll): Boolean {
        return poll.expiresAt?.isBefore(LocalDateTime.now()) == true
    }
    
    // ===== 投票结果 =====
    
    data class VoteResult(
        val success: Boolean,
        val message: String,
        val results: PollResultDto? = null
    )
}
```

### 4. WebSocketService.kt（实时推送）

```kotlin
package com.example.voting.service

import com.example.voting.dto.*
import com.fasterxml.jackson.databind.ObjectMapper
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Service
import org.springframework.web.socket.*
import org.springframework.web.socket.handler.TextWebSocketHandler
import jakarta.websocket.*
import jakarta.websocket.server.ServerEndpoint

/**
 * WebSocket 服务
 * 
 * 用于实时推送投票结果更新
 * 
 * 支持两种方式：
 * 1. WebSocket：全双工通信，适合需要双向交互的场景
 * 2. SSE：服务端推送，适合单向数据流
 */
@Service
class WebSocketService(
    private val objectMapper: ObjectMapper
) {
    private val logger = LoggerFactory.getLogger(WebSocketService::class.java)
    
    // 存储活跃连接
    // pollId -> Set<WebSocketSession>
    private val pollSubscriptions = ConcurrentHashMap<Long, MutableSet<WebSocketSession>>()
    
    /**
     * 广播投票结果更新
     */
    fun broadcastPollUpdate(pollId: Long, results: PollResultDto) {
        val message = WebSocketMessage(
            type = "poll_update",
            pollId = pollId,
            data = results
        )
        
        broadcastToPoll(pollId, message)
    }
    
    /**
     * 广播投票关闭消息
     */
    fun broadcastPollClosed(pollId: Long) {
        val message = WebSocketMessage(
            type = "poll_closed",
            pollId = pollId,
            data = null
        )
        
        broadcastToPoll(pollId, message)
        
        // 清理订阅
        pollSubscriptions.remove(pollId)
    }
    
    /**
     * 订阅投票
     */
    fun subscribeToPoll(pollId: Long, session: WebSocketSession) {
        pollSubscriptions
            .computeIfAbsent(pollId) { ConcurrentHashMap.newKeySet() }
            .add(session)
        
        logger.info("Session ${session.id} subscribed to poll $pollId")
    }
    
    /**
     * 取消订阅
     */
    fun unsubscribeFromPoll(pollId: Long, session: WebSocketSession) {
        pollSubscriptions[pollId]?.remove(session)
        
        logger.info("Session ${session.id} unsubscribed from poll $pollId")
    }
    
    /**
     * 广播消息到指定投票的所有订阅者
     */
    private fun broadcastToPoll(pollId: Long, message: WebSocketMessage) {
        val subscribers = pollSubscriptions[pollId] ?: return
        
        val messageJson = objectMapper.writeValueAsString(message)
        
        subscribers.removeIf { session ->
            try {
                if (session.isOpen) {
                    session.sendMessage(TextMessage(messageJson))
                    false
                } else {
                    true  // 移除已关闭的连接
                }
            } catch (e: Exception) {
                logger.error("Failed to send message to session ${session.id}", e)
                true
            }
        }
    }
    
    /**
     * 广播消息
     */
    private fun broadcastToAll(message: WebSocketMessage) {
        val messageJson = objectMapper.writeValueAsString(message)
        
        pollSubscriptions.values.forEach { sessions ->
            sessions.removeIf { session ->
                try {
                    if (session.isOpen) {
                        session.sendMessage(TextMessage(messageJson))
                        false
                    } else {
                        true
                    }
                } catch (e: Exception) {
                    true
                }
            }
        }
    }
    
    /**
     * WebSocket 消息格式
     */
    data class WebSocketMessage(
        val type: String,      // poll_update, poll_closed, poll_created
        val pollId: Long,
        val data: Any?
    )
}
```

```kotlin
package com.example.voting.controller

import com.example.voting.service.*
import com.example.voting.dto.*
import org.springframework.http.*
import org.springframework.web.bind.annotation.*
import jakarta.servlet.http.HttpServletRequest

/**
 * 投票控制器
 */
@RestController
@RequestMapping("/api/polls")
@CrossOrigin(origins = ["*"])
class PollController(
    private val pollService: PollService
) {
    
    /**
     * 创建投票
     */
    @PostMapping
    fun createPoll(@RequestBody request: CreatePollRequest): ResponseEntity<PollDto> {
        val poll = pollService.createPoll(request)
        return ResponseEntity.status(HttpStatusCode.Created).body(poll)
    }
    
    /**
     * 获取投票详情
     */
    @GetMapping("/{id}")
    fun getPoll(@PathVariable id: Long): ResponseEntity<PollDto> {
        val poll = pollService.getPoll(id)
            ?: return ResponseEntity.notFound().build()
        return ResponseEntity.ok(poll)
    }
    
    /**
     * 获取投票结果
     */
    @GetMapping("/{id}/results")
    fun getPollResults(
        @PathVariable id: Long,
        @RequestParam(required = false) voterId: String?
    ): ResponseEntity<PollResultDto> {
        val results = pollService.getPollResults(id)
            ?: return ResponseEntity.notFound().build()
        
        // 如果提供了 voterId，检查是否已投票
        val hasVoted = voterId?.let { 
            pollService.hasVoted(id, it) 
        } ?: false
        
        return ResponseEntity.ok(results.copy(hasVoted = hasVoted))
    }
    
    /**
     * 投票
     */
    @PostMapping("/{id}/vote")
    fun vote(
        @PathVariable id: Long,
        @RequestBody request: VoteRequest,
        httpRequest: HttpServletRequest
    ): ResponseEntity<VoteResponse> {
        // 获取投票者标识（优先使用用户ID，否则使用IP地址）
        val voterIdentifier = request.voterId 
            ?: httpRequest.getHeader("X-Forwarded-For") 
            ?: httpRequest.remoteAddr
            ?: "anonymous-${System.currentTimeMillis()}"
        
        val result = pollService.vote(id, request.optionIds, voterIdentifier)
        
        return if (result.success) {
            ResponseEntity.ok(VoteResponse(
                success = true,
                message = result.message,
                results = result.results
            ))
        } else {
            ResponseEntity.badRequest().body(VoteResponse(
                success = false,
                message = result.message,
                results = null
            ))
        }
    }
    
    /**
     * 检查是否已投票
     */
    @GetMapping("/{id}/has-voted")
    fun hasVoted(
        @PathVariable id: Long,
        @RequestParam voterId: String
    ): ResponseEntity<Map<String, Boolean>> {
        val hasVoted = pollService.hasVoted(id, voterId)
        return ResponseEntity.ok(mapOf("hasVoted" to hasVoted))
    }
    
    /**
     * 获取活跃投票列表
     */
    @GetMapping
    fun getActivePolls(
        @RequestParam(defaultValue = "0") page: Int,
        @RequestParam(defaultValue = "10") size: Int
    ): ResponseEntity<List<PollDto>> {
        val polls = pollService.getActivePolls(page, size)
        return ResponseEntity.ok(polls)
    }
    
    /**
     * 关闭投票（管理员操作）
     */
    @PostMapping("/{id}/close")
    fun closePoll(@PathVariable id: Long): ResponseEntity<Map<String, Boolean>> {
        val success = pollService.closePoll(id)
        return ResponseEntity.ok(mapOf("success" to success))
    }
}
```

```kotlin
package com.example.voting.dto

import com.example.voting.model.*
import java.time.LocalDateTime

/**
 * 创建投票请求
 */
data class CreatePollRequest(
    val title: String,                    // 投票标题
    val description: String? = null,      // 投票描述
    val options: List<String>,             // 选项列表（至少2个）
    val maxChoices: Int? = 1,             // 最大可选数量（1=单选）
    val showResultsImmediately: Boolean? = true,  // 是否立即显示结果
    val expiresAt: String? = null         // 截止时间（ISO格式）
)

/**
 * 投票 DTO
 */
data class PollDto(
    val id: Long,
    val title: String,
    val description: String?,
    val maxChoices: Int,
    val showResultsImmediately: Boolean,
    val isActive: Boolean,
    val expiresAt: String?,
    val options: List<OptionDto>,
    val createdAt: String,
    val totalVotes: Int  // 当前总票数
) {
    companion object {
        fun fromPoll(poll: Poll): PollDto {
            return PollDto(
                id = poll.id,
                title = poll.title,
                description = poll.description,
                maxChoices = poll.maxChoices,
                showResultsImmediately = poll.showResultsImmediately,
                isActive = poll.isActive,
                expiresAt = poll.expiresAt?.toString(),
                options = poll.options
                    .sortedBy { it.sortOrder }
                    .map { OptionDto.fromOption(it) },
                createdAt = poll.createdAt.toString(),
                totalVotes = poll.votes.size
            )
        }
    }
}

/**
 * 选项 DTO
 */
data class OptionDto(
    val id: Long,
    val text: String,
    val sortOrder: Int
) {
    companion object {
        fun fromOption(option: PollOption): OptionDto {
            return OptionDto(
                id = option.id,
                text = option.text,
                sortOrder = option.sortOrder
            )
        }
    }
}

/**
 * 投票请求
 */
data class VoteRequest(
    val optionIds: List<Long>,   // 选择的选项 ID 列表
    val voterId: String? = null  // 投票者标识（可选）
)

/**
 * 投票响应
 */
data class VoteResponse(
    val success: Boolean,
    val message: String,
    val results: PollResultDto?
)

/**
 * 投票结果 DTO
 */
data class PollResultDto(
    val id: Long,
    val title: String,
    val description: String?,
    val totalVotes: Int,
    val options: List<OptionResultDto>,
    val isActive: Boolean,
    val expiresAt: String?,
    val hasVoted: Boolean  // 当前用户是否已投票
)

/**
 * 选项结果 DTO
 */
```

```typescript
import React, { useEffect, useState } from 'react';
import { useParams } from 'react-router-dom';
import { Chart as ChartJS, ArcElement, Tooltip, Legend } from 'chart.js';
import { Doughnut } from 'react-chartjs-2';
import { pollApi } from '../services/api';

ChartJS.register(ArcElement, Tooltip, Legend);

const PollDetail: React.FC = () => {
  const { id } = useParams<{ id: string }>();
  const [poll, setPoll] = useState<PollDetailData | null>(null);
  const [results, setResults] = useState<PollResult | null>(null);
  const [selectedOptions, setSelectedOptions] = useState<number[]>([]);
  const [hasVoted, setHasVoted] = useState(false);
  const [loading, setLoading] = useState(true);

  // 生成匿名用户ID
  const voterId = localStorage.getItem('voterId') || 
    `anon-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
  localStorage.setItem('voterId', voterId);

  useEffect(() => {
    fetchPollData();
    connectWebSocket();
  }, [id]);

  const fetchPollData = async () => {
    try {
      const pollRes = await pollApi.getPoll(parseInt(id!));
      setPoll(pollRes.data);
      
      const resultsRes = await pollApi.getResults(parseInt(id!), voterId);
      setResults(resultsRes.data);
      setHasVoted(resultsRes.data.hasVoted);
    } catch (error) {
      console.error('Failed to fetch poll:', error);
    } finally {
      setLoading(false);
    }
  };

  // WebSocket 连接
  const connectWebSocket = () => {
    const ws = new WebSocket(`ws://localhost:8080/ws/poll/${id}`);
    
    ws.onmessage = (event) => {
      const message = JSON.parse(event.data);
      
      if (message.type === 'poll_update') {
        setResults(message.data);
        // 显示实时更新提示
        showUpdateNotification();
      } else if (message.type === 'poll_closed') {
        alert('投票已关闭');
        fetchPollData();
      }
    };
    
    ws.onerror = (error) => {
      console.error('WebSocket error:', error);
    };
    
    return () => ws.close();
  };

  // 投票
  const handleVote = async () => {
    if (selectedOptions.length === 0) {
      alert('请选择至少一个选项');
      return;
    }
    
    try {
      const response = await pollApi.vote(parseInt(id!), selectedOptions, voterId);
      if (response.data.success) {
        setResults(response.data.results);
        setHasVoted(true);
      } else {
        alert(response.data.message);
      }
    } catch (error) {
      console.error('Vote failed:', error);
      alert('投票失败，请重试');
    }
  };

  // 选择/取消选择选项
  const toggleOption = (optionId: number) => {
    if (hasVoted || !poll) return;
    
    if (selectedOptions.includes(optionId)) {
      setSelectedOptions(selectedOptions.filter(id => id !== optionId));
    } else {
      // 检查是否超过最大选择数
      if (selectedOptions.length >= poll.maxChoices) {
        alert(`最多只能选择 ${poll.maxChoices} 个选项`);
        return;
      }
      setSelectedOptions([...selectedOptions, optionId]);
    }
  };

  if (loading) return <div>加载中...</div>;
  if (!poll) return <div>投票不存在</div>;

  // Chart.js 数据配置
  const chartData = results ? {
    labels: results.options.map(o => o.text),
    datasets: [{
      data: results.options.map(o => o.voteCount),
      backgroundColor: [
        '#FF6384', '#36A2EB', '#FFCE56', '#4BC0C0', '#9966FF',
        '#FF9F40', '#FF6384', '#C9CBCF'
      ],
      borderWidth: 2,
      borderColor: '#fff'
    }]
  } : null;

  return (
    <div className="poll-detail">
      <h2>{poll.title}</h2>
      {poll.description && <p>{poll.description}</p>}
      
      <div className="poll-status">
        <span>总票数: {results?.totalVotes || 0}</span>
        <span>{results?.isActive ? '🟢 进行中' : '🔴 已结束'}</span>
      </div>
      
      {/* 投票选项 */}
      <div className="options">
        {poll.options.map((option) => {
          const result = results?.options.find(o => o.id === option.id);
          const isSelected = selectedOptions.includes(option.id);
          
          return (
            <div 
              key={option.id}
              className={`option ${isSelected ? 'selected' : ''}`}
              onClick={() => toggleOption(option.id)}
            >
              <span className="option-text">{option.text}</span>
              
              {/* 显示实时结果 */}
              {results && poll.showResultsImmediately && (
                <div className="option-result">
                  <div className="progress-bar">
                    <div 
                      className="progress-fill"
                      style={{ width: `${result?.percentage || 0}%` }}
                    />
                  </div>
                  <span className="vote-count">
                    {result?.voteCount || 0} 票 ({result?.percentage?.toFixed(1) || 0}%)
                  </span>
                </div>
              )}
            </div>
          );
        })}
      </div>
      
      {/* 投票按钮 */}
      {!hasVoted && results?.isActive && (
        <button onClick={handleVote} disabled={selectedOptions.length === 0}>
          投票 ({selectedOptions.length}/{poll.maxChoices})
        </button>
      )}
      
      {hasVoted && <p className="voted-message">✓ 您已投票</p>}
      
      {/* 图表 */}
      {results && poll.showResultsImmediately && results.totalVotes > 0 && (
        <div className="chart-container">
          <Doughnut data={chartData} />
        </div>
      )}
    </div>
  );
};

export default PollDetail;
```

## 面试要点

### 题目1：WebSocket 和 HTTP 有什么区别？

**标准答案**：

| 特性 | HTTP | WebSocket |
|------|------|-----------|
| 连接方式 | 请求-响应 | 双向握手 |
| 连接次数 | 每次请求新建 | 一次握手，长期保持 |
| 服务器推送 | 需要轮询/SSE | 原生支持 |
| 资源消耗 | 每次请求都有开销 | 低（保持连接） |
| 适用场景 | REST API | 实时应用 |

**在投票系统中的应用**：
- HTTP：创建投票、获取结果（投票）
- WebSocket：实时推送投票结果
