# Ktor 微服务方案三：用户认证系统

## 项目介绍

### 项目是什么
这是一个完整的用户认证授权微服务，包含用户注册、登录、JWT 认证、OAuth2 第三方登录（GitHub/Google）、角色权限管理等功能。

### 解决什么问题
- 为多个前端应用提供统一的身份认证服务
- 支持多种登录方式（账号密码、OAuth2）
- 提供 JWT Token 管理
- 实现基于角色的访问控制（RBAC）

### 核心技术亮点
| 技术 | 说明 |
|------|------|
| JWT | 无状态身份认证 |
| OAuth2 | 第三方登录授权 |
| BCrypt | 密码安全存储 |
| RBAC | 基于角色的权限控制 |
| Refresh Token | Token 刷新机制 |

### 适合场景
- 微服务架构中的身份认证服务
- 需要统一用户管理的多应用系统
- 学习 OAuth2 和 JWT 认证

### 4. Security.kt（JWT 安全配置）

```kotlin
package com.example.auth.plugin

import com.auth0.jwt.JWT
import com.auth0.jwt.algorithms.Algorithm
import com.auth0.jwt.exceptions.JWTVerificationException
import com.auth0.jwt.interfaces.DecodedJWT
import io.ktor.server.auth.*
import io.ktor.server.auth.jwt.*
import java.util.*

/**
 * JWT 配置和认证服务
 */
object JwtConfig {
    // ===== JWT 配置 =====
    
    // 密钥 - 生产环境应使用更长的随机字符串
    private const val SECRET = "your-super-secret-key-change-in-production-must-be-at-least-256-bits"
    
    // 颁发者
    private const val ISSUER = "auth-service"
    
    // 观众
    private const val AUDIENCE = "auth-service-clients"
    
    // Access Token 过期时间：15 分钟
    private const val ACCESS_TOKEN_EXPIRATION = 15 * 60 * 1000L
    
    // Refresh Token 过期时间：7 天
    private const val REFRESH_TOKEN_EXPIRATION = 7 * 24 * 60 * 60 * 1000L
    
    // 算法
    private val algorithm = Algorithm.HMAC256(SECRET)
    
    // ===== Token 生成 =====
    
    /**
     * 生成 Access Token
     */
    fun generateAccessToken(
        userId: Int,
        username: String,
        email: String,
        roles: List<String>
    ): String {
        val now = Date()
        val expiresAt = Date(now.time + ACCESS_TOKEN_EXPIRATION)
        
        return JWT.create()
            .withIssuer(ISSUER)
            .withAudience(AUDIENCE)
            .withSubject(userId.toString())
            .withClaim("username", username)
            .withClaim("email", email)
            .withClaim("roles", roles.joinToString(","))
            .withClaim("type", "access")  // Token 类型标识
            .withIssuedAt(now)
            .withExpiresAt(expiresAt)
            .sign(algorithm)
    }
    
    /**
     * 生成 Refresh Token
     */
    fun generateRefreshToken(
        userId: Int,
        tokenId: String
    ): String {
        val now = Date()
        val expiresAt = Date(now.time + REFRESH_TOKEN_EXPIRATION)
        
        return JWT.create()
            .withIssuer(ISSUER)
            .withAudience(AUDIENCE)
            .withSubject(userId.toString())
            .withClaim("tokenId", tokenId)
            .withClaim("type", "refresh")
            .withIssuedAt(now)
            .withExpiresAt(expiresAt)
            .sign(algorithm)
    }
    
    /**
     * 验证 Token
     */
    fun verifyToken(token: String): DecodedJWT? {
        return try {
            val verifier = JWT.require(algorithm)
                .withIssuer(ISSUER)
                .withAudience(AUDIENCE)
                .build()
            
            verifier.verify(token)
        } catch (e: JWTVerificationException) {
            null
        }
    }
    
    /**
     * 解码 Token（不验证）
     * 谨慎使用！
     */
    fun decodeToken(token: String): DecodedJWT? {
        return try {
            JWT.decode(token)
        } catch (e: Exception) {
            null
        }
    }
    
    /**
     * 获取 Token 过期时间
     */
    fun getExpirationTime(): Long = ACCESS_TOKEN_EXPIRATION
    
    fun getRefreshExpirationTime(): Long = REFRESH_TOKEN_EXPIRATION
}

/**
 * JWT 用户主体
 */
data class JwtUser(
    val userId: Int,
    val username: String,
    val email: String,
    val roles: List<String>
)

/**
 * 配置 Ktor JWT 认证
 */
fun Application.configureSecurity() {
    install(Authentication) {
        jwt("auth-jwt") {
            realm = "Auth Service"
            
            // 验证器
            verifier { token ->
                JwtConfig.verifyToken(token)
            }
            
            // 验证成功：构建 JwtUser
            validate { credential ->
                val jwt = credential.payload
                
                // 检查 Token 类型
                val type = jwt.getClaim("type").asString()
                if (type != "access") {
                    return@validate null
                }
                
                // 构建用户信息
                val userId = jwt.subject?.toIntOrNull() ?: return@validate null
                val username = jwt.getClaim("username").asString()
                val email = jwt.getClaim("email").asString()
                val rolesStr = jwt.getClaim("roles").asString()
                val roles = if (rolesStr.isNotEmpty()) rolesStr.split(",") else emptyList()
                
                JWTPrincipal(jwt.claims) to JwtUser(userId, username, email, roles)
            }
            
            // 认证失败
            challenge { _, _ ->
                call.respond(
                    io.ktor.http.HttpStatusCode.Unauthorized,
                    mapOf("error" to "Invalid or expired token")
                )
            }
        }
    }
}
```

### 6. UserRepository.kt

```kotlin
package com.example.auth.repository

import com.example.auth.model.*
import com.example.auth.plugin.database
import org.jetbrains.exposed.sql.*
import org.jetbrains.exposed.sql.transactions.*
import org.mindrot.jbcrypt.*
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

class UserRepository {
    
    /**
     * 根据 ID 查找用户
     */
    suspend fun findById(id: Int): User? {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                Users.select { Users.id eq id }
                    .map { it.toUser() }
                    .singleOrNull()
            }
        }
    }
    
    /**
     * 根据用户名查找用户
     */
    suspend fun findByUsername(username: String): User? {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                Users.select { Users.username eq username }
                    .map { it.toUser() }
                    .singleOrNull()
            }
        }
    }
    
    /**
     * 根据邮箱查找用户
     */
    suspend fun findByEmail(email: String): User? {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                Users.select { Users.email eq email }
                    .map { it.toUser() }
                    .singleOrNull()
            }
        }
    }
    
    /**
     * 根据 Provider 和 Provider ID 查找用户（OAuth 用）
     */
    suspend fun findByProvider(provider: String, providerId: String): User? {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                Users.select { 
                    (Users.provider eq provider) and (Users.providerId eq providerId)
                }.map { it.toUser() }
                 .singleOrNull()
            }
        }
    }
    
    /**
     * 创建本地用户（账号密码注册）
     */
    suspend fun createLocalUser(
        username: String,
        email: String,
        password: String
    ): User {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                val passwordHash = BCrypt.hashpw(password, BCrypt.gensalt(12))
                
                val id = Users.insert {
                    it[Users.username] = username
                    it[Users.email] = email
                    it[Users.passwordHash] = passwordHash
                    it[Users.provider] = "local"
                    it[Users.isEmailVerified] = false
                }[Users.id]
                
                Users.select { Users.id eq id }
                    .map { it.toUser() }
                    .single()
            }
        }
    }
    
    /**
     * 创建 OAuth 用户
     */
    suspend fun createOAuthUser(
        username: String,
        email: String,
        provider: String,
        providerId: String,
        avatarUrl: String?
    ): User {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                val id = Users.insert {
                    it[Users.username] = username
                    it[Users.email] = email
                    it[Users.provider] = provider
                    it[Users.providerId] = providerId
                    it[Users.avatarUrl] = avatarUrl
                    it[Users.isEmailVerified] = true  // OAuth 用户邮箱已验证
                }[Users.id]
                
                Users.select { Users.id eq id }
                    .map { it.toUser() }
                    .single()
            }
        }
    }
    
    /**
     * 验证密码
     */
    fun verifyPassword(user: User, password: String): Boolean {
        if (user.passwordHash == null) return false
        return BCrypt.checkpw(password, user.passwordHash)
    }
    
    /**
     * 更新密码
     */
    suspend fun updatePassword(userId: Int, newPassword: String): Boolean {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                val hashed = BCrypt.hashpw(newPassword, BCrypt.gensalt(12))
                val count = Users.update({ Users.id eq userId }) {
                    it[Users.passwordHash] = hashed
                    it[Users.updatedAt] = java.time.LocalDateTime.now()
                }
                count > 0
            }
        }
    }
    
    /**
     * 更新用户头像
     */
    suspend fun updateAvatar(userId: Int, avatarUrl: String): Boolean {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                val count = Users.update({ Users.id eq userId }) {
                    it[Users.avatarUrl] = avatarUrl
                    it[Users.updatedAt] = java.time.LocalDateTime.now()
                }
                count > 0
            }
        }
    }
    
    /**
     * 检查用户名是否存在
     */
    suspend fun existsByUsername(username: String): Boolean {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                Users.select { Users.username eq username }.count() > 0
            }
        }
    }
    
    /**
     * 检查邮箱是否存在
     */
    suspend fun existsByEmail(email: String): Boolean {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                Users.select { Users.email eq email }.count() > 0
            }
        }
    }
    
    /**
     * 获取用户角色
     */
    suspend fun getUserRoles(userId: Int): List<String> {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                UserRoles.select { UserRoles.userId eq userId }
                    .joinTo(Roles) { UserRoles.roleId eq Roles.id }
                    .map { it[Roles.name] }
            }
        }
    }
    
    /**
     * 分配角色
     */
    suspend fun assignRole(userId: Int, roleName: String): Boolean {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                // 查找角色 ID
                val roleId = Roles.select { Roles.name eq roleName }
                    .map { it[Roles.id] }
                    .singleOrNull() ?: return@transaction false
                
                // 分配角色
                UserRoles.insert {
                    it[UserRoles.userId] = userId
                    it[UserRoles.roleId] = roleId
                }
                true
            }
        }
    }
    
    private fun ResultRow.toUser(): User {
        return User(
            id = this[Users.id],
            username = this[Users.username],
            email = this[Users.email],
            passwordHash = this[Users.passwordHash],
            avatarUrl = this[Users.avatarUrl],
            isEmailVerified = this[Users.isEmailVerified],
            provider = this[Users.provider],
            providerId = this[Users.providerId],
            status = this[Users.status],
            createdAt = this[Users.createdAt],
            updatedAt = this[Users.updatedAt]
        )
    }
}
```

```kotlin
package com.example.auth.repository

import com.example.auth.model.*
import com.example.auth.plugin.database
import org.jetbrains.exposed.sql.*
import org.jetbrains.exposed.sql.transactions.*
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import java.time.LocalDateTime

class RefreshTokenRepository {
    
    /**
     * 保存 Refresh Token
     */
    suspend fun save(
        tokenId: String,
        userId: Int,
        tokenHash: String,
        deviceInfo: String?,
        ipAddress: String?,
        expiresAt: LocalDateTime
    ): RefreshToken {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                RefreshTokens.insert {
                    it[RefreshTokens.id] = tokenId
                    it[RefreshTokens.userId] = userId
                    it[RefreshTokens.tokenHash] = tokenHash
                    it[RefreshTokens.deviceInfo] = deviceInfo
                    it[RefreshTokens.ipAddress] = ipAddress
                    it[RefreshTokens.expiresAt] = expiresAt
                }
                
                RefreshTokens.select { RefreshTokens.id eq tokenId }
                    .map { it.toRefreshToken() }
                    .single()
            }
        }
    }
    
    /**
     * 查找 Token
     */
    suspend fun findById(id: String): RefreshToken? {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                RefreshTokens.select { RefreshTokens.id eq id }
                    .map { it.toRefreshToken() }
                    .singleOrNull()
            }
        }
    }
    
    /**
     * 查找用户的所有 Token
     */
    suspend fun findByUserId(userId: Int): List<RefreshToken> {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                RefreshTokens.select { RefreshTokens.userId eq userId }
                    .map { it.toRefreshToken() }
            }
        }
    }
    
    /**
     * 验证 Token（检查哈希和过期）
     */
    suspend fun isValid(tokenId: String, tokenHash: String): Boolean {
        val token = findById(tokenId) ?: return false
        
        // 检查是否已撤回
        if (token.isRevoked) return false
        
        // 检查是否过期
        if (token.expiresAt.isBefore(LocalDateTime.now())) return false
        
        // 检查哈希是否匹配
        if (token.tokenHash != tokenHash) return false
        
        return true
    }
    
    /**
     * 撤回 Token
     */
    suspend fun revoke(tokenId: String): Boolean {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                val count = RefreshTokens.update({ RefreshTokens.id eq tokenId }) {
                    it[RefreshTokens.isRevoked] = true
                }
                count > 0
            }
        }
    }
    
    /**
     * 撤回用户的所有 Token
     */
    suspend fun revokeAllForUser(userId: Int): Int {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                RefreshTokens.update({ RefreshTokens.userId eq userId }) {
                    it[RefreshTokens.isRevoked] = true
                }
            }
        }
    }
    
    /**
     * 更新最后使用时间
     */
    suspend fun updateLastUsed(tokenId: String): Boolean {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                val count = RefreshTokens.update({ RefreshTokens.id eq tokenId }) {
                    it[RefreshTokens.lastUsedAt] = LocalDateTime.now()
                }
                count > 0
            }
        }
    }
    
    /**
     * 删除过期 Token
     */
    suspend fun deleteExpired(): Int {
        return withContext(Dispatchers.IO) {
            transaction(database) {
                RefreshTokens.deleteWhere {
                    RefreshTokens.expiresAt less LocalDateTime.now()
                }
            }
        }
    }
    
    private fun ResultRow.toRefreshToken(): RefreshToken {
        return RefreshToken(
            id = this[RefreshTokens.id],
            userId = this[RefreshTokens.userId],
            tokenHash = this[RefreshTokens.tokenHash],
            deviceInfo = this[RefreshTokens.deviceInfo],
            ipAddress = this[RefreshTokens.ipAddress],
            expiresAt = this[RefreshTokens.expiresAt],
            isRevoked = this[RefreshTokens.isRevoked],
            createdAt = this[RefreshTokens.createdAt],
            lastUsedAt = this[RefreshTokens.lastUsedAt]
        )
    }
```

```kotlin
package com.example.auth.service

import com.example.auth.model.*
import com.example.auth.repository.*
import com.example.auth.dto.*

import com.google.uuid.UUID
import java.security.MessageDigest
import java.time.LocalDateTime

/**
 * 认证服务
 * 
 * 职责：
 * - 用户注册/登录
 * - Token 管理
 * - 密码重置
 */
class AuthService(
    private val userRepository: UserRepository,
    private val refreshTokenRepository: RefreshTokenRepository,
    private val jwtService: JwtService
) {
    
    /**
     * 用户注册
     */
    suspend fun register(
        username: String,
        email: String,
        password: String
    ): Result<AuthResponse> {
        // ===== 参数验证 =====
        if (username.length < 3 || username.length > 20) {
            return Result.failure(IllegalArgumentException("用户名长度必须在 3-20 个字符之间"))
        }
        
        if (!email.matches(Regex("^[A-Za-z0-9+_.-]+@[A-Za-z0-9.-]+$"))) {
            return Result.failure(IllegalArgumentException("邮箱格式不正确"))
        }
        
        if (password.length < 8) {
            return Result.failure(IllegalArgumentException("密码长度至少 8 个字符"))
        }
        
        // 检查唯一性
        if (userRepository.existsByUsername(username)) {
            return Result.failure(IllegalArgumentException("用户名已存在"))
        }
        
        if (userRepository.existsByEmail(email)) {
            return Result.failure(IllegalArgumentException("邮箱已被注册"))
        }
        
        // ===== 创建用户 =====
        val user = userRepository.createLocalUser(username, email, password)
        
        // ===== 分配默认角色 =====
        userRepository.assignRole(user.id, Role.USER.name)
        
        // ===== 获取用户角色 =====
        val roles = userRepository.getUserRoles(user.id)
        
        // ===== 生成 Token =====
        val tokens = generateTokens(user, roles, null, null)
        
        return Result.success(
            AuthResponse(
                accessToken = tokens.accessToken,
                refreshToken = tokens.refreshToken,
                expiresIn = JwtConfig.getExpirationTime() / 1000,
                tokenType = "Bearer",
                user = UserDto.fromUser(user, roles)
            )
        )
    }
    
    /**
     * 用户登录
     */
    suspend fun login(
        usernameOrEmail: String,
        password: String,
        deviceInfo: String?,
        ipAddress: String?
    ): Result<AuthResponse> {
        // ===== 查找用户 =====
        val user = if (usernameOrEmail.contains("@")) {
            userRepository.findByEmail(usernameOrEmail)
        } else {
            userRepository.findByUsername(usernameOrEmail)
        } ?: return Result.failure(IllegalArgumentException("用户不存在"))
        
        // ===== 检查用户状态 =====
        if (user.status != "active") {
            return Result.failure(IllegalArgumentException("账号已被禁用"))
        }
        
        // ===== 验证密码 =====
        if (!userRepository.verifyPassword(user, password)) {
            return Result.failure(IllegalArgumentException("密码错误"))
        }
        
        // ===== 获取角色 =====
        val roles = userRepository.getUserRoles(user.id)
        
        // ===== 生成 Token =====
        val tokens = generateTokens(user, roles, deviceInfo, ipAddress)
        
        return Result.success(
            AuthResponse(
                accessToken = tokens.accessToken,
                refreshToken = tokens.refreshToken,
                expiresIn = JwtConfig.getExpirationTime() / 1000,
                tokenType = "Bearer",
                user = UserDto.fromUser(user, roles)
            )
        )
    }
    
    /**
     * 刷新 Token
     */
    suspend fun refreshToken(refreshToken: String): Result<AuthResponse> {
        // ===== 验证 Token =====
        val decoded = JwtConfig.decodeToken(refreshToken)
            ?: return Result.failure(IllegalArgumentException("无效的 Token"))
        
        // ===== 检查 Token 类型 =====
        if (decoded.getClaim("type").asString() != "refresh") {
            return Result.failure(IllegalArgumentException("不是 Refresh Token"))
        }
        
        val tokenId = decoded.getClaim("tokenId").asString()
        val userId = decoded.subject?.toIntOrNull()
            ?: return Result.failure(IllegalArgumentException("无效的 Token"))
        
        if (userId == null) {
            return Result.failure(IllegalArgumentException("无效的 Token"))
        }
        
        // ===== 验证 Token =====
        val tokenHash = hashToken(refreshToken)
        if (!refreshTokenRepository.isValid(tokenId, tokenHash)) {
            return Result.failure(IllegalArgumentException("Token 已过期或已撤回"))
        }
        
        // ===== 获取用户 =====
        val user = userRepository.findById(userId)
            ?: return Result.failure(IllegalArgumentException("用户不存在"))
        
        if (user.status != "active") {
            return Result.failure(IllegalArgumentException("账号已被禁用"))
        }
        
        // ===== 获取角色 =====
        val roles = userRepository.getUserRoles(user.id)
        
        // ===== 撤回旧 Refresh Token =====
        refreshTokenRepository.revoke(tokenId)
        
        // ===== 生成新 Token =====
        val tokens = generateTokens(user, roles, null, null)
        
        return Result.success(
            AuthResponse(
                accessToken = tokens.accessToken,
                refreshToken = tokens.refreshToken,
                expiresIn = JwtConfig.getExpirationTime() / 1000,
                tokenType = "Bearer",
                user = UserDto.fromUser(user, roles)
            )
        )
    }
    
    /**
     * 登出（撤回所有 Token）
     */
    suspend fun logout(userId: Int): Result<Unit> {
        refreshTokenRepository.revokeAllForUser(userId)
        return Result.success(Unit)
    }
    
    /**
     * 获取用户信息
     */
    suspend fun getUserInfo(userId: Int): Result<UserDto> {
        val user = userRepository.findById(userId)
            ?: return Result.failure(NoSuchElementException("用户不存在"))
        
        val roles = userRepository.getUserRoles(userId)
        
        return Result.success(UserDto.fromUser(user, roles))
    }
    
    // ==================== 私有方法 ====================
    
    /**
     * 生成 Access Token 和 Refresh Token
     */
    private suspend fun generateTokens(
        user: User,
        roles: List<String>,
        deviceInfo: String?,
        ipAddress: String?
    ): TokenPair {
        // 生成 Access Token
        val accessToken = JwtConfig.generateAccessToken(
            userId = user.id,
            username = user.username,
            email = user.email,
            roles = roles
        )
        
        // 生成 Refresh Token
        val tokenId = UUID.randomUUID().toString()
        val refreshToken = JwtConfig.generateRefreshToken(user.id, tokenId)
        val refreshHash = hashToken(refreshToken)
        
        // 计算过期时间
        val expiresAt = LocalDateTime.now().plusSeconds(
            JwtConfig.getRefreshExpirationTime() / 1000
        )
        
        // 保存 Refresh Token
        refreshTokenRepository.save(
            tokenId = tokenId,
            userId = user.id,
            tokenHash = refreshHash,
            deviceInfo = deviceInfo,
            ipAddress = ipAddress,
            expiresAt = expiresAt
        )
        
        return TokenPair(accessToken, refreshToken)
    }
    
    /**
     * Hash Token（用于存储）
     */
    private fun hashToken(token: String): String {
        val digest = MessageDigest.getInstance("SHA-256")
        val hash = digest.digest(token.toByteArray())
        return hash.joinToString("") { "%02x".format(it) }
    }
    
    data class TokenPair(
        val accessToken: String,
        val refreshToken: String
    )
}
```

```kotlin
class OAuthService(

     */
    fun getGitHubAuthUrl(state: String): String {
        val redirectUri = "http://localhost:8080/api/oauth/github/callback"
        return "${githubConfig.authorizeUrl}" +
            "?client_id=${githubConfig.clientId}" +
            "&redirect_uri=$redirectUri" +
            "&scope=read:user,user:email" +
            "&state=$state"
    }

     */

        return "${googleConfig.authorizeUrl}" +
            "?client_id=${googleConfig.clientId}" +
            "&redirect_uri=$redirectUri" +
            "&scope=openid,email,profile" +
            "&response_type=code" +
            "&state=$state"
    }

     * GitHub OAuth 回调
     */
    suspend fun handleGitHubCallback(
        code: String,
        state: String,
        deviceInfo: String?,
        ipAddress: String?
    ): Result<OAuthAuthResponse> {
        return withContext(Dispatchers.IO) {
            try {
                // ===== 1. 用 code 换取 access_token =====
                val accessToken = exchangeCodeForToken(code, githubConfig)
                
                // ===== 2. 用 access_token 获取用户信息 =====
                val githubUser = getGitHubUserInfo(accessToken)
                
                // ===== 3. 查找或创建用户 =====
                var user = userRepository.findByProvider("github", githubUser.id)
                
                if (user == null) {
                    // 新用户：创建
                    val email = githubUser.email ?: "${githubUser.id}@github.placeholder"
                    user = userRepository.createOAuthUser(
                        username = githubUser.login,
                        email = email,
                        provider = "github",
                        providerId = githubUser.id,
                        avatarUrl = githubUser.avatarUrl
                    )
                    
                    // 分配默认角色
                    userRepository.assignRole(user.id, Role.USER.name)
                }
                
                // ===== 4. 生成 Token =====
                val roles = userRepository.getUserRoles(user.id)
                val tokens = generateTokens(user, roles, deviceInfo, ipAddress)
                
                Result.success(
                    OAuthAuthResponse(
                        accessToken = tokens.accessToken,
                        refreshToken = tokens.refreshToken,
                        expiresIn = JwtConfig.getExpirationTime() / 1000,
                        tokenType = "Bearer",
                        user = UserDto.fromUser(user, roles),
                        isNewUser = userRepository.findByProvider("github", githubUser.id) == null
                    )
                )
            } catch (e: Exception) {
                Result.failure(e)
            }
        }
    }
    
    /**
     * 用 code 换取 access_token
     */
    private suspend fun exchangeCodeForToken(code: String, config: OAuthConfig): String {
        val response: TokenResponse = httpClient.post(config.tokenUrl) {
            contentType(ContentType.Application.Json)
            setBody(mapOf(
                "client_id" to config.clientId,
                "client_secret" to config.clientSecret,
                "code" to code
            ))
        }.body()
        
        return response.accessToken
    }

    private suspend fun getGitHubUserInfo(accessToken: String): GitHubUser {

            tokenId = tokenId,
            userId = user.id,
            tokenHash = refreshHash,

            ipAddress = ipAddress,
```

```kotlin
            val ipAddress = call.request.origin.remoteAddress
            
            val result = oAuthService.handleGitHubCallback(code, state, deviceInfo, ipAddress)
            
            result.fold(
                onSuccess = { response ->
                    // 返回 Token（也可以重定向到前端页面）
                    call.respond(io.ktor.http.HttpStatusCode.OK, response)
                },
                onFailure = { error ->
                    call.respond(
                        io.ktor.http.HttpStatusCode.BadRequest,
                        ErrorResponse(error.message ?: "OAuth 登录失败")
                    )
                }
            )
        }
        
        /**
         * GET /apioauth/google - 获取 Google 授权 URL
         */
        get("/google") {
            val state = UUID.randomUUID().toString()
            val url = oAuthService.getGoogleAuthUrl(state)
            
            call.respond(OAuthUrlResponse(url, state))
        }
    }
}
```

## 代码关键点说明

### 1. JWT Token 结构

**关键点**：JWT Token 包含标准声明和自定义声明。

```kotlin
// JWT Token 结构
data class JWTClaims(
    // 标准声明
    iss: String,      // 颁发者
    sub: String,       // 主题（用户 ID）
    aud: String,       // 观众
    exp: Long,        // 过期时间
    iat: Long,        // 签发时间
    
    // 自定义声明
    username: String,
    email: String,
    roles: String,    // "admin,user" 格式
    type: String      // "access" 或 "refresh"
)
```

### 2. Access Token vs Refresh Token

**关键点**：两种 Token 有不同的用途和生命周期。

| 特性 | Access Token | Refresh Token |
|------|-------------|---------------|
| 有效期 | 15 分钟 | 7 天 |
| 存储 | 内存 | HttpOnly Cookie |
| 用途 | API 认证 | 刷新 Access Token |
| 包含信息 | 用户 ID、角色 | Token ID |

### 3. RBAC 权限控制

**关键点**：基于角色的访问控制。

```kotlin
// 检查用户是否有权限
fun hasPermission(roles: List<String>, requiredPermission: String): Boolean {
    // ADMIN 角色拥有所有权限
    if (roles.contains("ADMIN")) return true
    
    // 检查具体权限
    return permissions[roles.firstOrNull()]?.contains(requiredPermission) ?: false
}

// 使用示例
if (hasPermission(userRoles, "manage:users")) {
    // 执行管理操作
}
```

## 面试要点

### 题目1：JWT 的优缺点是什么？

**标准答案**：

**优点**：
1. **无状态**：服务端不需要存储 Token，减小存储压力
2. **可扩展**：可以在 Token 中包含用户信息
3. **跨域**：适合分布式系统和微服务
4. **性能**：验证 Token 比查询数据库快

**缺点**：
1. **无法撤回**：Token 过期前无法主动失效
2. **数据量**：Token 越大，网络传输越慢
3. **安全**：Token 泄露后可以被使用
4. **时间_pages页面**：必须等待 Token 过期

**解决方案**：
- 短期 Access Token + Refresh Token
- Token 黑名单（Redis）
- 敏感操作需要额外验证

### 题目2：Kotlin 协程作用域（Coroutine Scope）是什么？

**标准答案**：

协程作用域定义了协程的执行范围。

```kotlin
// 全局作用域 - 应用级别
GlobalScope.launch {
    // 全局协程，整个应用生命周期
}

// 视图模型作用域（Android）
viewModelScope.launch {
    // 与 ViewModel 生命周期绑定
}

// 自定义作用域
val myScope = CoroutineScope(Dispatchers.IO + SupervisorJob())
myScope.launch {
    // 自定义协程
}
```

**作用域传播**：
```kotlin
suspend fun parent() {
    // 这个协程会在 parent 的作用域中执行
    child()
}

suspend fun child() {
    // 子协程会自动继承父协程的上下文
}
```

### 题目3：Kotlin 的 lateinit 和 lazy 有什么区别？

**标准答案**：

| 特性 | lateinit | lazy |
|------|----------|------|
| 适用类型 | var | val |
| 初始化时机 | 代码块中赋值 | 首次访问时 |
| 线程安全 | 否 | 是（默认） |
| 失败处理 | UninitializedPropertyAccessException | - |
| 用途 | 需要重新赋值的依赖注入 | 只读的单例资源 |

```kotlin
// lateinit - 非空，但暂时无法初始化
class MyService {
    lateinit var database: Database  // 稍后在构造函数或 init 中初始化
    
    fun initialize() {
        database = connect()  // 赋值
    }
    
    fun doSomething() {
        println(database.name)  // 使用
    }
}

// lazy - 延迟初始化，只读
class MyService {
    val config: Config by lazy {
        loadConfig()  // 首次访问时执行
    }
}
```
