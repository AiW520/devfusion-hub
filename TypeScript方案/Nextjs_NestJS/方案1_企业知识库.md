# 方案四：Next.js + NestJS 企业知识库

## 项目介绍

### 项目是什么
这是一个企业级知识库系统，支持文档创建、分类管理、全文搜索、权限控制、版本管理等功能。

### 解决什么问题
- 企业内部知识分散、难以查找
- 文档版本混乱、权限不清
- 展示完整的全栈企业级架构

### 核心技术亮点
- **Next.js 14** 前端框架
- **NestJS** 企业级后端框架
- **PostgreSQL + Prisma** 数据库
- **Elasticsearch** 全文搜索
- **JWT + RBAC** 权限系统
- **模块化架构** 清晰分层

### 为什么选这个方案
1. **NestJS** 是企业级 Node.js 框架，面试必问
2. **依赖注入** 是核心概念，体现架构设计能力
3. **装饰器** 是 TypeScript 高级用法
4. **模块化** 适合复杂业务系统

```typescript
/**
 * NestJS 应用入口
 * 
 * NestJS 使用依赖注入和装饰器来组织代码
 * 这与传统 Express 应用有本质区别
 */

import { NestFactory } from '@nestjs/core';
import { ValidationPipe, VersioningType } from '@nestjs/common';
import { SwaggerModule, DocumentBuilder } from '@nestjs/swagger';
import { AppModule } from './app.module';
import { ConfigService } from '@nestjs/config';
import { HttpExceptionFilter } from './common/filters/http-exception.filter';
import { TransformInterceptor } from './common/interceptors/transform.interceptor';

async function bootstrap() {
  // 创建应用实例
  const app = await NestFactory.create(AppModule);
  
  // 获取配置服务
  const configService = app.get(ConfigService);
  
  // ============ 全局中间件配置 ============
  
  // 启用 CORS
  app.enableCors({
    origin: configService.get('CORS_ORIGINS')?.split(',') || ['http://localhost:3000'],
    credentials: true,
  });
  
  // API 版本控制
  app.enableVersioning({
    type: VersioningType.URI,  // /api/v1/...
    defaultVersion: '1',
  });
  
  // 全局前缀
  app.setGlobalPrefix('api');
  
  // ============ 全局管道 ============
  
  // ValidationPipe 用于验证请求数据
  app.useGlobalPipes(
    new ValidationPipe({
      whitelist: true,         // 去除不在 DTO 中定义的字段
      transform: true,         // 自动类型转换
      transformOptions: {
        enableImplicitConversion: true,  // 隐式类型转换
      },
      forbidNonWhitelisted: true,  // 禁止额外字段
    }),
  );
  
  // ============ 全局拦截器 ============
  
  // TransformInterceptor 统一响应格式
  app.useGlobalInterceptors(new TransformInterceptor());
  
  // ============ 全局过滤器 ============
  
  // 统一异常处理
  app.useGlobalFilters(new HttpExceptionFilter());
  
  // ============ Swagger API 文档 ============
  
  const swaggerConfig = new DocumentBuilder()
    .setTitle('知识库 API')
    .setDescription('企业知识库系统 API 文档')
    .setVersion('1.0')
    .addBearerAuth()  // 添加 JWT 认证
    .build();
  
  const document = SwaggerModule.createDocument(app, swaggerConfig);
  SwaggerModule.setup('docs', app, document);
  
  // ============ 启动服务器 ============
  
  const port = configService.get('PORT') || 3000;
  await app.listen(port);
  
  console.log(`🚀 服务器已启动: http://localhost:${port}`);
  console.log(`📚 API 文档: http://localhost:${port}/docs`);
}

bootstrap();
```

**backend/src/auth/auth.service.ts**
```typescript
/**
 * 认证服务
 * 
 * 依赖注入是 NestJS 的核心概念
 * 通过构造函数注入依赖，NestJS 会自动解析
 */

import { Injectable, UnauthorizedException, ConflictException } from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { ConfigService } from '@nestjs/config';
import * as bcrypt from 'bcryptjs';
import { UsersService } from '../users/users.service';
import { CreateUserDto, UserResponseDto } from '../users/dto';
import { LoginDto } from './dto/login.dto';

@Injectable()
export class AuthService {
  constructor(
    // 依赖注入：NestJS 自动解析这些服务
    private readonly usersService: UsersService,
    private readonly jwtService: JwtService,
    private readonly configService: ConfigService,
  ) {}

  /**
   * 用户注册
   */
  async register(createUserDto: CreateUserDto): Promise<{ user: UserResponseDto; token: string }> {
    // 1. 检查邮箱是否已存在
    const existingUser = await this.usersService.findByEmail(createUserDto.email);
    if (existingUser) {
      throw new ConflictException('该邮箱已被注册');
    }
    
    // 2. 检查用户名是否已存在
    const existingUsername = await this.usersService.findByUsername(createUserDto.username);
    if (existingUsername) {
      throw new ConflictException('该用户名已被使用');
    }
    
    // 3. 加密密码
    const hashedPassword = await bcrypt.hash(createUserDto.password, 10);
    
    // 4. 创建用户
    const user = await this.usersService.create({
      ...createUserDto,
      password: hashedPassword,
    });
    
    // 5. 生成 Token
    const token = await this.generateToken(user);
    
    return {
      user: this.sanitizeUser(user),
      token,
    };
  }

  /**
   * 用户登录
   */
  async login(loginDto: LoginDto): Promise<{ user: UserResponseDto; token: string }> {
    // 1. 查找用户（包含密码用于验证）
    const user = await this.usersService.findByEmailWithPassword(loginDto.email);
    if (!user) {
      throw new UnauthorizedException('邮箱或密码错误');
    }
    
    // 2. 验证密码
    const isPasswordValid = await bcrypt.compare(loginDto.password, user.password);
    if (!isPasswordValid) {
      throw new UnauthorizedException('邮箱或密码错误');
    }
    
    // 3. 检查用户是否激活
    if (!user.isActive) {
      throw new UnauthorizedException('账户已被禁用');
    }
    
    // 4. 生成 Token
    const token = await this.generateToken(user);
    
    return {
      user: this.sanitizeUser(user),
      token,
    };
  }

  /**
   * 获取当前用户信息
   */
  async getProfile(userId: string): Promise<UserResponseDto> {
    const user = await this.usersService.findById(userId);
    if (!user) {
      throw new UnauthorizedException('用户不存在');
    }
    return this.sanitizeUser(user);
  }

  /**
   * 生成 JWT Token
   */
  private async generateToken(user: any): Promise<string> {
    const payload = {
      sub: user.id,          // subject：用户 ID
      email: user.email,
      username: user.username,
      role: user.role,
    };
    
    return this.jwtService.sign(payload);
  }

  /**
   * 移除敏感信息
   */
  private sanitizeUser(user: any): UserResponseDto {
    const { password, ...result } = user;
    return result;
  }
}
```

```typescript
/**
 * JWT 策略
 * 
 * Passport 的策略模式：
 * 1. 定义如何验证用户
 * 2. 验证成功后返回用户信息
 */

import { Injectable, UnauthorizedException } from '@nestjs/common';
import { PassportStrategy } from '@nestjs/passport';
import { ExtractJwt, Strategy } from 'passport-jwt';
import { ConfigService } from '@nestjs/config';
import { UsersService } from '../../users/users.service';

// JWT Payload 类型
interface JwtPayload {
  sub: string;
  email: string;
  username: string;
  role: string;
}

@Injectable()
export class JwtStrategy extends PassportStrategy(Strategy) {
  constructor(
    private readonly usersService: UsersService,
    configService: ConfigService,
  ) {
    super({
      // 从请求头提取 Token
      jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),
      
      // 忽略过期（NestJS Passport 会自动处理）
      ignoreExpiration: false,
      
      // JWT 密钥
      secretOrKey: configService.get('JWT_SECRET'),
    });
  }

  /**
   * 验证并返回用户
   * 
   * Passport 会自动：
   * 1. 验证 JWT 签名
   * 2. 检查过期时间
   * 3. 调用 validate 方法
   */
  async validate(payload: JwtPayload) {
    const user = await this.usersService.findById(payload.sub);
    
    if (!user || !user.isActive) {
      throw new UnauthorizedException('用户不存在或已被禁用');
    }
    
    // 返回的用户会附加到 req.user
    return {
      id: payload.sub,
      email: payload.email,
      username: payload.username,
      role: payload.role,
    };
  }
}
```

```typescript
/**
 * JWT 认证守卫
 * 
 * 守卫是请求管道的一部分
 * 用于在处理请求之前进行权限检查
 */

import { Injectable, ExecutionContext } from '@nestjs/common';
import { AuthGuard } from '@nestjs/passport';
import { Reflector } from '@nestjs/core';
import { IS_PUBLIC_KEY } from '../decorators/public.decorator';

@Injectable()
export class JwtAuthGuard extends AuthGuard('jwt') {
  constructor(private reflector: Reflector) {
    super();
  }

  /**
   * canActivate 判断请求是否可以继续
   */
  canActivate(context: ExecutionContext) {
    // 检查是否是公开路由
    const isPublic = this.reflector.getAllAndOverride<boolean>(IS_PUBLIC_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);
    
    if (isPublic) {
      return true;
    }
    
    // 其他路由需要 JWT 认证
    return super.canActivate(context);
  }
}
```

```typescript
/**
 * 文档服务
 * 
 * 展示 NestJS 的依赖注入和模块化
 */

import { Injectable, NotFoundException, ForbiddenException } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { PrismaService } from 'nestjs-prisma';
import { CreateDocumentDto, UpdateDocumentDto, DocumentQueryDto } from './dto';
import { SearchService } from '../search/search.service';

@Injectable()
export class DocumentsService {
  constructor(
    private readonly prisma: PrismaService,
    private readonly searchService: SearchService,
    private readonly configService: ConfigService,
  ) {}

  /**
   * 创建文档
   */
  async create(
    createDto: CreateDocumentDto,
    authorId: string,
  ) {
    const { categoryId, tags, ...data } = createDto;
    
    // 生成 slug
    const slug = await this.generateSlug(data.title, authorId);
    
    // 创建文档
    const document = await this.prisma.document.create({
      data: {
        ...data,
        slug,
        authorId,
        categoryId: categoryId || null,
        tags: tags || [],
        status: data.status || 'DRAFT',
      },
      include: {
        author: {
          select: { id: true, username: true, avatar: true },
        },
        category: {
          select: { id: true, name: true, slug: true },
        },
      },
    });
    
    // 创建初始版本
    await this.prisma.documentVersion.create({
      data: {
        documentId: document.id,
        version: 1,
        title: document.title,
        content: document.content,
      },
    });
    
    // 同步到搜索引擎
    await this.searchService.indexDocument(document);
    
    return document;
  }

  /**
   * 获取文档列表
   */
  async findAll(query: DocumentQueryDto, userId?: string) {
    const {
      page = 1,
      limit = 10,
      categoryId,
      status,
      search,
      authorId,
      tag,
    } = query;
    
    // 构建查询条件
    const where: any = {};
    
    // 公开文档或作者本人
    if (userId) {
      where.OR = [
        { isPublic: true },
        { authorId: userId },
        { accessList: { some: { userId } } },
      ];
    } else {
      where.isPublic = true;
    }
    
    // 状态筛选（未登录只能看已发布）
    if (status) {
      where.status = status;
    } else if (!userId) {
      where.status = 'PUBLISHED';
    }
    
    // 分类筛选
    if (categoryId) {
      where.categoryId = categoryId;
    }
    
    // 作者筛选
    if (authorId) {
      where.authorId = authorId;
    }
    
    // 标签筛选
    if (tag) {
      where.tags = { has: tag };
    }
    
    // 搜索
    if (search) {
      where.OR = [
        { title: { contains: search, mode: 'insensitive' } },
        { content: { contains: search, mode: 'insensitive' } },
      ];
    }
    
    // 执行查询
    const [documents, total] = await Promise.all([
      this.prisma.document.findMany({
        where,
        include: {
          author: {
            select: { id: true, username: true, avatar: true },
          },
          category: {
            select: { id: true, name: true, slug: true },
          },
        },
        orderBy: { updatedAt: 'desc' },
        skip: (page - 1) * limit,
        take: limit,
      }),
      this.prisma.document.count({ where }),
    ]);
    
    return {
      documents,
      meta: {
        page,
        limit,
        total,
        totalPages: Math.ceil(total / limit),
      },
    };
  }

  /**
   * 获取单个文档
   */
  async findOne(id: string, userId?: string) {
    const document = await this.prisma.document.findUnique({
      where: { id },
      include: {
        author: {
          select: { id: true, username: true, avatar: true, bio: true },
        },
        category: true,
        comments: {
          where: { parentId: null },
          include: {
            author: {
              select: { id: true, username: true, avatar: true },
            },
            replies: {
              include: {
                author: {
                  select: { id: true, username: true, avatar: true },
                },
              },
            },
          },
          orderBy: { createdAt: 'desc' },
        },
        versions: {
          orderBy: { version: 'desc' },
          take: 10,
        },
      },
    });
    
    if (!document) {
      throw new NotFoundException('文档不存在');
    }
    
    // 检查访问权限
    const hasAccess = document.isPublic || 
                      document.authorId === userId ||
                      (userId && await this.hasDocumentAccess(id, userId));
    
    if (!hasAccess) {
      throw new ForbiddenException('无权访问此文档');
    }
    
    // 增加阅读数
    await this.prisma.document.update({
      where: { id },
      data: { viewCount: { increment: 1 } },
    });
    
    return document;
  }

  /**
   * 更新文档
   */
  async update(
    id: string,
    updateDto: UpdateDocumentDto,
    userId: string,
  ) {
    // 检查文档存在和权限
    const document = await this.prisma.document.findUnique({
      where: { id },
      include: { versions: { orderBy: { version: 'desc' }, take: 1 } },
    });
    
    if (!document) {
      throw new NotFoundException('文档不存在');
    }
    
    // 检查权限
    const hasAccess = document.authorId === userId || 
                      await this.hasDocumentPermission(id, userId, 'EDIT');
    
    if (!hasAccess) {
      throw new ForbiddenException('无权修改此文档');
    }
    
    const { categoryId, tags, ...data } = updateDto;
    
    // 创建新版本
    const newVersion = (document.versions[0]?.version || 0) + 1;
    
    // 更新文档并创建版本记录
    const [updated] = await this.prisma.$transaction([
      this.prisma.document.update({
        where: { id },
        data: {
          ...data,
          categoryId: categoryId || undefined,
          tags: tags || undefined,
        },
        include: {
          author: {
            select: { id: true, username: true, avatar: true },
          },
          category: true,
        },
      }),
      this.prisma.documentVersion.create({
        data: {
          documentId: id,
          version: newVersion,
          title: data.title || document.title,
          content: data.content || document.content,
        },
      }),
    ]);
    
    // 更新搜索引擎
    await this.searchService.indexDocument(updated);
    
    return updated;
  }

  /**
   * 删除文档
   */
  async remove(id: string, userId: string) {
    const document = await this.prisma.document.findUnique({
      where: { id },
    });
    
    if (!document) {
      throw new NotFoundException('文档不存在');
    }
    
    // 只有作者可以删除
    if (document.authorId !== userId) {
      throw new ForbiddenException('无权删除此文档');
    }
    
    // 删除文档（级联删除版本、评论等）
    await this.prisma.document.delete({ where: { id } });
    
    // 从搜索引擎移除
    await this.searchService.removeDocument(id);
    
    return { success: true };
  }

  /**
   * 生成唯一 slug
   */
  private async generateSlug(title: string, authorId: string): Promise<string> {
    const baseSlug = title
      .toLowerCase()
      .replace(/[^a-z0-9\u4e00-\u9fa5]+/g, '-')
      .replace(/^-|-$/g, '');
    
    let slug = baseSlug;
    let counter = 1;
    
    while (
      await this.prisma.document.findUnique({
        where: { authorId_slug: { authorId, slug } },
      })
    ) {
      slug = `${baseSlug}-${counter}`;
      counter++;
    }
    
    return slug;
  }

  /**
   * 检查文档访问权限
   */
  private async hasDocumentAccess(documentId: string, userId: string): Promise<boolean> {
    const access = await this.prisma.documentAccess.findUnique({
      where: {
        documentId_userId: { documentId, userId },
      },
    });
    return !!access;
  }

  /**
   * 检查文档特定权限
   */
  private async hasDocumentPermission(
    documentId: string, 
    userId: string, 
    permission: string
  ): Promise<boolean> {
    const access = await this.prisma.documentAccess.findUnique({
      where: {
        documentId_userId: { documentId, userId },
      },
    });
    
    if (!access) return false;
    
    const permissionLevels: Record<string, number> = {
      VIEW: 1,
      COMMENT: 2,
      EDIT: 3,
      ADMIN: 4,
    };
    
    return permissionLevels[access.permission] >= permissionLevels[permission];
  }
}
```

```typescript
      message,
```

### 问题 1: NestJS 的依赖注入是如何工作的？

**参考答案**:
```
1. 容器（Container）
   - NestJS 创建 IoC 容器管理所有实例

2. 依赖解析
   - 构造函数参数声明依赖
   - NestJS 自动实例化并注入

3. 作用域
   - DEFAULT: 单例
   - REQUEST: 每个请求创建新实例
   - TRANSIENT: 每次注入创建新实例

4. 装饰器
   - @Injectable() 标记可注入
   - @Inject() 手动指定 token
   - @Optional() 标记可选依赖
```
