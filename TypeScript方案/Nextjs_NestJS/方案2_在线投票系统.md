# 方案四：Next.js + NestJS 在线投票系统

## 项目介绍

### 项目是什么
这是一个实时投票系统，支持创建投票、实时投票统计、多选/单选、截止时间、结果可视化等功能。

### 核心技术亮点
- **NestJS** 后端框架
- **WebSocket** 实时投票
- **Chart.js** 数据可视化
- **TypeScript** 严格类型

### 3. 投票 WebSocket 网关

**backend/src/polls/polls.gateway.ts**
```typescript
/**
 * 投票 WebSocket 网关
 * 
 * 实现实时投票功能
 */

import {
  WebSocketGateway,
  WebSocketServer,
  SubscribeMessage,
  OnGatewayConnection,
  OnGatewayDisconnect,
  ConnectedSocket,
  MessageBody,
} from '@nestjs/websockets';
import { Server, Socket } from 'socket.io';
import { UseGuards } from '@nestjs/common';
import { WsJwtGuard } from '../auth/guards/ws-jwt.guard';

interface VotePayload {
  pollId: string;
  optionIds: string[];
}

@WebSocketGateway({
  cors: {
    origin: '*',
  },
  namespace: '/polls',
})
export class PollsGateway implements OnGatewayConnection, OnGatewayDisconnect {
  @WebSocketServer()
  server: Server;
  
  // 在线用户映射
  private onlineUsers = new Map<string, string>();  // socketId -> userId
  private userSockets = new Map<string, Set<string>>();  // userId -> Set<socketId>

  /**
   * 客户端连接
   */
  handleConnection(client: Socket) {
    // 从 handshake 获取 token
    const token = client.handshake.auth.token || 
                  client.handshake.headers.authorization?.split(' ')[1];
    
    if (token) {
      try {
        // 验证 token（需要在 guard 中实现）
        const userId = this.verifyToken(token);
        
        this.onlineUsers.set(client.id, userId);
        
        if (!this.userSockets.has(userId)) {
          this.userSockets.set(userId, new Set());
        }
        this.userSockets.get(userId)!.add(client.id);
        
        console.log(`用户 ${userId} 已连接 (${client.id})`);
      } catch {
        // token 无效，不允许连接
        client.disconnect();
      }
    }
  }

  /**
   * 客户端断开
   */
  handleDisconnect(client: Socket) {
    const userId = this.onlineUsers.get(client.id);
    
    if (userId) {
      this.onlineUsers.delete(client.id);
      
      const sockets = this.userSockets.get(userId);
      if (sockets) {
        sockets.delete(client.id);
        if (sockets.size === 0) {
          this.userSockets.delete(userId);
        }
      }
      
      console.log(`用户 ${userId} 已断开`);
    }
  }

  /**
   * 加入投票房间
   */
  @SubscribeMessage('join:poll')
  handleJoinPoll(
    @ConnectedSocket() client: Socket,
    @MessageBody() pollId: string,
  ) {
    client.join(`poll:${pollId}`);
    console.log(`客户端 ${client.id} 加入投票 ${pollId}`);
    
    // 返回当前在线人数
    const room = this.server.sockets.adapter.rooms.get(`poll:${pollId}`);
    const onlineCount = room?.size || 0;
    
    client.emit('poll:online', { pollId, count: onlineCount });
  }

  /**
   * 离开投票房间
   */
  @SubscribeMessage('leave:poll')
  handleLeavePoll(
    @ConnectedSocket() client: Socket,
    @MessageBody() pollId: string,
  ) {
    client.leave(`poll:${pollId}`);
    console.log(`客户端 ${client.id} 离开投票 ${pollId}`);
  }

  /**
   * 广播投票更新
   * 服务端调用，向所有订阅者推送投票结果
   */
  broadcastVoteUpdate(pollId: string, options: { id: string; voteCount: number }[]) {
    this.server.to(`poll:${pollId}`).emit('poll:update', {
      pollId,
      options,
      timestamp: Date.now(),
    });
  }

  /**
   * 广播新投票创建
   */
  broadcastNewPoll(poll: any) {
    this.server.emit('poll:new', poll);
  }

  /**
   * 广播投票结束
   */
  broadcastPollEnded(pollId: string, finalResults: any) {
    this.server.to(`poll:${pollId}`).emit('poll:ended', {
      pollId,
      results: finalResults,
    });
  }

  /**
   * 获取房间在线人数
   */
  getRoomOnlineCount(pollId: string): number {
    const room = this.server.sockets.adapter.rooms.get(`poll:${pollId}`);
    return room?.size || 0;
  }

  /**
   * 验证 token（简化实现）
   */
  private verifyToken(token: string): string {
    // 实际应该使用 JWT 验证
    // 这里返回简化的 userId
    return token;
  }
}
```

### 4. 投票服务

**backend/src/polls/polls.service.ts**
```typescript
/**
 * 投票服务
 */

import { Injectable, NotFoundException, ForbiddenException } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';
import { PollsGateway } from './polls.gateway';
import { CreatePollDto, VoteDto } from './dto';

@Injectable()
export class PollsService {
  constructor(
    private readonly prisma: PrismaService,
    private readonly pollsGateway: PollsGateway,
  ) {}

  /**
   * 创建投票
   */
  async create(createDto: CreatePollDto, authorId: string) {
    const { options, ...pollData } = createDto;
    
    const poll = await this.prisma.poll.create({
      data: {
        ...pollData,
        authorId,
        options: {
          create: options.map((text, index) => ({
            text,
            voteCount: 0,
          })),
        },
      },
      include: {
        author: {
          select: { id: true, username: true },
        },
        options: true,
      },
    });
    
    // 广播新投票
    this.pollsGateway.broadcastNewPoll(poll);
    
    return poll;
  }

  /**
   * 获取投票列表
   */
  async findAll(page = 1, limit = 10) {
    const [polls, total] = await Promise.all([
      this.prisma.poll.findMany({
        where: { isActive: true },
        include: {
          author: {
            select: { id: true, username: true },
          },
          options: {
            select: { id: true, text: true, voteCount: true },
          },
          _count: {
            select: { votes: true },
          },
        },
        orderBy: { createdAt: 'desc' },
        skip: (page - 1) * limit,
        take: limit,
      }),
      this.prisma.poll.count({ where: { isActive: true } }),
    ]);
    
    return {
      polls,
      meta: { page, limit, total, totalPages: Math.ceil(total / limit) },
    };
  }

  /**
   * 获取单个投票
   */
  async findOne(id: string) {
    const poll = await this.prisma.poll.findUnique({
      where: { id },
      include: {
        author: {
          select: { id: true, username: true },
        },
        options: true,
        _count: {
          select: { votes: true },
        },
      },
    });
    
    if (!poll) {
      throw new NotFoundException('投票不存在');
    }
    
    // 获取在线人数
    const onlineCount = this.pollsGateway.getRoomOnlineCount(id);
    
    return { ...poll, onlineCount };
  }

  /**
   * 投票
   */
  async vote(pollId: string, voteDto: VoteDto, userId: string) {
    const poll = await this.prisma.poll.findUnique({
      where: { id: pollId },
      include: { options: true },
    });
    
    if (!poll) {
      throw new NotFoundException('投票不存在');
    }
    
    // 检查是否已投票
    const existingVote = await this.prisma.vote.findUnique({
      where: {
        pollId_userId: { pollId, userId },
      },
    });
    
    if (existingVote) {
      throw new ForbiddenException('您已经投过票了');
    }
    
    // 检查投票是否已开始
    if (new Date() < poll.startsAt) {
      throw new ForbiddenException('投票尚未开始');
    }
    
    // 检查投票是否已结束
    if (poll.endsAt && new Date() > poll.endsAt) {
      throw new ForbiddenException('投票已结束');
    }
    
    // 检查选项是否有效
    const optionIds = voteDto.optionIds;
    if (optionIds.length > poll.maxChoices) {
      throw new ForbiddenException(`最多只能选择 ${poll.maxChoices} 个选项`);
    }
    
    const validOptionIds = poll.options.map(o => o.id);
    for (const id of optionIds) {
      if (!validOptionIds.includes(id)) {
        throw new ForbiddenException('无效的选项');
      }
    }
    
    // 记录投票
    await this.prisma.$transaction([
      // 创建投票记录
      this.prisma.vote.create({
        data: {
          pollId,
          userId,
          optionIds,
        },
      }),
      // 更新每个选项的投票数
      ...optionIds.map(optionId =>
        this.prisma.pollOption.update({
          where: { id: optionId },
          data: { voteCount: { increment: 1 } },
        })
      ),
    ]);
    
    // 获取更新后的选项投票数
    const updatedOptions = await this.prisma.pollOption.findMany({
      where: { pollId },
      select: { id: true, voteCount: true },
    });
    
    // 广播更新
    this.pollsGateway.broadcastVoteUpdate(pollId, updatedOptions);
    
    return { success: true };
  }

  /**
   * 获取投票结果
   */
  async getResults(id: string) {
    const poll = await this.prisma.poll.findUnique({
      where: { id },
      include: {
        options: {
          select: { id: true, text: true, voteCount: true },
        },
        _count: {
          select: { votes: true },
        },
      },
    });
    
    if (!poll) {
      throw new NotFoundException('投票不存在');
    }
    
    // 计算百分比
    const totalVotes = poll._count.votes;
    const results = poll.options.map(option => ({
      ...option,
      percentage: totalVotes > 0 
        ? Math.round((option.voteCount / totalVotes) * 100) 
        : 0,
    }));
    
    return {
      pollId: id,
      totalVotes,
      results,
      userVote: null, // 如果需要，可以在查询时传入 userId
    };
  }

  /**
   * 结束投票
   */
  async endPoll(id: string, userId: string) {
    const poll = await this.prisma.poll.findUnique({
      where: { id },
      include: {
        options: true,
        _count: { select: { votes: true } },
      },
    });
    
    if (!poll) {
      throw new NotFoundException('投票不存在');
    }
    
    if (poll.authorId !== userId) {
      throw new ForbiddenException('只有创建者可以结束投票');
    }
    
    await this.prisma.poll.update({
      where: { id },
      data: { isActive: false },
    });
    
    // 广播结束
    const results = await this.getResults(id);
    this.pollsGateway.broadcastPollEnded(id, results);
    
    return { success: true };
  }
}
```

### 5. 前端组件

```typescript
// frontend/src/components/PollVote.tsx
'use client';

import { useEffect, useState } from 'react';
import { io, Socket } from 'socket.io-client';
import { Pie, Bar } from 'react-chartjs-2';
import {
  Chart as ChartJS,
  ArcElement,
  Tooltip,
  Legend,
  CategoryScale,
  LinearScale,
  BarElement,
} from 'chart.js';

ChartJS.register(ArcElement, Tooltip, Legend, CategoryScale, LinearScale, BarElement);

interface PollOption {
  id: string;
  text: string;
  voteCount: number;
}

interface Poll {
  id: string;
  title: string;
  description?: string;
  maxChoices: number;
  options: PollOption[];
  totalVotes: number;
  hasVoted: boolean;
}

export function PollVote({ poll: initialPoll }: { poll: Poll }) {
  const [poll, setPoll] = useState(initialPoll);
  const [selectedOptions, setSelectedOptions] = useState<string[]>([]);
  const [socket, setSocket] = useState<Socket | null>(null);
  const [isConnected, setIsConnected] = useState(false);

  useEffect(() => {
    // 连接 WebSocket
    const socket = io(process.env.NEXT_PUBLIC_WS_URL + '/polls', {
      auth: { token: localStorage.getItem('token') },
    });

    socket.on('connect', () => {
      setIsConnected(true);
      socket.emit('join:poll', initialPoll.id);
    });

    socket.on('disconnect', () => {
      setIsConnected(false);
    });

    // 监听投票更新
    socket.on('poll:update', (data: { pollId: string; options: PollOption[] }) => {
      if (data.pollId === initialPoll.id) {
        setPoll(prev => ({
          ...prev,
          options: data.options,
          totalVotes: data.options.reduce((sum, o) => sum + o.voteCount, 0),
        }));
      }
    });

    // 监听投票结束
    socket.on('poll:ended', (data: { pollId: string; results: any }) => {
      if (data.pollId === initialPoll.id) {
        setPoll(prev => ({ ...prev, hasVoted: true }));
      }
    });

    setSocket(socket);

    return () => {
      socket.emit('leave:poll', initialPoll.id);
      socket.disconnect();
    };
  }, [initialPoll.id]);

  const handleOptionClick = (optionId: string) => {
    if (poll.hasVoted) return;
    
    setSelectedOptions(prev => {
      if (prev.includes(optionId)) {
        return prev.filter(id => id !== optionId);
      }
      
      if (prev.length >= poll.maxChoices) {
        return prev;
      }
      
      return [...prev, optionId];
    });
  };

  const handleSubmitVote = async () => {
    if (selectedOptions.length === 0) return;
    
    const res = await fetch(`/api/polls/${poll.id}/vote`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ optionIds: selectedOptions }),
    });
    
    if (res.ok) {
      setPoll(prev => ({ ...prev, hasVoted: true }));
    }
  };

  // 计算百分比
  const getPercentage = (option: PollOption) => {
    if (poll.totalVotes === 0) return 0;
    return Math.round((option.voteCount / poll.totalVotes) * 100);
  };

  // 图表数据
  const chartData = {
    labels: poll.options.map(o => o.text),
    datasets: [{
      data: poll.options.map(o => o.voteCount),
      backgroundColor: [
        '#FF6384',
        '#36A2EB',
        '#FFCE56',
        '#4BC0C0',
        '#9966FF',
        '#FF9F40',
      ],
    }],
  };

  return (
    <div className="poll-vote">
      <header className="poll-header">
        <h1>{poll.title}</h1>
        {poll.description && <p>{poll.description}</p>}
        <div className="poll-status">
          <span>🗳️ {poll.totalVotes} 票</span>
          {isConnected && <span className="online">● 在线</span>}
        </div>
      </header>

      {!poll.hasVoted ? (
        <div className="poll-options">
          <p className="poll-hint">
            可选择 {poll.maxChoices} 个选项
          </p>
          
          {poll.options.map(option => (
            <button
              key={option.id}
              className={`poll-option ${selectedOptions.includes(option.id) ? 'selected' : ''}`}
              onClick={() => handleOptionClick(option.id)}
            >
              <span className="checkbox">
                {selectedOptions.includes(option.id) && '✓'}
              </span>
              <span className="option-text">{option.text}</span>
            </button>
          ))}
          
          <button
            className="vote-btn"
            onClick={handleSubmitVote}
            disabled={selectedOptions.length === 0}
          >
            投票
          </button>
        </div>
      ) : (
        <div className="poll-results">
          <div className="results-chart">
            <Pie data={chartData} />
          </div>
          
          <div className="results-bars">
            {poll.options.map(option => (
              <div key={option.id} className="result-bar">
                <div className="result-label">
                  <span>{option.text}</span>
                  <span>{option.voteCount} 票 ({getPercentage(option)}%)</span>
                </div>
                <div className="result-track">
                  <div
                    className="result-fill"
                    style={{ width: `${getPercentage(option)}%` }}
                  />
                </div>
              </div>
            ))}
          </div>
        </div>
      )}
    </div>
  );
}
```
