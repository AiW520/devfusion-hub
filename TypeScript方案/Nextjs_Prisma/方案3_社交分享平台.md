# 方案二：Next.js + Prisma 社交分享平台

## 项目介绍

### 项目是什么
这是一个类似 Twitter/Threads 的社交分享平台，支持发布动态、关注用户、点赞评论、话题标签等功能。

### 核心技术亮点
- **Next.js 14 + Prisma**
- **Infinite Scroll 无限滚动**
- **图片上传与处理**
- **实时通知**

```prisma
  id          String    @id @default(cuid())
  username    String    @unique
  email       String    @unique
  password    String
  name        String?
  bio         String?
  avatar      String?
  coverImage  String?

  following   Follow[]  @relation("following")

  // 内容

model Follow {
  id          String   @id @default(cuid())
  followerId   String
  followingId  String
  follower     User     @relation("following", fields: [followerId], references: [id])
  following    User     @relation("followers", fields: [followingId], references: [id])
  
  createdAt   DateTime @default(now())
  
  @@unique([followerId, followingId])
  @@index([followerId])
  @@index([followingId])
}

model Post {
  id          String    @id @default(cuid())
  content     String
  images      String[]  // 图片 URL 数组
  
  authorId    String
  author      User      @relation(fields: [authorId], references: [id])
  
  // 互动
  comments    Comment[]
  likes       Like[]
  
  // 话题
  hashtags    Hashtag[]
  
  // 统计
  likeCount   Int       @default(0)
  commentCount Int      @default(0)
  repostCount  Int       @default(0)
  
  // 引用转发
  repostedFromId String?
  repostedFrom   Post?   @relation("Reposts", fields: [repostedFromId], references: [id])
  reposts       Post[]   @relation("Reposts")
  
  // 引用
  quotedPostId  String?
  
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt
  
  @@index([authorId])
  @@index([createdAt])
}

model Comment {
  id          String   @id @default(cuid())
  content     String
  postId      String
  post        Post     @relation(fields: [postId], references: [id], onDelete: Cascade)
  authorId    String
  author      User     @relation(fields: [authorId], references: [id])
  
  likes       Like[]
  
  // 回复
  replies     Comment[]
  parentId    String?
  parent      Comment? @relation("CommentReplies", fields: [parentId], references: [id])
  
  likeCount   Int      @default(0)
  
  createdAt   DateTime @default(now())
  
  @@index([postId])
  @@index([authorId])
}

model Like {
  id          String   @id @default(cuid())
  userId      String
  user        User     @relation(fields: [userId], references: [id])
  
  // Polymorphic: 可以点赞帖子或评论
  postId      String?
  post        Post?    @relation(fields: [postId], references: [id], onDelete: Cascade)
  commentId   String?
  comment     Comment? @relation(fields: [commentId], references: [id], onDelete: Cascade)
  
  createdAt   DateTime @default(now())
  
  @@unique([userId, postId])
  @@unique([userId, commentId])
}

model Hashtag {
  id          String   @id @default(cuid())
  name        String   @unique  // 不带 #
  posts       Post[]
  
  @@index([name])
}

model Notification {
  id          String   @id @default(cuid())
  userId      String   // 接收通知的用户
  user        User     @relation(fields: [userId], references: [id])
  
  type        NotificationType
  actorId     String   // 触发通知的用户
  actor       User     @relation("NotificationActor", fields: [actorId], references: [id])
  
  // 关联内容
  postId      String?
  commentId   String?
  followId    String?
  
  read        Boolean  @default(false)
  
  createdAt   DateTime @default(now())
  
  @@index([userId, read])
}

enum NotificationType {
  LIKE
  COMMENT
  FOLLOW
  MENTION
}
```

### Server Actions

```typescript
// src/actions/post.ts
'use server';

import { prisma } from '@/lib/prisma';
import { requireAuth } from '@/lib/auth';
import { revalidatePath } from 'next/cache';
import { z } from 'zod';

const createPostSchema = z.object({
  content: z.string().min(1).max(500),
  images: z.array(z.string().url()).max(4).optional(),
  quotedPostId: z.string().optional(),
});

/**
 * 创建动态
 */
export async function createPost(data: z.infer<typeof createPostSchema>) {
  const user = await requireAuth();
  
  const validation = createPostSchema.safeParse(data);
  if (!validation.success) {
    return { error: '内容不能为空' };
  }
  
  // 提取话题标签
  const hashtags = validation.data.content.match(/#\w+/g) || [];
  
  // 创建帖子
  const post = await prisma.post.create({
    data: {
      content: validation.data.content,
      images: validation.data.images || [],
      authorId: user.id,
      quotedPostId: validation.data.quotedPostId,
      hashtags: {
        connectOrCreate: hashtags.map(tag => ({
          where: { name: tag.slice(1).toLowerCase() },
          create: { name: tag.slice(1).toLowerCase() },
        })),
      },
    },
    include: {
      author: {
        select: { id: true, username: true, name: true, avatar: true },
      },
      hashtags: true,
      _count: { select: { likes: true, comments: true } },
    },
  });
  
  // 检查 @提及并创建通知
  const mentions = validation.data.content.match(/@\w+/g) || [];
  for (const mention of mentions) {
    const mentionedUser = await prisma.user.findUnique({
      where: { username: mention.slice(1) },
    });
    
    if (mentionedUser) {
      await prisma.notification.create({
        data: {
          userId: mentionedUser.id,
          actorId: user.id,
          type: 'MENTION',
          postId: post.id,
        },
      });
    }
  }
  
  revalidatePath('/');
  
  return { success: true, post };
}

/**
 * 获取时间线（关注的人 + 自己）
 */
export async function getTimeline(page = 1, limit = 20) {
  const user = await requireAuth();
  
  const posts = await prisma.post.findMany({
    where: {
      OR: [
        { authorId: user.id },
        { author: { followers: { some: { followerId: user.id } } } },
      ],
      repostedFromId: null,  // 不显示转发的转发
    },
    include: {
      author: {
        select: { id: true, username: true, name: true, avatar: true },
      },
      hashtags: true,
      quotedPost: {
        include: {
          author: { select: { id: true, username: true } },
        },
      },
      _count: {
        select: { likes: true, comments: true, reposts: true },
      },
      likes: {
        where: { userId: user.id },
        select: { id: true },
      },
    },
    orderBy: { createdAt: 'desc' },
    take: limit,
    skip: (page - 1) * limit,
  });
  
  return posts;
}

/**
 * 获取用户动态
 */
export async function getUserPosts(username: string, page = 1) {
  const currentUser = await requireAuth().catch(() => null);
  
  const user = await prisma.user.findUnique({
    where: { username },
  });
  
  if (!user) return [];
  
  // 检查是否可见
  if (user.isPrivate && user.id !== currentUser?.id) {
    const isFollowing = await prisma.follow.findUnique({
      where: {
        followerId_followingId: {
          followerId: currentUser?.id || '',
          followingId: user.id,
        },
      },
    });
    
    if (!isFollowing) return [];
  }
  
  const posts = await prisma.post.findMany({
    where: {
      authorId: user.id,
      repostedFromId: null,
    },
    include: {
      author: { select: { id: true, username: true, name: true, avatar: true } },
      hashtags: true,
      _count: { select: { likes: true, comments: true } },
      likes: currentUser ? {
        where: { userId: currentUser.id },
        select: { id: true },
      } : false,
    },
    orderBy: { createdAt: 'desc' },
    take: 20,
    skip: (page - 1) * 20,
  });
  
  return posts;
}

/**
 * 点赞/取消点赞
 */
export async function toggleLike(postId: string) {
  const user = await requireAuth();
  
  // 检查是否已点赞
  const existing = await prisma.like.findUnique({
    where: {
      userId_postId: { userId: user.id, postId },
    },
  });
  
  if (existing) {
    // 取消点赞
    await prisma.$transaction([
      prisma.like.delete({ where: { id: existing.id } }),
      prisma.post.update({
        where: { id: postId },
        data: { likeCount: { decrement: 1 } },
      }),
    ]);
    
    return { liked: false };
  } else {
    // 点赞
    await prisma.$transaction([
      prisma.like.create({
        data: { userId: user.id, postId },
      }),
      prisma.post.update({
        where: { id: postId },
        data: { likeCount: { increment: 1 } },
      }),
      // 创建通知
      prisma.notification.create({
        data: {
          userId: (await prisma.post.findUnique({ where: { id: postId } }))!.authorId,
          actorId: user.id,
          type: 'LIKE',
          postId,
        },
      }),
    ]);
    
    return { liked: true };
  }
}

/**
 * 关注/取消关注
 */
export async function toggleFollow(userId: string) {
  const currentUser = await requireAuth();
  
  if (currentUser.id === userId) {
    return { error: '不能关注自己' };
  }
  
  const existing = await prisma.follow.findUnique({
    where: {
      followerId_followingId: {
        followerId: currentUser.id,
        followingId: userId,
      },
    },
  });
  
  if (existing) {
    await prisma.follow.delete({ where: { id: existing.id } });
    return { following: false };
  } else {
    await prisma.$transaction([
      prisma.follow.create({
        data: { followerId: currentUser.id, followingId: userId },
      }),
      prisma.notification.create({
        data: {
          userId,
          actorId: currentUser.id,
          type: 'FOLLOW',
        },
      }),
    ]);
    return { following: true };
  }
}

/**
 * 转发
 */
export async function repost(postId: string, content?: string) {
  const user = await requireAuth();
  
  const originalPost = await prisma.post.findUnique({
    where: { id: postId },
  });
  
  if (!originalPost) {
    return { error: '原帖不存在' };
  }
  
  // 创建转发
  await prisma.post.create({
    data: {
      content: content || '',
      authorId: user.id,
      repostedFromId: postId,
    },
  });
  
  // 增加原帖转发数
  await prisma.post.update({
    where: { id: postId },
    data: { repostCount: { increment: 1 } },
  });
  
  // 创建通知
  if (originalPost.authorId !== user.id) {
    await prisma.notification.create({
      data: {
        userId: originalPost.authorId,
        actorId: user.id,
        type: 'REPOST',
        postId,
      },
    });
  }
  
  revalidatePath('/');
  
  return { success: true };
}
```

### 组件示例

```typescript
// src/components/PostCard.tsx
'use client';

import { useState, useTransition } from 'react';
import Link from 'next/link';
import Image from 'next/image';
import { toggleLike, repost } from '@/actions/post';
import { formatDistanceToNow } from 'date-fns';
import { zhCN } from 'date-fns/locale';

interface PostCardProps {
  post: {
    id: string;
    content: string;
    images: string[];
    createdAt: Date;
    author: {
      id: string;
      username: string;
      name?: string;
      avatar?: string;
    };
    hashtags: { name: string }[];
    likeCount: number;
    commentCount: number;
    repostCount: number;
    isLiked?: boolean;
    quotedPost?: {
      id: string;
      content: string;
      author: { username: string };
    };
  };
  currentUserId?: string;
}

export function PostCard({ post, currentUserId }: PostCardProps) {
  const [isPending, startTransition] = useTransition();
  const [liked, setLiked] = useState(post.isLiked || false);
  const [likeCount, setLikeCount] = useState(post.likeCount);
  const [showComments, setShowComments] = useState(false);
  
  const handleLike = () => {
    setLiked(!liked);
    setLikeCount(liked ? likeCount - 1 : likeCount + 1);
    
    startTransition(async () => {
      await toggleLike(post.id);
    });
  };
  
  const handleRepost = () => {
    startTransition(async () => {
      await repost(post.id);
    });
  };
  
  // 高亮话题标签
  const renderContent = () => {
    let content = post.content;
    
    // 话题标签
    content = content.replace(
      /#(\w+)/g,
      '<a href="/hashtag/$1" class="hashtag">#$1</a>'
    );
    
    // @提及
    content = content.replace(
      /@(\w+)/g,
      '<a href="/$1" class="mention">@$1</a>'
    );
    
    return content;
  };
  
  return (
    <article className="post-card">
      {/* 头像 */}
      <Link href={`/${post.author.username}`}>
        <Image
          src={post.author.avatar || '/default-avatar.png'}
          alt={post.author.username}
          width={48}
          height={48}
          className="post-card__avatar"
        />
      </Link>
      
      {/* 内容 */}
      <div className="post-card__content">
        {/* 作者信息 */}
        <header className="post-card__header">
          <Link href={`/${post.author.username}`} className="post-card__author">
            <span className="post-card__name">{post.author.name || post.author.username}</span>
            <span className="post-card__username">@{post.author.username}</span>
          </Link>
          <span className="post-card__dot">·</span>
          <time className="post-card__time">
            {formatDistanceToNow(new Date(post.createdAt), { addSuffix: true, locale: zhCN })}
          </time>
        </header>
        
        {/* 动态内容 */}
        <div 
          className="post-card__text"
          dangerouslySetInnerHTML={{ __html: renderContent() }}
        />
        
        {/* 图片 */}
        {post.images && post.images.length > 0 && (
          <div className="post-card__images">
            {post.images.map((image, index) => (
              <Image
                key={index}
                src={image}
                alt=""
                width={300}
                height={200}
                className="post-card__image"
              />
            ))}
          </div>
        )}
        
        {/* 引用帖子 */}
        {post.quotedPost && (
          <Link href={`/post/${post.quotedPost.id}`} className="post-card__quoted">
            <span>@{post.quotedPost.author.username}</span>
            <p>{post.quotedPost.content}</p>
          </Link>
        )}
        
        {/* 互动按钮 */}
        <footer className="post-card__actions">
          {/* 评论 */}
          <button 
            className="post-card__action"
            onClick={() => setShowComments(!showComments)}
          >
            💬 {post.commentCount}
          </button>
          
          {/* 转发 */}
          <button 
            className="post-card__action"
            onClick={handleRepost}
            disabled={isPending}
          >
            🔁 {post.repostCount}
          </button>
          
          {/* 点赞 */}
          <button 
            className={`post-card__action ${liked ? 'post-card__action--liked' : ''}`}
            onClick={handleLike}
            disabled={isPending}
          >
            {liked ? '❤️' : '🤍'} {likeCount}
          </button>
          
          {/* 分享 */}
          <button className="post-card__action">
            📤
          </button>
        </footer>
      </div>
    </article>
  );
}
```

### 问题: 如何实现 Infinite Scroll？

```typescript
// 1. 使用 Intersection Observer
'use client';
import { useEffect, useRef, useCallback } from 'react';

export function useInfiniteScroll(
  onLoadMore: () => void,
  hasMore: boolean
) {
  const observerRef = useRef<IntersectionObserver>();
  const loadMoreRef = useCallback(
    (node: HTMLDivElement | null) => {
      if (observerRef.current) {
        observerRef.current.disconnect();
      }
      
      observerRef.current = new IntersectionObserver(entries => {
        if (entries[0].isIntersecting && hasMore) {
          onLoadMore();
        }
      });
      
      if (node) {
        observerRef.current.observe(node);
      }
    },
    [onLoadMore, hasMore]
  );
  
  return loadMoreRef;
}

// 2. 组件中使用
function Timeline() {
  const [posts, setPosts] = useState([]);
  const [page, setPage] = useState(1);
  const [hasMore, setHasMore] = useState(true);
  
  const loadMore = async () => {
    const newPosts = await getTimeline(page);
    setPosts([...posts, ...newPosts]);
    setHasMore(newPosts.length === 20);
    setPage(page + 1);
  };
  
  const loadMoreRef = useInfiniteScroll(loadMore, hasMore);
  
  return (
    <div>
      {posts.map(post => (
        <PostCard key={post.id} post={post} />
      ))}
      {hasMore && <div ref={loadMoreRef}>加载更多...</div>}
    </div>
  );
}
```
