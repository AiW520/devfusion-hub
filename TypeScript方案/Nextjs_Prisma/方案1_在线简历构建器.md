# 方案二：Next.js + Prisma 在线简历构建器

## 项目介绍

### 项目是什么
这是一个专业的在线简历构建器，用户可以拖拽式编辑简历模块、实时预览、自动生成 PDF。

### 解决什么问题
- 求职者制作简历困难的问题
- 展示 Next.js App Router 和服务端组件
- 学习 Prisma ORM 和数据库设计

### 核心技术亮点
- **Next.js 14 App Router**: 服务端组件、React Server Components
- **Prisma ORM**: 类型安全的数据库操作
- **React DnD**: 拖拽式简历编辑
- **Server Actions**: 服务端操作
- **Zod**: 数据验证

### 适合什么场景
- SaaS 简历平台
- 在线表单构建器
- 文档编辑器

/**
 * Prisma Schema 定义
 * 
 * Prisma 是一种现代的 ORM，相比传统 SQL 更类型安全
 * 支持自动生成 TypeScript 类型
 */

/**
 * 数据源配置
 * 指定使用什么数据库
 */
datasource db {
  provider = "mysql"      // 数据库类型：mysql, postgresql, sqlite, mongodb
  url      = env("DATABASE_URL")  // 数据库连接 URL
}

/**
 * 生成器配置
 * 指定生成的客户端配置
 */
generator client {
  provider = "prisma-client-js"  // 使用 Prisma 官方的 JS 客户端
}

/**
 * 用户模型
 */
model User {
  id            String    @id @default(cuid())  // 默认生成唯一 ID
  email         String    @unique               // 唯一索引
  name          String?
  password      String                       // 密码（加密存储）
  image         String?
  
  // 关联：用户可以有多个简历
  resumes       Resume[]
  
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
  
  @@map("users")  // 数据库表名映射
}

/**
 * 简历模型
 */
model Resume {
  id            String    @id @default(cuid())
  
  // 简历基本信息
  title         String                       // 简历标题
  slug          String    @unique            // URL 友好的标识
  
  // 简历数据（JSON 格式存储简历内容）
  // 这样设计更灵活，可以存储任意结构的简历模块
  content       Json
  
  // 样式配置（主题、字体、颜色等）
  theme         String    @default("default")
  fontFamily    String    @default("Inter")
  fontSize      Int       @default(14)
  primaryColor  String    @default("#3B82F6")
  
  // 是否公开分享
  isPublic      Boolean   @default(false)
  shareSlug     String?   @unique            // 公开分享的短链接
  
  // 关联
  userId        String
  user          User      @relation(fields: [userId], references: [id], onDelete: Cascade)
  
  // 简历版本控制
  version       Int       @default(1)
  
  // 时间戳
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
  publishedAt   DateTime?
  
  // 索引
  @@index([userId])
  @@map("resumes")
}

/**
 * 模板模型（可选，用于模板市场）
 */
model Template {
  id            String    @id @default(cuid())
  name          String
  description   String?
  thumbnail     String
  content       Json      // 模板内容
  
  // 是否为官方模板
  isOfficial    Boolean   @default(false)
  
  // 使用次数
  useCount      Int       @default(0)
  
  createdAt     DateTime  @default(now())
  
  @@map("templates")
}

/**
 * 访问日志（可选，用于分析）
 */
model AccessLog {
  id            String    @id @default(cuid())
  resumeId      String
  ip            String?
  userAgent     String?
  
  createdAt     DateTime  @default(now())
  
  @@index([resumeId])
  @@map("access_logs")
}

  log: process.env.NODE_ENV === 'development'

  global.prisma = prisma;
}

 * 断开数据库连接
 * 通常在应用关闭时调用

/**
 * 创建新简历
 */
export async function createResume(formData: FormData) {
  try {
    // 1. 获取当前用户
    const user = await getCurrentUser();
    if (!user) {
      return { error: '请先登录' };
    }
    
    // 2. 验证输入
    const title = formData.get('title') as string;
    const validation = createResumeSchema.safeParse({ title });
    
    if (!validation.success) {
      return { error: validation.error.errors[0].message };
    }
    
    // 3. 生成唯一 slug
    const baseSlug = generateSlug(title);
    let slug = baseSlug;
    let counter = 1;
    
    // 检查 slug 是否已存在
    while (await prisma.resume.findUnique({ where: { slug } })) {
      slug = `${baseSlug}-${counter}`;
      counter++;
    }
    
    // 4. 创建简历
    const resume = await prisma.resume.create({
      data: {
        title,
        slug,
        userId: user.id,
        // 默认简历内容模板
        content: {
          personal: {
            name: '',
            title: '',
            email: '',
            phone: '',
            location: '',
            website: '',
            summary: '',
          },
          experience: [],
          education: [],
          skills: [],
          projects: [],
        },
      },
    });
    
    // 5. 重新验证页面数据（清除缓存）
    revalidatePath('/dashboard/resumes');
    
    // 6. 跳转到编辑页面
    redirect(`/dashboard/resumes/${resume.id}`);
    
  } catch (error) {
    console.error('创建简历失败:', error);
    return { error: '创建失败，请重试' };
  }
}

/**
 * 获取单个简历（包含完整内容）
 */
export async function getResume(id: string) {
  const user = await getCurrentUser();
  
  const resume = await prisma.resume.findUnique({
    where: { id },
    include: {
      user: {
        select: {
          id: true,
          name: true,
          email: true,
          image: true,
        },
      },
    },
  });
  
  // 检查权限：只有作者可以查看
  if (!resume || (user && resume.userId !== user.id && !resume.isPublic)) {
    return null;
  }
  
  return resume;
}

/**
 * 更新简历
 */
export async function updateResume(data: z.infer<typeof updateResumeSchema>) {
  try {
    const user = await getCurrentUser();
    if (!user) {
      return { error: '请先登录' };
    }
    
    // 验证输入
    const validation = updateResumeSchema.safeParse(data);
    if (!validation.success) {
      return { error: validation.error.errors[0].message };
    }
    
    // 检查所有权
    const existing = await prisma.resume.findUnique({
      where: { id: data.id },
    });
    
    if (!existing || existing.userId !== user.id) {
      return { error: '无权限修改' };
    }
    
    // 更新简历
    const updateData: any = {
      updatedAt: new Date(),
    };
    
    if (data.title) updateData.title = data.title;
    if (data.content) updateData.content = data.content;
    if (data.theme) updateData.theme = data.theme;
    if (data.fontFamily) updateData.fontFamily = data.fontFamily;
    if (data.fontSize) updateData.fontSize = data.fontSize;
    if (data.primaryColor) updateData.primaryColor = data.primaryColor;
    if (typeof data.isPublic === 'boolean') {
      updateData.isPublic = data.isPublic;
      if (data.isPublic && !existing.publishedAt) {
        updateData.publishedAt = new Date();
      }
    }
    
    const resume = await prisma.resume.update({
      where: { id: data.id },
      data: updateData,
    });
    
    // 清除缓存
    revalidatePath(`/dashboard/resumes/${data.id}`);
    
    return { success: true, resume };
    
  } catch (error) {
    console.error('更新简历失败:', error);
    return { error: '更新失败，请重试' };
  }
}

export async function duplicateResume(id: string) {

    const user = await getCurrentUser();

      return { error: '请先登录' };
    }

        theme: original.theme,
        fontFamily: original.fontFamily,
        fontSize: original.fontSize,

    });

/**
 * 添加简历模块
 */
export async function addResumeModule(data: z.infer<typeof addModuleSchema>) {
  try {
    const user = await getCurrentUser();
    if (!user) {
      return { error: '请先登录' };
    }
    
    const validation = addModuleSchema.safeParse(data);
    if (!validation.success) {
      return { error: validation.error.errors[0].message };
    }
    
    // 获取简历
    const resume = await prisma.resume.findUnique({
      where: { id: data.resumeId },
    });
    
    if (!resume || resume.userId !== user.id) {
      return { error: '简历不存在' };
    }
    
    // 更新内容
    const content = resume.content as Record<string, any>;
    const moduleType = data.moduleType;
    
    // 初始化模块数组（如果不存在）
    if (!content[moduleType]) {
      content[moduleType] = [];
    }
    
    // 创建新模块
    const newModule = createEmptyModule(moduleType, data.position);
    content[moduleType].push(newModule);
    
    // 保存
    await prisma.resume.update({
      where: { id: data.resumeId },
      data: { content },
    });
    
    revalidatePath(`/dashboard/resumes/${data.resumeId}`);
    
    return { success: true, module: newModule };
    
  } catch (error) {
    console.error('添加模块失败:', error);
    return { error: '添加失败' };
  }
}

/**
 * 创建空模块
 */
function createEmptyModule(type: string, position?: number) {
  const id = `module_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  
  switch (type) {
    case 'experience':
      return {
        id,
        company: '',
        position: '',
        location: '',
        startDate: '',
        endDate: '',
        current: false,
        description: '',
      };
    case 'education':
      return {
        id,
        school: '',
        degree: '',
        field: '',
        startDate: '',
        endDate: '',
        gpa: '',
        description: '',
      };
    case 'skills':
      return {
        id,
        category: '技术技能',
        items: [],
      };
    case 'projects':
      return {
        id,
        name: '',
        description: '',
        url: '',
        technologies: [],
        startDate: '',
        endDate: '',
      };
    default:
      return { id };
  }
}

interface EditorProps {
  resume: Resume & { content: ResumeContent };
}

type ViewMode = 'edit' | 'preview';
type ActivePanel = 'modules' | 'style' | null;

export const Editor: React.FC<EditorProps> = ({ resume }) => {
  // 状态
  const [content, setContent] = useState<ResumeContent>(resume.content as ResumeContent);
  const [styleSettings, setStyleSettings] = useState({
    theme: resume.theme,
    fontFamily: resume.fontFamily,
    fontSize: resume.fontSize,
    primaryColor: resume.primaryColor,
  });
  const [viewMode, setViewMode] = useState<ViewMode>('edit');
  const [activePanel, setActivePanel] = useState<ActivePanel>('modules');
  const [isSaving, setIsSaving] = useState(false);
  const [lastSaved, setLastSaved] = useState<Date | null>(null);
  
  // 使用 useTransition 处理非紧急更新
  // isPending 表示是否有过渡中的操作
  const [isPending, startTransition] = useTransition();
  
  /**
   * 更新内容
   */
  const updateContent = useCallback((
    moduleType: keyof ResumeContent,
    moduleId: string | null,
    data: any
  ) => {
    setContent((prev) => {
      const module = prev[moduleType];
      
      if (Array.isArray(module)) {
        // 数组类型（experience, education 等）

          // 更新指定项

              item.id === moduleId ? { ...item, ...data } : item
            ),
          };
        } else {
          // 添加新项
          return {
            ...prev,
            [moduleType]: [...module, { id: `new_${Date.now()}`, ...data }],
          };
        }
      } else {
        // 对象类型（personal）
        return {
          ...prev,
          [moduleType]: { ...(module || {}), ...data },
        };
      }
    });
  }, []);
  
  /**
   * 删除模块项
   */
  const deleteModuleItem = useCallback((
    moduleType: keyof ResumeContent,
    moduleId: string
  ) => {
    setContent((prev) => {
      const module = prev[moduleType];
      if (Array.isArray(module)) {
        return {
          ...prev,
          [moduleType]: module.filter((item: any) => item.id !== moduleId),
        };
      }
      return prev;
    });
  }, []);
  
  /**
   * 保存到服务器
   * 使用 startTransition 使 UI 保持响应
   */
  const save = useCallback(async () => {
    setIsSaving(true);
    
    try {
      const result = await updateResume({
        id: resume.id,
        content,
        ...styleSettings,
      });
      
      if (result.error) {
        alert(result.error);
      } else {
        setLastSaved(new Date());
      }
    } catch (error) {
      console.error('保存失败:', error);
      alert('保存失败，请重试');
    } finally {
      setIsSaving(false);
    }
  }, [resume.id, content, styleSettings]);
  
  /**
   * 拖拽排序（待实现）
   */
  const handleReorder = useCallback((
    moduleType: keyof ResumeContent,
    fromIndex: number,
    toIndex: number
  ) => {
    setContent((prev) => {
      const module = prev[moduleType];
      if (Array.isArray(module)) {
        const newModule = [...module];
        const [removed] = newModule.splice(fromIndex, 1);
        newModule.splice(toIndex, 0, removed);
        return { ...prev, [moduleType]: newModule };
      }
      return prev;
    });
  }, []);
  
  return (
    <div className="editor">
      {/* 工具栏 */}
      <header className="editor__toolbar">
        <div className="editor__toolbar-left">
          <button
            className={`editor__btn ${viewMode === 'edit' ? 'active' : ''}`}
            onClick={() => setViewMode('edit')}
          >
            编辑
          </button>
          <button
            className={`editor__btn ${viewMode === 'preview' ? 'active' : ''}`}
            onClick={() => setViewMode('preview')}
          >
            预览
          </button>
        </div>
        
        <div className="editor__toolbar-center">
          <input
            type="text"
            className="editor__title-input"
            value={resume.title}
            // TODO: 实现标题更新
          />
        </div>
        
        <div className="editor__toolbar-right">
          <span className="editor__save-status">
            {isSaving ? '保存中...' : lastSaved ? `已保存 ${lastSaved.toLocaleTimeString()}` : ''}
          </span>
          <button
            className="editor__btn editor__btn--primary"
            onClick={save}
            disabled={isSaving}
          >
            保存
          </button>
        </div>
      </header>
      
      {/* 主体内容 */}
      <div className="editor__body">
        {viewMode === 'edit' ? (
          <>
            {/* 左侧面板 */}
            <aside className="editor__sidebar">
              <nav className="editor__sidebar-tabs">
                <button
                  className={`editor__tab ${activePanel === 'modules' ? 'active' : ''}`}
                  onClick={() => setActivePanel(activePanel === 'modules' ? null : 'modules')}
                >
                  📋 模块
                </button>
                <button
                  className={`editor__tab ${activePanel === 'style' ? 'active' : ''}`}
                  onClick={() => setActivePanel(activePanel === 'style' ? null : 'style')}
                >
                  🎨 样式
                </button>
              </nav>
              
              {activePanel === 'modules' && (
                <ModuleList
                  content={content}
                  onUpdate={updateContent}
                  onDelete={deleteModuleItem}
                  onReorder={handleReorder}
                />
              )}
              
              {activePanel === 'style' && (
                <PropertyPanel
                  settings={styleSettings}
                  onChange={setStyleSettings}
                />
              )}
            </aside>
            
            {/* 中间编辑区 */}
            <main className="editor__main">
              {/* 简历编辑内容 */}
              <div className="editor__canvas">
                {/* 个人基本信息 */}
                <section className="resume-section">
                  <h2 className="resume-section__title">基本信息</h2>
                  <div className="resume-section__content">
                    <input
                      type="text"
                      placeholder="姓名"
                      value={content.personal?.name || ''}
                      onChange={(e) => updateContent('personal', null, { name: e.target.value })}
                    />
                    <input
                      type="text"
                      placeholder="职位"
                      value={content.personal?.title || ''}
                      onChange={(e) => updateContent('personal', null, { title: e.target.value })}
                    />
                    {/* 更多字段... */}
                  </div>
                </section>
                
                {/* 工作经验 */}
                <section className="resume-section">
                  <h2 className="resume-section__title">工作经验</h2>
                  <div className="resume-section__content">
                    {content.experience?.map((exp, index) => (
                      <div key={exp.id} className="resume-item">
                        <input
                          type="text"
                          placeholder="公司名称"
                          value={exp.company}
                          onChange={(e) => updateContent('experience', exp.id, { company: e.target.value })}
                        />
                        <input
                          type="text"
                          placeholder="职位"
                          value={exp.position}
                          onChange={(e) => updateContent('experience', exp.id, { position: e.target.value })}
                        />
                        <textarea
                          placeholder="工作描述"
                          value={exp.description}
                          onChange={(e) => updateContent('experience', exp.id, { description: e.target.value })}
                        />
                        <button onClick={() => deleteModuleItem('experience', exp.id)}>
                          删除
                        </button>
                      </div>
                    ))}
                    <button
                      className="add-item-btn"
                      onClick={() => updateContent('experience', null, {
                        company: '',
                        position: '',
                        location: '',
                        startDate: '',
                        endDate: '',
                        current: false,
                        description: '',
                      })}
                    >
                      + 添加工作经验
                    </button>
                  </div>
                </section>
                
                {/* 教育经历 */}
                <section className="resume-section">
                  <h2 className="resume-section__title">教育经历</h2>
                  {/* 类似结构 */}
                </section>
                
                {/* 技能 */}
                <section className="resume-section">
                  <h2 className="resume-section__title">技能</h2>
                  {/* 类似结构 */}
                </section>
              </div>
            </main>
          </>
        ) : (
          /* 预览模式 */
          <Preview content={content} settings={styleSettings} />
        )}
      </div>
    </div>
  );
};

## 面试要点

### 问题 1: Next.js App Router vs Pages Router 有什么区别？

**参考答案**:
```
App Router (Next.js 13+):
- React Server Components (RSC) 默认开启
- 布局系统更灵活，支持嵌套布局
- 服务端组件可以直接访问数据库
- 使用 Server Actions 处理表单
- 自动代码分割，更好的加载性能

Pages Router (传统方式):
- 页面级服务渲染
- API Routes 处理后端逻辑
- getServerSideProps / getStaticProps
- 混合水合（部分客户端渲染）
```

---

### 问题 2: Prisma 相比传统 ORM 有什么优势？

**参考答案**:
```
1. 类型安全
   - 自动生成 TypeScript 类型
   - 编译时检查查询语法
   
2. 简洁的 API
   - 链式调用，接近自然语言
   - 支持嵌套关联查询
   
3. 迁移系统
   - 数据库版本控制
   - 自动生成迁移脚本
   
4. 性能
   - 查询构建优化
   - 连接池管理
   
5. 开发体验
   - TypeScript 优先
   - 智能补全
   - 错误提示
```

---

### 问题 3: 什么是 React Server Components？

**参考答案**:
```
React Server Components 是 React 18 引入的新特性：
- 默认在服务端渲染
- 无法使用 useState、useEffect 等 Hooks
- 可以直接访问服务端资源（数据库、文件系统）
- 不会打包到客户端 JavaScript 中
- 减少客户端 bundle 大小

Next.js App Router 完全支持 RSC
```
