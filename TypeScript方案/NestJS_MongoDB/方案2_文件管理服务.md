# 方案五：NestJS + MongoDB 文件管理服务

## 项目介绍

### 项目是什么
这是一个企业级文件管理服务，支持文件上传下载、文件夹管理、权限控制、版本历史、在线预览等功能。

### 核心技术亮点
- **NestJS** 后端框架
- **MongoDB GridFS** 大文件存储
- **JWT + ACL** 权限控制
- **文件预览** (图片、PDF、视频)

## 完整可运行代码

### 1. 文件模型

```typescript
// backend/src/files/schemas/file.schema.ts

import { Prop, Schema, SchemaFactory } from '@nestjs/mongoose';
import { Document, Types } from 'mongoose';

export type FileDocument = File & Document;

@Schema({ timestamps: true })
export class File {
  @Prop({ required: true })
  filename: string;  // 原始文件名

  @Prop({ required: true })
  originalName: string;  // 显示的文件名

  @Prop({ required: true })
  mimeType: string;  // MIME 类型

  @Prop({ required: true })
  size: number;  // 文件大小(bytes)

  @Prop({ type: String, index: true })
  gridFsId: string;  // GridFS 文件 ID

  @Prop({ type: String, index: true })
  folderId: string | null;  // 所属文件夹

  @Prop({ type: Types.ObjectId, ref: 'User', required: true })
  ownerId: Types.ObjectId;

  @Prop({ type: [String], default: [] })
  sharedWith: string[];  // 共享用户 ID 列表

  @Prop({ type: Boolean, default: false })
  isPublic: boolean;  // 是否公开

  @Prop({ type: String, enum: ['owner', 'read', 'write'], default: 'owner' })
  sharePermission: string;

  @Prop({ type: Object })
  metadata: Record<string, any>;  // 自定义元数据

  @Prop({ type: Number, default: 1 })
  version: number;  // 当前版本

  @Prop({ type: Boolean, default: false })
  isDeleted: boolean;  // 软删除

  @Prop({ type: Date })
  deletedAt: Date;
}

export const FileSchema = SchemaFactory.createForClass(File);

// 复合索引
FileSchema.index({ ownerId: 1, folderId: 1 });
FileSchema.index({ 'sharedWith': 1 });
FileSchema.index({ filename: 'text', originalName: 'text' });
```

### 2. 文件服务

```typescript
// backend/src/files/files.service.ts

import { Injectable, NotFoundException, ForbiddenException } from '@nestjs/common';
import { InjectModel } from '@nestjs/mongoose';
import { Model, GridFSBucket, Types } from 'mongoose';
import { Readable } from 'stream';
import { File as FileModel, FileDocument } from './schemas/file.schema';
import { CreateFileDto, UpdateFileDto } from './dto';
import { createReadStream, unlink } from 'fs/promises';
import { join } from 'path';

@Injectable()
export class FilesService {
  private gridFsBucket: GridFSBucket;

  constructor(
    @InjectModel(File.name) private fileModel: Model<FileDocument>,
  ) {
    // 初始化 GridFS
    // this.gridFsBucket = new GridFSBucket(this.db, { bucketName: 'files' });
  }

  /**
   * 上传文件
   */
  async upload(
    file: Express.Multer.File,
    dto: CreateFileDto,
    userId: string,
  ): Promise<FileDocument> {
    // 使用本地存储（简化实现）
    // 生产环境使用 GridFS 或云存储 (S3)
    const fileDoc = await this.fileModel.create({
      filename: file.filename,
      originalName: dto.originalName || file.originalname,
      mimeType: file.mimetype,
      size: file.size,
      folderId: dto.folderId || null,
      ownerId: new Types.ObjectId(userId),
      metadata: dto.metadata || {},
    });

    return fileDoc;
  }

  /**
   * 批量上传
   */
  async uploadMany(
    files: Express.Multer.File[],
    folderId: string | null,
    userId: string,
  ) {
    const fileDocs = await Promise.all(
      files.map(file =>
        this.fileModel.create({
          filename: file.filename,
          originalName: file.originalname,
          mimeType: file.mimetype,
          size: file.size,
          folderId,
          ownerId: new Types.ObjectId(userId),
        })
      )
    );

    return fileDocs;
  }

  /**
   * 获取文件列表
   */
  async findAll(
    userId: string,
    folderId?: string | null,
    options: { page?: number; limit?: number; search?: string } = {}
  ) {
    const { page = 1, limit = 20, search } = options;

    // 查询：自己上传的 或 共享给我的
    const query: any = {
      isDeleted: false,
      $or: [
        { ownerId: new Types.ObjectId(userId) },
        { sharedWith: userId },
        { isPublic: true },
      ],
    };

    if (folderId !== undefined) {
      query.folderId = folderId;
    }

    if (search) {
      query.$text = { $search: search };
    }

    const [files, total] = await Promise.all([
      this.fileModel
        .find(query)
        .populate('ownerId', 'username email')
        .sort({ updatedAt: -1 })
        .skip((page - 1) * limit)
        .limit(limit),
      this.fileModel.countDocuments(query),
    ]);

    return { files, meta: { page, limit, total } };
  }

  /**
   * 下载文件
   */
  async download(fileId: string, userId: string) {
    const file = await this.fileModel.findById(fileId);

    if (!file || file.isDeleted) {
      throw new NotFoundException('文件不存在');
    }

    // 检查权限
    const hasAccess =
      file.isPublic ||
      file.ownerId.toString() === userId ||
      file.sharedWith.includes(userId);

    if (!hasAccess) {
      throw new ForbiddenException('无权访问此文件');
    }

    return file;
  }

  /**
   * 软删除文件
   */
  async softDelete(fileId: string, userId: string) {
    const file = await this.fileModel.findById(fileId);

    if (!file) {
      throw new NotFoundException('文件不存在');
    }

    if (file.ownerId.toString() !== userId) {
      throw new ForbiddenException('只有文件所有者可以删除');
    }

    file.isDeleted = true;
    file.deletedAt = new Date();
    await file.save();

    return { success: true };
  }

  /**
   * 共享文件
   */
  async share(fileId: string, targetUserIds: string[], permission: string, userId: string) {
    const file = await this.fileModel.findById(fileId);

    if (!file) {
      throw new NotFoundException('文件不存在');
    }

    if (file.ownerId.toString() !== userId) {
      throw new ForbiddenException('只有文件所有者可以共享');
    }

    file.sharedWith = [...new Set([...file.sharedWith, ...targetUserIds])];
    file.sharePermission = permission;
    await file.save();

    return file;
  }

  /**
   * 取消共享
   */
  async unshare(fileId: string, targetUserId: string, userId: string) {
    const file = await this.fileModel.findById(fileId);

    if (!file) {
      throw new NotFoundException('文件不存在');
    }

    if (file.ownerId.toString() !== userId) {
      throw new ForbiddenException('只有文件所有者可以取消共享');
    }

    file.sharedWith = file.sharedWith.filter(id => id !== targetUserId);
    await file.save();

    return file;
  }
}
```
