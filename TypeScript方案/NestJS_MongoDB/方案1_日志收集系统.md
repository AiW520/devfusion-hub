# 方案五：NestJS + MongoDB 日志收集系统

## 项目介绍

### 项目是什么
这是一个分布式日志收集和分析系统，类似 ELK Stack，支持实时日志收集、搜索过滤、告警通知、可视化仪表盘。

### 核心技术亮点
- **NestJS** 企业级后端
- **MongoDB** 日志存储（文档型数据）
- **Redis** 缓存和消息队列
- **Socket.io** 实时推送
- **聚合管道** 数据分析

```typescript
// backend/src/logs/schemas/log.schema.ts

import { Prop, Schema, SchemaFactory } from '@nestjs/mongoose';
import { Document, Types } from 'mongoose';

export type LogDocument = Log & Document;

@Schema({
  timestamps: true,  // 自动添加 createdAt, updatedAt
  collection: 'logs',
})
export class Log {
  @Prop({ required: true, index: true })
  message: string;  // 日志消息
  
  @Prop({ required: true, index: true })
  level: string;  // info, warn, error, debug
  
  @Prop({ type: Object })
  context: Record<string, any>;  // 上下文信息
  
  @Prop({ type: Object })
  metadata: Record<string, any>;  // 元数据
  
  @Prop({ required: true, index: true })
  service: string;  // 服务名称
  
  @Prop({ required: true, index: true })
  environment: string;  // 环境: production, staging, development
  
  @Prop({ index: true })
  traceId: string;  // 链路追踪 ID
  
  @Prop({ index: true })
  spanId: string;  // Span ID
  
  @Prop({ type: Date, default: Date.now, index: true })
  timestamp: Date;
  
  @Prop({ type: Object })
  user?: {
    id: string;
    email?: string;
    ip?: string;
  };
  
  @Prop({ type: Object })
  request?: {
    method: string;
    url: string;
    headers?: Record<string, any>;
    body?: any;
    response?: {
      statusCode: number;
      body?: any;
    };
  };
  
  @Prop({ type: Number })
  duration?: number;  // 请求耗时(ms)
  
  @Prop({ type: Boolean, default: false })
  isAlert: boolean;  // 是否为告警日志
  
  @Prop({ type: [String], index: true })
  tags: string[];  // 标签
}

export const LogSchema = SchemaFactory.createForClass(Log);

// 创建复合索引
LogSchema.index({ service: 1, level: 1, timestamp: -1 });
LogSchema.index({ traceId: 1 });
LogSchema.index({ 'user.id': 1 });
```

### 2. 日志服务

```typescript
// backend/src/logs/logs.service.ts

import { Injectable, Logger } from '@nestjs/common';
import { InjectModel } from '@nestjs/mongoose';
import { Model } from 'mongoose';
import { Log, LogDocument } from './schemas/log.schema';
import { CreateLogDto, LogQueryDto } from './dto';
import { LogsGateway } from './logs.gateway';

@Injectable()
export class LogsService {
  private readonly logger = new Logger(LogsService.name);
  
  constructor(
    @InjectModel(Log.name) private logModel: Model<LogDocument>,
    private readonly logsGateway: LogsGateway,
  ) {}

  /**
   * 创建日志
   */
  async create(createLogDto: CreateLogDto): Promise<LogDocument> {
    const log = new this.logModel(createLogDto);
    const saved = await log.save();
    
    // 实时推送
    this.logsGateway.emitNewLog(saved);
    
    // 检查告警规则
    if (await this.shouldAlert(saved)) {
      await this.triggerAlert(saved);
    }
    
    return saved;
  }

  /**
   * 批量创建日志
   */
  async createMany(logs: CreateLogDto[]): Promise<number> {
    const result = await this.logModel.insertMany(logs);
    return result.length;
  }

  /**
   * 分页查询日志
   */
  async findAll(query: LogQueryDto) {
    const {
      page = 1,
      limit = 50,
      level,
      service,
      environment,
      search,
      traceId,
      startDate,
      endDate,
      tags,
    } = query;
    
    // 构建查询条件
    const filter: any = {};
    
    if (level) filter.level = level;
    if (service) filter.service = service;
    if (environment) filter.environment = environment;
    if (traceId) filter.traceId = traceId;
    
    if (search) {
      filter.$or = [
        { message: { $regex: search, $options: 'i' } },
        { 'context.error': { $regex: search, $options: 'i' } },
      ];
    }
    
    if (startDate || endDate) {
      filter.timestamp = {};
      if (startDate) filter.timestamp.$gte = new Date(startDate);
      if (endDate) filter.timestamp.$lte = new Date(endDate);
    }
    
    if (tags && tags.length > 0) {
      filter.tags = { $all: tags };
    }
    
    const [logs, total] = await Promise.all([
      this.logModel
        .find(filter)
        .sort({ timestamp: -1 })
        .skip((page - 1) * limit)
        .limit(limit)
        .lean(),
      this.logModel.countDocuments(filter),
    ]);
    
    return {
      logs,
      meta: { page, limit, total, totalPages: Math.ceil(total / limit) },
    };
  }

  /**
   * 获取日志统计
   */
  async getStats(service?: string, timeRange: '1h' | '24h' | '7d' = '24h') {
    // 计算时间范围
    const now = new Date();
    let startTime: Date;
    
    switch (timeRange) {
      case '1h':
        startTime = new Date(now.getTime() - 60 * 60 * 1000);
        break;
      case '24h':
        startTime = new Date(now.getTime() - 24 * 60 * 60 * 1000);
        break;
      case '7d':
        startTime = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000);
        break;
    }
    
    const match: any = {
      timestamp: { $gte: startTime, $lte: now },
    };
    if (service) match.service = service;
    
    // 聚合查询
    const stats = await this.logModel.aggregate([
      { $match: match },
      {
        $group: {
          _id: '$level',
          count: { $sum: 1 },
        },
      },
    ]);
    
    // 按级别分组
    const levelCounts = {
      error: 0,
      warn: 0,
      info: 0,
      debug: 0,
    };
    
    stats.forEach(s => {
      if (s._id in levelCounts) {
        levelCounts[s._id] = s.count;
      }
    });
    
    // 每小时趋势
    const trend = await this.logModel.aggregate([
      { $match: match },
      {
        $group: {
          _id: {
            $dateToString: { format: '%Y-%m-%d %H:00', date: '$timestamp' },
          },
          error: {
            $sum: { $cond: [{ $eq: ['$level', 'error'] }, 1, 0] },
          },
          warn: {
            $sum: { $cond: [{ $eq: ['$level', 'warn'] }, 1, 0] },
          },
          info: {
            $sum: { $cond: [{ $eq: ['$level', 'info'] }, 1, 0] },
          },
          total: { $sum: 1 },
        },
      },
      { $sort: { _id: 1 } },
    ]);
    
    // 服务分布
    const serviceStats = await this.logModel.aggregate([
      { $match: match },
      {
        $group: {
          _id: '$service',
          count: { $sum: 1 },
          errorCount: {
            $sum: { $cond: [{ $eq: ['$level', 'error'] }, 1, 0] },
          },
        },
      },
      { $sort: { count: -1 } },
      { $limit: 10 },
    ]);
    
    return {
      levelCounts,
      trend,
      serviceStats,
      timeRange,
      generatedAt: now,
    };
  }

  /**
   * 获取错误日志详情
   */
  async getErrorDetails(errorId: string) {
    return this.logModel.findById(errorId).lean();
  }

  /**
   * 检查是否应该告警
   */
  private async shouldAlert(log: LogDocument): Promise<boolean> {
    // 简单实现：ERROR 级别告警
    if (log.level === 'error') {
      // 检查是否在短时间内有大量错误
      const oneMinuteAgo = new Date(Date.now() - 60 * 1000);
      const recentErrors = await this.logModel.countDocuments({
        level: 'error',
        service: log.service,
        timestamp: { $gte: oneMinuteAgo },
      });
      
      // 1分钟内超过5个错误则告警
      return recentErrors >= 5;
    }
    
    return false;
  }

  /**
   * 触发告警
   */
  private async triggerAlert(log: LogDocument) {
    this.logger.warn(`触发告警: ${log.service} 服务在短时间内产生大量错误`);
    
    // 更新日志为告警状态
    await this.logModel.updateOne(
      { _id: log._id },
      { $set: { isAlert: true } }
    );
    
    // 推送告警
    this.logsGateway.emitAlert(log);
  }
}
```

### 3. 日志网关

```typescript
// backend/src/logs/logs.gateway.ts

import {
  WebSocketGateway,
  WebSocketServer,
  SubscribeMessage,
  OnGatewayInit,
} from '@nestjs/websockets';
import { Server } from 'socket.io';
import { Injectable } from '@nestjs/common';

@Injectable()
@WebSocketGateway({
  cors: { origin: '*' },
  namespace: '/logs',
})
export class LogsGateway {
  @WebSocketServer()
  server: Server;

  /**
   * 推送新日志
   */
  emitNewLog(log: any) {
    this.server.emit('log:new', log);
  }

  /**
   * 推送告警
   */
  emitAlert(log: any) {
    this.server.emit('log:alert', log);
  }

  /**
   * 推送统计更新
   */
  emitStatsUpdate(stats: any) {
    this.server.emit('stats:update', stats);
  }
}
```
