# 方案三：Next.js + Supabase 待办清单应用

## 项目介绍

### 项目是什么
这是一个现代化的待办清单应用，支持多清单、标签、协作、番茄钟等功能。

### 核心技术亮点
- **Supabase 实时数据库**
- **React Query 数据管理**
- **拖拽排序**
- **PWA 离线支持**

```sql
-- 待办清单应用数据库设计

-- 用户（使用 Supabase Auth）

-- 清单
CREATE TABLE lists (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    color TEXT DEFAULT '#3B82F6',
    icon TEXT DEFAULT '📋',
    is_archived BOOLEAN DEFAULT FALSE,
    sort_order INT DEFAULT 0,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 标签
CREATE TABLE tags (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    color TEXT NOT NULL,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(user_id, name)
);

-- 待办事项
CREATE TABLE todos (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    list_id UUID REFERENCES lists(id) ON DELETE CASCADE,
    title TEXT NOT NULL,
    description TEXT,
    is_completed BOOLEAN DEFAULT FALSE,
    priority INT DEFAULT 2 CHECK (priority BETWEEN 1 AND 4), -- 1=低, 2=中, 3=高, 4=紧急
    due_date TIMESTAMPTZ,
    completed_at TIMESTAMPTZ,
    sort_order INT DEFAULT 0,
    estimated_pomodoros INT DEFAULT 1,
    actual_pomodoros INT DEFAULT 0,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 待办标签关联
CREATE TABLE todo_tags (
    todo_id UUID REFERENCES todos(id) ON DELETE CASCADE,
    tag_id UUID REFERENCES tags(id) ON DELETE CASCADE,
    PRIMARY KEY (todo_id, tag_id)
);

-- 番茄钟记录
CREATE TABLE pomodoro_sessions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
    todo_id UUID REFERENCES todos(id) ON DELETE SET NULL,
    started_at TIMESTAMPTZ NOT NULL,
    ended_at TIMESTAMPTZ,
    duration_minutes INT DEFAULT 25,
    completed BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 索引
CREATE INDEX idx_todos_list ON todos(list_id);
CREATE INDEX idx_todos_user ON todos(user_id);
CREATE INDEX idx_todos_completed ON todos(is_completed);
CREATE INDEX idx_todos_due_date ON todos(due_date);

-- RLS
ALTER TABLE lists ENABLE ROW LEVEL SECURITY;
ALTER TABLE tags ENABLE ROW LEVEL SECURITY;
ALTER TABLE todos ENABLE ROW LEVEL SECURITY;
ALTER TABLE todo_tags ENABLE ROW LEVEL SECURITY;
ALTER TABLE pomodoro_sessions ENABLE ROW LEVEL SECURITY;

CREATE POLICY "用户只能操作自己的数据" ON lists
    FOR ALL USING (auth.uid() = user_id);

CREATE POLICY "用户只能操作自己的数据" ON tags
    FOR ALL USING (auth.uid() = user_id);

CREATE POLICY "用户只能操作自己的数据" ON todos
    FOR ALL USING (auth.uid() = user_id);

CREATE POLICY "用户只能操作自己的数据" ON todo_tags
    FOR ALL USING (
        todo_id IN (SELECT id FROM todos WHERE user_id = auth.uid())
    );

CREATE POLICY "用户只能操作自己的数据" ON pomodoro_sessions
    FOR ALL USING (auth.uid() = user_id);
```

### 2. 待办事项管理

```typescript
// src/lib/todos.ts
import { supabase } from './supabase';
import type { Database } from '@/types/database';

type Todo = Database['public']['Tables']['todos']['Row'];
type TodoInsert = Database['public']['Tables']['todos']['Insert'];

interface TodoWithTags extends Todo {
  tags: Database['public']['Tables']['tags']['Row'][];
  list?: Database['public']['Tables']['lists']['Row'];
}

/**
 * 获取所有清单
 */
export async function getLists() {
  const { data, error } = await supabase
    .from('lists')
    .select('*')
    .eq('is_archived', false)
    .order('sort_order');
  
  return { data, error };
}

/**
 * 获取清单下的待办
 */
export async function getTodosByList(listId?: string) {
  let query = supabase
    .from('todos')
    .select(`
      *,
      tags:todo_tags(
        tags(*)
      ),
      list:lists(*)
    `)
    .eq('is_completed', false)
    .order('sort_order', { ascending: true });

  if (listId) {
    query = query.eq('list_id', listId);
  }

  const { data, error } = await query;
  
  // 处理嵌套的 tags
  const todos = data?.map((todo) => ({
    ...todo,
    tags: (todo as any).tags?.map((t: any) => t.tags) || [],
  })) || [];
  
  return { data: todos, error };
}

/**
 * 获取已完成待办
 */
export async function getCompletedTodos(listId?: string) {
  let query = supabase
    .from('todos')
    .select(`
      *,
      tags:todo_tags(
        tags(*)
      )
    `)
    .eq('is_completed', true)
    .order('completed_at', { ascending: false });

  if (listId) {
    query = query.eq('list_id', listId);
  }

  const { data, error } = await query;
  
  const todos = data?.map((todo) => ({
    ...todo,
    tags: (todo as any).tags?.map((t: any) => t.tags) || [],
  })) || [];
  
  return { data: todos, error };
}

/**
 * 创建待办
 */
export async function createTodo(todo: TodoInsert & { tagIds?: string[] }) {
  const { tagIds, ...todoData } = todo;
  
  // 创建待办
  const { data: newTodo, error } = await supabase
    .from('todos')
    .insert(todoData)
    .select()
    .single();
  
  if (error || !newTodo) return { data: null, error };
  
  // 添加标签关联
  if (tagIds && tagIds.length > 0) {
    await supabase.from('todo_tags').insert(
      tagIds.map((tagId) => ({
        todo_id: newTodo.id,
        tag_id: tagId,
      }))
    );
  }
  
  return { data: newTodo, error };
}

/**
 * 更新待办
 */
export async function updateTodo(
  id: string,
  updates: Partial<TodoInsert> & { tagIds?: string[] }
) {
  const { tagIds, ...todoData } = updates;
  
  // 更新待办
  const { data: updatedTodo, error } = await supabase
    .from('todos')
    .update({
      ...todoData,
      updated_at: new Date().toISOString(),
    })
    .eq('id', id)
    .select()
    .single();
  
  if (error || !updatedTodo) return { data: null, error };
  
  // 更新标签关联
  if (tagIds !== undefined) {
    // 先删除旧关联
    await supabase.from('todo_tags').delete().eq('todo_id', id);
    
    // 添加新关联
    if (tagIds.length > 0) {
      await supabase.from('todo_tags').insert(
        tagIds.map((tagId) => ({
          todo_id: id,
          tag_id: tagId,
        }))
      );
    }
  }
  
  return { data: updatedTodo, error };
}

/**
 * 标记完成/未完成
 */
export async function toggleTodoComplete(id: string) {
  const { data: todo, error: fetchError } = await supabase
    .from('todos')
    .select('is_completed')
    .eq('id', id)
    .single();
  
  if (fetchError || !todo) return { data: null, error: fetchError };
  
  const { data, error } = await supabase
    .from('todos')
    .update({
      is_completed: !todo.is_completed,
      completed_at: !todo.is_completed ? new Date().toISOString() : null,
    })
    .eq('id', id)
    .select()
    .single();
  
  return { data, error };
}

/**
 * 删除待办
 */
export async function deleteTodo(id: string) {
  const { error } = await supabase
    .from('todos')
    .delete()
    .eq('id', id);
  
  return { error };
}

/**
 * 重新排序
 */
export async function reorderTodos(
  todoIds: string[],
  listId?: string
) {
  const updates = todoIds.map((id, index) => ({
    id,
    sort_order: index,
  }));
  
  const { error } = await supabase.from('todos').upsert(updates);
  
  return { error };
}

/**
 * 获取统计数据
 */
export async function getStats() {
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  
  const tomorrow = new Date(today);
  tomorrow.setDate(tomorrow.getDate() + 1);
  
  const weekStart = new Date(today);
  weekStart.setDate(weekStart.getDate() - weekStart.getDay());
  
  const [totalResult, completedTodayResult, overdueResult, weekResult] = await Promise.all([
    supabase.from('todos').select('*', { count: 'exact', head: true }),
    supabase.from('todos').select('*', { count: 'exact', head: true })
      .eq('is_completed', true)
      .gte('completed_at', today.toISOString()),
    supabase.from('todos').select('*', { count: 'exact', head: true })
      .eq('is_completed', false)
      .lt('due_date', today.toISOString()),
    supabase.from('todos').select('*', { count: 'exact', head: true })
      .eq('is_completed', true)
      .gte('completed_at', weekStart.toISOString()),
  ]);
  
  return {
    total: totalResult.count || 0,
    completedToday: completedTodayResult.count || 0,
    overdue: overdueResult.count || 0,
    completedThisWeek: weekResult.count || 0,
  };
}
```

### 3. React Query Hooks

```typescript
// src/hooks/useTodos.ts
'use client';

import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { useCallback } from 'react';
import * as todosApi from '@/lib/todos';
import type { Database } from '@/types/database';

type Todo = Database['public']['Tables']['todos']['Row'];

export function useLists() {
  return useQuery({
    queryKey: ['lists'],
    queryFn: todosApi.getLists,
  });
}

export function useTodos(listId?: string) {
  return useQuery({
    queryKey: ['todos', listId],
    queryFn: () => todosApi.getTodosByList(listId),
    enabled: true,
  });
}

export function useCompletedTodos(listId?: string) {
  return useQuery({
    queryKey: ['todos', listId, 'completed'],
    queryFn: () => todosApi.getCompletedTodos(listId),
  });
}

export function useCreateTodo() {
  const queryClient = useQueryClient();
  
  return useMutation({
    mutationFn: (todo: any) => todosApi.createTodo(todo),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['todos'] });
      queryClient.invalidateQueries({ queryKey: ['stats'] });
    },
  });
}

export function useUpdateTodo() {
  const queryClient = useQueryClient();
  
  return useMutation({
    mutationFn: ({ id, updates }: { id: string; updates: any }) =>
      todosApi.updateTodo(id, updates),
    onSuccess: (_, { id }) => {
      queryClient.invalidateQueries({ queryKey: ['todos'] });
      queryClient.invalidateQueries({ queryKey: ['todo', id] });
    },
  });
}

export function useToggleTodo() {
  const queryClient = useQueryClient();
  
  return useMutation({
    mutationFn: (id: string) => todosApi.toggleTodoComplete(id),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['todos'] });
      queryClient.invalidateQueries({ queryKey: ['stats'] });
    },
  });
}

export function useDeleteTodo() {
  const queryClient = useQueryClient();
  
  return useMutation({
    mutationFn: (id: string) => todosApi.deleteTodo(id),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['todos'] });
      queryClient.invalidateQueries({ queryKey: ['stats'] });
    },
  });
}

export function useReorderTodos() {
  const queryClient = useQueryClient();
  
  return useMutation({
    mutationFn: ({ todoIds, listId }: { todoIds: string[]; listId?: string }) =>
      todosApi.reorderTodos(todoIds, listId),
    onMutate: async ({ todoIds, listId }) => {
      // 乐观更新
      await queryClient.cancelQueries({ queryKey: ['todos', listId] });
      
      const previousTodos = queryClient.getQueryData(['todos', listId]);
      
      queryClient.setQueryData(['todos', listId], (old: any) => {
        if (!old?.data) return old;
        
        const todoMap = new Map(old.data.map((t: Todo) => [t.id, t]));
        const reordered = todoIds.map((id, index) => ({
          ...todoMap.get(id),
          sort_order: index,
        }));
        
        return {
          ...old,
          data: reordered,
        };
      });
      
      return { previousTodos };
    },
    onError: (_, __, context) => {
      if (context?.previousTodos) {
        queryClient.setQueryData(['todos'], context.previousTodos);
      }
    },
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ['todos'] });
    },
  });
}

export function useStats() {
  return useQuery({
    queryKey: ['stats'],
    queryFn: todosApi.getStats,
    refetchInterval: 30000, // 每30秒刷新
  });
}
```

### 4. 待办组件

```typescript
// src/components/TodoList.tsx
'use client';

import { useState } from 'react';
import { useTodos, useToggleTodo, useDeleteTodo, useReorderTodos } from '@/hooks/useTodos';
import { TodoItem } from './TodoItem';
import { TodoForm } from './TodoForm';
import styles from './TodoList.module.css';

interface TodoListProps {
  listId?: string;
  title: string;
}

export function TodoList({ listId, title }: TodoListProps) {
  const { data: todosResult, isLoading, error } = useTodos(listId);
  const toggleTodo = useToggleTodo();
  const deleteTodo = useDeleteTodo();
  const reorderTodos = useReorderTodos();
  
  const [showCompleted, setShowCompleted] = useState(false);
  const [draggedId, setDraggedId] = useState<string | null>(null);
  
  const todos = todosResult?.data || [];
  const incompleteTodos = todos.filter(t => !t.is_completed);
  const completedTodos = todos.filter(t => t.is_completed);
  
  // 拖拽开始
  const handleDragStart = (id: string) => {
    setDraggedId(id);
  };
  
  // 拖拽结束
  const handleDragEnd = () => {
    setDraggedId(null);
  };
  
  // 拖拽悬停
  const handleDragOver = (e: React.DragEvent, targetId: string) => {
    e.preventDefault();
    
    if (!draggedId || draggedId === targetId) return;
    
    const draggedIndex = incompleteTodos.findIndex(t => t.id === draggedId);
    const targetIndex = incompleteTodos.findIndex(t => t.id === targetId);
    
    if (draggedIndex === -1 || targetIndex === -1) return;
    
    const newOrder = [...incompleteTodos];
    const [removed] = newOrder.splice(draggedIndex, 1);
    newOrder.splice(targetIndex, 0, removed);
    
    reorderTodos.mutate({
      todoIds: newOrder.map(t => t.id),
      listId,
    });
  };
  
  if (isLoading) {
    return (
      <div className={styles.loading}>
        <div className={styles.spinner} />
        <p>加载中...</p>
      </div>
    );
  }
  
  if (error) {
    return (
      <div className={styles.error}>
        <p>加载失败: {error.message}</p>
      </div>
    );
  }
  
  return (
    <div className={styles.todoList}>
      <header className={styles.header}>
        <h2 className={styles.title}>{title}</h2>
        <span className={styles.count}>
          {incompleteTodos.length} 项待办
        </span>
      </header>
      
      {/* 新建表单 */}
      <TodoForm listId={listId} />
      
      {/* 待办列表 */}
      <div className={styles.items}>
        {incompleteTodos.map((todo, index) => (
          <TodoItem
            key={todo.id}
            todo={todo}
            index={index}
            onToggle={() => toggleTodo.mutate(todo.id)}
            onDelete={() => deleteTodo.mutate(todo.id)}
            onDragStart={() => handleDragStart(todo.id)}
            onDragEnd={handleDragEnd}
            onDragOver={(e) => handleDragOver(e, todo.id)}
            isDragging={draggedId === todo.id}
          />
        ))}
        
        {incompleteTodos.length === 0 && (
          <div className={styles.empty}>
            <p>🎉 所有任务已完成！</p>
            <p>或者添加一个新任务</p>
          </div>
        )}
      </div>
      
      {/* 已完成区域 */}
      {completedTodos.length > 0 && (
        <div className={styles.completed}>
          <button
            className={styles.completedToggle}
            onClick={() => setShowCompleted(!showCompleted)}
          >
            {showCompleted ? '▼' : '▶'} 已完成 ({completedTodos.length})
          </button>
          
          {showCompleted && (
            <div className={styles.items}>
              {completedTodos.map((todo) => (
                <TodoItem
                  key={todo.id}
                  todo={todo}
                  onToggle={() => toggleTodo.mutate(todo.id)}
                  onDelete={() => deleteTodo.mutate(todo.id)}
                />
              ))}
            </div>
          )}
        </div>
      )}
    </div>
  );
}
```
