# 方案三：Next.js + Supabase 实时聊天应用

## 项目介绍

### 项目是什么
这是一个实时聊天应用，类似 Slack/Discord，支持一对一聊天、群聊、实时消息、在线状态等功能。

### 核心技术亮点
- **Supabase**: BaaS 后端即服务
- **Realtime**: 实时消息推送
- **Next.js 14 App Router**
- **Tailwind CSS**

### 为什么选 Supabase
- 开箱即用的身份认证
- 实时数据库订阅
- 存储服务（文件上传）
- 自动生成 REST API
- Row Level Security (RLS)

### 2. Supabase 数据库设计

```sql
-- supabase/migrations/001_initial.sql

-- 启用 UUID 扩展
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- 用户配置（扩展 Supabase auth.users）
CREATE TABLE public.profiles (
    id UUID REFERENCES auth.users(id) ON DELETE CASCADE PRIMARY KEY,
    username TEXT UNIQUE NOT NULL,
    full_name TEXT,
    avatar_url TEXT,
    status TEXT DEFAULT 'offline' CHECK (status IN ('online', 'offline', 'away')),
    last_seen_at TIMESTAMPTZ DEFAULT NOW(),
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 群组/频道
CREATE TABLE public.conversations (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    name TEXT,
    is_direct BOOLEAN DEFAULT FALSE,
    is_group BOOLEAN DEFAULT FALSE,
    avatar_url TEXT,
    created_by UUID REFERENCES public.profiles(id),
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 群组成员
CREATE TABLE public.conversation_members (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    conversation_id UUID REFERENCES public.conversations(id) ON DELETE CASCADE,
    user_id UUID REFERENCES public.profiles(id) ON DELETE CASCADE,
    role TEXT DEFAULT 'member' CHECK (role IN ('owner', 'admin', 'member')),
    joined_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(conversation_id, user_id)
);

-- 消息
CREATE TABLE public.messages (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    conversation_id UUID REFERENCES public.conversations(id) ON DELETE CASCADE,
    sender_id UUID REFERENCES public.profiles(id),
    content TEXT NOT NULL,
    content_type TEXT DEFAULT 'text' CHECK (content_type IN ('text', 'image', 'file', 'system')),
    metadata JSONB DEFAULT '{}',
    is_edited BOOLEAN DEFAULT FALSE,
    is_deleted BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 消息阅读状态
CREATE TABLE public.message_reads (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    message_id UUID REFERENCES public.messages(id) ON DELETE CASCADE,
    user_id UUID REFERENCES public.profiles(id) ON DELETE CASCADE,
    read_at TIMESTAMPTZ DEFAULT NOW(),
    UNIQUE(message_id, user_id)
);

-- 在线状态（使用 Supabase Realtime Presence）
-- 不需要数据库表，Presence 会自动管理

-- ================== 索引 ==================
CREATE INDEX idx_messages_conversation ON messages(conversation_id);
CREATE INDEX idx_messages_created_at ON messages(created_at DESC);
CREATE INDEX idx_conversation_members_user ON conversation_members(user_id);
CREATE INDEX idx_messages_sender ON messages(sender_id);

-- ================== 视图：对话列表 ==================
CREATE VIEW public.conversation_list AS
SELECT 
    c.id,
    c.name,
    c.is_direct,
    c.is_group,
    c.avatar_url,
    c.created_at,
    c.updated_at,
    -- 获取最后一条消息
    (
        SELECT content FROM messages 
        WHERE conversation_id = c.id AND is_deleted = FALSE
        ORDER BY created_at DESC LIMIT 1
    ) as last_message,
    (
        SELECT created_at FROM messages 
        WHERE conversation_id = c.id
        ORDER BY created_at DESC LIMIT 1
    ) as last_message_at,
    (
        SELECT COUNT(*) FROM messages m
        JOIN conversation_members cm ON cm.conversation_id = m.conversation_id
        WHERE m.conversation_id = c.id 
        AND m.sender_id != auth.uid()
        AND m.id NOT IN (
            SELECT message_id FROM message_reads WHERE user_id = auth.uid()
        )
    ) as unread_count,
    -- 成员信息
    (
        SELECT json_agg(json_build_object(
            'id', p.id,
            'username', p.username,
            'avatar_url', p.avatar_url,
            'status', p.status
        ))
        FROM conversation_members cm
        JOIN profiles p ON p.id = cm.user_id
        WHERE cm.conversation_id = c.id
    ) as members
FROM conversations c
JOIN conversation_members cm ON cm.conversation_id = c.id
WHERE cm.user_id = auth.uid();

-- ================== RLS 策略 ==================
ALTER TABLE profiles ENABLE ROW LEVEL SECURITY;
ALTER TABLE conversations ENABLE ROW LEVEL SECURITY;
ALTER TABLE conversation_members ENABLE ROW LEVEL SECURITY;
ALTER TABLE messages ENABLE ROW LEVEL SECURITY;
ALTER TABLE message_reads ENABLE ROW LEVEL SECURITY;

-- profiles 策略
CREATE POLICY "公开个人资料" ON profiles FOR SELECT USING (true);
CREATE POLICY "更新自己的资料" ON profiles FOR UPDATE USING (auth.uid() = id);

-- conversations 策略
CREATE POLICY "查看参与的对话" ON conversations FOR SELECT
USING (
    id IN (
        SELECT conversation_id FROM conversation_members 
        WHERE user_id = auth.uid()
    )
);

CREATE POLICY "创建对话" ON conversations FOR INSERT
WITH CHECK (auth.uid() = created_by);

-- conversation_members 策略
CREATE POLICY "查看成员" ON conversation_members FOR SELECT
USING (
    conversation_id IN (
        SELECT conversation_id FROM conversation_members 
        WHERE user_id = auth.uid()
    )
);

CREATE POLICY "加入对话" ON conversation_members FOR INSERT
WITH CHECK (auth.uid() = user_id);

-- messages 策略
CREATE POLICY "发送消息" ON messages FOR INSERT
WITH CHECK (
    auth.uid() = sender_id AND
    conversation_id IN (
        SELECT conversation_id FROM conversation_members 
        WHERE user_id = auth.uid()
    )
);

CREATE POLICY "查看消息" ON messages FOR SELECT
USING (
    conversation_id IN (
        SELECT conversation_id FROM conversation_members 
        WHERE user_id = auth.uid()
    )
);

CREATE POLICY "修改自己的消息" ON messages FOR UPDATE
USING (auth.uid() = sender_id);

-- ================== 触发器 ==================

-- 创建用户时自动创建 profile
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO public.profiles (id, username, full_name, avatar_url)
    VALUES (
        NEW.id,
        COALESCE(NEW.raw_user_meta_data->>'username', 'user_' || LEFT(NEW.id::text, 8)),
        NEW.raw_user_meta_data->>'full_name',
        NEW.raw_user_meta_data->>'avatar_url'
    );
    RETURN NEW;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_auth_user_created
    AFTER INSERT ON auth.users
    FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();

-- 自动更新 updated_at
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER profiles_updated_at BEFORE UPDATE ON profiles
    FOR EACH ROW EXECUTE FUNCTION update_updated_at();

CREATE TRIGGER conversations_updated_at BEFORE UPDATE ON conversations
    FOR EACH ROW EXECUTE FUNCTION update_updated_at();

CREATE TRIGGER messages_updated_at BEFORE UPDATE ON messages
    FOR EACH ROW EXECUTE FUNCTION update_updated_at();

-- 启用实时订阅
ALTER PUBLICATION supabase_realtime ADD TABLE messages;
ALTER PUBLICATION supabase_realtime ADD TABLE conversation_members;
ALTER PUBLICATION supabase_realtime ADD TABLE profiles;
```

### 5. 认证 Hook

```typescript
// src/hooks/useAuth.ts
'use client';

import { useEffect, useState } from 'react';
import { User, Session } from '@supabase/supabase-js';
import { supabase } from '@/lib/supabase';
import { useRouter } from 'next/navigation';
import type { Profile } from '@/types/database';

export function useAuth() {
  const [user, setUser] = useState<User | null>(null);
  const [profile, setProfile] = useState<Profile | null>(null);
  const [isLoading, setIsLoading] = useState(true);
  const [isInitialized, setIsInitialized] = useState(false);
  const router = useRouter();

  useEffect(() => {
    // 获取初始会话
    const initAuth = async () => {
      const { data: { session } } = await supabase.auth.getSession();
      
      if (session?.user) {
        setUser(session.user);
        await fetchProfile(session.user.id);
      }
      
      setIsLoading(false);
      setIsInitialized(true);
    };

    initAuth();

    // 监听认证状态变化
    const { data: { subscription } } = supabase.auth.onAuthStateChange(
      async (event, session) => {
        setUser(session?.user ?? null);
        
        if (session?.user) {
          await fetchProfile(session.user.id);
        } else {
          setProfile(null);
        }
        
        // 处理特定事件
        if (event === 'SIGNED_OUT') {
          router.push('/login');
        }
      }
    );

    return () => {
      subscription.unsubscribe();
    };
  }, [router]);

  const fetchProfile = async (userId: string) => {
    const { data, error } = await supabase
      .from('profiles')
      .select('*')
      .eq('id', userId)
      .single();
    
    if (!error && data) {
      setProfile(data);
    }
  };

  const signOut = async () => {
    await supabase.auth.signOut();
    setUser(null);
    setProfile(null);
    router.push('/login');
  };

  const updateProfile = async (updates: Partial<Profile>) => {
    if (!user) return;
    
    const { data, error } = await supabase
      .from('profiles')
      .update(updates)
      .eq('id', user.id)
      .select()
      .single();
    
    if (!error && data) {
      setProfile(data);
    }
    
    return { data, error };
  };

  return {
    user,
    profile,
    isLoading,
    isInitialized,
    isAuthenticated: !!user,
    signOut,
    updateProfile,
  };
}
```

### 6. 实时聊天 Hook

```typescript
// src/hooks/useRealtimeChat.ts
'use client';

import { useEffect, useState, useCallback } from 'react';
import { supabase } from '@/lib/supabase';
import { useAuth } from './useAuth';
import type { Message, MessageWithSender } from '@/types/database';

export function useRealtimeChat(conversationId: string | null) {
  const { user } = useAuth();
  const [messages, setMessages] = useState<MessageWithSender[]>([]);
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  // 加载消息
  const loadMessages = useCallback(async () => {
    if (!conversationId) return;
    
    setIsLoading(true);
    setError(null);
    
    const { data, error } = await supabase
      .from('messages')
      .select(`
        *,
        sender:profiles(id, username, avatar_url, status)
      `)
      .eq('conversation_id', conversationId)
      .eq('is_deleted', false)
      .order('created_at', { ascending: true })
      .limit(100);
    
    if (error) {
      setError(error.message);
    } else {
      setMessages(data || []);
    }
    
    setIsLoading(false);
  }, [conversationId]);

  // 初始加载
  useEffect(() => {
    loadMessages();
  }, [loadMessages]);

  // 实时订阅
  useEffect(() => {
    if (!conversationId) return;

    // 订阅新消息
    const channel = supabase
      .channel(`messages:${conversationId}`)
      .on(
        'postgres_changes',
        {
          event: 'INSERT',
          schema: 'public',
          table: 'messages',
          filter: `conversation_id=eq.${conversationId}`,
        },
        async (payload) => {
          // 获取发送者信息
          const senderId = payload.new.sender_id;
          if (senderId && senderId !== user?.id) {
            const { data: sender } = await supabase
              .from('profiles')
              .select('id, username, avatar_url, status')
              .eq('id', senderId)
              .single();
            
            setMessages((prev) => [
              ...prev,
              { ...payload.new, sender } as MessageWithSender,
            ]);
          } else {
            setMessages((prev) => [
              ...prev,
              { ...payload.new, sender: null } as MessageWithSender,
            ]);
          }
          
          // 标记已读
          if (user) {
            await markAsRead(payload.new.id);
          }
        }
      )
      .on(
        'postgres_changes',
        {
          event: 'UPDATE',
          schema: 'public',
          table: 'messages',
          filter: `conversation_id=eq.${conversationId}`,
        },
        (payload) => {
          setMessages((prev) =>
            prev.map((msg) =>
              msg.id === payload.new.id
                ? { ...msg, ...payload.new }
                : msg
            )
          );
        }
      )
      .subscribe();

    return () => {
      supabase.removeChannel(channel);
    };
  }, [conversationId, user]);

  // 发送消息
  const sendMessage = useCallback(
    async (content: string, contentType: 'text' | 'image' | 'file' = 'text') => {
      if (!conversationId || !user) return null;

      const { data, error } = await supabase
        .from('messages')
        .insert({
          conversation_id: conversationId,
          sender_id: user.id,
          content,
          content_type: contentType,
        })
        .select(`
          *,
          sender:profiles(id, username, avatar_url, status)
        `)
        .single();

      if (error) {
        setError(error.message);
        return null;
      }

      // 乐观更新
      setMessages((prev) => [...prev, data as MessageWithSender]);
      
      return data;
    },
    [conversationId, user]
  );

  // 编辑消息
  const editMessage = useCallback(
    async (messageId: string, newContent: string) => {
      const { data, error } = await supabase
        .from('messages')
        .update({
          content: newContent,
          is_edited: true,
        })
        .eq('id', messageId)
        .select()
        .single();

      if (!error && data) {
        setMessages((prev) =>
          prev.map((msg) =>
            msg.id === messageId ? { ...msg, ...data } : msg
          )
        );
      }

      return { data, error };
    },
    []
  );

  // 删除消息（软删除）
  const deleteMessage = useCallback(async (messageId: string) => {
    const { data, error } = await supabase
      .from('messages')
      .update({ is_deleted: true })
      .eq('id', messageId)
      .select()
      .single();

    if (!error && data) {
      setMessages((prev) =>
        prev.map((msg) =>
          msg.id === messageId ? { ...msg, content: '[已删除]' } : msg
        )
      );
    }

    return { data, error };
  }, []);

  // 标记已读
  const markAsRead = async (messageId: string) => {
    if (!user) return;
    
    await supabase.from('message_reads').upsert({
      message_id: messageId,
      user_id: user.id,
    });
  };

  return {
    messages,
    isLoading,
    error,
    sendMessage,
    editMessage,
    deleteMessage,
    loadMessages,
  };
}
```

```typescript
// src/components/ChatWindow.tsx
'use client';

import { useState, useRef, useEffect } from 'react';
import { useRealtimeChat } from '@/hooks/useRealtimeChat';
import { useAuth } from '@/hooks/useAuth';
import { MessageList } from './MessageList';
import { MessageInput } from './MessageInput';
import { UserAvatar } from './UserAvatar';
import type { ConversationListItem, Profile } from '@/types/database';
import styles from './ChatWindow.module.css';

interface ChatWindowProps {
  conversation: ConversationListItem;
}

export function ChatWindow({ conversation }: ChatWindowProps) {
  const { profile } = useAuth();
  const {
    messages,
    isLoading,
    error,
    sendMessage,
    editMessage,
    deleteMessage,
  } = useRealtimeChat(conversation.id);
  
  const [isTyping, setIsTyping] = useState(false);
  const messagesEndRef = useRef<HTMLDivElement>(null);

  // 获取对话名称
  const getConversationName = (): string => {
    if (conversation.name) return conversation.name;
    
    // 直接消息：显示对方用户名
    if (conversation.is_direct && conversation.members) {
      const otherMember = conversation.members.find(
        (m) => m.id !== profile?.id
      );
      return otherMember?.username || 'Unknown';
    }
    
    return '群聊';
  };

  // 获取在线状态
  const getOnlineStatus = (): 'online' | 'offline' | 'away' | null => {
    if (!conversation.is_direct || !conversation.members) return null;
    
    const otherMember = conversation.members.find(
      (m) => m.id !== profile?.id
    );
    return otherMember?.status as 'online' | 'offline' | 'away' || null;
  };

  // 自动滚动到底部
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [messages]);

  // 处理发送消息
  const handleSend = async (content: string) => {
    await sendMessage(content);
  };

  // 处理编辑消息
  const handleEdit = async (messageId: string, newContent: string) => {
    await editMessage(messageId, newContent);
  };

  // 处理删除消息
  const handleDelete = async (messageId: string) => {
    await deleteMessage(messageId);
  };

  return (
    <div className={styles.chatWindow}>
      {/* 头部 */}
      <header className={styles.header}>
        <div className={styles.headerInfo}>
          <UserAvatar
            name={getConversationName()}
            avatar={conversation.avatar_url}
            status={getOnlineStatus()}
            size="md"
          />
          <div className={styles.headerText}>
            <h2 className={styles.headerName}>{getConversationName()}</h2>
            {conversation.is_direct && (
              <span className={styles.headerStatus}>
                {getOnlineStatus() === 'online' ? '在线' : '离线'}
              </span>
            )}
          </div>
        </div>
        
        <div className={styles.headerActions}>
          <button className={styles.headerBtn}>📞</button>
          <button className={styles.headerBtn}>🎥</button>
          <button className={styles.headerBtn}>⚙️</button>
        </div>
      </header>

      {/* 消息列表 */}
      <MessageList
        messages={messages}
        currentUserId={profile?.id}
        isLoading={isLoading}
```

```typescript
// src/components/MessageList.tsx
'use client';

import { useState } from 'react';
import { formatDistanceToNow } from 'date-fns';
import { zhCN } from 'date-fns/locale';
import { UserAvatar } from './UserAvatar';
import type { MessageWithSender } from '@/types/database';
import styles from './MessageList.module.css';

interface MessageListProps {
  messages: MessageWithSender[];
  currentUserId?: string;
  isLoading: boolean;
  error: string | null;
  onEdit: (messageId: string, newContent: string) => void;
  onDelete: (messageId: string) => void;
  messagesEndRef: React.RefObject<HTMLDivElement>;
}

export function MessageList({
  messages,
  currentUserId,
  isLoading,
  error,
  onEdit,
  onDelete,
  messagesEndRef,
}: MessageListProps) {
  const [editingId, setEditingId] = useState<string | null>(null);
  const [editContent, setEditContent] = useState('');

  // 开始编辑
  const startEdit = (message: MessageWithSender) => {
    setEditingId(message.id);
    setEditContent(message.content);
  };

  // 取消编辑
  const cancelEdit = () => {
    setEditingId(null);
    setEditContent('');
  };

  // 提交编辑
  const submitEdit = (messageId: string) => {
    onEdit(messageId, editContent);
    cancelEdit();
  };

  if (isLoading) {
    return (
      <div className={styles.loading}>
        <div className={styles.spinner} />
        <p>加载消息中...</p>
      </div>
    );
  }

  if (error) {
    return (
      <div className={styles.error}>
        <p>加载失败: {error}</p>
        <button onClick={() => window.location.reload()}>重试</button>
      </div>
    );
  }

  if (messages.length === 0) {
    return (
      <div className={styles.empty}>
        <p>还没有消息，开始聊天吧！</p>
      </div>
    );
  }

  // 按日期分组
  const groupedMessages = messages.reduce((groups, message) => {
    const date = new Date(message.created_at).toDateString();
    if (!groups[date]) {
      groups[date] = [];
    }
    groups[date].push(message);
    return groups;
  }, {} as Record<string, MessageWithSender[]>);

  return (
    <div className={styles.messageList}>
      {Object.entries(groupedMessages).map(([date, msgs]) => (
        <div key={date} className={styles.dateGroup}>
          <div className={styles.dateDivider}>
            <span>{formatDistanceToNow(new Date(date), { addSuffix: true, locale: zhCN })}</span>
          </div>
          
          {msgs.map((message) => {
            const isOwn = message.sender_id === currentUserId;
            const isEditing = editingId === message.id;
            
            return (
              <div
                key={message.id}
                className={`${styles.message} ${isOwn ? styles.messageOwn : ''}`}
              >
                {!isOwn && (
                  <UserAvatar
                    name={message.sender?.username || '?'}
                    avatar={message.sender?.avatar_url}
                    status={message.sender?.status as any}
                    size="sm"
                  />
                )}
                
                <div className={styles.messageContent}>
                  {!isOwn && (
                    <div className={styles.messageHeader}>
                      <span className={styles.messageSender}>
                        {message.sender?.username}
                      </span>
                      <span className={styles.messageTime}>
                        {formatDistanceToNow(new Date(message.created_at), { locale: zhCN })}
                      </span>
                    </div>
                  )}
                  
                  <div className={styles.messageBubble}>
                    {isEditing ? (
                      <div className={styles.editForm}>
                        <textarea
                          value={editContent}
                          onChange={(e) => setEditContent(e.target.value)}
                          className={styles.editInput}
                        />
                        <div className={styles.editActions}>
                          <button onClick={() => submitEdit(message.id)}>保存</button>
                          <button onClick={cancelEdit}>取消</button>
                        </div>
                      </div>
                    ) : (
                      <p className={styles.messageText}>
                        {message.content}
                        {message.is_edited && <span className={styles.edited}>(已编辑)</span>}
                      </p>
                    )}
                  </div>
                  
                  {isOwn && (
                    <div className={styles.messageActions}>
                      <button onClick={() => startEdit(message)}>编辑</button>
                      <button onClick={() => onDelete(message.id)}>删除</button>
                    </div>
                  )}
                </div>
              </div>
            );
          })}
        </div>
      ))}
      
      <div ref={messagesEndRef} />
    </div>
  );
}
```

### 问题: Supabase 与传统后端的区别？

**参考答案**:
```
Supabase 是 BaaS (Backend as a Service)：
1. 开箱即用的认证
2. PostgreSQL 数据库 + 自动 API
3. 实时订阅 (Realtime)
4. 文件存储 (Storage)
5. Row Level Security

优势：
- 快速开发
- 免维护
- 成本低

劣势：
- 灵活性受限
- 复杂业务逻辑困难
- 供应商锁定
```
