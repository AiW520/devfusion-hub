# 方案六：T3 Stack 在线商城

## 项目介绍

### 项目是什么
这是一个功能完整的电商商城系统，支持商品浏览、购物车、订单管理、支付集成、用户评价等功能。

### 核心技术亮点
- **T3 Stack** 全栈技术
- **Stripe** 支付集成
- **购物车状态管理** (Zustand)
- **图片上传** (Cloudinary)
- **邮件通知** (Resend)

### 2. tRPC 路由

```typescript
// src/server/api/routers/product.ts

import { z } from 'zod';
import { createTRPCRouter, publicProcedure, protectedProcedure } from '../trpc';

export const productRouter = createTRPCRouter({
  // 获取商品列表
  getAll: publicProcedure
    .input(z.object({
      limit: z.number().min(1).max(50).default(12),
      cursor: z.string().nullish(),
      categorySlug: z.string().optional(),
      search: z.string().optional(),
      sortBy: z.enum(['newest', 'price-asc', 'price-desc', 'best-selling']).default('newest'),
    }))
    .query(async ({ ctx, input }) => {
      const { limit, cursor, categorySlug, search, sortBy } = input;
      
      const where: any = { isActive: true };
      
      if (categorySlug) {
        const category = await ctx.prisma.category.findUnique({
          where: { slug: categorySlug },
          include: { children: true },
        });
        
        if (category) {
          where.categoryId = {
            in: [category.id, ...category.children.map(c => c.id)],
          };
        }
      }
      
      if (search) {
        where.OR = [
          { name: { contains: search, mode: 'insensitive' } },
          { description: { contains: search, mode: 'insensitive' } },
        ];
      }
      
      const orderBy: any = { createdAt: 'desc' };
      if (sortBy === 'price-asc') orderBy = { price: 'asc' };
      if (sortBy === 'price-desc') orderBy = { price: 'desc' };
      if (sortBy === 'best-selling') orderBy = { soldCount: 'desc' };
      
      const products = await ctx.prisma.product.findMany({
        take: limit + 1,
        cursor: cursor ? { id: cursor } : undefined,
        where,
        orderBy,
        include: {
          category: { select: { name: true, slug: true } },
        },
      });
      
      let nextCursor: typeof cursor | undefined;
      if (products.length > limit) {
        const next = products.pop();
        nextCursor = next!.id;
      }
      
      return { products, nextCursor };
    }),

  // 获取单个商品
  getBySlug: publicProcedure
    .input(z.object({ slug: z.string() }))
    .query(async ({ ctx, input }) => {
      const product = await ctx.prisma.product.findUnique({
        where: { slug: input.slug },
        include: {
          category: true,
          reviews: {
            take: 10,
            orderBy: { createdAt: 'desc' },
            include: {
              user: { select: { name: true, image: true } },
            },
          },
        },
      });
      
      if (!product || !product.isActive) {
        throw new TRPCError({ code: 'NOT_FOUND', message: '商品不存在' });
      }
      
      return product;
    }),

  // 获取分类
  getCategories: publicProcedure.query(async ({ ctx }) => {
    return ctx.prisma.category.findMany({
      where: { parentId: null },
      include: { children: true },
      orderBy: { sortOrder: 'asc' },
    });
  }),
});
```

```typescript
// src/server/api/routers/cart.ts

export const cartRouter = createTRPCRouter({
  // 获取购物车
  getCart: protectedProcedure.query(async ({ ctx }) => {
    const items = await ctx.prisma.cartItem.findMany({
      where: { userId: ctx.session.user.id },
      include: {
        product: {
          select: {
            id: true,
            name: true,
            slug: true,
            price: true,
            images: true,
            stock: true,
          },
        },
      },
    });
    
    const subtotal = items.reduce((sum, item) => {
      return sum + item.product.price * item.quantity;
    }, 0);
    
    return { items, subtotal };
  }),

  // 添加到购物车
  addItem: protectedProcedure
    .input(z.object({
      productId: z.string(),
      quantity: z.number().min(1).max(99).default(1),
    }))
    .mutation(async ({ ctx, input }) => {
      // 检查商品存在且有库存
      const product = await ctx.prisma.product.findUnique({
        where: { id: input.productId },
      });
      
      if (!product || !product.isActive) {
        throw new TRPCError({ code: 'NOT_FOUND', message: '商品不存在' });
      }
      
      if (product.stock < input.quantity) {
        throw new TRPCError({ code: 'BAD_REQUEST', message: '库存不足' });
      }
      
      // 查找或创建购物车项
      const existingItem = await ctx.prisma.cartItem.findUnique({
        where: {
          userId_productId: {
            userId: ctx.session.user.id,
            productId: input.productId,
          },
        },
      });
      
      if (existingItem) {
        return ctx.prisma.cartItem.update({
          where: { id: existingItem.id },
          data: { quantity: existingItem.quantity + input.quantity },
        });
      } else {
        return ctx.prisma.cartItem.create({
          data: {
            userId: ctx.session.user.id,
            productId: input.productId,
            quantity: input.quantity,
          },
        });
      }
    }),

  // 更新数量
  updateQuantity: protectedProcedure
    .input(z.object({
      productId: z.string(),
      quantity: z.number().min(1).max(99),
    }))
    .mutation(async ({ ctx, input }) => {
      const item = await ctx.prisma.cartItem.findUnique({
        where: {
          userId_productId: {
            userId: ctx.session.user.id,
            productId: input.productId,
          },
        },
      });
      
      if (!item) {
        throw new TRPCError({ code: 'NOT_FOUND', message: '购物车项不存在' });
      }
      
      return ctx.prisma.cartItem.update({
        where: { id: item.id },
        data: { quantity: input.quantity },
      });
    }),

  // 删除购物车项
  removeItem: protectedProcedure
    .input(z.object({ productId: z.string() }))
    .mutation(async ({ ctx, input }) => {
      await ctx.prisma.cartItem.deleteMany({
        where: {
          userId: ctx.session.user.id,
          productId: input.productId,
        },
      });
      
      return { success: true };
    }),

  // 清空购物车
  clearCart: protectedProcedure.mutation(async ({ ctx }) => {
    await ctx.prisma.cartItem.deleteMany({
      where: { userId: ctx.session.user.id },
    });
    
    return { success: true };
  }),
});
```

```typescript
// src/server/api/routers/order.ts

export const orderRouter = createTRPCRouter({
  // 创建订单
  create: protectedProcedure
    .input(z.object({
      addressId: z.string(),
      note: z.string().optional(),
      paymentMethod: z.enum(['stripe', 'alipay', 'wechat']),
    }))
    .mutation(async ({ ctx, input }) => {
      // 获取购物车
      const cartItems = await ctx.prisma.cartItem.findMany({
        where: { userId: ctx.session.user.id },
        include: { product: true },
      });
      
      if (cartItems.length === 0) {
        throw new TRPCError({ code: 'BAD_REQUEST', message: '购物车为空' });
      }
      
      // 获取地址
      const address = await ctx.prisma.address.findFirst({
        where: {
          id: input.addressId,
          userId: ctx.session.user.id,
        },
      });
      
      if (!address) {
        throw new TRPCError({ code: 'BAD_REQUEST', message: '地址不存在' });
      }
      
      // 计算金额
      const subtotal = cartItems.reduce(
        (sum, item) => sum + item.product.price * item.quantity, 
        0
      );
      
      // 运费计算（满99免运费）
      const shippingFee = subtotal >= 9900 ? 0 : 1000;
      const total = subtotal + shippingFee;
      
      // 生成订单号
      const orderNumber = `ORD${Date.now()}${Math.random().toString(36).substr(2, 6).toUpperCase()}`;
      
      // 创建订单
      const order = await ctx.prisma.$transaction(async (tx) => {
        // 创建订单
        const newOrder = await tx.order.create({
          data: {
            orderNumber,
            userId: ctx.session.user.id,
            
            shippingName: address.name,
            shippingPhone: address.phone,
            shippingStreet: address.street,
            shippingCity: address.city,
            shippingState: address.state,
            shippingPostalCode: address.postalCode,
            shippingCountry: address.country,
            
            subtotal,
            shippingFee,
            total,
            note: input.note,
            
            items: {
              create: cartItems.map(item => ({
                productId: item.productId,
                name: item.product.name,
                price: item.product.price,
                quantity: item.quantity,
              })),
            },
          },
        });
        
        // 更新库存
        for (const item of cartItems) {
          await tx.product.update({
            where: { id: item.productId },
            data: {
              stock: { decrement: item.quantity },
              soldCount: { increment: item.quantity },
            },
          });
        }
        
        // 清空购物车
        await tx.cartItem.deleteMany({
          where: { userId: ctx.session.user.id },
        });
        
        return newOrder;
      });
      
      return order;
    }),

  // 获取订单列表
  getOrders: protectedProcedure
    .input(z.object({
      limit: z.number().min(1).max(20).default(10),
      cursor: z.string().nullish(),
    }))
    .query(async ({ ctx, input }) => {
      const orders = await ctx.prisma.order.findMany({
        where: { userId: ctx.session.user.id },
        take: input.limit + 1,
        cursor: input.cursor ? { id: input.cursor } : undefined,
        orderBy: { createdAt: 'desc' },
        include: {
          items: {
            include: {
              product: {
                select: { id: true, slug: true, images: true },
              },
            },
          },
        },
      });
      
      let nextCursor: typeof input.cursor | undefined;
      if (orders.length > input.limit) {
        const next = orders.pop();
        nextCursor = next!.id;
      }
      
      return { orders, nextCursor };
    }),

  // 获取订单详情
  getById: protectedProcedure
    .input(z.object({ id: z.string() }))
    .query(async ({ ctx, input }) => {
      const order = await ctx.prisma.order.findUnique({
        where: { id: input.id },
        include: {
          items: {
            include: {
              product: true,
            },
          },
        },
      });
      
      if (!order || order.userId !== ctx.session.user.id) {
        throw new TRPCError({ code: 'NOT_FOUND', message: '订单不存在' });
      }
      
      return order;
    }),
});
```
