## 项目介绍

### 项目是什么
这是一个专为开发者设计的博客平台，支持 Markdown 编写、代码高亮、SEO 优化、订阅通知、Giscus 评论系统。

### 核心技术亮点
- **T3 Stack** 全栈
- **MDX** Markdown + React
- **Prism.js** 代码高亮

### 2. MDX 处理

```typescript
// src/server/posts.ts

import { prisma } from './db';
import { serialize } from 'next-mdx-remote/serialize';
import remarkGfm from 'remark-gfm';
import rehypeSlug from 'rehype-slug';
import rehypePrism from 'rehim-prism-plus';
import readingTime from 'reading-time';

/**
 * 获取所有已发布的文章
 */
export async function getPublishedPosts(options: {
  limit?: number;
  tag?: string;
  category?: string;
} = {}) {
  const { limit = 10, tag, category } = options;
  
  const where: any = {
    isPublished: true,
  };
  
  if (tag) {
    where.tags = { has: tag };
  }
  
  if (category) {
    where.category = category;
  }
  
  const posts = await prisma.post.findMany({
    where,
    select: {
      id: true,
      title: true,
      slug: true,
      excerpt: true,
      coverImage: true,
      tags: true,
      category: true,
      readingTime: true,
      viewCount: true,
      likeCount: true,
      publishedAt: true,
      author: {
        select: { name: true, image: true },
      },
    },
    orderBy: { publishedAt: 'desc' },
    take: limit,
  });
  
  return posts;
}

/**
 * 获取单个文章（包含渲染后的 MDX）
 */
export async function getPostBySlug(slug: string) {
  const post = await prisma.post.findUnique({
    where: { slug },
    include: {
      author: {
        select: { 
          id: true, 
          name: true, 
          image: true,
          bio: true,
          github: true,
        },
      },
    },
  });
  
  if (!post) return null;
  
  // 渲染 MDX
  const mdxSource = await serialize(post.content, {
    mdxOptions: {
      remarkPlugins: [remarkGfm],
      rehypePlugins: [
        rehypeSlug,
        [rehypePrism, { ignoreMissing: true }],
      ],
    },
  });
  
  // 计算阅读时间
  const stats = readingTime(post.content);
  
  return {
    ...post,
    mdxSource,
    readingTime: Math.ceil(stats.minutes),
  };
}

/**
 * 处理发布文章
 */
export async function publishPost(
  id: string, 
  content: string,
  options: {
    title: string;
    slug: string;
    excerpt?: string;
    tags?: string[];
    category?: string;
    coverImage?: string;
  }
) {
  // 计算阅读时间
  const stats = readingTime(content);
  
  // 生成摘要（如果未提供）
  const autoExcerpt = options.excerpt || 
    content
      .replace(/```[\s\S]*?```/g, '')  // 移除代码块
      .replace(/`[^`]+`/g, '')          // 移除行内代码
      .replace(/[#*_~[\]()]/g, '')     // 移除格式符号
      .substring(0, 200);
  
  const post = await prisma.post.update({
    where: { id },
    data: {
      title: options.title,
      slug: options.slug,
      content,
      excerpt: autoExcerpt,
      tags: options.tags || [],
      category: options.category,
      coverImage: options.coverImage,
      readingTime: Math.ceil(stats.minutes),
      isPublished: true,
      publishedAt: new Date(),
    },
  });
  
  return post;
}
```

```typescript
// src/components/MDXComponents.tsx

import Link from 'next/link';
import Image from 'next/image';
import { Highlight, themes } from 'prism-react-renderer';

export const components = {
  // 链接
  a: ({ href, children, ...props }: any) => {
    const isInternal = href?.startsWith('/');
    
    if (isInternal) {
      return <Link href={href}>{children}</Link>;
    }
    
    return (
      <a href={href} target="_blank" rel="noopener noreferrer" {...props}>
        {children}
      </a>
    );
  },
  
  // 图片
  img: ({ src, alt, ...props }: any) => (
    <Image
      src={src}
      alt={alt || ''}
      width={800}
      height={450}
      className="rounded-lg"
      {...props}
    />
  ),
  
  // 代码块
  pre: ({ children, ...props }: any) => (
    <div className="code-block" {...props}>
      {children}
    </div>
  ),
  
  code: ({ className, children, ...props }: any) => {
    const match = /language-(\w+)/.exec(className || '');
    const language = match?.[1] || 'text';
    const code = String(children).replace(/\n$/, '');
    
    // 如果是内联代码
    if (!match) {
      return <code className="inline-code">{code}</code>;
    }
    
    return (
      <Highlight
        theme={themes.nightOwl}
        code={code}
        language={language}
      >
        {({ className, style, tokens, getLineProps, getTokenProps }) => (
          <pre className={className} style={{ ...style, padding: '1rem' }}>
            {tokens.map((line, i) => (
              <div key={i} {...getLineProps({ line })}>
                <span className="line-number">{i + 1}</span>
                {line.map((token, key) => (
                  <span key={key} {...getTokenProps({ token })} />
                ))}
              </div>
            ))}
          </pre>
        )}
      </Highlight>
    );
  },
  
  // 引用
  blockquote: ({ children, ...props }: any) => (
    <blockquote className="callout callout-info" {...props}>
      {children}
    </blockquote>
  ),
  
  // 表格
  table: ({ children, ...props }: any) => (
    <div className="table-wrapper">
      <table {...props}>{children}</table>
    </div>
  ),
};
```
