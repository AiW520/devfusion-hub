# Tauri 桌面应用方案：方案3 - 截图工具

## 项目介绍

### 项目是什么
这是一个基于 Tauri + React + TypeScript 构建的跨平台截图工具，支持全屏截图、区域截图、窗口截图、延时截图、图片标注、图片编辑和快捷键截图功能。

### 解决什么问题
- 替代 Windows Snipping Tool / macOS Grab
- 轻量级截图工具，无需安装厚重软件
- 支持图片标注（箭头、矩形、文字、马赛克）
- 截图后自动复制到剪贴板
- 支持全局快捷键

### 核心技术亮点
| 特性 | 技术实现 |
|------|----------|
| 截图核心 | Rust 的 screenshots 库 |
| 图像处理 | image 库 + base64 编码 |
| 画布编辑 | HTML5 Canvas + Fabric.js |
| 全局快捷键 | tauri-plugin-global-shortcut |
| 剪贴板 | tauri-plugin-clipboard |
| 截图历史 | 本地存储 |
| 前端框架 | React 18 + TypeScript 5 |
| 后端核心 | Rust + Tauri 2.0 |

### 适合什么场景
- 程序员写文档时的截图需求
- 需要快速截图和简单标注的用户
- 对截图工具有轻量化需求的用户
- 需要自定义快捷键的用户

```typescript
      isEditing: false,

        isCapturing: true,
        captureMode: mode,

        isCapturing: false,
      }),

            image: string;
            width: number;
            height: number;

            mode: get().captureMode,
            delay: get().delaySeconds,

            isCapturing: false,

        });
      },

      deleteScreenshot: (id) => {
        const history = get().screenshotHistory;
        set({
          screenshotHistory: history.filter((s) => s.id !== id),
        });
      },
      
      /** 清空历史 */
      clearHistory: () => set({ screenshotHistory: [] }),
      
      // ==================== 编辑器操作 ====================
      
      /** 设置当前工具 */
      setCurrentTool: (tool) => set({ currentTool: tool }),
      
      /** 设置线条颜色 */
      setStrokeColor: (color) => set({ strokeColor: color }),
      
      /** 设置线条粗细 */
      setStrokeWidth: (width) => set({ strokeWidth: width }),
      
      /** 设置字体大小 */
      setFontSize: (size) => set({ fontSize: size }),
      
      /** 设置编辑状态 */
      setEditing: (editing) => set({ isEditing: editing }),
      
      // ==================== UI 操作 ====================
      
      /** 切换历史记录显示 */
      setHistoryOpen: (open) => set({ isHistoryOpen: open }),
      
      /** 加载历史截图 */

// ==================== 辅助函数 ====================

/**
 * 生成缩略图
 * 将 Base64 图片转换为小尺寸缩略图
 */
async function generateThumbnail(base64Image: string): Promise<string> {
  return new Promise((resolve) => {
    const img = new Image();
    img.onload = () => {
      // 创建 canvas
      const canvas = document.createElement('canvas');
      const ctx = canvas.getContext('2d');
      if (!ctx) {
        resolve(base64Image);
        return;
      }
      
      // 缩放比例（最大 200px）
      const maxSize = 200;
      const scale = Math.min(maxSize / img.width, maxSize / img.height, 1);
      canvas.width = img.width * scale;
      canvas.height = img.height * scale;
      
      // 绘制缩略图
      ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
      
      // 输出为 Base64
      resolve(canvas.toDataURL('image/jpeg', 0.7));
    };
    img.src = base64Image;
  });
}

/**
 * 调用 Tauri 命令的辅助函数
 */
async function invoke<T>(cmd: string, args?: Record<string, unknown>): Promise<T> {
  // @ts-ignore - Tauri API
  const { invoke } = window.__TAURI__;
  return invoke(cmd, args);
}
```

```typescript
// useScreenshot.ts - 截图 Hook
// 作用：封装截图相关的逻辑

import { useCallback, useEffect } from 'react';
import { useScreenshotStore, CaptureMode } from '../stores/screenshotStore';

/**
 * 截图 Hook
 * 提供截图功能和相关操作
 */
export function useScreenshot() {
  // 获取状态和方法
  const {
    isCapturing,
    captureMode,
    delaySeconds,
    startCapture,
    cancelCapture,
    captureScreen,
  } = useScreenshotStore();

  /**
   * 执行截图
   * 根据模式执行不同的截图操作
   */
  const takeScreenshot = useCallback(async () => {
    try {
      // 开始截图流程
      startCapture(captureMode);
      
      // 如果设置了延时，等待
      if (delaySeconds > 0) {
        await new Promise((resolve) => setTimeout(resolve, delaySeconds * 1000));
      }
      
      // 执行截图
      await captureScreen();
      
      return true;
    } catch (error) {
      console.error('截图失败:', error);
      cancelCapture();
      return false;
    }
  }, [captureMode, delaySeconds, startCapture, cancelCapture, captureScreen]);

  /**
   * 快捷键监听
   */
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      // ESC 取消截图
      if (e.key === 'Escape' && isCapturing) {
        cancelCapture();
      }
    };

    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [isCapturing, cancelCapture]);

  return {
    isCapturing,
    takeScreenshot,
    cancelCapture,
  };
}

/**
 * 全屏截图 Hook
 */
export function useFullscreenCapture() {
  const { takeScreenshot } = useScreenshot();
  
  const captureFullscreen = useCallback(async () => {
    useScreenshotStore.getState().startCapture('fullscreen');
    await takeScreenshot();
  }, [takeScreenshot]);
  
  return { captureFullscreen };
}

/**
 * 区域截图 Hook
 */
export function useRegionCapture() {
  const { takeScreenshot } = useScreenshot();
  
  const captureRegion = useCallback(async () => {
    useScreenshotStore.getState().startCapture('region');
    await takeScreenshot();
  }, [takeScreenshot]);
  
  return { captureRegion };
}

/**
 * 延时截图 Hook
 */
export function useDelayedCapture(seconds: number) {
  const { takeScreenshot } = useScreenshot();
  
  const captureWithDelay = useCallback(async () => {
    useScreenshotStore.getState().startCapture('delayed');
    useScreenshotStore.getState().captureScreen();
  }, [takeScreenshot]);
  
  return { captureWithDelay };
}
```

### 3. 截图覆盖层组件：src/components/CaptureOverlay.tsx
```tsx
// CaptureOverlay.tsx - 截图覆盖层组件
// 作用：在截图时覆盖整个屏幕，允许用户选择区域

import React, { useState, useCallback, useEffect, useRef } from 'react';
import { invoke } from '@tauri-apps/api/core';

interface SelectionRect {
  x: number;
  y: number;
  width: number;
  height: number;
}

interface CaptureOverlayProps {
  onCapture: (imageData: string, width: number, height: number) => void;
  onCancel: () => void;
  mode: 'region' | 'fullscreen';
}

export const CaptureOverlay: React.FC<CaptureOverlayProps> = ({
  onCapture,
  onCancel,
  mode,
}) => {
  // 截图背景（屏幕截图）
  const [backgroundImage, setBackgroundImage] = useState<string | null>(null);
  
  // 选择区域
  const [selection, setSelection] = useState<SelectionRect | null>(null);
  
  // 鼠标位置
  const [mousePos, setMousePos] = useState({ x: 0, y: 0 });
  
  // 选区起点
  const [startPoint, setStartPoint] = useState<{ x: number; y: number } | null>(null);
  
  // 组件引用
  const containerRef = useRef<HTMLDivElement>(null);
  
  // ==================== 加载屏幕截图 ====================
  
  useEffect(() => {
    // 获取屏幕截图作为背景
    const loadScreenCapture = async () => {
      try {
        const result = await invoke<{ image: string }>('capture_fullscreen');
        setBackgroundImage(`data:image/png;base64,${result.image}`);
      } catch (error) {
        console.error('获取屏幕截图失败:', error);
        // 即使失败也继续，允许手动框选
      }
    };
    
    if (mode === 'region') {
      loadScreenCapture();
    }
  }, [mode]);
  
  // ==================== 鼠标事件处理 ====================
  
  /**
   * 鼠标按下：开始选择
   */
  const handleMouseDown = useCallback((e: React.MouseEvent) => {
    const rect = containerRef.current?.getBoundingClientRect();
    if (!rect) return;
    
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;
    
    setStartPoint({ x, y });
    setSelection({ x, y, width: 0, height: 0 });
  }, []);
  
  /**
   * 鼠标移动：更新选择区域
   */
  const handleMouseMove = useCallback((e: React.MouseEvent) => {
    const rect = containerRef.current?.getBoundingClientRect();
    if (!rect) return;
    
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;
    
    setMousePos({ x, y });
    
    if (startPoint) {
      // 计算选区（确保宽度和高度为正）
      const newSelection: SelectionRect = {
        x: Math.min(startPoint.x, x),
        y: Math.min(startPoint.y, y),
        width: Math.abs(x - startPoint.x),
        height: Math.abs(y - startPoint.y),
      };
      setSelection(newSelection);
    }
  }, [startPoint]);
  
  /**
   * 鼠标释放：完成选择并截图
   */
  const handleMouseUp = useCallback(async () => {
    if (!selection || selection.width < 10 || selection.height < 10) {
      // 选择区域太小，不截图
      return;
    }
    
    try {
      // 调用 Rust 裁剪选区
      const result = await invoke<{
        image: string;
        width: number;
        height: number;
      }>('crop_region', {
        x: Math.round(selection.x),
        y: Math.round(selection.y),
        width: Math.round(selection.width),
        height: Math.round(selection.height),
      });
      
      onCapture(result.image, result.width, result.height);
    } catch (error) {
      console.error('裁剪失败:', error);
      onCancel();
    }
  }, [selection, onCapture, onCancel]);
  
  /**
   * 键盘事件：ESC 取消
   */
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key === 'Escape') {
        onCancel();
      } else if (e.key === 'Enter' && selection && selection.width > 10) {
        // Enter 确认当前选区
        handleMouseUp();
      }
    };
    
    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [selection, onCancel, handleMouseUp]);
  
  // ==================== 坐标信息 ====================
  
  // 格式化坐标显示
  const formatCoord = (num: number) => `${Math.round(num)}px`;
  
  // ==================== 渲染 ====================
  
  return (
    <div
      ref={containerRef}
      className="fixed inset-0 z-50 cursor-crosshair"
      style={{
        // 背景样式
        backgroundColor: 'rgba(0, 0, 0, 0.3)',
        backgroundImage: backgroundImage ? `url(${backgroundImage})` : undefined,
        backgroundSize: 'cover',
        backgroundPosition: 'center',
      }}
      onMouseDown={handleMouseDown}
      onMouseMove={handleMouseMove}
      onMouseUp={handleMouseUp}
    >
      {/* 半透明遮罩 */}
      <div className="absolute inset-0 bg-black/30" />
      
      {/* 选区预览 */}
      {selection && (
        <>
          {/* 裁剪预览（选区正常显示） */}
          <div
            className="absolute border-2 border-blue-500 bg-transparent"
            style={{
              left: selection.x,
              top: selection.y,
              width: selection.width,
              height: selection.height,
              boxShadow: '0 0 0 9999px rgba(0, 0, 0, 0.5)',
            }}
          />
          
          {/* 选区内容预览 */}
          <div
            className="absolute border border-white/50"
            style={{
              left: selection.x,
              top: selection.y,
              width: selection.width,
              height: selection.height,
              overflow: 'hidden',
              backgroundImage: backgroundImage ? `url(${backgroundImage})` : undefined,
              backgroundPosition: `-${selection.x}px -${selection.y}px`,
              backgroundSize: `${window.innerWidth}px ${window.innerHeight}px`,
            }}
          />
        </>
      )}
      
      {/* 坐标信息 */}
      {selection && (
        <div
          className="absolute bg-gray-900/90 text-white text-xs px-2 py-1 rounded pointer-events-none"
          style={{
            left: selection.x + selection.width + 10,
            top: selection.y,
          }}
        >
          <div>X: {formatCoord(selection.x)}</div>
          <div>Y: {formatCoord(selection.y)}</div>
          <div>W: {formatCoord(selection.width)}</div>
          <div>H: {formatCoord(selection.height)}</div>
        </div>
      )}
      
      {/* 十字准星 */}
      <div
        className="absolute pointer-events-none"
        style={{
          left: mousePos.x,
          top: mousePos.y,
          transform: 'translate(-50%, -50%)',
        }}
      >
        <div className="w-6 h-0.5 bg-white/50" />
        <div className="w-0.5 h-6 bg-white/50 absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2" />
      </div>
      
      {/* 提示文字 */}
      <div className="absolute top-4 left-1/2 -translate-x-1/2 bg-gray-900/90 text-white px-4 py-2 rounded-lg text-sm">
        {selection 
          ? '释放鼠标确认选区，按 ESC 取消'
          : '拖动鼠标选择区域，按 ESC 取消'}
      </div>
      
      {/* 取消按钮 */}
      <button
        onClick={onCancel}
        className="absolute bottom-4 right-4 bg-gray-800 hover:bg-gray-700 text-white px-4 py-2 rounded-lg text-sm transition-colors"
      >
        取消 (ESC)
      </button>
    </div>
  );
};
```

```tsx
// ImageEditor.tsx - 图片编辑器组件
// 作用：提供图片标注和编辑功能

import React, { useRef, useEffect, useState, useCallback } from 'react';
import { useScreenshotStore, Tool, Screenshot, COLOR_PRESETS, STROKE_WIDTHS, FONT_SIZES } from '../stores/screenshotStore';
import { Toolbar } from './Toolbar';

interface DrawState {
  isDrawing: boolean;
  startX: number;
  startY: number;
  currentX: number;
  currentY: number;
}

export const ImageEditor: React.FC = () => {
  // 画布引用
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const contextRef = useRef<CanvasRenderingContext2D | null>(null);
  
  // 图片引用
  const imageRef = useRef<HTMLImageElement | null>(null);
  
  // 绘制状态
  const [drawState, setDrawState] = useState<DrawState>({
    isDrawing: false,
    startX: 0,
    startY: 0,
    currentX: 0,
    currentY: 0,
  });
  
  // 撤销历史
  const [history, setHistory] = useState<ImageData[]>([]);
  const [historyIndex, setHistoryIndex] = useState(-1);
  
  // 获取状态
  const {
    currentScreenshot,
    currentTool,
    strokeColor,
    strokeWidth,
    fontSize,
    saveScreenshot,
    setEditing,
  } = useScreenshotStore();
  
  // ==================== 画布初始化 ====================
  
  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas || !currentScreenshot) return;
    
    // 设置画布尺寸
    canvas.width = currentScreenshot.width;
    canvas.height = currentScreenshot.height;
    
    // 获取 2D 上下文
    const ctx = canvas.getContext('2d');
    if (!ctx) return;
    contextRef.current = ctx;
    
    // 加载图片
    const img = new Image();
    img.onload = () => {
      imageRef.current = img;
      ctx.drawImage(img, 0, 0);
      
      // 保存初始状态
      saveToHistory();
    };
    img.src = currentScreenshot.imageData;
  }, [currentScreenshot]);
  
  // ==================== 历史记录 ====================
  
  /**
   * 保存到历史
   */
  const saveToHistory = useCallback(() => {
    const canvas = canvasRef.current;
    const ctx = contextRef.current;
    if (!canvas || !ctx) return;
    
    const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
    
    // 如果当前不是最新位置，截断后面的历史
    const newHistory = history.slice(0, historyIndex + 1);
    newHistory.push(imageData);
    
    // 限制历史长度
    if (newHistory.length > 50) {
      newHistory.shift();
    }
    
    setHistory(newHistory);
    setHistoryIndex(newHistory.length - 1);
  }, [history, historyIndex]);
  
  /**
   * 撤销
   */
  const undo = useCallback(() => {
    if (historyIndex <= 0) return;
    
    const ctx = contextRef.current;
    if (!ctx) return;
    
    const newIndex = historyIndex - 1;
    ctx.putImageData(history[newIndex], 0, 0);
    setHistoryIndex(newIndex);
  }, [history, historyIndex]);
  
  /**
   * 重做
   */
  const redo = useCallback(() => {
    if (historyIndex >= history.length - 1) return;
    
    const ctx = contextRef.current;
    if (!ctx) return;
    
    const newIndex = historyIndex + 1;
    ctx.putImageData(history[newIndex], 0, 0);
    setHistoryIndex(newIndex);
  }, [history, historyIndex]);
  
  // 快捷键：Ctrl+Z 撤销，Ctrl+Y 重做
  useEffect(() => {
    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.ctrlKey || e.metaKey) {
        if (e.key === 'z') {
          e.preventDefault();
          undo();
        } else if (e.key === 'y') {
          e.preventDefault();
          redo();
        }
      }
    };
    
    window.addEventListener('keydown', handleKeyDown);
    return () => window.removeEventListener('keydown', handleKeyDown);
  }, [undo, redo]);
  
  // ==================== 绘图逻辑 ====================
  
  /**
   * 获取鼠标在画布上的坐标
   */
  const getCanvasCoords = useCallback((e: React.MouseEvent) => {
    const canvas = canvasRef.current;
    if (!canvas) return { x: 0, y: 0 };
    
    const rect = canvas.getBoundingClientRect();
    const scaleX = canvas.width / rect.width;
    const scaleY = canvas.height / rect.height;
    
    return {
      x: (e.clientX - rect.left) * scaleX,
      y: (e.clientY - rect.top) * scaleY,
    };
  }, []);
  
  /**
   * 绘制形状（基于当前绘制状态）
   */
  const renderShape = useCallback(() => {
    const ctx = contextRef.current;
    if (!ctx) return;
    
    const { startX, startY, currentX, currentY } = drawState;
    const canvas = canvasRef.current;
    if (!canvas) return;
    
    // 恢复之前的状态
    if (historyIndex >= 0) {
      ctx.putImageData(history[historyIndex], 0, 0);
    }
    
    // 设置样式
    ctx.strokeStyle = strokeColor;
    ctx.fillStyle = strokeColor;
    ctx.lineWidth = strokeWidth;
    ctx.lineCap = 'round';
    ctx.lineJoin = 'round';
    
    switch (currentTool) {
      case 'arrow':
        // 绘制箭头
        drawArrow(ctx, startX, startY, currentX, currentY);
        break;
        
      case 'rectangle':
        // 绘制矩形
        ctx.strokeRect(startX, startY, currentX - startX, currentY - startY);
        break;
        
      case 'ellipse':
        // 绘制椭圆
        drawEllipse(ctx, startX, startY, currentX, currentY);
        break;
        
      case 'highlight':
        // 绘制高亮（半透明矩形）
        ctx.fillStyle = `${strokeColor}40`;  // 40 = 25% 透明度
        ctx.fillRect(startX, startY, currentX - startX, currentY - startY);
        break;
    }
  }, [drawState, currentTool, strokeColor, strokeWidth, history, historyIndex]);
  
  /**
   * 绘制箭头
   */
  const drawArrow = (
    ctx: CanvasRenderingContext2D,
    fromX: number,
    fromY: number,
    toX: number,
    toY: number
  ) => {
    const headLength = strokeWidth * 4;  // 箭头长度
    const angle = Math.atan2(toY - fromY, toX - fromX);
    
    // 绘制线条
    ctx.beginPath();
    ctx.moveTo(fromX, fromY);
    ctx.lineTo(toX, toY);
    ctx.stroke();
    
    // 绘制箭头头部
    ctx.beginPath();
    ctx.moveTo(toX, toY);
    ctx.lineTo(
      toX - headLength * Math.cos(angle - Math.PI / 6),
      toY - headLength * Math.sin(angle - Math.PI / 6)
    );
    ctx.moveTo(toX, toY);
    ctx.lineTo(
      toX - headLength * Math.cos(angle + Math.PI / 6),
      toY - headLength * Math.sin(angle + Math.PI / 6)
    );
    ctx.stroke();
  };
  
  /**
   * 绘制椭圆
   */
  const drawEllipse = (
    ctx: CanvasRenderingContext2D,
    x1: number,
    y1: number,
    x2: number,
    y2: number
  ) => {
    const centerX = (x1 + x2) / 2;
    const centerY = (y1 + y2) / 2;
    const radiusX = Math.abs(x2 - x1) / 2;
    const radiusY = Math.abs(y2 - y1) / 2;
    
    ctx.beginPath();
    ctx.ellipse(centerX, centerY, radiusX, radiusY, 0, 0, 2 * Math.PI);
    ctx.stroke();
  };
  
  /**
   * 添加马赛克效果
   */
  const addBlur = useCallback((x: number, y: number, width: number, height: number) => {
    const ctx = contextRef.current;
    const canvas = canvasRef.current;
    if (!ctx || !canvas) return;
    
    // 缩小区域
    const smallCanvas = document.createElement('canvas');
    const smallCtx = smallCanvas.getContext('2d');
    if (!smallCtx) return;
    
    smallCanvas.width = Math.max(1, Math.floor(width / 10));
    smallCanvas.height = Math.max(1, Math.floor(height / 10));
    
    // 绘制缩小后的区域
    smallCtx.drawImage(canvas, x, y, width, height, 0, 0, smallCanvas.width, smallCanvas.height);
    
    // 放大回原区域
    ctx.imageSmoothingEnabled = false;
    ctx.drawImage(smallCanvas, 0, 0, smallCanvas.width, smallCanvas.height, x, y, width, height);
    ctx.imageSmoothingEnabled = true;
  }, []);
  
  /**
   * 添加文字
   */
  const addText = useCallback((x: number, y: number, text: string) => {
    const ctx = contextRef.current;
    if (!ctx) return;
    
    ctx.font = `${fontSize}px Arial, sans-serif`;
    ctx.fillStyle = strokeColor;
    ctx.fillText(text, x, y);
  }, [fontSize, strokeColor]);
  
  // ==================== 鼠标事件处理 ====================
  
  const handleMouseDown = useCallback((e: React.MouseEvent) => {
    const coords = getCanvasCoords(e);
    
    if (currentTool === 'text') {
      // 文字工具：弹出输入框
      const text = prompt('输入文字:');
      if (text) {
        addText(coords.x, coords.y, text);
        saveToHistory();
      }
      return;
    }
    
    if (currentTool === 'blur') {
      // 马赛克工具：先保存历史
      saveToHistory();
      return;
    }
    
    // 其他工具：开始绘制
    setDrawState({
      isDrawing: true,
      startX: coords.x,
      startY: coords.y,
      currentX: coords.x,
      currentY: coords.y,
    });
  }, [currentTool, getCanvasCoords, addText, saveToHistory]);
  
  const handleMouseMove = useCallback((e: React.MouseEvent) => {
    if (!drawState.isDrawing) return;
    
    const coords = getCanvasCoords(e);
    
    if (currentTool === 'blur') {
      // 马赛克：实时应用
      const width = strokeWidth * 4;
      const height = strokeWidth * 4;
      addBlur(
        coords.x - width / 2,
        coords.y - height / 2,
        width,
        height
      );
      return;
    }
    
    // 更新绘制状态并重绘
    setDrawState((prev) => ({
      ...prev,
      currentX: coords.x,
      currentY: coords.y,
    }));
    
    renderShape();
  }, [drawState.isDrawing, currentTool, getCanvasCoords, addBlur, renderShape]);
  
  const handleMouseUp = useCallback(() => {
    if (!drawState.isDrawing) return;
    
    if (currentTool !== 'blur') {
      // 保存历史（除了马赛克工具）
      saveToHistory();
    }
    
    setDrawState((prev) => ({
      ...prev,
      isDrawing: false,
    }));
  }, [drawState.isDrawing, currentTool, saveToHistory]);
  
  // ==================== 保存和关闭 ====================
  
  /**
   * 保存图片
   */
  const handleSave = useCallback(() => {
    const canvas = canvasRef.current;
    if (!canvas || !currentScreenshot) return;
    
    // 导出图片数据
    const imageData = canvas.toDataURL('image/png');
    
    // 更新截图对象
    const updatedScreenshot: Screenshot = {
      ...currentScreenshot,
      imageData,
      width: canvas.width,
      height: canvas.height,
    };
    
    // 保存到历史记录
    saveScreenshot(updatedScreenshot);
    
    // 关闭编辑器
    setEditing(false);
  }, [currentScreenshot, saveScreenshot, setEditing]);
  
  /**
   * 复制到剪贴板
   */
  const handleCopy = useCallback(async () => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    
    try {
      const blob = await new Promise<Blob | null>((resolve) => 
        canvas.toBlob(resolve, 'image/png')
      );
      
      if (blob) {
        await navigator.clipboard.write([
          new ClipboardItem({ 'image/png': blob })
        ]);
        alert('已复制到剪贴板');
      }
    } catch (error) {
      console.error('复制失败:', error);
      alert('复制失败，请重试');
    }
  }, []);
  
  /**
   * 关闭编辑器
   */
  const handleClose = useCallback(() => {
    if (historyIndex > 0) {
      if (!confirm('有未保存的修改，确定要关闭吗？')) {
        return;
      }
    }
    setEditing(false);
  }, [historyIndex, setEditing]);
  
  // ==================== 渲染 ====================
  
  if (!currentScreenshot) {
    return null;
  }
  
  return (
    <div className="fixed inset-0 z-40 bg-gray-900 flex flex-col">
      {/* 工具栏 */}
      <div className="bg-gray-800 border-b border-gray-700 px-4 py-2 flex items-center justify-between">
        <div className="flex items-center gap-4">
          <h2 className="text-white font-medium">图片编辑器</h2>
          
          {/* 撤销/重做 */}
          <div className="flex items-center gap-1">
            <button
              onClick={undo}
              disabled={historyIndex <= 0}
              className="p-2 text-gray-400 hover:text-white disabled:opacity-50 transition-colors"
              title="撤销 (Ctrl+Z)"
            >
              <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M3 10h10a8 8 0 018 8v2M3 10l6 6m-6-6l6-6" />
              </svg>
            </button>
            <button
              onClick={redo}
              disabled={historyIndex >= history.length - 1}
              className="p-2 text-gray-400 hover:text-white disabled:opacity-50 transition-colors"
              title="重做 (Ctrl+Y)"
            >
              <svg className="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M21 10h-10a8 8 0 00-8 8v2m18-10l-6 6m6-6l-6-6" />
              </svg>
            </button>
          </div>
        </div>
        
        <div className="flex items-center gap-2">
          <button
            onClick={handleCopy}
            className="px-4 py-2 bg-gray-700 hover:bg-gray-600 text-white rounded text-sm transition-colors"
          >
            复制到剪贴板
          </button>
          <button
            onClick={handleSave}
            className="px-4 py-2 bg-indigo-600 hover:bg-indigo-700 text-white rounded text-sm transition-colors"

            保存

          <button
            onClick={handleClose}
            className="px-4 py-2 text-gray-400 hover:text-white transition-colors"
          >
            关闭
          </button>
        </div>
      </div>
      
      {/* 主内容区 */}
      <div className="flex-1 flex overflow-hidden">
        {/* 工具栏 */}
        <Toolbar />
        
        {/* 画布区 */}
        <div className="flex-1 flex items-center justify-center overflow-auto p-8 bg-gray-900">
          <canvas
            ref={canvasRef}
            className="max-w-full max-h-full shadow-2xl cursor-crosshair"

            onMouseMove={handleMouseMove}

  );
```

```typescript
// time.ts - 时间格式化工具

/**
 * 格式化时间为 HH:mm:ss 或 HH:mm 格式
 */
export function formatTime(timestamp: number): string {
  const date = new Date(timestamp);
  const hours = date.getHours();
  const minutes = date.getMinutes();
  const seconds = date.getSeconds();
  
  if (hours > 0) {
    return `${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
  }
  
  return `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
}

/**
 * 格式化时长（秒）为 mm:ss 格式
 */
export function formatDuration(seconds: number): string {
  const mins = Math.floor(seconds / 60);
  const secs = Math.floor(seconds % 60);
  return `${mins.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;
}
```

```rust
            width: image.width(),
            height: image.height(),

    #[cfg(not(feature = "screenshots"))]
```

### 面试题 2：什么是 Rust 的错误处理哲学？

**标准答案**：
1. **编译时强制处理**：错误必须显式处理，不能忽略
2. **可恢复 vs 不可恢复**：
   - 可恢复：`Result<T, E>`
   - 不可恢复：`panic!`
3. **组合子模式**：使用 `map`、`and_then`、`or_else` 等
4. **`?` 操作符**：自动传播错误

```rust
// 传统错误处理
match read_file() {
    Ok(data) => process(data),
    Err(e) => return Err(e),
}

// 使用 ? 操作符（更简洁）
fn process() -> Result<Data, Error> {
    let data = read_file()?;
    Ok(data)
}
```

### 面试题 5：Rust 的特征（Trait）系统与泛型的关系？

**标准答案**：
1. **Trait 定义行为**：
```rust
trait Drawable {
    fn draw(&self);
}
```

2. **泛型约束**：
```rust
// 静态分发（编译时确定）
fn draw_all<T: Drawable>(items: Vec<T>) {
    for item in items {
        item.draw();
    }
}

// 动态分发（运行时确定）
fn draw_all(items: Vec<&dyn Drawable>) {
    for item in items {
        item.draw();
    }
}
```

3. **选择**：
   - 性能优先：泛型 + Trait Bound
   - 灵活性优先：`dyn Trait`
