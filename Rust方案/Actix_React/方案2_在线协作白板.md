# Actix + React 方案：方案2 - 在线协作白板

## 项目介绍

### 项目是什么
这是一个基于 Rust + Actix Web + React + WebSocket 构建的实时协作白板，支持多人同时编辑、图形绘制、文本标注、实时同步和房间管理。

### 解决什么问题
- 远程团队协作
- 头脑风暴和流程图绘制
- 在线教学和演示
- 设计评审和反馈

### 核心技术亮点
| 特性 | 技术实现 |
|------|----------|
| 后端框架 | Actix-web 4 |
| 实时通信 | WebSocket |
| 数据库 | PostgreSQL + Redis |
| 前端框架 | React 18 + TypeScript |
| 画布 | Fabric.js / Konva |
| 状态同步 | CRDT 算法 |
| 认证 | JWT |

### 适合什么场景
- 在线教育平台
- 远程协作工具
- 设计评审系统
- 会议白板

### 3. WebSocket 会话：backend/src/ws/session.rs
```rust
// ws/session.rs - WebSocket 会话管理

use actix::{Actor, ActorContext, Addr, AsyncContext, Handler, Message, StreamHandler};
use actix_web_actors::ws;
use std::collections::{HashMap, HashSet};
use std::sync::Arc;
use tokio::sync::{broadcast, RwLock};
use parking_lot::RwLock as PLRwLock;
use uuid::Uuid;
use chrono::Utc;

use super::messages::*;

// ==================== 会话管理器 ====================

/// 全局会话管理器
pub struct SessionManager {
    /// 房间 -> 用户会话映射
    rooms: Arc<RwLock<HashMap<String, HashSet<String>>>>,
    
    /// 用户 ID -> 会话地址
    sessions: Arc<RwLock<HashMap<String, Addr<WhiteboardSession>>>>,
    
    /// 用户信息缓存
    users: Arc<RwLock<HashMap<String, UserPresence>>>>,
    
    /// 广播频道
    broadcast_tx: broadcast::Sender<BroadcastMessage>,
}

impl SessionManager {
    pub fn new() -> Self {
        let (broadcast_tx, _) = broadcast::channel(1000);
        
        Self {
            rooms: Arc::new(RwLock::new(HashMap::new())),
            sessions: Arc::new(RwLock::new(HashMap::new())),
            users: Arc::new(RwLock::new(HashMap::new())),
            broadcast_tx,
        }
    }
    
    /// 加入房间
    pub async fn join_room(
        &self,
        room_id: &str,
        user_id: &str,
        username: &str,
        addr: Addr<WhiteboardSession>,
    ) {
        // 记录会话
        self.sessions.write().await.insert(user_id.to_string(), addr);
        
        // 生成用户颜色
        let colors = ["#FF6B6B", "#4ECDC4", "#45B7D1", "#96CEB4", "#FFEAA7", "#DDA0DD"];
        let color_idx = user_id.as_bytes().iter().fold(0usize, |acc, b| acc.wrapping_add(*b as usize)) % colors.len();
        
        // 创建用户状态
        let user = UserPresence {
            user_id: user_id.to_string(),
            username: username.to_string(),
            color: colors[color_idx].to_string(),
            cursor_x: None,
            cursor_y: None,
        };
        
        // 记录用户
        self.users.write().await.insert(user_id.to_string(), user.clone());
        
        // 加入房间
        let mut rooms = self.rooms.write().await;
        rooms
            .entry(room_id.to_string())
            .or_insert_with(HashSet::new)
            .insert(user_id.to_string());
    }
    
    /// 离开房间
    pub async fn leave_room(&self, room_id: &str, user_id: &str) {
        // 从房间移除
        if let Some(room) = self.rooms.write().await.get_mut(room_id) {
            room.remove(user_id);
        }
        
        // 移除用户
        self.users.write().await.remove(user_id);
    }
    
    /// 广播消息到房间
    pub async fn broadcast_to_room(&self, room_id: &str, message: &WsMessage, exclude_user: Option<&str>) {
        let rooms = self.rooms.read().await;
        
        if let Some(users) = rooms.get(room_id) {
            for user_id in users {
                if let Some(exclude) = exclude_user {
                    if user_id == exclude {
                        continue;
                    }
                }
                
                if let Some(addr) = self.sessions.read().await.get(user_id) {
                    let _ = addr.try_send(WsMessage::clone(&message));
                }
            }
        }
    }
    
    /// 获取房间用户列表
    pub async fn get_room_users(&self, room_id: &str) -> Vec<UserPresence> {
        let rooms = self.rooms.read().await;
        let users = self.users.read().await;
        
        rooms
            .get(room_id)
            .map(|user_ids| {
                user_ids
                    .iter()
                    .filter_map(|id| users.get(id).cloned())
                    .collect()
            })
            .unwrap_or_default()
    }
    
    /// 获取房间人数
    pub async fn get_room_count(&self, room_id: &str) -> usize {
        self.rooms
            .read()
            .await
            .get(room_id)
            .map(|s| s.len())
            .unwrap_or(0)
    }
}

impl Default for SessionManager {
    fn default() -> Self {
        Self::new()
    }
}

/// 广播消息
#[derive(Debug, Clone)]
pub struct BroadcastMessage {
    pub room_id: String,
    pub message: WsMessage,
    pub exclude_user: Option<String>,
}

/// 全局会话管理器实例
pub static SESSION_MANAGER: once_cell::sync::Lazy<Arc<SessionManager>> = 
    once_cell::sync::Lazy::new(|| Arc::new(SessionManager::new()));

// ==================== Whiteboard Session Actor ====================

/// 白板会话 Actor
pub struct WhiteboardSession {
    /// 会话 ID
    pub id: String,
    
    /// 用户 ID
    pub user_id: Option<String>,
    
    /// 当前房间
    pub room_id: Option<String>,
    
    /// 心跳时间
    pub last_heartbeat: i64,
}

impl WhiteboardSession {
    pub fn new() -> Self {
        Self {
            id: Uuid::new_v4().to_string(),
            user_id: None,
            room_id: None,
            last_heartbeat: Utc::now().timestamp(),
        }
    }
}

impl Actor for WhiteboardSession {
    type Context = ws::WebsocketContext<Self>;
    
    fn started(&mut self, ctx: &mut Self::Context) {
        tracing::info!("WebSocket 连接建立: {}", self.id);
        
        // 启动心跳
        self.start_heartbeat(ctx);
    }
    
    fn stopped(&mut self, _ctx: &mut Self::Context) {
        tracing::info!("WebSocket 连接断开: {}", self.id);
    }
}

impl WhiteboardSession {
    /// 启动心跳
    fn start_heartbeat(&self, ctx: &mut <Self as Actor>::Context) {
        ctx.run_interval(
            tokio::time::Duration::from_secs(30),
            |act, ctx| {
                if Utc::now().timestamp() - act.last_heartbeat > 60 {
                    tracing::warn!("心跳超时，断开连接");
                    ctx.stop();
                    return;
                }
                
                ctx.ping(b"ping");
            }
        );
    }
}

/// 处理 WebSocket 消息
impl StreamHandler<Result<ws::Message, ws::ProtocolError>> for WhiteboardSession {
    fn handle(&mut self, msg: Result<ws::Message, ws::ProtocolError>, ctx: &mut Self::Context) {
        match msg {
            Ok(ws::Message::Ping(msg)) => {
                self.last_heartbeat = Utc::now().timestamp();
                ctx.pong(&msg);
            }
            Ok(ws::Message::Pong(_)) => {
                self.last_heartbeat = Utc::now().timestamp();
            }
            Ok(ws::Message::Text(text)) => {
                if let Ok(message) = serde_json::from_str::<WsMessage>(&text) {
                    self.handle_message(message, ctx);
                }
            }
            Ok(ws::Message::Close(reason)) => {
                ctx.close(reason);
                ctx.stop();
            }
            _ => {}
        }
    }
}

impl WhiteboardSession {
    /// 处理消息
    fn handle_message(&mut self, msg: WsMessage, ctx: &mut <Self as Actor>::Context) {
        let manager = SESSION_MANAGER.clone();
        
        match msg {
            WsMessage::Join { room_id, user_id, username } => {
                // 记录当前用户
                self.user_id = Some(user_id.clone());
                self.room_id = Some(room_id.clone());
                
                // 加入房间
                let addr = ctx.address();
                let user_id_clone = user_id.clone();
                let room_id_clone = room_id.clone();
                
                // 异步加入
                tokio::spawn(async move {
                    manager.join_room(&room_id_clone, &user_id_clone, &username, addr).await;
                });
                
                // 发送房间状态
                let users = manager.get_room_users(&room_id).await;
                let response = WsMessage::RoomState { 
                    shapes: vec![], 
                    users,
                };
                ctx.text(serde_json::to_string(&response).unwrap());
                
                // 广播用户加入
                let join_msg = WsMessage::UserJoined {
                    user: UserPresence {
                        user_id,
                        username,
                        color: "#4ECDC4".to_string(),
                        cursor_x: None,
                        cursor_y: None,
                    }
                };
                let _ = manager.broadcast_tx.send(BroadcastMessage {
                    room_id: room_id.clone(),
                    message: join_msg,
                    exclude_user: None,
                });
            }
            
            WsMessage::Leave { room_id } => {
                if let Some(user_id) = &self.user_id {
                    tokio::spawn(async move {
                        manager.leave_room(&room_id, user_id).await;
                    });
                }
                self.room_id = None;
            }
            
            WsMessage::AddShape { shape } => {
                if let Some(room_id) = &self.room_id {
                    let broadcast = BroadcastMessage {
                        room_id: room_id.clone(),
                        message: WsMessage::ShapeAdded { shape },
                        exclude_user: self.user_id.clone(),
                    };
                    let _ = manager.broadcast_tx.send(broadcast);
                }
            }
            
            WsMessage::UpdateShape { shape_id, updates } => {
                if let Some(room_id) = &self.room_id {
                    let broadcast = BroadcastMessage {
                        room_id: room_id.clone(),
                        message: WsMessage::ShapeUpdated { shape_id, updates },
                        exclude_user: self.user_id.clone(),
                    };
                    let _ = manager.broadcast_tx.send(broadcast);
                }
            }
            
            WsMessage::DeleteShape { shape_id } => {
                if let Some(room_id) = &self.room_id {
                    let broadcast = BroadcastMessage {
                        room_id: room_id.clone(),
                        message: WsMessage::ShapeDeleted { shape_id },
                        exclude_user: self.user_id.clone(),
                    };
                    let _ = manager.broadcast_tx.send(broadcast);
                }
            }
            
            WsMessage::CursorMove { user_id, x, y } => {
                if let Some(room_id) = &self.room_id {
                    let broadcast = BroadcastMessage {
                        room_id: room_id.clone(),
                        message: WsMessage::CursorUpdated { user_id, x, y },
                        exclude_user: self.user_id.clone(),
                    };
                    let _ = manager.broadcast_tx.send(broadcast);
                }
            }
            
            _ => {}
        }
    }
}
```

```rust
// handlers/rooms.rs - 房间管理处理器

use actix_web::{web, HttpResponse, Result};
use serde::{Deserialize, Serialize};
use sqlx::PgPool;
use uuid::Uuid;

use crate::errors::AppError;

// ==================== 请求/响应 ====================

/// 房间信息
#[derive(Debug, Serialize)]
pub struct RoomInfo {
    pub id: String,
    pub name: String,
    pub owner_id: String,
    pub created_at: String,
    pub user_count: i64,
}

/// 创建房间请求
#[derive(Debug, Deserialize)]
pub struct CreateRoomRequest {
    pub name: String,
}

/// 房间列表响应
#[derive(Debug, Serialize)]
pub struct RoomListResponse {
    pub rooms: Vec<RoomInfo>,
}

// ==================== 处理器 ====================

/// 创建房间
pub async fn create_room(
    pool: web::Data<PgPool>,
    body: web::Json<CreateRoomRequest>,
) -> Result<HttpResponse, AppError> {
    let room_id = Uuid::new_v4().to_string();
    
    sqlx::query(
        r#"
        INSERT INTO rooms (id, name, owner_id)
        VALUES ($1, $2, $3)
        "#,
    )
    .bind(&room_id)
    .bind(&body.name)
    .bind("system") // 简化：使用 system 作为所有者
    .execute(pool.get_ref())
    .await?;
    
    Ok(HttpResponse::Created().json(serde_json::json!({
        "id": room_id,
        "name": body.name,
    })))
}

/// 获取房间列表
pub async fn list_rooms(
    pool: web::Data<PgPool>,
) -> Result<HttpResponse, AppError> {
    let rooms: Vec<(String, String, String)> = sqlx::query_as(
        r#"
        SELECT r.id, r.name, r.created_at::text
        FROM rooms r
        ORDER BY r.created_at DESC
        LIMIT 50
        "#
    )
    .fetch_all(pool.get_ref())
    .await?;
    
    let room_infos: Vec<RoomInfo> = rooms
        .into_iter()
        .map(|(id, name, created_at)| RoomInfo {
            id,
            name,
            owner_id: "system".to_string(),
            created_at,
            user_count: 0,
        })
        .collect();
    
    Ok(HttpResponse::Ok().json(RoomListResponse {
        rooms: room_infos,
    }))
}

/// 获取房间详情
pub async fn get_room(
    pool: web::Data<PgPool>,
    path: web::Path<String>,
) -> Result<HttpResponse, AppError> {
    let room_id = path.into_inner();
    
    let room: Option<(String, String, String)> = sqlx::query_as(
        "SELECT id, name, created_at::text FROM rooms WHERE id = $1"
    )
    .bind(&room_id)
    .fetch_optional(pool.get_ref())
    .await?;
    
    match room {
        Some((id, name, created_at)) => {
            Ok(HttpResponse::Ok().json(RoomInfo {
                id,
                name,
                owner_id: "system".to_string(),
                created_at,
                user_count: 0,
            }))
        }
        None => Err(AppError::NotFound("房间不存在".to_string())),
    }
}
```

```tsx
// Whiteboard.tsx - 在线协作白板组件

import React, { useEffect, useRef, useState, useCallback } from 'react';
import { v4 as uuidv4 } from 'uuid';
import { WsMessage, Shape, ShapeType } from '../types';

// 颜色预设
const COLORS = [
  '#000000', '#ffffff', '#ff0000', '#00ff00',
  '#0000ff', '#ffff00', '#ff00ff', '#00ffff',
];

// 工具类型
type Tool = 'select' | 'rectangle' | 'circle' | 'line' | 'arrow' | 'text' | 'freehand';

interface WhiteboardProps {
  roomId: string;
  userId: string;
  username: string;
  wsUrl: string;
}

export const Whiteboard: React.FC<WhiteboardProps> = ({
  roomId,
  userId,
  username,
  wsUrl,
}) => {
  // Canvas refs
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const containerRef = useRef<HTMLDivElement>(null);
  
  // 状态
  const [tool, setTool] = useState<Tool>('select');
  const [color, setColor] = useState('#000000');
  const [strokeWidth, setStrokeWidth] = useState(2);
  const [shapes, setShapes] = useState<Shape[]>([]);
  const [users, setUsers] = useState<UserPresence[]>([]);
  const [selectedShape, setSelectedShape] = useState<string | null>(null);
  
  // WebSocket
  const wsRef = useRef<WebSocket | null>(null);
  const [connected, setConnected] = useState(false);
  
  // 绘图状态
  const [isDrawing, setIsDrawing] = useState(false);
  const [startPos, setStartPos] = useState({ x: 0, y: 0 });
  const [currentPos, setCurrentPos] = useState({ x: 0, y: 0 });
  const [freehandPoints, setFreehandPoints] = useState<{x: number, y: number}[]>([]);
  
  // 初始化 WebSocket
  useEffect(() => {
    const ws = new WebSocket(wsUrl);
    
    ws.onopen = () => {
      setConnected(true);
      // 加入房间
      ws.send(JSON.stringify({
        type: 'Join',
        data: {
          room_id: roomId,
          user_id: userId,
          username: username,
        },
      }));
    };
    
    ws.onmessage = (event) => {
      const msg: WsMessage = JSON.parse(event.data);
      handleMessage(msg);
    };
    
    ws.onclose = () => {
      setConnected(false);
    };
    
    wsRef.current = ws;
    
    return () => {
      ws.close();
    };
  }, [roomId, userId, username, wsUrl]);
  
  // 处理 WebSocket 消息
  const handleMessage = useCallback((msg: WsMessage) => {
    switch (msg.type) {
      case 'RoomState':
        if ('shapes' in msg.data && 'users' in msg.data) {
          setShapes(msg.data.shapes);
          setUsers(msg.data.users);
        }
        break;
      case 'ShapeAdded':
        if ('shape' in msg.data) {
          setShapes(prev => [...prev, msg.data.shape]);
        }
        break;
      case 'ShapeUpdated':
        if ('shape_id' in msg.data && 'updates' in msg.data) {
          setShapes(prev => prev.map(s => 
            s.id === msg.data.shape_id ? { ...s, ...msg.data.updates } : s
          ));
        }
        break;
      case 'ShapeDeleted':
        if ('shape_id' in msg.data) {
          setShapes(prev => prev.filter(s => s.id !== msg.data.shape_id));
        }
        break;
      case 'UserJoined':
        if ('user' in msg.data) {
          setUsers(prev => [...prev.filter(u => u.user_id !== msg.data.user.user_id), msg.data.user]);
        }
        break;
      case 'UserLeft':
        if ('user_id' in msg.data) {
          setUsers(prev => prev.filter(u => u.user_id !== msg.data.user_id));
        }
        break;
    }
  }, []);
  
  // 绘制
  useEffect(() => {
    const canvas = canvasRef.current;
    const ctx = canvas?.getContext('2d');
    if (!canvas || !ctx) return;
    
    // 清空画布
    ctx.fillStyle = '#ffffff';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    
    // 绘制所有图形
    shapes.forEach(shape => {
      ctx.strokeStyle = shape.stroke || '#000000';
      ctx.fillStyle = shape.fill || 'transparent';
      ctx.lineWidth = shape.stroke_width || 2;
      
      switch (shape.shape_type) {
        case 'rectangle':
          ctx.strokeRect(shape.x, shape.y, shape.width || 100, shape.height || 100);
          break;
        case 'circle':
          ctx.beginPath();
          ctx.arc(
            shape.x + (shape.width || 50) / 2,
            shape.y + (shape.height || 50) / 2,
            (shape.radius || 50),
            0,
            Math.PI * 2
          );
          ctx.stroke();
          break;
        case 'line':
          ctx.beginPath();
          ctx.moveTo(shape.x, shape.y);
          ctx.lineTo(shape.x + (shape.width || 0), shape.y + (shape.height || 0));
          ctx.stroke();
          break;
        case 'freehand':
          if (shape.points) {
            ctx.beginPath();
            shape.points.forEach((point, i) => {
              if (i === 0) {
                ctx.moveTo(point.x, point.y);
              } else {
                ctx.lineTo(point.x, point.y);
              }
            });
            ctx.stroke();
          }
          break;
        case 'text':
          ctx.font = `${shape.font_size || 16}px sans-serif`;
          ctx.fillText(shape.text || '', shape.x, shape.y);
          break;
      }
      
      // 绘制选中框
      if (selectedShape === shape.id) {
        ctx.setLineDash([5, 5]);
        ctx.strokeRect(shape.x - 5, shape.y - 5, (shape.width || 50) + 10, (shape.height || 50) + 10);
        ctx.setLineDash([]);
      }
    });
    
    // 绘制当前正在创建的图形
    if (isDrawing) {
      ctx.strokeStyle = color;
      ctx.fillStyle = 'transparent';
      ctx.lineWidth = strokeWidth;
      
      switch (tool) {
        case 'rectangle':
          ctx.strokeRect(
            startPos.x,
            startPos.y,
            currentPos.x - startPos.x,
            currentPos.y - startPos.y
          );
          break;
        case 'circle':
          const radius = Math.sqrt(
            Math.pow(currentPos.x - startPos.x, 2) + Math.pow(currentPos.y - startPos.y, 2)
          );
          ctx.beginPath();
          ctx.arc(startPos.x, startPos.y, radius, 0, Math.PI * 2);
          ctx.stroke();
          break;
        case 'line':
          ctx.beginPath();
          ctx.moveTo(startPos.x, startPos.y);
          ctx.lineTo(currentPos.x, currentPos.y);
          ctx.stroke();
          break;
        case 'freehand':
          if (freehandPoints.length > 0) {
            ctx.beginPath();
            freehandPoints.forEach((point, i) => {
              if (i === 0) {
                ctx.moveTo(point.x, point.y);
              } else {
                ctx.lineTo(point.x, point.y);
              }
            });
            ctx.stroke();
          }
          break;
      }
    }
  }, [shapes, selectedShape, isDrawing, tool, startPos, currentPos, color, strokeWidth, freehandPoints]);
  
  // 鼠标事件
  const handleMouseDown = (e: React.MouseEvent) => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    
    const rect = canvas.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;
    
    if (tool === 'select') {
      // 选择图形
      const clicked = shapes.find(s => 
        x >= s.x && x <= s.x + (s.width || 50) &&
        y >= s.y && y <= s.y + (s.height || 50)
      );
      setSelectedShape(clicked?.id || null);
    } else {
      // 开始绘制
      setIsDrawing(true);
      setStartPos({ x, y });
      setCurrentPos({ x, y });
      
      if (tool === 'freehand') {
        setFreehandPoints([{ x, y }]);
      }
    }
  };
  
  const handleMouseMove = (e: React.MouseEvent) => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    
    const rect = canvas.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;
    
    setCurrentPos({ x, y });
    
    if (isDrawing && tool === 'freehand') {
      setFreehandPoints(prev => [...prev, { x, y }]);
    }
    
    // 发送光标位置
    if (wsRef.current?.readyState === WebSocket.OPEN) {
      wsRef.current.send(JSON.stringify({
        type: 'CursorMove',
        data: { user_id: userId, x, y },
      }));
    }
  };
  
  const handleMouseUp = () => {
    if (!isDrawing) return;
    
    setIsDrawing(false);
    
    // 创建新图形
    const shape: Shape = {
      id: uuidv4(),
      shape_type: tool === 'select' ? 'rectangle' : tool,
      x: startPos.x,
      y: startPos.y,
      width: tool === 'freehand' ? undefined : currentPos.x - startPos.x,
      height: tool === 'freehand' ? undefined : currentPos.y - startPos.y,
      fill: tool === 'freehand' ? undefined : undefined,
      stroke: color,
      stroke_width: strokeWidth,
      text: tool === 'text' ? 'Text' : undefined,
      font_size: tool === 'text' ? 16 : undefined,
      points: tool === 'freehand' ? freehandPoints : undefined,
      created_by: userId,
      version: 1,
      timestamp: Date.now(),
    };
    
    // 发送到服务器
    if (wsRef.current?.readyState === WebSocket.OPEN) {
      wsRef.current.send(JSON.stringify({
        type: 'AddShape',
        data: { shape },
      }));
    }
    
    setFreehandPoints([]);
  };
  
  // 删除选中图形
  const deleteSelected = () => {
    if (!selectedShape) return;
    
    if (wsRef.current?.readyState === WebSocket.OPEN) {
      wsRef.current.send(JSON.stringify({
        type: 'DeleteShape',
        data: { shape_id: selectedShape },
      }));
    }
    
    setSelectedShape(null);
  };
  
  return (
    <div className="flex flex-col h-screen">
      {/* 工具栏 */}
      <div className="bg-gray-100 border-b p-2 flex items-center gap-4">
        {/* 连接状态 */}
        <div className={`px-2 py-1 rounded ${connected ? 'bg-green-100 text-green-700' : 'bg-red-100 text-red-700'}`}>
          {connected ? '已连接' : '未连接'}
        </div>
        
        {/* 工具选择 */}
        <div className="flex gap-1">
          {(['select', 'rectangle', 'circle', 'line', 'arrow', 'text', 'freehand'] as Tool[]).map(t => (
            <button
              key={t}
              onClick={() => setTool(t)}
              className={`px-3 py-1 rounded ${tool === t ? 'bg-indigo-500 text-white' : 'bg-white'}`}
            >
              {t === 'select' && '↖'}
              {t === 'rectangle' && '▢'}
              {t === 'circle' && '○'}
              {t === 'line' && '/'}
              {t === 'arrow' && '→'}
              {t === 'text' && 'T'}
              {t === 'freehand' && '✎'}
            </button>
          ))}
        </div>
        
        {/* 颜色选择 */}
        <div className="flex gap-1">
          {COLORS.map(c => (
            <button
              key={c}
              onClick={() => setColor(c)}
              className={`w-6 h-6 rounded border ${color === c ? 'ring-2 ring-offset-2' : ''}`}
              style={{ backgroundColor: c }}
            />
          ))}
        </div>
        
        {/* 线宽 */}
        <select
          value={strokeWidth}
          onChange={e => setStrokeWidth(Number(e.target.value))}
          className="border rounded px-2 py-1"
        >
          <option value={1}>1px</option>
          <option value={2}>2px</option>
          <option value={4}>4px</option>
          <option value={8}>8px</option>
        </select>
        
        {/* 删除 */}
        <button
          onClick={deleteSelected}
          disabled={!selectedShape}
          className="px-3 py-1 bg-red-500 text-white rounded disabled:opacity-50"
        >
          删除
        </button>
        
        {/* 在线用户 */}
        <div className="ml-auto flex gap-2">
          {users.map(u => (
            <div
              key={u.user_id}
              className="w-8 h-8 rounded-full flex items-center justify-center text-white text-sm"
              style={{ backgroundColor: u.color }}
              title={u.username}
            >
              {u.username[0].toUpperCase()}
            </div>
          ))}
        </div>
      </div>
      
      {/* 画布 */}
      <div ref={containerRef} className="flex-1 overflow-hidden bg-gray-200">
        <canvas
          ref={canvasRef}
          width={1920}
          height={1080}
          className="bg-white shadow-lg cursor-crosshair"
          onMouseDown={handleMouseDown}
          onMouseMove={handleMouseMove}
          onMouseUp={handleMouseUp}
          onMouseLeave={handleMouseUp}
        />
      </div>
    </div>
  );
};

// 用户在线状态
interface UserPresence {
  user_id: string;
  username: string;
  color: string;
  cursor_x?: number;
```
