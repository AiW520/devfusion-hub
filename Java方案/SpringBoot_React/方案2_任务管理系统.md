# Spring Boot + React - 任务管理系统

## 一、项目介绍

### 1.1 项目概述
任务管理系统是一个前后端分离的全栈项目，后端使用 Spring Boot + Spring Data JPA，前端使用 React + TypeScript，支持项目管理、任务分配、进度跟踪、团队协作等功能。

### 1.2 解决的问题
- **任务分配混乱**：缺乏统一的任务管理平台
- **进度不透明**：无法实时了解项目进度
- **团队协作困难**：缺少有效的协作工具
- **效率低下**：重复劳动多，沟通成本高

### 1.3 核心技术亮点
- **JWT 认证**：无状态身份验证
- **权限管理**：基于角色的访问控制（RBAC）
- **实时通知**：WebSocket 推送
- **Kanban 看板**：拖拽式任务管理
- **看板视图**：可视化任务状态

package com.taskmanager.service;

import com.taskmanager.entity.Project;
import com.taskmanager.entity.User;
import com.taskmanager.exception.ResourceNotFoundException;
import com.taskmanager.exception.UnauthorizedException;
import com.taskmanager.repository.ProjectRepository;
import com.taskmanager.repository.UserRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;
import java.util.Set;

/**
 * 项目服务
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class ProjectService {

    private final ProjectRepository projectRepository;
    private final UserRepository userRepository;

    /**
     * 创建项目
     */
    @Transactional
    public Project createProject(Long ownerId, Project project, Set<Long> memberIds) {
        log.info("创建项目：{}", project.getName());

        // 查找负责人
        User owner = userRepository.findById(ownerId)
                .orElseThrow(() -> new ResourceNotFoundException("用户不存在"));

        project.setOwner(owner);

        // 添加成员
        if (memberIds != null && !memberIds.isEmpty()) {
            Set<User> members = userRepository.findAllById(memberIds).stream().collect(java.util.stream.Collectors.toSet());
            project.setMembers(members);
        }

        // 添加负责人为成员
        project.getMembers().add(owner);

        return projectRepository.save(project);
    }

    /**
     * 获取用户参与的项目列表
     */
    @Transactional(readOnly = true)
    public List<Project> getUserProjects(Long userId) {
        return projectRepository.findProjectsByMemberId(userId);
    }

    /**
     * 获取项目详情
     */
    @Transactional(readOnly = true)
    public Project getProjectById(Long projectId, Long userId) {
        Project project = projectRepository.findById(projectId)
                .orElseThrow(() -> new ResourceNotFoundException("项目不存在"));

        // 验证用户是否有权限访问项目
        boolean hasAccess = project.getMembers().stream()
                .anyMatch(m -> m.getId().equals(userId));
        
        if (!hasAccess && !project.getOwner().getId().equals(userId)) {
            throw new UnauthorizedException("无权限访问此项目");
        }

        return project;
    }

    /**
     * 更新项目
     */
    @Transactional
    public Project updateProject(Long projectId, Long userId, Project updates) {
        Project project = getProjectById(projectId, userId);

        // 只有项目经理和管理员可以更新
        if (!project.getOwner().getId().equals(userId)) {
            User user = userRepository.findById(userId)
                    .orElseThrow(() -> new ResourceNotFoundException("用户不存在"));
            if (user.getRole() != User.Role.ADMIN) {
                throw new UnauthorizedException("无权限更新此项目");
            }
        }

        // 更新字段
        if (updates.getName() != null) project.setName(updates.getName());
        if (updates.getDescription() != null) project.setDescription(updates.getDescription());
        if (updates.getStartDate() != null) project.setStartDate(updates.getStartDate());
        if (updates.getEndDate() != null) project.setEndDate(updates.getEndDate());
        if (updates.getStatus() != null) project.setStatus(updates.getStatus());

        return projectRepository.save(project);
    }

    /**
     * 删除项目
     */
    @Transactional
    public void deleteProject(Long projectId, Long userId) {
        Project project = projectRepository.findById(projectId)
                .orElseThrow(() -> new ResourceNotFoundException("项目不存在"));

        // 只有项目负责人可以删除
        if (!project.getOwner().getId().equals(userId)) {
            throw new UnauthorizedException("无权限删除此项目");
        }

        projectRepository.delete(project);
    }

    /**
     * 添加项目成员
     */
    @Transactional
    public Project addMember(Long projectId, Long userId, Long newMemberId) {
        Project project = getProjectById(projectId, userId);

        // 只有项目经理可以添加成员
        if (!project.getOwner().getId().equals(userId)) {
            throw new UnauthorizedException("无权限添加成员");
        }

        User newMember = userRepository.findById(newMemberId)
                .orElseThrow(() -> new ResourceNotFoundException("用户不存在"));

        project.getMembers().add(newMember);

        return projectRepository.save(project);
    }
}

package com.taskmanager.service;

import com.taskmanager.entity.Project;
import com.taskmanager.entity.Task;
import com.taskmanager.entity.User;
import com.taskmanager.exception.ResourceNotFoundException;
import com.taskmanager.exception.UnauthorizedException;
import com.taskmanager.repository.ProjectRepository;
import com.taskmanager.repository.TaskRepository;
import com.taskmanager.repository.UserRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;

/**
 * 任务服务
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class TaskService {

    private final TaskRepository taskRepository;
    private final ProjectRepository projectRepository;
    private final UserRepository userRepository;

    /**
     * 创建任务
     */
    @Transactional
    public Task createTask(Long projectId, Long creatorId, Task task) {
        log.info("创建任务：{}", task.getTitle());

        Project project = projectRepository.findById(projectId)
                .orElseThrow(() -> new ResourceNotFoundException("项目不存在"));

        User creator = userRepository.findById(creatorId)
                .orElseThrow(() -> new ResourceNotFoundException("用户不存在"));

        task.setProject(project);
        task.setCreator(creator);
        task.setStatus(Task.TaskStatus.TODO);

        return taskRepository.save(task);
    }

    /**
     * 获取项目的所有任务
     */
    @Transactional(readOnly = true)
    public List<Task> getProjectTasks(Long projectId) {
        return taskRepository.findByProjectIdOrderByPriorityDescCreatedAtDesc(projectId);
    }

    /**
     * 获取任务详情
     */
    @Transactional(readOnly = true)
    public Task getTaskById(Long taskId) {
        return taskRepository.findByIdWithDetails(taskId)
                .orElseThrow(() -> new ResourceNotFoundException("任务不存在"));
    }

    /**
     * 更新任务
     */
    @Transactional
    public Task updateTask(Long taskId, Long userId, Task updates) {
        Task task = taskRepository.findById(taskId)
                .orElseThrow(() -> new ResourceNotFoundException("任务不存在"));

        // 更新基本字段
        if (updates.getTitle() != null) task.setTitle(updates.getTitle());
        if (updates.getDescription() != null) task.setDescription(updates.getDescription());
        if (updates.getPriority() != null) task.setPriority(updates.getPriority());
        if (updates.getStartDate() != null) task.setStartDate(updates.getStartDate());
        if (updates.getDueDate() != null) task.setDueDate(updates.getDueDate());
        if (updates.getEstimatedHours() != null) task.setEstimatedHours(updates.getEstimatedHours());

        // 更新负责人
        if (updates.getAssignee() != null) {
            task.setAssignee(updates.getAssignee());
        }

        // 更新状态
        if (updates.getStatus() != null) {
            task.setStatus(updates.getStatus());
        }

        return taskRepository.save(task);
    }

    /**
     * 更新任务状态（看板拖拽）
     */
    @Transactional
    public Task updateTaskStatus(Long taskId, Task.TaskStatus newStatus) {
        Task task = taskRepository.findById(taskId)
                .orElseThrow(() -> new ResourceNotFoundException("任务不存在"));

        task.setStatus(newStatus);
        return taskRepository.save(task);
    }

    /**
     * 分配任务
     */
    @Transactional
    public Task assignTask(Long taskId, Long assigneeId) {
        Task task = taskRepository.findById(taskId)
                .orElseThrow(() -> new ResourceNotFoundException("任务不存在"));

        User assignee = userRepository.findById(assigneeId)
                .orElseThrow(() -> new ResourceNotFoundException("用户不存在"));

        task.setAssignee(assignee);
        return taskRepository.save(task);
    }

    /**
     * 删除任务
     */
    @Transactional
    public void deleteTask(Long taskId, Long userId) {
        Task task = taskRepository.findById(taskId)
                .orElseThrow(() -> new ResourceNotFoundException("任务不存在"));

        // 只有创建者和项目经理可以删除
        if (!task.getCreator().getId().equals(userId) && 
            !task.getProject().getOwner().getId().equals(userId)) {
            throw new UnauthorizedException("无权限删除此任务");
        }

        taskRepository.delete(task);
    }

    /**
     * 获取分配给我的任务
     */
    @Transactional(readOnly = true)
    public List<Task> getMyTasks(Long userId) {
        return taskRepository.findByAssigneeIdOrderByPriorityDescDueDateAsc(userId);
    }
}

package com.taskmanager.controller;

import com.taskmanager.entity.Project;

import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

import java.util.List;
import java.util.Map;
import java.util.Set;

/**
 * 项目控制器
 */
@Slf4j
@RestController
@RequestMapping("/api/projects")
@RequiredArgsConstructor
public class ProjectController {

    private final ProjectService projectService;
    private final AuthService authService;

    /**
     * 创建项目
     * POST /api/projects
     */
    @PostMapping
    public ResponseEntity<Project> createProject(@RequestBody Map<String, Object> request) {
        User currentUser = authService.getCurrentUser();
        
        Project project = new Project();
        project.setName((String) request.get("name"));
        project.setDescription((String) request.get("description"));
        
        @SuppressWarnings("unchecked")
        List<Integer> memberIds = (List<Integer>) request.get("memberIds");
        Set<Long> memberIdSet = memberIds != null ? 
                memberIds.stream().map(Integer::longValue).collect(java.util.stream.Collectors.toSet()) : null;

        Project created = projectService.createProject(currentUser.getId(), project, memberIdSet);
        
        return ResponseEntity.status(HttpStatus.CREATED).body(created);
    }

    /**
     * 获取我的项目列表
     * GET /api/projects
     */
    @GetMapping
    public ResponseEntity<List<Project>> getMyProjects() {
        User currentUser = authService.getCurrentUser();
        List<Project> projects = projectService.getUserProjects(currentUser.getId());
        return ResponseEntity.ok(projects);
    }

    /**
     * 获取项目详情
     * GET /api/projects/{id}
     */
    @GetMapping("/{id}")
    public ResponseEntity<Project> getProject(@PathVariable Long id) {
        User currentUser = authService.getCurrentUser();
        Project project = projectService.getProjectById(id, currentUser.getId());
        return ResponseEntity.ok(project);
    }

    /**
     * 更新项目
     * PUT /api/projects/{id}
     */
    @PutMapping("/{id}")
    public ResponseEntity<Project> updateProject(
            @PathVariable Long id,
            @RequestBody Project updates) {
        User currentUser = authService.getCurrentUser();
        Project project = projectService.updateProject(id, currentUser.getId(), updates);
        return ResponseEntity.ok(project);
    }

    /**
     * 删除项目
     * DELETE /api/projects/{id}
     */
    @DeleteMapping("/{id}")

        User currentUser = authService.getCurrentUser();
        projectService.deleteProject(id, currentUser.getId());
        return ResponseEntity.noContent().build();

     * 添加项目成员

        Long newMemberId = request.get("userId");

package com.taskmanager.repository;

## 三、前端核心代码

### 3.1 看板组件 (src/components/KanbanBoard.tsx)

```tsx
import React, { useState, useEffect } from 'react';
import { DragDropContext, Droppable, Draggable, DropResult } from 'react-beautiful-dnd';
import { getProjectTasks, updateTaskStatus } from '../services/api';

interface Task {
  id: number;
  title: string;
  status: 'TODO' | 'IN_PROGRESS' | 'REVIEW' | 'DONE';
  priority: 'LOW' | 'MEDIUM' | 'HIGH' | 'URGENT';
  assignee?: { id: number; displayName: string };
  dueDate?: string;
}

interface KanbanBoardProps {
  projectId: number;
}

const KanbanBoard: React.FC<KanbanBoardProps> = ({ projectId }) => {
  const [tasks, setTasks] = useState<Task[]>([]);
  const [loading, setLoading] = useState(true);

  // 任务状态列
  const columns = [
    { id: 'TODO', title: '待办' },
    { id: 'IN_PROGRESS', title: '进行中' },
    { id: 'REVIEW', title: '审核中' },
    { id: 'DONE', title: '已完成' },
  ];

  // 获取任务列表
  const fetchTasks = async () => {
    try {
      const response = await getProjectTasks(projectId);
      setTasks(response.data);
    } catch (err) {
      console.error('获取任务失败', err);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => {
    fetchTasks();
  }, [projectId]);

  // 拖拽结束处理
  const handleDragEnd = async (result: DropResult) => {
    const { destination, source, draggableId } = result;

    // 没有目标位置或没有移动
    if (!destination || (destination.droppableId === source.droppableId && destination.index === source.index)) {
      return;
    }

    // 更新本地状态
    const newTasks = [...tasks];
    const taskIndex = tasks.findIndex(t => t.id === Number(draggableId));
    const task = { ...tasks[taskIndex] };
    task.status = destination.droppableId as Task['status'];
    newTasks[taskIndex] = task;
    setTasks(newTasks);

    // 发送到服务器
    try {
      await updateTaskStatus(Number(draggableId), destination.droppableId);
    } catch (err) {
      // 失败则回滚
      setTasks(tasks);
      console.error('更新任务状态失败', err);
    }
  };

  // 获取指定列的任务
  const getTasksByStatus = (status: string) => {
    return tasks.filter(task => task.status === status);
  };

  // 优先级颜色
  const priorityColors: Record<string, string> = {
    LOW: '#52c41a',
    MEDIUM: '#1890ff',
    HIGH: '#faad14',
    URGENT: '#f5222d',
  };

  if (loading) {
    return <div className="loading">加载中...</div>;
  }

  return (
    <DragDropContext onDragEnd={handleDragEnd}>
      <div className="kanban-board">
        {columns.map(column => (
          <div key={column.id} className="kanban-column">
            <div className="column-header">
              <h3>{column.title}</h3>
              <span className="task-count">{getTasksByStatus(column.id).length}</span>
            </div>
            
            <Droppable droppableId={column.id}>
              {(provided, snapshot) => (
                <div
                  ref={provided.innerRef}
                  {...provided.droppableProps}
                  className={`column-content ${snapshot.isDraggingOver ? 'dragging' : ''}`}
                >
                  {getTasksByStatus(column.id).map((task, index) => (
                    <Draggable key={task.id} draggableId={String(task.id)} index={index}>
                      {(provided, snapshot) => (
                        <div
                          ref={provided.innerRef}
                          {...provided.draggableProps}
                          {...provided.dragHandleProps}
                          className={`task-card ${snapshot.isDragging ? 'dragging' : ''}`}
                        >
                          <div className="task-priority" style={{ backgroundColor: priorityColors[task.priority] }} />
                          <h4>{task.title}</h4>
                          <div className="task-meta">
                            {task.assignee && (
                              <span className="assignee">{task.assignee.displayName}</span>
                            )}
                            {task.dueDate && (
                              <span className="due-date">截止: {task.dueDate}</span>
                            )}
                          </div>
                        </div>
                      )}
                    </Draggable>
                  ))}
                  {provided.placeholder}
                </div>
              )}
            </Droppable>
          </div>
        ))}
      </div>
    </DragDropContext>
  );
};

export default KanbanBoard;
```

## 五、面试要点

### 5.1 看板系统设计

**面试题：如何设计一个看板系统（Kanban）？**

**参考答案：**
1. **数据模型**：Task 有 status 字段（TODO, IN_PROGRESS, REVIEW, DONE）
2. **前端实现**：使用拖拽库（如 react-beautiful-dnd）
3. **后端实现**：提供更新状态的 API
4. **状态管理**：前端本地更新 + 服务器同步
5. **乐观更新**：先更新本地，失败则回滚

### 5.2 权限控制

**面试题：如何实现基于角色的访问控制（RBAC）？**

**参考答案：**
1. **角色定义**：ADMIN, MANAGER, MEMBER
2. **实体关联**：User 有 role 字段
3. **权限判断**：
   - 前端：根据 role 显示/隐藏功能
   - 后端：使用 @PreAuthorize 注解
4. **业务判断**：在 Service 层判断用户权限

```java
@PreAuthorize("hasRole('ADMIN') or hasRole('MANAGER')")
public Project updateProject(...) {
    // 只有管理员和项目经理可以更新
}
```

### 5.3 乐观更新

**面试题：什么是乐观更新？如何实现？**

**参考答案：**
- **乐观更新**：先更新本地 UI，再发送请求到服务器
- **失败处理**：请求失败则回滚到之前的状态
- **适用场景**：大多数用户操作，特别是需要快速响应的场景

```tsx
// 先更新本地
setTasks(newTasks);

// 发送请求
try {
  await updateTaskStatus(id, newStatus);
} catch {
  // 失败则回滚
  setTasks(oldTasks);
}
```
