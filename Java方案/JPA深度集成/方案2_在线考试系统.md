# JPA深度集成 - 在线考试系统

## 一、项目介绍

### 1.1 项目概述
在线考试系统是一个基于 Spring Data JPA + PostgreSQL 的教育类应用，支持在线题库管理、试卷生成、自动评分、成绩统计等功能。

### 1.2 解决的问题
- **传统考试痛点**：纸质试卷成本高、批改慢、数据难统计
- **题库管理困难**：题目分散、重复率高、难以维护
- **防作弊需求**：需要支持随机抽题、乱序、限时等功能
- **成绩分析**：需要快速统计成绩、生成报告

### 1.3 核心技术亮点
- **复杂关联查询**：题库-试卷-考试-成绩的多表关联
- **动态查询**：支持按难度、类型、知识点筛选题目
- **事务管理**：考试提交、成绩计算的事务控制
- **JPA 锁机制**：防止并发提交导致的数据不一致
- **Criteria API**：动态构建查询条件
- **@Formula 注解**：计算字段（如总分数、平均分）
- **JPA 监听器**：自动记录考试日志

### 1.4 适用场景
- 教育培训机构
- 企业内部考核
- 在线学习平台
- 需要复杂数据关联和统计分析的系统

```java
package com.example.exam.entity;

import jakarta.persistence.*;
import lombok.Data;
import org.springframework.data.annotation.CreatedDate;
import org.springframework.data.annotation.LastModifiedDate;
import org.springframework.data.jpa.domain.support.AuditingEntityListener;

import java.time.LocalDateTime;

/**
 * 基础实体类
 */
@Data
@MappedSuperclass
@EntityListeners(AuditingEntityListener.class)
public abstract class BaseEntity {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @CreatedDate
    @Column(name = "created_time", nullable = false, updatable = false)
    private LocalDateTime createdTime;

    @LastModifiedDate
    @Column(name = "updated_time", nullable = false)
    private LocalDateTime updatedTime;

    @Column(name = "is_deleted", nullable = false)
    private Boolean deleted = false;

    @Version
    @Column(name = "version")
    private Integer version;
}
```

```java
package com.example.exam.entity;

import jakarta.persistence.*;
import lombok.AllArgsConstructor;
import lombok.Data;
import lombok.EqualsAndHashCode;
import lombok.NoArgsConstructor;
import org.hibernate.annotations.SQLDelete;
import org.hibernate.annotations.Where;

import java.util.Set;

/**
 * 题目实体类
 */
@Data
@NoArgsConstructor
@AllArgsConstructor
@EqualsAndHashCode(callSuper = true, onlyExplicitlyIncluded = true)
@Entity
@Table(name = "question", indexes = {
    @Index(name = "idx_question_subject", columnList = "subject_id"),
    @Index(name = "idx_question_type", columnList = "question_type"),
    @Index(name = "idx_question_difficulty", columnList = "difficulty")
})
@SQLDelete(sql = "UPDATE question SET is_deleted = true WHERE id = ?")
@Where(clause = "is_deleted = false")
public class Question extends BaseEntity {

    /**
     * 题目标题
     */
    @Column(name = "title", nullable = false, columnDefinition = "TEXT")
    private String title;

    /**
     * 题目类型
     * SINGLE_CHOICE-单选题  MULTI_CHOICE-多选题  TRUE_FALSE-判断题  FILL_BLANK-填空题  SHORT_ANSWER-简答题
     */
    @Enumerated(EnumType.STRING)
    @Column(name = "question_type", nullable = false, length = 20)
    private QuestionType questionType;

    /**
     * 难度等级
     * EASY-简单  MEDIUM-中等  HARD-困难
     */
    @Enumerated(EnumType.STRING)
    @Column(name = "difficulty", nullable = false, length = 10)
    private Difficulty difficulty;

    /**
     * 分数
     */
    @Column(name = "score", nullable = false)
    private Integer score;

    /**
     * 题目选项（JSON 格式）
     * 单选题/多选题使用，格式：[{"label": "A", "content": "选项内容"}, ...]
     */
    @Column(name = "options", columnDefinition = "jsonb")
    private String options;

    /**
     * 正确答案
     * 单选题：A/B/C/D  多选题：AB/ACD  判断题：true/false  填空题：答案1,答案2  简答题：关键词
     */
    @Column(name = "correct_answer", nullable = false, length = 500)
    private String correctAnswer;

    /**
     * 题目解析
     */
    @Column(name = "analysis", columnDefinition = "TEXT")
    private String analysis;

    /**
     * 知识点标签（JSON 格式）
     * 例如：["Spring IoC", "依赖注入", "Bean 生命周期"]
     */
    @Column(name = "tags", columnDefinition = "jsonb")
    private String tags;

    /**
     * 使用次数（统计题目被使用的次数）
     */
    @Column(name = "usage_count", nullable = false)
    private Integer usageCount = 0;

    /**
     * 所属科目
     */
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "subject_id", nullable = false)
    private Subject subject;

    /**
     * 该题目在哪些试卷中（多对多关联，通过 PaperQuestion 中间表）
     */
    @OneToMany(mappedBy = "question", cascade = CascadeType.ALL, orphanRemoval = true)

    /**
     * 题目类型枚举
     */
    public enum QuestionType {
        SINGLE_CHOICE("单选题"),
        MULTI_CHOICE("多选题"),
        TRUE_FALSE("判断题"),
        FILL_BLANK("填空题"),
        SHORT_ANSWER("简答题");

        private final String description;

        QuestionType(String description) {
            this.description = description;
        }

        public String getDescription() {
            return description;
        }
    }

    /**
     * 难度枚举
     */
    public enum Difficulty {
        EASY("简单"),
        MEDIUM("中等"),
        HARD("困难");

        private final String description;

        Difficulty(String description) {

        public String getDescription() {
            return description;
        }
    }

    /**
     * 增加使用次数
     */

    }
```

```java
package com.example.exam.entity;

import jakarta.persistence.*;
import lombok.AllArgsConstructor;
import lombok.Data;
import lombok.EqualsAndHashCode;
import lombok.NoArgsConstructor;
import org.hibernate.annotations.SQLDelete;
import org.hibernate.annotations.Where;

import java.time.LocalDateTime;
import java.util.Set;

/**
 * 试卷实体类
 */
@Data
@NoArgsConstructor
@AllArgsConstructor
@EqualsAndHashCode(callSuper = true, onlyExplicitlyIncluded = true)
@Entity
@Table(name = "paper", indexes = {
    @Index(name = "idx_paper_subject", columnList = "subject_id"),
    @Index(name = "idx_paper_status", columnList = "status")
})
@SQLDelete(sql = "UPDATE paper SET is_deleted = true WHERE id = ?")
@Where(clause = "is_deleted = false")
public class Paper extends BaseEntity {

    /**
     * 试卷名称
     */
    @Column(name = "name", nullable = false, length = 200)
    @EqualsAndHashCode.Include
    private String name;

    /**
     * 试卷描述
     */
    @Column(name = "description", columnDefinition = "TEXT")
    private String description;

    /**
     * 总分
     * 通过计算所有题目的分数总和得出
     */
    @Column(name = "total_score", nullable = false)
    private Integer totalScore;

    /**
     * 考试时长（分钟）
     */
    @Column(name = "duration", nullable = false)
    private Integer duration;

    /**
     * 及格分数
     */
    @Column(name = "passing_score", nullable = false)
    private Integer passingScore;

    /**
     * 题目数量
     */
    @Column(name = "question_count", nullable = false)
    private Integer questionCount;

    /**
     * 是否随机抽题
     */
    @Column(name = "is_random", nullable = false)
    private Boolean isRandom = false;

    /**
     * 试卷状态
     * DRAFT-草稿  PUBLISHED-已发布  ARCHIVED-已归档
     */
    @Enumerated(EnumType.STRING)
    @Column(name = "status", nullable = false, length = 20)
    private PaperStatus status = PaperStatus.DRAFT;

    /**
     * 所属科目
     */
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "subject_id", nullable = false)
    private Subject subject;

    /**
     * 试卷题目列表（关联关系）
     */
    @OneToMany(mappedBy = "paper", cascade = CascadeType.ALL, orphanRemoval = true)
    private Set<PaperQuestion> paperQuestions;

    /**
     * 创建人
     */
    @Column(name = "creator", length = 50)
    private String creator;

    /**
     * 发布时间
     */
    @Column(name = "publish_time")
    private LocalDateTime publishTime;

    /**
     * 试卷状态枚举
     */
    public enum PaperStatus {
        DRAFT("草稿"),
        PUBLISHED("已发布"),
        ARCHIVED("已归档");

        private final String description;

        PaperStatus(String description) {
            this.description = description;
        }

        public String getDescription() {
            return description;
        }
    }

    /**
     * 发布试卷
     */
    public void publish() {
        if (this.status == PaperStatus.PUBLISHED) {
            throw new IllegalStateException("试卷已发布，不能重复发布");
        }
        this.status = PaperStatus.PUBLISHED;
        this.publishTime = LocalDateTime.now();
    }

    /**
     * 归档试卷
     */
    public void archive() {
        if (this.status == PaperStatus.ARCHIVED) {
            throw new IllegalStateException("试卷已归档");
        }
        this.status = PaperStatus.ARCHIVED;
    }
}
```

```java
package com.example.exam.entity;

import jakarta.persistence.*;
import lombok.AllArgsConstructor;
import lombok.Data;
import lombok.EqualsAndHashCode;
import lombok.NoArgsConstructor;
import org.hibernate.annotations.SQLDelete;
import org.hibernate.annotations.Where;

import java.time.LocalDateTime;

/**
 * 考试实体类
 * 记录每次考试的信息
 */
@Data
@NoArgsConstructor
@AllArgsConstructor
@EqualsAndHashCode(callSuper = true, onlyExplicitlyIncluded = true)
@Entity
@Table(name = "exam", indexes = {
    @Index(name = "idx_exam_paper", columnList = "paper_id"),
    @Index(name = "idx_exam_student", columnList = "student_id"),
    @Index(name = "idx_exam_status", columnList = "status")
})
@SQLDelete(sql = "UPDATE exam SET is_deleted = true WHERE id = ?")
@Where(clause = "is_deleted = false")
public class Exam extends BaseEntity {

    /**
     * 考试编号（唯一）
     */
    @Column(name = "exam_no", unique = true, nullable = false, length = 50)
    @EqualsAndHashCode.Include
    private String examNo;

    /**
     * 学生 ID 或学号
     */
    @Column(name = "student_id", nullable = false, length = 50)
    private String studentId;

    /**
     * 学生姓名
     */
    @Column(name = "student_name", nullable = false, length = 100)
    private String studentName;

    /**
     * 使用的试卷
     */
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "paper_id", nullable = false)
    private Paper paper;

    /**
     * 考试开始时间
     */
    @Column(name = "start_time", nullable = false)
    private LocalDateTime startTime;

    /**
     * 考试结束时间
     */
    @Column(name = "end_time")
    private LocalDateTime endTime;

    /**
     * 考试状态
     * NOT_STARTED-未开始  IN_PROGRESS-进行中  SUBMITTED-已提交  GRADED-已评分
     */
    @Enumerated(EnumType.STRING)
    @Column(name = "status", nullable = false, length = 20)
    private ExamStatus status = ExamStatus.NOT_STARTED;

    /**
     * 得分
     * 提交后由系统自动计算
     */
    @Column(name = "score")
    private Integer score;

    /**
     * 是否及格
     */
    @Column(name = "is_passed")
    private Boolean isPassed;

    /**
     * 考试状态枚举
     */
    public enum ExamStatus {
        NOT_STARTED("未开始"),
        IN_PROGRESS("进行中"),
        SUBMITTED("已提交"),
        GRADED("已评分");

        private final String description;

        ExamStatus(String description) {
            this.description = description;
        }

        public String getDescription() {
            return description;
        }
    }

    /**
     * 开始考试
     */
    public void start() {
        if (this.status != ExamStatus.NOT_STARTED) {
            throw new IllegalStateException("考试已经开始或已结束");
        }
        this.status = ExamStatus.IN_PROGRESS;
        this.startTime = LocalDateTime.now();
    }

    /**
     * 提交考试
     */
    public void submit() {
        if (this.status != ExamStatus.IN_PROGRESS) {
            throw new IllegalStateException("考试未开始或已提交");
        }
        this.status = ExamStatus.SUBMITTED;
        this.endTime = LocalDateTime.now();
    }

    /**
     * 设置成绩
     */
    public void setGrade(Integer score, Boolean isPassed) {
        this.score = score;
        this.isPassed = isPassed;
        this.status = ExamStatus.GRADED;
    }

    /**
     * 检查是否超时
     */
    public boolean isTimeout() {
        if (this.status != ExamStatus.IN_PROGRESS) {
            return false;
        }
        LocalDateTime deadline = this.startTime.plusMinutes(this.paper.getDuration());
        return LocalDateTime.now().isAfter(deadline);
    }
```

```java
/**
 * 试卷服务层
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class PaperService {

    private final PaperRepository paperRepository;
    private final SubjectRepository subjectRepository;
    private final QuestionRepository questionRepository;

    /**
     * 创建试卷
     */
    @Transactional
    public Paper createPaper(Long subjectId, Paper paper, List<Long> questionIds) {
        log.info("创建试卷，科目ID：{}，名称：{}，题目数量：{}",
                 subjectId, paper.getName(), questionIds.size());

        // 查找科目
        Subject subject = subjectRepository.findById(subjectId)
                .orElseThrow(() -> new ResourceNotFoundException("科目不存在，ID：" + subjectId));

        paper.setSubject(subject);

        // 添加题目
        if (questionIds != null && !questionIds.isEmpty()) {
            List<Question> questions = questionRepository.findAllById(questionIds);

            // 创建关联关系
            Set<PaperQuestion> paperQuestions = new HashSet<>();
            int totalScore = 0;
            int order = 1;

            for (Question question : questions) {
                PaperQuestion pq = new PaperQuestion();
                pq.setPaperId(null);  // 稍后设置
                pq.setQuestionId(question.getId());
                pq.setQuestionOrder(order++);
                pq.setScore(question.getScore());
                pq.setQuestion(question);

                paperQuestions.add(pq);
                totalScore += question.getScore();
            }

            paper.setPaperQuestions(paperQuestions);
            paper.setTotalScore(totalScore);
            paper.setQuestionCount(questions.size());
        }

        Paper saved = paperRepository.save(paper);

        // 更新关联关系中的试卷 ID
        if (saved.getPaperQuestions() != null) {
            saved.getPaperQuestions().forEach(pq -> pq.setPaperId(saved.getId()));
            saved = paperRepository.save(saved);
        }

        log.info("试卷创建成功，ID：{}，总分：{}，题目数：{}",
                 saved.getId(), saved.getTotalScore(), saved.getQuestionCount());

        return saved;
    }

    /**
     * 自动生成试卷（随机抽题）
     */
    @Transactional
    public Paper autoGeneratePaper(Long subjectId, Paper paper, Map<String, Integer> config) {
        log.info("自动生成试卷，科目ID：{}", subjectId);

        // 查找科目
        Subject subject = subjectRepository.findById(subjectId)
                .orElseThrow(() -> new ResourceNotFoundException("科目不存在，ID：" + subjectId));

        paper.setSubject(subject);
        paper.setIsRandom(true);

        // 根据配置随机抽题
        Set<PaperQuestion> paperQuestions = new HashSet<>();
        int totalScore = 0;
        int order = 1;

        // 配置示例：{"SINGLE_CHOICE_EASY": 10, "SINGLE_CHOICE_MEDIUM": 5, ...}
        for (Map.Entry<String, Integer> entry : config.entrySet()) {
            String key = entry.getKey();  // 例如：SINGLE_CHOICE_EASY
            int count = entry.getValue();  // 数量

            // 解析类型和难度
            String[] parts = key.split("_");
            Question.QuestionType type = Question.QuestionType.valueOf(parts[0]);
            Question.Difficulty difficulty = Question.Difficulty.valueOf(parts[1]);

            // 随机抽题
            List<Question> questions = questionRepository.findRandomQuestions(
                    subjectId,
                    type.name(),
                    difficulty.name(),
                    count
            );

            // 创建关联关系
            for (Question question : questions) {
                PaperQuestion pq = new PaperQuestion();
                pq.setQuestionId(question.getId());
                pq.setQuestionOrder(order++);
                pq.setScore(question.getScore());
                pq.setQuestion(question);

                paperQuestions.add(pq);
                totalScore += question.getScore();

                // 增加题目使用次数
                question.incrementUsageCount();
            }
        }

        paper.setPaperQuestions(paperQuestions);
        paper.setTotalScore(totalScore);
        paper.setQuestionCount(paperQuestions.size());

        Paper saved = paperRepository.save(paper);
        log.info("试卷自动生成成功，ID：{}，总分：{}，题目数：{}",
                 saved.getId(), saved.getTotalScore(), saved.getQuestionCount());

        return saved;
    }

    @Transactional
    public Paper publishPaper(Long id) {
```

#### 2.7.4 评分 Service (GradingService.java)

```java
package com.example.exam.service;

import com.example.exam.entity.*;
import com.example.exam.repository.ExamAnswerRepository;
import com.example.exam.repository.PaperQuestionRepository;
import lombok.Data;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

import java.util.List;

/**
 * 评分服务层
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class GradingService {

    private final ExamAnswerRepository examAnswerRepository;
    private final PaperQuestionRepository paperQuestionRepository;

    /**
     * 评分
     */
    @org.springframework.transaction.annotation.Transactional(readOnly = true)
    public GradingResult grade(Exam exam) {
        log.info("开始评分，考试ID：{}", exam.getId());

        // 查找考试答案
        List<ExamAnswer> answers = examAnswerRepository.findByExamId(exam.getId());

        // 查找试卷题目
        List<PaperQuestion> paperQuestions = paperQuestionRepository.findByPaperId(exam.getPaper().getId());

        int totalScore = 0;

        // 评分
        for (ExamAnswer answer : answers) {
            // 查找对应的试卷题目
            PaperQuestion paperQuestion = paperQuestions.stream()
                    .filter(pq -> pq.getQuestionId().equals(answer.getQuestionId()))
                    .findFirst()
                    .orElse(null);

            if (paperQuestion != null) {
                // 评分逻辑
                GradingDetail detail = gradeAnswer(answer, paperQuestion.getQuestion());
                answer.setIsCorrect(detail.isCorrect());
                answer.setScore(detail.getScore());

                totalScore += detail.getScore();
            }
        }

        // 更新答案
        examAnswerRepository.saveAll(answers);

        // 判断是否及格
        boolean isPassed = totalScore >= exam.getPaper().getPassingScore();

        log.info("评分完成，考试ID：{}，得分：{}，是否及格：{}",
                 exam.getId(), totalScore, isPassed);

        return new GradingResult(totalScore, isPassed);
    }

    /**
     * 评分单个答案
     */
    private GradingDetail gradeAnswer(ExamAnswer answer, Question question) {
        boolean isCorrect = false;
        int score = 0;

        switch (question.getQuestionType()) {
            case SINGLE_CHOICE:
                // 单选题：答案必须完全匹配
                isCorrect = question.getCorrectAnswer().equalsIgnoreCase(answer.getStudentAnswer());
                score = isCorrect ? question.getScore() : 0;
                break;

            case MULTI_CHOICE:
                // 多选题：答案必须完全匹配（顺序不重要）
                isCorrect = compareMultiChoice(question.getCorrectAnswer(), answer.getStudentAnswer());
                score = isCorrect ? question.getScore() : 0;
                break;

            case TRUE_FALSE:
                // 判断题：true/false 匹配
                isCorrect = question.getCorrectAnswer().equalsIgnoreCase(answer.getStudentAnswer());
                score = isCorrect ? question.getScore() : 0;
                break;

            case FILL_BLANK:
                // 填空题：逗号分隔的多个答案
                isCorrect = compareFillBlank(question.getCorrectAnswer(), answer.getStudentAnswer());
                score = isCorrect ? question.getScore() : 0;
                break;

            case SHORT_ANSWER:
                // 简答题：关键词匹配
                isCorrect = compareShortAnswer(question.getCorrectAnswer(), answer.getStudentAnswer());
                score = isCorrect ? question.getScore() : 0;
                break;
        }

        return new GradingDetail(isCorrect, score);
    }

    /**
     * 比较多选题答案
     */
    private boolean compareMultiChoice(String correctAnswer, String studentAnswer) {
        // 排序后比较
        char[] correct = correctAnswer.toCharArray();
        char[] student = studentAnswer.toCharArray();
        java.util.Arrays.sort(correct);
        java.util.Arrays.sort(student);
        return java.util.Arrays.equals(correct, student);
    }

    /**
     * 比较填空题答案
     */
    private boolean compareFillBlank(String correctAnswer, String studentAnswer) {
        String[] correct = correctAnswer.split(",");
        String[] student = studentAnswer.split(",");

        if (correct.length != student.length) {
            return false;
        }

        for (int i = 0; i < correct.length; i++) {
            if (!correct[i].trim().equalsIgnoreCase(student[i].trim())) {
                return false;
            }
        }

        return true;
    }

    /**
     * 比较简答题答案（关键词匹配）
     */
    private boolean compareShortAnswer(String keywords, String studentAnswer) {
        String[] keywordArray = keywords.split(",");
        String lowerAnswer = studentAnswer.toLowerCase();

        for (String keyword : keywordArray) {
            if (!lowerAnswer.contains(keyword.trim().toLowerCase())) {
                return false;
            }
        }

        return true;
    }

    /**
     * 评分结果
     */
    @Data
    public static class GradingResult {
        private final Integer score;
        private final Boolean isPassed;
    }

    /**
     * 评分详情
     */
    @Data
    private static class GradingDetail {
        private final boolean isCorrect;
        private final int score;
    }
}
```

```java
/**
```

**可能踩的坑：**
- 计算字段不支持更新

### 5.1 JPA 多对多关联

**面试题：JPA 中如何实现多对多关联？**

**参考答案：**
- 使用 `@ManyToMany` 注解
- 需要 `@JoinTable` 指定中间表
- 可以通过中间表实体添加额外属性

**示例：**
```java
// 方式1：使用 @ManyToMany
@ManyToMany
@JoinTable(
    name = "paper_question",
    joinColumns = @JoinColumn(name = "paper_id"),
    inverseJoinColumns = @JoinColumn(name = "question_id")
)
private Set<Question> questions;

// 方式2：使用中间表实体（推荐，可以添加额外属性）
@OneToMany(mappedBy = "paper")
private Set<PaperQuestion> paperQuestions;
```

### 5.2 @Formula 使用场景

**面试题：@Formula 注解的使用场景和限制？**

**参考答案：**
- **使用场景**：
  - 计算字段（如正确率、百分比）
  - 统计字段（如总数、平均值）
- **限制**：
  - 计算字段是只读的
  - SQL 语法依赖于数据库
  - 每次查询都会计算，可能影响性能
