# Kafka Streams - 实时数据大屏

## 一、项目介绍

### 1.1 项目概述
实时数据大屏是一个基于 Kafka Streams 的流处理项目，用于实时处理和分析海量数据，支持实时统计、监控告警、数据可视化等功能。

### 1.2 解决的问题
- **实时性要求高**：需要毫秒级的数据处理能力
- **海量数据处理**：需要处理每秒数万条消息
- **实时统计**：需要实时计算各种业务指标
- **监控告警**：需要实时监控系统状态

### 1.3 核心技术亮点
- **Kafka Streams**：轻量级流处理库，简化流处理开发
- **Spring Boot + Kafka**：Java 原生支持，生态完善
- **实时聚合**：窗口计算、滚动聚合
- **状态管理**：Kafka Streams 内置状态存储
- **Exactly-Once**：端到端精确一次语义
- **Redis**：热点数据缓存

### 1.4 适用场景
- 实时数据大屏
- 实时监控系统
- 实时推荐系统
- 实时风控系统

#### 3.3.2 订单流处理器

```java
package com.dashboard.streams;

import com.dashboard.common.model.DashboardMetrics;
import com.dashboard.common.model.OrderEvent;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.apache.kafka.common.serialization.Serdes;
import org.apache.kafka.streams.KafkaStreams;
import org.apache.kafka.streams.StreamsBuilder;
import org.apache.kafka.streams.StreamsConfig;
import org.apache.kafka.streams.kstream.*;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.data.redis.core.RedisTemplate;

import java.math.BigDecimal;
import java.time.Duration;
import java.time.Instant;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.HashMap;
import java.util.Map;
import java.util.Properties;

/**
 * 订单流处理器
 * 
 * 功能：
 * 1. 消费订单事件
 * 2. 按时间窗口统计订单数、订单金额
 * 3. 按状态统计各状态订单数
 * 4. 统计热销商品排行
 * 5. 将结果写入 Redis
 */
@Slf4j
@Configuration
@RequiredArgsConstructor
public class OrderStreamProcessor {

    private final RedisTemplate<String, Object> redisTemplate;

    // Kafka 配置
    @Value("${spring.kafka.bootstrap-servers}")
    private String bootstrapServers;

    // Redis Key 前缀
    private static final String KEY_PREFIX = "dashboard:order:";

    /**
     * 配置 Kafka Streams
     */
    @Bean
    public StreamsBuilder streamsBuilder() {
        StreamsBuilder builder = new StreamsBuilder();
        
        // 构建订单流处理拓扑
        buildOrderStreamTopology(builder);
        
        return builder;
    }

    /**
     * 构建订单流处理拓扑
     */
    private void buildOrderStreamTopology(StreamsBuilder builder) {
        // 1. 从 Kafka 主题读取订单事件
        KStream<String, OrderEvent> orderStream = builder.stream(
                "order-events",
                // 指定 key serde 和 value serde
                Consumed.with(Serdes.String(), new OrderEventSerde())
        );

        // 2. 按状态分组统计
        countByStatus(orderStream);

        // 3. 按时间窗口统计（每分钟）
        countByTimeWindow(orderStream);

        // 4. 统计热销商品
        countHotProducts(orderStream);

        // 5. 统计热卖店铺
        countHotStores(orderStream);

        log.info("订单流处理拓扑构建完成");
    }

    /**
     * 按订单状态统计
     */
    private void countByStatus(KStream<String, OrderEvent> orderStream) {
        // 按状态分组
        KGroupedStream<String, OrderEvent> groupedByStatus = orderStream
                .groupBy((key, value) -> value.getStatus());

        // 聚合统计
        KTable<String, Long> statusCounts = groupedByStatus.count();

        // 输出到 Redis
        statusCounts.toStream().foreach((status, count) -> {
            String redisKey = KEY_PREFIX + "status:" + status;
            redisTemplate.opsForValue().set(redisKey, count);
            log.debug("订单状态统计更新：{} = {}", status, count);
        });
    }

    /**
     * 按时间窗口统计（滚动窗口）
     * 
     * 滚动窗口（Tumbling Window）：
     * - 固定大小、不重叠
     * - 每分钟一个窗口
     */
    private void countByTimeWindow(KStream<String, OrderEvent> orderStream) {
        // 定义 1 分钟滚动窗口
        TimeWindows tumblingWindow = TimeWindows.ofSizeAndGrace(
                Duration.ofMinutes(1),  // 窗口大小
                Duration.ofSeconds(10)   // 容忍延迟
        );

        // 按时间窗口聚合
        KTable<Windowed<String>, Long> windowedCounts = orderStream
                .groupBy((key, value) -> value.getStatus())
                .windowedBy(tumblingWindow)
                .count();

        // 输出到 Redis
        windowedCounts.toStream().foreach((windowedStatus, count) -> {
            String redisKey = KEY_PREFIX + "window:" + windowedStatus.key() + 
                    ":" + windowedStatus.window().start();
            redisTemplate.opsForValue().set(redisKey, count);
            log.debug("时间窗口统计更新：{} = {}", windowedStatus.key(), count);
        });

        // 统计今日累计
        countTodayTotal(orderStream);
    }

    /**
     * 统计今日累计
     */
    private void countTodayTotal(KStream<String, OrderEvent> orderStream) {
        // 过滤今日订单
        KStream<String, OrderEvent> todayOrders = orderStream
                .filter((key, order) -> isToday(order.getEventTime()));

        // 统计订单数
        KTable<String, Long> todayOrderCount = todayOrders
                .groupBy((key, value) -> "total")
                .count();

        todayOrderCount.toStream().foreach((key, count) -> {
            String redisKey = KEY_PREFIX + "today:order-count";
            redisTemplate.opsForValue().set(redisKey, count);
        });

        // 统计订单金额
        KTable<String, BigDecimal> todayOrderAmount = todayOrders
                .groupBy((key, value) -> new KeyValue<>("total", value.getAmount()))
                .aggregate(
                        () -> BigDecimal.ZERO,
                        (key, order, total) -> total.add(order.getAmount()),
                        Materialized.with(Serdes.String(), new BigDecimalSerde())
                );

        todayOrderAmount.toStream().foreach((key, amount) -> {
            String redisKey = KEY_PREFIX + "today:order-amount";
            redisTemplate.opsForValue().set(redisKey, amount);
        });
    }

    /**
     * 统计热销商品排行
     * 
     * 使用会话窗口（Session Window）：
     * - 根据活动间隙动态调整
     * - 适合用户行为分析
     */
    private void countHotProducts(KStream<String, OrderEvent> orderStream) {
        // 按商品 ID 分组
        KGroupedStream<String, BigDecimal> groupedByProduct = orderStream
                .filter((key, order) -> "COMPLETE".equals(order.getStatus()))
                .map((key, order) -> new KeyValue<>(order.getProductId(), order.getAmount()));

        // 聚合金额
        KTable<String, BigDecimal> productSales = groupedByProduct
                .aggregate(
                        () -> BigDecimal.ZERO,
                        (key, amount, total) -> total.add(amount),
                        Materialized.with(Serdes.String(), new BigDecimalSerde())
                );

        // 取 Top 10
        productSales.toStream()
                .mapValues(sales -> sortTopProducts(productSales.getKeyValueStore()))
                .foreach((key, topProducts) -> {
                    String redisKey = KEY_PREFIX + "hot-products";
                    redisTemplate.opsForValue().set(redisKey, topProducts);
                });
    }

    /**
     * 统计热卖店铺
     */
    private void countHotStores(KStream<String, OrderEvent> orderStream) {
        // 按店铺 ID 分组
        KGroupedStream<String, BigDecimal> groupedByStore = orderStream
                .filter((key, order) -> "COMPLETE".equals(order.getStatus()))
                .map((key, order) -> new KeyValue<>(order.getStoreId(), order.getAmount()));

        // 聚合金额
        KTable<String, BigDecimal> storeSales = groupedByStore
                .aggregate(
                        () -> BigDecimal.ZERO,
                        (key, amount, total) -> total.add(amount),
                        Materialized.with(Serdes.String(), new BigDecimalSerde())
                );

        // 输出到 Redis
        storeSales.toStream().foreach((storeId, sales) -> {
            String redisKey = KEY_PREFIX + "store:" + storeId + ":sales";
            redisTemplate.opsForValue().set(redisKey, sales);
        });
    }

    /**
     * 判断时间是否是今天
     */
    private boolean isToday(Instant instant) {
        return LocalDate.now().equals(instant.atZone(java.time.ZoneId.systemDefault()).toLocalDate());
    }

    /**
     * 排序获取 Top 商品
     */
    private String sortTopProducts(ReadOnlyKeyValueStore<String, BigDecimal> store) {
        // 从状态存储中获取所有商品销售数据并排序
        Map<String, BigDecimal> allProducts = new HashMap<>();
        store.all().forEachRemaining(entry -> allProducts.put(entry.key(), entry.value()));

        return allProducts.entrySet().stream()
                .sorted(Map.Entry.<String, BigDecimal>comparingByValue().reversed())
                .limit(10)
                .toList()
                .toString();
    }

    /**
     * Kafka Streams 配置
     */
    @Bean
    public KafkaStreams kafkaStreams(StreamsBuilder streamsBuilder) {
        Properties props = new Properties();
        props.put(StreamsConfig.APPLICATION_ID_CONFIG, "order-stream-processor");
        props.put(StreamsConfig.BOOTSTRAP_SERVERS_CONFIG, bootstrapServers);
        props.put(StreamsConfig.DEFAULT_KEY_SERDE_CLASS_CONFIG, Serdes.String().getClass());
        props.put(StreamsConfig.DEFAULT_VALUE_SERDE_CLASS_CONFIG, Serdes.String().getClass());
        props.put(StreamsConfig.COMMIT_INTERVAL_MS_CONFIG, 1000);  // 每秒提交
        props.put(StreamsConfig.PROCESSING_GUARANTEE_CONFIG, StreamsConfig.EXACTLY_ONCE_V2);  // 精确一次

        KafkaStreams streams = new KafkaStreams(streamsBuilder.build(), props);
        streams.start();

        log.info("Kafka Streams 启动完成");

        return streams;
    }
}
```

```java
    public StreamsBuilder userActionStreamsBuilder() {
        StreamsBuilder builder = new StreamsBuilder();
        
        buildUserActionTopology(builder);
        
        return builder;
    }

    /**
     * 构建用户行为流处理拓扑
     */
    private void buildUserActionTopology(StreamsBuilder builder) {
        // 1. 读取用户行为事件
        KStream<String, UserActionEvent> actionStream = builder.stream(
                "user-action-events",
                Consumed.with(Serdes.String(), new UserActionEventSerde())
        );

        // 2. 统计 UV（独立访客数）
        countUv(actionStream);

        // 3. 统计 PV（页面浏览量）
        countPv(actionStream);

        // 4. 按行为类型统计
        countByActionType(actionStream);

        // 5. 统计搜索关键词
        countSearchKeywords(actionStream);

        log.info("用户行为流处理拓扑构建完成");
    }

    /**
     * 统计独立访客数（UV）
     * 
     * 使用会话窗口：
     * - 30 分钟无活动则会话结束
     * - 超过 30 分钟认为是新访客
     */
    private void countUv(KStream<String, UserActionEvent> actionStream) {
        // 按用户 ID 分组
        KGroupedStream<String, UserActionEvent> groupedByUser = actionStream
                .groupBy((key, action) -> action.getUserId());

        // 会话窗口（30 分钟无活动则会话结束）
        SessionWindows sessionWindow = SessionWindows.with(Duration.ofMinutes(30));

        // 统计会话数（UV）
        KTable<Windowed<String>, Long> sessionCounts = groupedByUser
                .windowedBy(sessionWindow)
                .count();

        // 统计今日 UV
        actionStream
                .filter((key, action) -> isToday(action.getEventTime()))
                .map((key, action) -> new KeyValue<>(action.getUserId(), 1L))
                .groupBy((key, value) -> KeyValue.pair("uv", value))
                .count()
                .toStream()
                .foreach((key, count) -> {
                    String redisKey = KEY_PREFIX + "today:uv";
                    redisTemplate.opsForValue().set(redisKey, count);
                    log.debug("今日 UV 更新：{}", count);
                });
    }

    /**
     * 统计页面浏览量（PV）
     */
    private void countPv(KStream<String, UserActionEvent> actionStream) {
        // 过滤浏览行为
        KStream<String, UserActionEvent> viewStream = actionStream
                .filter((key, action) -> "VIEW".equals(action.getActionType()));

        // 过滤今日
        KStream<String, UserActionEvent> todayViews = viewStream
                .filter((key, action) -> isToday(action.getEventTime()));

        // 按固定 key 统计
        todayViews
                .map((key, action) -> new KeyValue<>("pv", 1L))
                .groupBy((key, value) -> key)
                .count()
                .toStream()
                .foreach((key, count) -> {
                    String redisKey = KEY_PREFIX + "today:pv";
                    redisTemplate.opsForValue().set(redisKey, count);
                    log.debug("今日 PV 更新：{}", count);
                });
    }

    /**
     * 按行为类型统计
     */
    private void countByActionType(KStream<String, UserActionEvent> actionStream) {
        // 过滤今日
        KStream<String, UserActionEvent> todayActions = actionStream
                .filter((key, action) -> isToday(action.getEventTime()));

        // 按行为类型分组
        KGroupedStream<String, Long> groupedByType = todayActions
                .map((key, action) -> new KeyValue<>(action.getActionType(), 1L))
                .groupBy((key, value) -> key);

        // 统计数量
        KTable<String, Long> actionCounts = groupedByType.count();

        // 输出到 Redis
        actionCounts.toStream().foreach((actionType, count) -> {
            String redisKey = KEY_PREFIX + "today:action:" + actionType;
            redisTemplate.opsForValue().set(redisKey, count);
            log.debug("行为统计更新：{} = {}", actionType, count);
        });
    }

    /**
     * 统计搜索关键词
     */
    private void countSearchKeywords(KStream<String, UserActionEvent> actionStream) {
        // 过滤搜索行为
        KStream<String, UserActionEvent> searchStream = actionStream
                .filter((key, action) -> "SEARCH".equals(action.getActionType()));

        // 过滤今日
        KStream<String, UserActionEvent> todaySearches = searchStream
                .filter((key, action) -> isToday(action.getEventTime()));

        // 按关键词分组并统计
        todaySearches
                .map((key, action) -> new KeyValue<>(action.getKeyword(), 1L))
                .groupBy((key, value) -> key)
                .count()
                .toStream()
                .foreach((keyword, count) -> {
                    String redisKey = KEY_PREFIX + "keyword:" + keyword;
                    redisTemplate.opsForValue().set(redisKey, count);
                    log.debug("关键词统计更新：{} = {}", keyword, count);
                });
    }

    /**
     * 判断时间是否是今天
     */
    private boolean isToday(Instant instant) {
        return LocalDate.now().equals(instant.atZone(java.time.ZoneId.systemDefault()).toLocalDate());
    }

    @Bean
    public KafkaStreams userActionKafkaStreams(StreamsBuilder streamsBuilder) {
        Properties props = new Properties();
```

## 五、面试要点

### 5.1 Kafka Streams 基础

**面试题：什么是 Kafka Streams？它有什么特点？**

**参考答案：**
- **Kafka Streams**：Apache Kafka 提供的流处理库
- **特点**：
  1. 轻量级：无需单独集群，直接嵌入应用
  2. 事件时间处理：支持处理时间和事件时间
  3. 精确一次语义：端到端 exactly-once
  4. 状态管理：内置状态存储
  5. 窗口计算：支持滚动、滑动、会话窗口
  6. 高可用：Kafka 集群保证高可用

### 5.2 时间窗口

**面试题：Kafka Streams 支持哪些时间窗口？有什么区别？**

**参考答案：**
1. **滚动窗口（Tumbling Window）**
   - 固定大小、不重叠
   - 适用于周期性统计

2. **滑动窗口（Sliding Window）**
   - 固定大小、可重叠
   - 适用于平滑聚合

3. **会话窗口（Session Window）**
   - 根据活动间隙动态调整
   - 适用于用户行为分析

### 5.3 状态管理

**面试题：Kafka Streams 如何实现状态管理？**

**参考答案：**
- **状态存储**：Kafka Streams 内置 RocksDB 作为状态存储
- **聚合操作**：`groupBy` + `aggregate` / `count` / `reduce`
- **容错**：状态变更记录到 Kafka 主题，失败可恢复
- **查询**：通过 Interactive Queries 查询本地状态

### 5.4 Exactly-Once 语义

**面试题：Kafka Streams 如何保证 Exactly-Once？**

**参考答案：**
- **事务性输出**：使用事务发送结果到输出主题
- **幂等生产者**：保证消息不重复
- **消费者偏移量管理**：使用事务提交偏移量
- **配置**：`processing.guarantee=exactly_once_v2`

### 5.5 实时统计场景

**面试题：如何用 Kafka Streams 实现实时 UV 统计？**

**参考答案：**
1. 按用户 ID 分组
2. 使用会话窗口（30 分钟无活动则会话结束）
3. 统计会话数（每个用户一个会话 = 一个 UV）
4. 过滤今日数据

这个实时数据大屏项目涵盖了 Kafka Streams 流处理的核心知识点：

1. **Kafka Streams**：轻量级流处理库

通过这个项目，你可以掌握流处理技术的核心概念和实战技巧，为面试加分。
