# Gin+React 方案一：博客平台项目

## 📚 项目介绍

### 1.1 项目概述

这是一个现代化的博客平台，采用前后端分离架构，后端使用 Go + Gin 框架提供 RESTful API，前端使用 React + TypeScript 构建响应式界面。项目支持用户注册登录、文章发布与编辑、分类标签管理、评论互动等核心功能。

**解决什么问题：**
- 提供一个可展示的完整全栈项目
- 掌握 JWT 认证、RESTful API 设计、中间件开发
- 理解前后端分离架构的优势

#### 2.2.1 程序入口 main.go

```go
package main

import (
	"fmt"
	"log"
	"os"

	// 导入项目内部包
	"blog-backend/config"
	"blog-backend/internal/handler"
	"blog-backend/internal/middleware"
	"blog-backend/internal/model"
	"blog-backend/internal/repository"
	"blog-backend/internal/service"

	// 第三方库
	"github.com/gin-gonic/gin"
	"github.com/joho/godotenv"      // 环境变量加载
	"gorm.io/driver/postgres"       // PostgreSQL 驱动
	"gorm.io/gorm"                  // ORM 库
	"gorm.io/gorm/logger"           // 日志模式
)

// main 是应用程序的入口函数
// 执行顺序：加载配置 → 初始化数据库 → 初始化各层组件 → 注册路由 → 启动服务器
func main() {
	// ========== 第1步：加载环境变量 ==========
	// 从 .env 文件读取配置，这是12-Factor App的最佳实践
	// 在项目根目录创建 .env 文件，包含数据库连接信息等
	if err := godotenv.Load(); err != nil {
		log.Printf("⚠️  未找到 .env 文件，将使用系统环境变量")
	}

	// ========== 第2步：加载配置 ==========
	// 从环境变量读取配置，支持灵活部署
	cfg := config.Load()

	// ========== 第3步：初始化数据库连接 ==========
	// 使用 GORM 连接 PostgreSQL
	// DSN 格式：user:password@host:port/dbname?sslmode=disable
	dsn := fmt.Sprintf(
		"host=%s user=%s password=%s dbname=%s port=%s sslmode=disable TimeZone=Asia/Shanghai",
		cfg.DBHost, cfg.DBUser, cfg.DBPassword, cfg.DBName, cfg.DBPort,
	)

	// 初始化 GORM DB 对象
	// Config 参数说明：
	// - Logger: 自定义日志输出，显示 SQL 语句（开发环境）
	// - TranslateErrorCallback: 错误翻译回调
	db, err := gorm.Open(postgres.Open(dsn), &gorm.Config{
		Logger: logger.Default.LogMode(logger.Info), // 显示所有 SQL（生产环境改为 logger.Silent）
	})
	if err != nil {
		// 数据库连接失败是致命错误，直接退出
		log.Fatalf("❌ 数据库连接失败: %v", err)
	}
	log.Println("✅ 数据库连接成功")

	// ========== 第4步：数据库自动迁移 ==========
	// AutoMigrate 会根据模型定义自动创建/更新表结构
	// 首次运行创建表，后续运行添加新字段
	// 注意：生产环境应使用专门的迁移工具（如 golang-migrate）
	if err := db.AutoMigrate(
		&model.User{},      // 用户表
		&model.Article{},   // 文章表
		&model.Comment{},   // 评论表
		&model.Tag{},        // 标签表
	); err != nil {
		log.Fatalf("❌ 数据库迁移失败: %v", err)
	}
	log.Println("✅ 数据库迁移完成")

	// ========== 第5步：初始化各层组件 ==========
	// 采用依赖注入模式，层与层之间通过接口解耦
	// 这样可以方便地进行单元测试（Mock）

	// 初始化 Repository 层（数据访问层）
	userRepo := repository.NewUserRepository(db)
	articleRepo := repository.NewArticleRepository(db)
	commentRepo := repository.NewCommentRepository(db)

	// 初始化 Service 层（业务逻辑层）
	authService := service.NewAuthService(userRepo, cfg.JWTSecret)
	articleService := service.NewArticleService(articleRepo)
	commentService := service.NewCommentService(commentRepo)

	// 初始化 Handler 层（HTTP处理层）
	authHandler := handler.NewAuthHandler(authService)
	articleHandler := handler.NewArticleHandler(articleService)
	commentHandler := handler.NewCommentHandler(commentService)

	// ========== 第6步：初始化 Gin 路由器 ==========
	// 设置 Gin 为发布模式（关闭调试日志）
	if cfg.IsProduction() {
		gin.SetMode(gin.ReleaseMode)
	}

	// 创建默认路由器
	r := gin.New()

	// 注册全局中间件
	// 中间件执行顺序：Logger → Recovery → CORS
	r.Use(gin.Logger(),                        // 记录每个请求的日志
		gin.Recovery(),                       // 捕获 panic，防止服务器崩溃
		middleware.CORS(),                    // 处理跨域请求
	)

	// ========== 第7步：注册路由 ==========
	registerRoutes(r, authHandler, articleHandler, commentHandler)

	// ========== 第8步：启动服务器 ==========
	// 使用环境变量或默认端口
	port := cfg.AppPort
	if port == "" {
		port = "8080"
	}

	log.Printf("🚀 服务器启动中，监听端口: %s", port)
	log.Printf("📖 API文档: http://localhost:%s/api/v1/", port)
	log.Printf("📖 健康检查: http://localhost:%s/api/v1/health", port)

	// 启动 HTTP 服务器
	// 第一个参数是地址，第二个是处理器
	if err := r.Run(":" + port); err != nil {
		log.Fatalf("❌ 服务器启动失败: %v", err)
	}
}

// registerRoutes 注册所有路由
// 采用路由分组管理，API版本控制
func registerRoutes(
	r *gin.Engine,
	authHandler *handler.AuthHandler,
	articleHandler *handler.ArticleHandler,
	commentHandler *handler.CommentHandler,
) {
	// 创建 API v1 版本分组
	// 所有 API 路径都以 /api/v1 开头，便于版本迭代
	api := r.Group("/api/v1")

	// 健康检查接口（无需认证）
	api.GET("/health", func(c *gin.Context) {
		// 返回 JSON 格式的健康状态
		c.JSON(200, gin.H{
			"status":  "healthy",
			"service": "blog-backend",
			"version": "1.0.0",
		})
	})

	// 公开接口（无需认证）
	public := api.Group("/public")
	{
		// 用户认证接口
		public.POST("/auth/register", authHandler.Register)   // 用户注册
		public.POST("/auth/login", authHandler.Login)          // 用户登录
	}

	// 需要认证的接口（使用 JWT 中间件）
	// gin.HandlerFunc 是 Gin 中间件的类型
	authRequired := api.Group("")
	authRequired.Use(middleware.JWTAuth()) // JWT 中间件验证 token
	{
		// 用户相关
		users := authRequired.Group("/users")
		{
			users.GET("/me", authHandler.GetCurrentUser)       // 获取当前用户信息
			users.PUT("/me", authHandler.UpdateProfile)        // 更新个人信息
		}

		// 文章管理
		articles := authRequired.Group("/articles")
		{
			articles.POST("", articleHandler.Create)          // 创建文章
			articles.PUT("/:id", articleHandler.Update)        // 更新文章
			articles.DELETE("/:id", articleHandler.Delete)      // 删除文章
		}

		// 评论管理
		comments := authRequired.Group("/comments")
		{
			comments.POST("", commentHandler.Create)          // 创建评论
			comments.DELETE("/:id", commentHandler.Delete)     // 删除评论
		}
	}

	// 公开的文章接口（无需认证也能查看）
	articlesPublic := api.Group("/articles")
	{
		articlesPublic.GET("", articleHandler.List)            // 文章列表
		articlesPublic.GET("/:id", articleHandler.Get)         // 文章详情
		articlesPublic.GET("/:id/comments", commentHandler.ListByArticle) // 文章评论
	}
}
```

```go
package config

import (
	"os"      // 操作系统功能，用于读取环境变量
	"strconv" // 字符串转换
)

// Config 应用配置结构体
// 使用结构体集中管理所有配置，便于维护和传递
type Config struct {
	// 数据库配置
	DBHost     string // 数据库主机地址
	DBPort     string // 数据库端口
	DBUser     string // 数据库用户名
	DBPassword string // 数据库密码
	DBName     string // 数据库名称

	// Redis 配置
	RedisHost string // Redis 主机地址
	RedisPort string // Redis 端口
	RedisPass string // Redis 密码

	// 应用配置
	AppPort   string // HTTP 服务端口
	JWTSecret string // JWT 密钥，生产环境必须使用复杂的随机字符串
	JWTExpire int    // JWT 过期时间（小时）

	// 环境配置
	AppEnv string // 运行环境：development / production / staging
}

// Load 从环境变量加载配置
// 这是一种12-Factor App的配置管理方式：配置与代码分离
func Load() *Config {
	return &Config{
		// 数据库配置（带默认值）
		DBHost:     getEnv("DB_HOST", "localhost"),
		DBPort:     getEnv("DB_PORT", "5432"),
		DBUser:     getEnv("DB_USER", "postgres"),
		DBPassword: getEnv("DB_PASSWORD", "postgres"),
		DBName:     getEnv("DB_NAME", "blog"),

		// Redis 配置（带默认值）
		RedisHost: getEnv("REDIS_HOST", "localhost"),
		RedisPort: getEnv("REDIS_PORT", "6379"),
		RedisPass: getEnv("REDIS_PASS", ""),

		// 应用配置
		AppPort:   getEnv("APP_PORT", "8080"),
		JWTSecret: getEnv("JWT_SECRET", "your-secret-key-change-in-production"),
		JWTExpire: getEnvInt("JWT_EXPIRE", 72), // 默认72小时 = 3天

		// 环境配置
		AppEnv: getEnv("APP_ENV", "development"),
	}
}

// getEnv 获取环境变量，附带默认值
// 参数说明：
// - key: 环境变量名
// - defaultVal: 当环境变量不存在时使用的默认值
func getEnv(key, defaultVal string) string {
	// os.Getenv 尝试读取环境变量
	// 如果不存在或为空字符串，返回默认值
	if value := os.Getenv(key); value != "" {
		return value
	}
	return defaultVal
}

// getEnvInt 获取整数类型的环境变量
// 用于端口号、过期时间等配置
func getEnvInt(key string, defaultVal int) int {
	// 先按字符串读取
	if value := os.Getenv(key); value != "" {
		// strconv.Atoi 将字符串转换为整数
		if intVal, err := strconv.Atoi(value); err == nil {
			return intVal
		}
	}
	return defaultVal
}

// IsProduction 判断是否为生产环境
// 用于在生产环境关闭调试信息
func (c *Config) IsProduction() bool {
	return c.AppEnv == "production"
}

// IsDevelopment 判断是否为开发环境
func (c *Config) IsDevelopment() bool {
	return c.AppEnv == "development"
}
```

```go
// ========== User 用户模型 ==========

// User 用户结构体
// GORM 通过结构体标签（tag）定义数据库字段映射
type User struct {

	Username  string    `gorm:"uniqueIndex;size:50;not null"` // 用户名，唯一索引，必填
	Email     string    `gorm:"uniqueIndex;size:100;not null"` // 邮箱，唯一索引，必填
	Password  string    `gorm:"size:255;not null"`            // 密码，bcrypt加密后存储
	Nickname  string    `gorm:"size:100"`                     // 昵称
	Avatar    string    `gorm:"size:500"`                     // 头像URL
	Bio       string    `gorm:"size:500"`                     // 个人简介
	Status    int       `gorm:"default:1"`                   // 账号状态：1正常 0禁用
	CreatedAt time.Time `gorm:"autoCreateTime"`               // 创建时间，自动填充
	UpdatedAt time.Time `gorm:"autoUpdateTime"`              // 更新时间，自动填充

	// 关联关系：用户与文章为一对多关系
	Articles []Article `gorm:"foreignKey:AuthorID"` // 外键关联
}

	return "users"
}

	Content    string    `gorm:"type:text;not null"`            // 文章内容，支持长文本
	Summary    string    `gorm:"size:500"`                       // 文章摘要
```

#### 2.2.4 JWT 认证中间件 middleware/auth.go

```go
package middleware

import (
	"fmt"
	"net/http"
	"strings"

	"blog-backend/pkg/jwt"
	"blog-backend/pkg/response"

	"github.com/gin-gonic/gin"
)

// JWTAuth JWT认证中间件
// 功能：验证请求头中的 JWT token，解析用户信息并注入到上下文
// 执行时机：在 Handler 处理请求之前执行

// 1. 从请求头获取 Authorization
// 2. 验证 token 格式（Bearer xxx）
// 3. 解析并验证 token
// 4. 将用户ID注入到 gin.Context
// 5. 验证失败则返回 401

func JWTAuth() gin.HandlerFunc {
	return func(c *gin.Context) {
		// ========== 第1步：从请求头获取 Token ==========
		// Authorization 格式：Bearer <token>
		authHeader := c.GetHeader("Authorization")

		// 如果没有携带 Authorization 头，返回 401
		if authHeader == "" {
			response.Unauthorized(c, "请先登录")
			c.Abort() // 阻止后续 Handler 执行
			return
		}

		// ========== 第2步：验证 Token 格式 ==========
		// Bearer 方案：HTTP 标准授权方案，token 前需要有 "Bearer " 前缀
		parts := strings.SplitN(authHeader, " ", 2)
		if len(parts) != 2 || parts[0] != "Bearer" {
			response.Error(c, http.StatusBadRequest, "Token 格式错误，请使用 Bearer <token>")
			c.Abort()
			return
		}

		// 提取 token 字符串
		tokenString := parts[1]

		// ========== 第3步：解析并验证 Token ==========
		// 调用 jwt.ParseToken 验证 token 是否有效
		// - 检查签名是否正确（防止篡改）
		// - 检查是否过期
		// - 检查是否为恶意伪造
		claims, err := jwt.ParseToken(tokenString)
		if err != nil {
			// 根据不同错误类型返回不同提示
			if err.Error() == "token已过期" {
				response.Unauthorized(c, "登录已过期，请重新登录")
			} else {
				response.Unauthorized(c, "无效的 Token")
			}
			c.Abort()
			return
		}

		// ========== 第4步：将用户信息注入上下文 ==========
		// gin.Context 是 Gin 中的请求上下文，用于在中间件和处理器之间传递数据
		// 使用 Set 方法存储数据，Get 方法读取数据
		c.Set("userID", claims.UserID)    // 将用户ID存入上下文
		c.Set("username", claims.Username) // 将用户名存入上下文

		// ========== 第5步：继续处理请求 ==========
		// c.Next() 表示调用下一个中间件或 Handler
		// 这行代码必须调用，否则请求会在这里停止
		c.Next()
	}
}

// GetUserID 从上下文中获取当前用户ID
// 这是一个辅助函数，避免在每个 Handler 中重复类型断言
func GetUserID(c *gin.Context) (uint, error) {
	// 从上下文获取 userID，返回的是 interface{}
	userID, exists := c.Get("userID")
	if !exists {
		return 0, fmt.Errorf("用户ID不存在")
	}

	// 类型断言：将 interface{} 转换为 uint
```

```go
type RegisterRequest struct {
	Username string `json:"username" binding:"required,min=3,max=50"`  // 用户名，必填，长度3-50
	Email    string `json:"email" binding:"required,email"`             // 邮箱，必填，格式验证
	Password string `json:"password" binding:"required,min=6,max=100"`   // 密码，必填，最小6位
	Nickname string `json:"nickname"`                                    // 昵称，可选
}

// Register 用户注册
// 业务流程：
// 1. 验证用户名/邮箱是否已被注册
// 2. 密码强度验证（已通过 binding tag）
// 3. 密码加密（bcrypt）
// 4. 创建用户记录
// 5. 返回 JWT Token
func (s *AuthService) Register(req *RegisterRequest) (*model.User, string, error) {
	// ========== 第1步：检查用户是否已存在 ==========
	// 查询数据库，如果用户已存在，返回错误
	existingUser, _ := s.userRepo.FindByUsername(req.Username)
	if existingUser != nil {
		return nil, "", ErrUserAlreadyExists
	}

	existingEmail, _ := s.userRepo.FindByEmail(req.Email)
	if existingEmail != nil {
		return nil, "", errors.New("该邮箱已被注册")
	}

	// ========== 第2步：密码加密 ==========
	// bcrypt 是业界标准的密码哈希算法
	// - 自动加盐，防止彩虹表攻击
	// - 可配置计算强度（cost）
	// - 计算缓慢，防止暴力破解
	//
	// bcrypt.GenerateFromPassword 参数说明：
	// - 第一个参数是密码（字节数组）
	// - 第二个参数是 cost 值，越大越安全但越慢
	// - 生产环境建议 10-12，开发环境可以小一些加快测试
	hash, err := bcrypt.GenerateFromPassword([]byte(req.Password), bcrypt.DefaultCost)
	if err != nil {
		return nil, "", err
	}

	// ========== 第3步：创建用户对象 ==========
	user := &model.User{
		Username: req.Username,
		Email:    req.Email,
		Password: string(hash), // 存储哈希值，而非原始密码
		Nickname: req.Nickname,
		Status:   1, // 1 = 正常状态
	}

	// ========== 第4步：保存到数据库 ==========
	if err := s.userRepo.Create(user); err != nil {
		return nil, "", err
	}

	// ========== 第5步：生成 JWT Token ==========
	// 用户注册成功后直接登录，返回 token
	token, err := s.generateToken(user)
	if err != nil {
		return user, "", err
	}

	return user, token, nil
}

// Login 用户登录
// 业务流程：
// 1. 根据用户名查询用户
// 2. 验证密码（bcrypt.CompareHashAndPassword）
// 3. 生成 JWT Token
func (s *AuthService) Login(req *LoginRequest) (*model.User, string, error) {
	// ========== 第1步：查询用户 ==========
	// 支持用户名或邮箱登录
	user, err := s.userRepo.FindByUsername(req.Username)
	if err != nil {
		// 用户不存在
		return nil, "", ErrUserNotFound
	}

	// 如果用邮箱登录没找到，尝试邮箱查询
	if user == nil {
		user, err = s.userRepo.FindByEmail(req.Username)
		if err != nil || user == nil {
			return nil, "", ErrUserNotFound
		}
	}

	// ========== 第2步：验证密码 ==========
	// bcrypt.CompareHashAndPassword 对比哈希值和明文密码
	// 返回 nil 表示密码正确
	if err := bcrypt.CompareHashAndPassword([]byte(user.Password), []byte(req.Password)); err != nil {
		return nil, "", ErrInvalidPassword
	}

	// ========== 第3步：检查账号状态 ==========
	if user.Status != 1 {
		return nil, "", errors.New("账号已被禁用")
	}

	// ========== 第4步：生成 JWT Token ==========
	token, err := s.generateToken(user)
	if err != nil {
		return nil, "", err
	}

	return user, token, nil
}
```

#### 2.2.6 JWT 工具包 pkg/jwt/jwt.go

```go
package jwt

import (
	"errors"
	"time"

	"github.com/golang-jwt/jwt/v5" // JWT 库
)

// 错误定义
var (
	ErrInvalidToken = errors.New("无效的 Token")
	ErrExpiredToken = errors.New("Token 已过期")
)

// Claims 自定义声明
// 包含用户身份信息
type Claims struct {
	UserID   uint   `json:"userID"`   // 用户ID
	Username string `json:"username"` // 用户名
	jwt.StandardClaims              // 标准声明（包含过期时间、签发时间等）
}

// GenerateToken 生成 JWT Token
// 参数：
// - claims: Token 包含的用户信息
// - secret: 签名密钥（应该从配置读取，不要硬编码）
//
// 返回：签名的 token 字符串
func GenerateToken(claims Claims, secret string) (string, error) {
	// 创建 Token 对象
	// 第一个参数是签名方法（HMAC SHA256）
	// 第二个参数是 Token 的声明（claims）
	token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)

	// 签名
	// 使用密钥对 Token 进行签名，生成最终的 token 字符串
	// 如果密钥为空或格式错误，这里会返回错误
	return token.SignedString([]byte(secret))
}

// ParseToken 解析并验证 JWT Token
// 参数：
// - tokenString: 待解析的 token 字符串
//
// 返回：解析后的 claims
// 错误：
// - ErrInvalidToken: Token 格式错误或签名验证失败
// - ErrExpiredToken: Token 已过期
func ParseToken(tokenString string) (*Claims, error) {
	// 解析 Token
	// 第二个参数是密钥回调函数，用于获取签名密钥
	// 这里使用固定密钥，生产环境建议使用动态密钥（支持密钥轮换）
	token, err := jwt.ParseWithClaims(tokenString, &Claims{}, func(token *jwt.Token) (interface{}, error) {
		// 验证签名方法是否为 HS256
		// 这是安全检查，防止攻击者使用其他算法伪造 Token
		if _, ok := token.Method.(*jwt.SigningMethodHMAC); !ok {
			return nil, ErrInvalidToken
		}
		// 返回签名密钥
		return []byte("your-secret-key-change-in-production"), nil
	})

	if err != nil {
		// 解析失败，根据错误类型返回不同信息
		if errors.Is(err, jwt.ErrTokenExpired) {
			return nil, ErrExpiredToken
		}
		return nil, ErrInvalidToken
	}

	// 提取 Claims
	// 类型断言：token.Claims 是 jwt.Claims 接口，实际类型是 Claims
	if claims, ok := token.Claims.(*Claims); ok && token.Valid {
		return claims, nil
	}

	return nil, ErrInvalidToken
}

// RefreshToken 刷新 Token
// 在 Token 即将过期时，生成一个新的 Token
func RefreshToken(tokenString string, secret string) (string, error) {
	// 先解析旧 Token（不验证过期时间）
	// 这样即使用户的 Token 刚刚过期，也能刷新
	token, err := jwt.ParseWithClaims(tokenString, &Claims{}, func(token *jwt.Token) (interface{}, error) {
		return []byte(secret), nil
	})

	if err != nil {
		return "", ErrInvalidToken
	}

	// 提取旧 claims
	claims, ok := token.Claims.(*Claims)
	if !ok {
		return "", ErrInvalidToken
	}

	// 检查 Token 是否真的快过期了（可选的安全检查）
	// 如果 Token 还有超过一半的有效期，不允许刷新
	// expiringDuration := time.Until(time.Unix(claims.ExpiresAt, 0))
	// if expiringDuration > 36*time.Hour {
	//     return "", errors.New("Token 尚未需要刷新")
	// }

	// 生成新的 Claims
	newClaims := Claims{
		UserID:   claims.UserID,
		Username: claims.Username,
		StandardClaims: jwt.StandardClaims{
			ExpiresAt: time.Now().Add(72 * time.Hour).Unix(), // 新的过期时间
			IssuedAt:  time.Now().Unix(),
			Issuer:    "blog-backend",
		},
	}

	// 生成新 Token
	return GenerateToken(newClaims, secret)
}
```
