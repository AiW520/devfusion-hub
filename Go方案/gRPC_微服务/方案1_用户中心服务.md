# gRPC 微服务方案一：用户中心服务

## 📚 项目介绍

### 1.1 项目概述

这是一个基于 gRPC 的微服务用户中心系统，提供用户注册、登录、信息管理、权限验证等核心功能。项目展示了 Protobuf 协议设计、gRPC 服务开发、TLS 加密通信、微服务注册发现等核心能力。

**核心功能：**
- 用户注册与登录（支持多种认证方式）
- 用户信息 CRUD
- JWT Token 生成与验证
- 微服务间认证
- 用户权限管理
- 分布式会话管理

### 2.2 Protobuf 定义

#### 2.2.1 proto/user.proto

```protobuf
// 语法声明：使用 Protobuf v3 语法
// 这是标准的声明，必须放在文件第一行
syntax = "proto3";

// 包声明：防止命名冲突
// 类似于其他语言的 namespace
package user;

// 选项配置
// go_package 指定生成的 Go 代码的包路径
// 格式：<module>/<path>;<alias>
option go_package = "user-center/proto/user;user";

// 导入其他 proto 文件
import "proto/common.proto";

// ========== 用户服务定义 ==========

// Service 定义服务接口
// gRPC 会根据这个定义生成客户端和服务端代码
service UserService {
    // 注册用户
    // RPC 方法定义：
    // - 方法名：Register
    // - 请求类型：RegisterRequest
    // - 响应类型：RegisterResponse
    // - 选项：开启客户端流/服务端流（这里都是普通方法）
    rpc Register(RegisterRequest) returns (RegisterResponse);

    // 用户登录
    rpc Login(LoginRequest) returns (LoginResponse);

    // 获取用户信息
    rpc GetUser(GetUserRequest) returns (GetUserResponse);

    // 更新用户信息
    rpc UpdateUser(UpdateUserRequest) returns (UpdateUserResponse);

    // 删除用户
    rpc DeleteUser(DeleteUserRequest) returns (DeleteUserResponse);

    // 批量获取用户信息（服务端流）
    // 适用于数据量大、需要分批返回的场景
    rpc GetUsers(GetUsersRequest) returns (stream GetUserResponse);

    // 验证 Token（微服务间调用）
    rpc ValidateToken(ValidateTokenRequest) returns (ValidateTokenResponse);
}

// ========== 消息类型定义 ==========

// RegisterRequest 注册请求
// 字段格式：<类型> <名称> = <标签>;
// 标签是唯一标识，用于二进制编码
// 标签 1-15 使用单字节编码，适合频繁使用的字段
message RegisterRequest {
    string username = 1;              // 用户名
    string email = 2;                 // 邮箱
    string password = 3;             // 密码（客户端应加密传输）
    string nickname = 4;             // 昵称（可选）
}

// RegisterResponse 注册响应
message RegisterResponse {
    // 嵌套的 CommonResponse，使用组合模式
    CommonResponse common = 1;        // 通用响应（状态码、消息）
    User user = 2;                   // 用户信息（不含密码）
}

// LoginRequest 登录请求
message LoginRequest {
    // 支持用户名或邮箱登录
    string account = 1;              // 账号（用户名或邮箱）
    string password = 2;            // 密码
}

// LoginResponse 登录响应
message LoginResponse {
    CommonResponse common = 1;
    string access_token = 2;         // 访问令牌
    int64 expires_in = 3;            // 过期时间（秒）
    string token_type = 4;           // 令牌类型（Bearer）
    User user = 5;                   // 用户信息
}

// GetUserRequest 获取用户请求
message GetUserRequest {
    // 联合主键：可以用 ID 或用户名查询
    uint64 id = 1;                   // 用户ID（可选）
    string username = 2;            // 用户名（可选）
}

// GetUserResponse 获取用户响应
message GetUserResponse {
    CommonResponse common = 1;
    User user = 2;
}

// GetUsersRequest 批量获取用户请求
message GetUsersRequest {
    repeated uint64 ids = 1;         // 用户ID列表
}

// UpdateUserRequest 更新用户请求
message UpdateUserRequest {
    uint64 id = 1;                   // 用户ID
    string nickname = 2;            // 新昵称
    string avatar = 3;              // 新头像
    string email = 4;               // 新邮箱
    string password = 5;            // 新密码
}

// UpdateUserResponse 更新用户响应
message UpdateUserResponse {
    CommonResponse common = 1;
    User user = 2;
}

// DeleteUserRequest 删除用户请求
message DeleteUserRequest {
    uint64 id = 1;
}

// ValidateTokenRequest 验证 Token 请求
message ValidateTokenRequest {
    string token = 1;               // 待验证的 Token
}

// ValidateTokenResponse 验证 Token 响应
message ValidateTokenResponse {
    bool valid = 1;                 // 是否有效
    uint64 user_id = 2;             // 用户ID（有效时返回）
    string username = 3;            // 用户名（有效时返回）
}

// ========== 数据模型定义 ==========

// User 用户信息
// 注意：这里定义的是跨服务共享的用户模型
message User {
    uint64 id = 1;                   // 用户ID
    string username = 2;           // 用户名
    string email = 3;              // 邮箱
    string nickname = 4;            // 昵称
    string avatar = 5;              // 头像URL
    string status = 6;             // 状态：active/inactive/banned
    int64 created_at = 7;           // 创建时间戳
    int64 updated_at = 8;           // 更新时间戳
}
```

### 2.3 服务端实现

#### 2.3.1 主程序 main.go

```go
package main

import (
	"context"
	"fmt"
	"log"
	"net"
	"os"
	"os/signal"
	"syscall"

	"user-center/config"
	"user-center/internal/handler"
	"user-center/internal/middleware"
	pb "user-center/proto/user"
	"user-center/pkg/redis"

	"google.golang.org/grpc"
	"google.golang.org/grpc/health"
	healthpb "google.golang.org/grpc/health/grpc_health_v1"
	"google.golang.org/grpc/reflection" // 启用 gRPC 反射，用于 grpcurl 测试
)

func main() {
	// ========== 第1步：加载配置 ==========
	cfg := config.Load()

	// ========== 第2步：初始化 Redis ==========
	rdb := redis.NewClient(&redis.Config{
		Addr: cfg.RedisAddr(),
		Pass: cfg.RedisPass,
	})

	// ========== 第3步：创建 gRPC Server ==========
	// grpc.NewServer 的选项：
	// - UnaryInterceptor: 单次 RPC 调用拦截器
	// - StreamInterceptor: 流式 RPC 拦截器
	grpcServer := grpc.NewServer(
		grpc.UnaryInterceptor(middleware.ChainUnaryServer(
			middleware.Logging(),     // 日志拦截器
			middleware.Recovery(),    // 异常恢复
			middleware.ContextTimeout(), // 超时控制
		)),
		grpc.StreamInterceptor(middleware.ChainStreamServer(
			middleware.StreamLogging(),
			middleware.StreamRecovery(),
		)),
	)

	// ========== 第4步：注册服务 ==========
	// 创建服务实例
	userHandler := handler.NewUserHandler(rdb, cfg.JWTSecret)

	// 注册 UserService
	// 第二个参数是 gRPC server 实例
	// 第三个参数是服务实现
	pb.RegisterUserServiceServer(grpcServer, userHandler)

	// 注册健康检查服务
	// 这是 Kubernetes/Consul 健康检查的标准方式
	healthServer := health.NewServer()
	healthpb.RegisterHealthServer(grpcServer, healthServer)

	// 启用 gRPC 反射（用于 grpcurl、Postman 等工具测试）
	// 生产环境应禁用
	reflection.Register(grpcServer)

	// ========== 第5步：启动监听 ==========
	// net.Listen 创建 TCP 监听
	addr := fmt.Sprintf(":%s", cfg.GRPCPort)
	lis, err := net.Listen("tcp", addr)
	if err != nil {
		log.Fatalf("❌ 监听端口失败: %v", err)
	}
	log.Printf("🚀 gRPC 服务器启动，监听端口: %s", addr)

	// ========== 第6步：优雅关闭 ==========
	// 启动服务在 goroutine 中
	go func() {
		if err := grpcServer.Serve(lis); err != nil {
			log.Fatalf("❌ gRPC 服务启动失败: %v", err)
		}
	}()

	// 设置健康检查状态
	healthServer.SetServingStatus("", healthpb.HealthCheckResponse_SERVING)

	// 等待中断信号
	quit := make(chan os.Signal, 1)
	signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
	<-quit

	log.Println("🛑 正在关闭 gRPC 服务器...")
	grpcServer.GracefulStop() // 优雅停止，不中断正在处理的请求
	log.Println("✅ gRPC 服务器已关闭")
}
```

#### 2.3.2 gRPC 处理器 handler/user.go

```go
package handler

import (
	"context"
	"errors"
	"io"
	"time"

	"user-center/internal/model"
	"user-center/internal/repository"
	"user-center/internal/service"
	pb "user-center/proto/user"
	"common"  // common proto 生成的包

	"github.com/golang-jwt/jwt/v5"
	"golang.org/x/crypto/bcrypt"
	"gorm.io/gorm"
)

/*
gRPC 处理器开发指南：

1. 上下文（context.Context）
   - 每个 RPC 方法的第一个参数
   - 用于传递超时、取消信号、metadata
   - 必须检查上下文是否已取消

2. 错误处理
   - 使用 status.Errorf 返回错误
   - 区分业务错误和系统错误
   - 错误应包含足够的调试信息

3. 响应格式
   - 所有响应都包含 CommonResponse
   - 业务数据放在单独的字段
*/

// UserHandler 用户服务处理器
type UserHandler struct {
	pb.UnimplementedUserServiceServer // 嵌入未实现的服务（防止编译错误）
	userService *service.UserService
	jwtSecret   string
}

// NewUserHandler 构造函数
func NewUserHandler(userRepo *repository.UserRepository, jwtSecret string) *UserHandler {
	return &UserHandler{
		userService: service.NewUserService(userRepo),
		jwtSecret:   jwtSecret,
	}
}

// Register 用户注册
// 实现 pb.UserServiceServer 接口
func (h *UserHandler) Register(ctx context.Context, req *pb.RegisterRequest) (*pb.RegisterResponse, error) {
	// ========== 第1步：参数验证 ==========
	// 简单的必填检查
	if req.Username == "" {
		return h.errorResponse(400, "用户名不能为空"), nil
	}
	if req.Email == "" {
		return h.errorResponse(400, "邮箱不能为空"), nil
	}
	if len(req.Password) < 6 {
		return h.errorResponse(400, "密码长度至少6位"), nil
	}

	// ========== 第2步：检查用户是否已存在 ==========
	existing, _ := h.userService.FindByUsername(req.Username)
	if existing != nil {
		return h.errorResponse(409, "用户名已存在"), nil
	}

	existingEmail, _ := h.userService.FindByEmail(req.Email)
	if existingEmail != nil {
		return h.errorResponse(409, "邮箱已被注册"), nil
	}

	// ========== 第3步：密码加密 ==========
	// bcrypt.GenerateFromPassword 加密密码
	// bcrypt.DefaultCost = 10 次迭代
	hashedPassword, err := bcrypt.GenerateFromPassword([]byte(req.Password), bcrypt.DefaultCost)
	if err != nil {
		return h.errorResponse(500, "密码加密失败"), nil
	}

	// ========== 第4步：创建用户 ==========
	user := &model.User{
		Username: req.Username,
		Email:    req.Email,
		Password: string(hashedPassword),
		Nickname: req.Nickname,
		Status:   "active",
	}

	if err := h.userService.Create(user); err != nil {
		return h.errorResponse(500, "创建用户失败"), nil
	}

	// ========== 第5步：生成 Token ==========
	token, _ := h.generateToken(user)

	// ========== 第6步：返回响应 ==========
	return &pb.RegisterResponse{
		Common: &common.CommonResponse{
			Code:    0,
			Message: "注册成功",
			Success: true,
		},
		User: h.toProtoUser(user),
	}, nil
}

// Login 用户登录
func (h *UserHandler) Login(ctx context.Context, req *pb.LoginRequest) (*pb.LoginResponse, error) {
	// 参数验证
	if req.Account == "" || req.Password == "" {
		return h.loginErrorResponse("账号或密码不能为空"), nil
	}

	// 查询用户
	var user *model.User
	var err error

	// 支持用户名或邮箱登录
	if containsEmail(req.Account) {
		user, err = h.userService.FindByEmail(req.Account)
	} else {
		user, err = h.userService.FindByUsername(req.Account)
	}

	if err != nil || user == nil {
		return h.loginErrorResponse("用户不存在"), nil
	}

	// 验证密码
	if err := bcrypt.CompareHashAndPassword([]byte(user.Password), []byte(req.Password)); err != nil {
		return h.loginErrorResponse("密码错误"), nil
	}

	// 检查账号状态
	if user.Status != "active" {
		return h.loginErrorResponse("账号已被禁用"), nil
	}

	// 生成 Token
	token, expiresIn, _ := h.generateTokenWithExpiry(user, 72*time.Hour)

	return &pb.LoginResponse{
		Common: &common.CommonResponse{
			Code:    0,
			Message: "登录成功",
			Success: true,
		},
		AccessToken: token,
		ExpiresIn:   int64(expiresIn.Seconds()),
		TokenType:   "Bearer",
		User:        h.toProtoUser(user),
	}, nil
}

// GetUser 获取单个用户
func (h *UserHandler) GetUser(ctx context.Context, req *pb.GetUserRequest) (*pb.GetUserResponse, error) {
	var user *model.User
	var err error

	// 根据 ID 或用户名查询
	if req.Id > 0 {
		user, err = h.userService.FindByID(req.Id)
	} else if req.Username != "" {
		user, err = h.userService.FindByUsername(req.Username)
	} else {
		return h.errorResponse(400, "请提供用户ID或用户名"), nil
	}

	if err != nil || user == nil {
		return h.errorResponse(404, "用户不存在"), nil
	}

	return &pb.GetUserResponse{
		Common: h.successResponse("获取成功"),
		User:   h.toProtoUser(user),
	}, nil
}

// UpdateUser 更新用户
func (h *UserHandler) UpdateUser(ctx context.Context, req *pb.UpdateUserRequest) (*pb.UpdateUserResponse, error) {
	// TODO: 权限检查 - 只有用户本人或管理员可以修改
	user, err := h.userService.FindByID(req.Id)
	if err != nil || user == nil {
		return h.errorResponse(404, "用户不存在"), nil
	}

	// 更新字段
	if req.Nickname != "" {
		user.Nickname = req.Nickname
	}
	if req.Avatar != "" {
		user.Avatar = req.Avatar
	}
	if req.Email != "" {
		user.Email = req.Email
	}
	if req.Password != "" {
		hashedPassword, _ := bcrypt.GenerateFromPassword([]byte(req.Password), bcrypt.DefaultCost)
		user.Password = string(hashedPassword)
	}

	if err := h.userService.Update(user); err != nil {
		return h.errorResponse(500, "更新失败"), nil
	}

	return &pb.UpdateUserResponse{
		Common: h.successResponse("更新成功"),
		User:   h.toProtoUser(user),
	}, nil
}

// GetUsers 批量获取用户（服务端流）
// 适用于一次返回大量数据的场景
// gRPC 会自动将消息分帧传输，客户端可以边接收边处理
func (h *UserHandler) GetUsers(req *pb.GetUsersRequest, stream pb.UserService_GetUsersServer) error {
	log.Printf("📋 开始批量获取 %d 个用户", len(req.Ids))

	for _, id := range req.Ids {
		// 模拟处理延迟
		// time.Sleep(100 * time.Millisecond)

		user, err := h.userService.FindByID(id)
		if err != nil || user == nil {
			continue // 跳过不存在的用户，或发送错误状态
		}

		// 通过流发送响应
		// 这是服务端流式 RPC 的核心方法
		if err := stream.Send(&pb.GetUserResponse{
			Common: h.successResponse(""),
			User:   h.toProtoUser(user),
		}); err != nil {
			log.Printf("❌ 发送流数据失败: %v", err)
			return err
		}
	}

	log.Printf("✅ 批量获取完成，共发送 %d 个用户", len(req.Ids))
	return nil
}

// ValidateToken 验证 Token（微服务间调用）
// 这是其他微服务调用本服务的典型场景
func (h *UserHandler) ValidateToken(ctx context.Context, req *pb.ValidateTokenRequest) (*pb.ValidateTokenResponse, error) {
	claims, err := h.parseToken(req.Token)
	if err != nil {
		return &pb.ValidateTokenResponse{
			Valid: false,
		}, nil
	}

	return &pb.ValidateTokenResponse{
		Valid:    true,
		UserId:   claims.UserID,
		Username: claims.Username,
	}, nil
}

// ========== 辅助方法 ==========

// generateToken 生成 JWT Token
func (h *UserHandler) generateToken(user *model.User) (string, error) {
	return h.generateTokenWithExpiry(user, 72*time.Hour)
}

// generateTokenWithExpiry 生成带过期时间的 Token
func (h *UserHandler) generateTokenWithExpiry(user *model.User, expiry time.Duration) (string, time.Duration, error) {
	now := time.Now()
	expiresAt := now.Add(expiry)

	claims := jwt.MapClaims{
		"user_id":  user.ID,
		"username": user.Username,
		"exp":      expiresAt.Unix(),
		"iat":      now.Unix(),
		"iss":      "user-center",
	}

	token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
	tokenString, err := token.SignedString([]byte(h.jwtSecret))
	return tokenString, expiry, err
}

// parseToken 解析 Token
func (h *UserHandler) parseToken(tokenString string) (jwt.MapClaims, error) {
	token, err := jwt.Parse(tokenString, func(token *jwt.Token) (interface{}, error) {
		if _, ok := token.Method.(*jwt.SigningMethodHMAC); !ok {
			return nil, errors.New("invalid signing method")
		}
		return []byte(h.jwtSecret), nil
	})

	if err != nil {
		return nil, err
	}

	if claims, ok := token.Claims.(jwt.MapClaims); ok && token.Valid {
		return claims, nil
	}

	return nil, errors.New("invalid token")
}

// toProtoUser 转换为 Protobuf User
func (h *UserHandler) toProtoUser(user *model.User) *pb.User {
	if user == nil {
		return nil
	}
	return &pb.User{
		Id:        user.ID,
		Username:  user.Username,
		Email:     user.Email,
		Nickname:  user.Nickname,
		Avatar:    user.Avatar,
		Status:    user.Status,
		CreatedAt: user.CreatedAt.Unix(),
		UpdatedAt: user.UpdatedAt.Unix(),
	}
}

	}

#### 2.3.3 拦截器 middleware/auth.go

```go
package middleware

import (
	"context"
	"log"
	"time"

	"github.com/golang-jwt/jwt/v5"
	"google.golang.org/grpc"
	"google.golang.org/grpc/metadata"
)

/*
gRPC 拦截器（Interceptor）

类似于 HTTP 中间件，用于在 RPC 调用前后执行通用逻辑

类型：
1. UnaryInterceptor：单次 RPC 调用拦截
2. StreamInterceptor：流式 RPC 调用拦截

常用场景：
- 日志记录
- 认证授权
- 错误处理
- 超时控制
- 链路追踪
*/

// ContextKey 上下文键类型（避免字符串硬编码）
type ContextKey string

const (
	UserIDKey   ContextKey = "user_id"
	UsernameKey ContextKey = "username"
)

// UnaryLoggingInterceptor 日志拦截器
func UnaryLoggingInterceptor() grpc.UnaryServerInterceptor {
	return func(ctx context.Context, req interface{}, info *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (interface{}, error) {
		// info 包含方法信息
		// - FullMethod: 完整方法名，如 /user.UserService/Login
		// - Server: 服务器实例
		// - StatsHandler: 统计处理器
		// - Interceptable: 拦截器链

		start := time.Now()

		// 从 context 获取请求 metadata
		md, _ := metadata.FromIncomingContext(ctx)
		traceID := md.Get("trace-id")

		log.Printf("📥 收到请求: %s | trace_id: %v", info.FullMethod, traceID)

		// 调用下一个处理器
		resp, err := handler(ctx, req)

		// 记录处理时间和结果
		latency := time.Since(start)
		if err != nil {
			log.Printf("📤 请求失败: %s | 耗时: %v | 错误: %v", info.FullMethod, latency, err)
		} else {
			log.Printf("📤 请求成功: %s | 耗时: %v", info.FullMethod, latency)
		}

		return resp, err
	}
}

// RecoveryInterceptor 异常恢复拦截器
// 防止 panic 导致整个服务崩溃
func RecoveryInterceptor() grpc.UnaryServerInterceptor {
	return func(ctx context.Context, req interface{}, info *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (interface{}, error) {
		defer func() {
			if r := recover(); r != nil {
				log.Printf("❌ Panic recovered: %v", r)
				// 可以在这里添加告警逻辑
				// 如：发送告警到监控系统
			}
		}()

		return handler(ctx, req)
	}
}

// TimeoutInterceptor 超时拦截器
func TimeoutInterceptor(timeout time.Duration) grpc.UnaryServerInterceptor {
	return func(ctx context.Context, req interface{}, info *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (interface{}, error) {
		// 创建带超时的 context
		ctx, cancel := context.WithTimeout(ctx, timeout)
		defer cancel()

		return handler(ctx, req)
	}
}

// AuthInterceptor 认证拦截器
// 验证请求头中的 Token
func AuthInterceptor(jwtSecret string) grpc.UnaryServerInterceptor {
	return func(ctx context.Context, req interface{}, info *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (interface{}, error) {
		// 不需要认证的方法（白名单）
		publicMethods := map[string]bool{
			"/user.UserService/Register": true,
			"/user.UserService/Login":    true,
			"/user.UserService/ValidateToken": true, // 微服务间调用可能不需要
		}

		if publicMethods[info.FullMethod] {
			return handler(ctx, req)
		}

		// 获取 metadata
		md, ok := metadata.FromIncomingContext(ctx)
		if !ok {
			return nil, errMissingMetadata
		}

		// 获取 Token
		tokens := md.Get("authorization")
		if len(tokens) == 0 {
			return nil, errMissingToken
		}

		tokenString := tokens[0]
		// 去掉 "Bearer " 前缀
		if len(tokenString) > 7 {
			tokenString = tokenString[7:]
		}

		// 验证 Token
		claims, err := parseToken(tokenString, jwtSecret)
		if err != nil {
			return nil, errInvalidToken
		}

		// 将用户信息注入 context
		ctx = context.WithValue(ctx, UserIDKey, claims["user_id"])
		ctx = context.WithValue(ctx, UsernameKey, claims["username"])

		return handler(ctx, req)
	}
}

// parseToken 解析 JWT Token
func parseToken(tokenString, secret string) (jwt.MapClaims, error) {
	token, err := jwt.Parse(tokenString, func(token *jwt.Token) (interface{}, error) {
		if _, ok := token.Method.(*jwt.SigningMethodHMAC); !ok {
			return nil, errInvalidSigningMethod
		}
		return []byte(secret), nil
	})

	if err != nil {
		return nil, err
	}

	if claims, ok := token.Claims.(jwt.MapClaims); ok && token.Valid {
		return claims, nil
	}

	return nil, errInvalidToken
}

// StreamLoggingInterceptor 流式日志拦截器
func StreamLoggingInterceptor() grpc.StreamServerInterceptor {
	return func(srv interface{}, ss grpc.ServerStream, info *grpc.StreamServerInfo, handler grpc.StreamHandler) error {
		log.Printf("📥 收到流请求: %s", info.FullMethod)
		return handler(srv, ss)
	}
}

// ChainUnaryServer 链式调用多个 Unary 拦截器
// 类似于 Go HTTP 的 Use 方法
func ChainUnaryServer(interceptors ...grpc.UnaryServerInterceptor) grpc.UnaryServerInterceptor {
	return func(ctx context.Context, req interface{}, info *grpc.UnaryServerInfo, handler grpc.UnaryHandler) (interface{}, error) {
		// 递归调用，形成拦截器链
		var i int
		var next grpc.UnaryHandler

		// 最后一个拦截器调用 handler
		next = func(ctx context.Context, req interface{}) (interface{}, error) {
			if i == len(interceptors)-1 {
				return handler(ctx, req)
			}
			i++
			return interceptors[i](ctx, req, info, next)
		}

		// 从第一个拦截器开始
		i = 0
		return interceptors[0](ctx, req, info, next)
	}
}

// ChainStreamServer 链式调用多个 Stream 拦截器
func ChainStreamServer(interceptors ...grpc.StreamServerInterceptor) grpc.StreamServerInterceptor {
	return func(srv interface{}, ss grpc.ServerStream, info *grpc.StreamServerInfo, handler grpc.StreamHandler) error {
		var i int
		var next grpc.StreamHandler

		next = func(srv interface{}, stream grpc.ServerStream) error {
			if i == len(interceptors)-1 {
				return handler(srv, stream)
			}
			i++
			return interceptors[i](srv, stream, info, next)
		}

		i = 0
		return interceptors[0](srv, ss, info, next)
	}
}
