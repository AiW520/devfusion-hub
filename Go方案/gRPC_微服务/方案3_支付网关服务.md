# gRPC 微服务方案三：支付网关服务

## 📚 项目介绍

### 1.1 项目概述

这是一个完整的支付网关微服务，负责对接支付宝、微信支付等第三方支付平台，提供统一的支付接口。项目展示了设计模式（策略模式、工厂模式）、接口隔离原则、支付安全等核心能力。

**核心功能：**
- 统一支付接口
- 支付宝支付集成（当面付、手机网站支付）
- 微信支付集成（Native 支付、JSAPI 支付）
- 支付回调处理
- 退款处理
- 交易查询与对账

### 1.2 核心技术亮点

```
设计模式：
├── 策略模式              # 支付渠道切换
├── 工厂模式              # 支付客户端创建
├── 适配器模式            # 第三方 SDK 适配
├── 模板方法模式          # 支付流程
└── 观察者模式            # 支付通知

安全特性：
├── 签名验证              # 防止篡改
├── 证书管理              # API 密钥
├── 加密传输              # TLS
└── 幂等性设计            # 防止重复扣款
```

### 1.3 适合场景

- 展示设计模式应用能力

#### 2.3.1 支付策略接口 channel/interface.go

```go
package channel

import (
	"context"
	"time"
)

/*
支付渠道接口设计

采用策略模式，支持灵活切换支付渠道。

设计原则：
1. 接口隔离：每个渠道只需实现自己需要的方法
2. 依赖倒置：上层依赖抽象接口，不依赖具体实现
3. 开闭原则：新增渠道不需要修改已有代码
*/

// PaymentRequest 统一支付请求
type PaymentRequest struct {
	OrderNo    string            // 商户订单号
	Subject    string            // 订单标题
	Amount     int64             // 金额（分）
	Currency   string            // 货币类型
	Method     PayMethodType     // 支付方式
	NotifyURL  string            // 回调地址
	ReturnURL  string            // 返回地址
	ClientIP   string            // 客户端IP
	UserID     uint64            // 用户ID
	Extra      map[string]string // 扩展参数
}

// PaymentResponse 统一支付响应
type PaymentResponse struct {
	PaymentNo  string            // 支付流水号
	TradeNo    string            // 第三方交易号
	PayURL     string            // 支付链接/二维码
	PayData    string            // APP调起参数
	ExpireTime time.Time         // 过期时间
}

// RefundRequest 统一退款请求
type RefundRequest struct {
	OrderNo     string // 原订单号
	TradeNo     string // 第三方交易号
	RefundNo    string // 退款单号
	RefundAmount int64  // 退款金额
	Reason      string // 退款原因
}

// RefundResponse 统一退款响应
type RefundResponse struct {
	RefundNo     string     // 退款单号
	RefundAmount int64      // 退款金额
	Status       int        // 退款状态
	RefundTime   time.Time  // 退款时间
}

// QueryRequest 统一查询请求
type QueryRequest struct {
	OrderNo string // 商户订单号
	TradeNo string // 第三方交易号
}

// QueryResponse 统一查询响应
type QueryResponse struct {
	PaymentNo string    // 支付流水号
	TradeNo   string    // 第三方交易号
	Status    int       // 支付状态
	Amount    int64      // 支付金额
	PaidTime  time.Time // 支付时间
}

// PayMethodType 支付方式类型
type PayMethodType int

const (
	// 支付宝
	PayMethodAlipayApp PayMethodType = iota + 1
	PayMethodAlipayWap
	PayMethodAlipayWeb
	PayMethodAlipayQR

	// 微信
	PayMethodWechatApp
	PayMethodWechatJSAPI
	PayMethodWechatNative
	PayMethodWechatH5
)

// PaymentChannel 支付渠道接口
// 定义所有支付渠道必须实现的方法
type PaymentChannel interface {
	// 获取渠道名称
	Name() string

	// 发起支付
	Pay(ctx context.Context, req *PaymentRequest) (*PaymentResponse, error)

	// 支付回调验证与解析
	// 返回解析后的支付结果
	ParseNotify(ctx context.Context, notifyData map[string]string) (*NotifyResult, error)

	// 申请退款
	Refund(ctx context.Context, req *RefundRequest) (*RefundResponse, error)

	// 退款回调验证与解析
	ParseRefundNotify(ctx context.Context, notifyData map[string]string) (*RefundNotifyResult, error)

	// 查询支付状态
	QueryPayment(ctx context.Context, req *QueryRequest) (*QueryResponse, error)

	// 查询退款状态
	QueryRefund(ctx context.Context, refundNo string) (*RefundResponse, error)

	// 关闭交易
	ClosePayment(ctx context.Context, orderNo string) error

	// 验证签名
	VerifySign(params map[string]string) bool

	// 构建签名
	BuildSign(params map[string]string) string
}

// NotifyResult 支付回调结果
type NotifyResult struct {
	OrderNo      string    // 商户订单号
	TradeNo      string    // 第三方交易号
	PaymentNo    string    // 支付流水号
	Status       int       // 支付状态
	Amount       int64     // 支付金额
	PaidTime     time.Time // 支付时间
	RawData      string    // 原始回调数据
}

// RefundNotifyResult 退款回调结果
type RefundNotifyResult struct {
	RefundNo     string    // 退款单号
	TradeNo      string    // 第三方交易号
	RefundAmount int64     // 退款金额
	Status       int       // 退款状态
	RefundTime   time.Time // 退款时间
	RawData      string    // 原始回调数据
}

// BaseChannel 基类，提供公共方法
// 使用组合而非继承，实现代码复用
type BaseChannel struct {
	AppID     string // 应用ID
	MerchantID string // 商户号
	AppKey    string // 应用密钥
}

// Name 返回渠道名称（子类覆盖）
func (b *BaseChannel) Name() string {
	return "base"
}

// GetExtraParam 获取扩展参数
func GetExtraParam(req *PaymentRequest, key string) string {
	if req.Extra == nil {
		return ""
	}
	return req.Extra[key]
}

// ToFen 金额转换（元转分）
func ToFen(yuan float64) int64 {
	return int64(yuan * 100)
}

// ToYuan 分转元
func ToYuan(fen int64) float64 {
	return float64(fen) / 100
}
```

#### 2.3.2 支付宝渠道实现 channel/alipay.go

```go
package channel

import (
	"context"
	"crypto/sha256"
	"encoding/json"
	"fmt"
	"net/url"
	"sort"
	"strings"
	"time"

	"github.com/smartwalle/alipay/v3" // 支付宝 Go SDK
)

/*
支付宝支付渠道实现

支持的支付方式：
- 当面付（扫码）
- 手机网站支付
- APP支付

回调处理：
- 使用 RSA2 签名验证
- 异步回调通知
*/

// AlipayChannel 支付宝渠道
type AlipayChannel struct {
	BaseChannel              // 嵌入基类
	client       *alipay.AliPay // 支付宝客户端
}

// NewAlipayChannel 创建支付宝渠道
func NewAlipayChannel(appID, merchantID, privateKey, alipayPublicKey string) (*AlipayChannel, error) {
	// 创建支付宝客户端
	client, err := alipay.New(appID, merchantID, privateKey, alipayPublicKey)
	if err != nil {
		return nil, fmt.Errorf("初始化支付宝客户端失败: %w", err)
	}

	// 设置为沙箱环境（开发测试用）
	// 生产环境删除这行
	client.SetSandbox(true)

	// 设置签名类型为 RSA2
	client.SetAliPayPublicKey(alipayPublicKey)

	return &AlipayChannel{
		BaseChannel: BaseChannel{
			AppID:     appID,
			MerchantID: merchantID,
			AppKey:    privateKey,
		},
		client: client,
	}, nil
}

// Name 返回渠道名称
func (a *AlipayChannel) Name() string {
	return "alipay"
}

// Pay 发起支付
func (a *AlipayChannel) Pay(ctx context.Context, req *PaymentRequest) (*PaymentResponse, error) {
	// 根据支付方式选择不同的接口
	switch req.Method {
	case PayMethodAlipayQR:
		return a.payQR(ctx, req)
	case PayMethodAlipayWap:
		return a.payWap(ctx, req)
	case PayMethodAlipayApp:
		return a.payApp(ctx, req)
	default:
		return nil, fmt.Errorf("不支持的支付方式: %v", req.Method)
	}
}

// payQR 扫码支付（当面付）
func (a *AlipayChannel) payQR(ctx context.Context, req *PaymentRequest) (*PaymentResponse, error) {
	// 创建交易请求
	param := alipay.TradePreCreate{}
	param.Subject = req.Subject
	param.OutTradeNo = req.OrderNo
	param.TotalAmount = fmt.Sprintf("%.2f", ToYuan(req.Amount))
	param.TimeoutExpress = "30m" // 30分钟过期

	// 调用当面付预下单接口
	resp, err := a.client.TradePrecreate(param)
	if err != nil {
		return nil, fmt.Errorf("预下单失败: %w", err)
	}

	// 检查响应
	if !resp.Response.Success {
		return nil, fmt.Errorf("预下单失败: %s", resp.Response.SubMsg)
	}

	// 返回二维码链接
	return &PaymentResponse{
		PaymentNo:  generatePaymentNo("AL"),
		TradeNo:    resp.Response.TradeNo,
		PayURL:     resp.Response.QRCode,
		ExpireTime: time.Now().Add(30 * time.Minute),
	}, nil
}

// payWap 手机网站支付
func (a *AlipayChannel) payWap(ctx context.Context, req *PaymentRequest) (*PaymentResponse, error) {
	param := alipay.TradeWapPay{}
	param.Subject = req.Subject
	param.OutTradeNo = req.OrderNo
	param.TotalAmount = fmt.Sprintf("%.2f", ToYuan(req.Amount))
	param.TimeoutExpress = "30m"
	param.ProductCode = "QUICK_WAP_WAY"
	param.ReturnURL = req.ReturnURL

	// 生成支付链接
	payURL, err := a.client.TradeWapPay(param)
	if err != nil {
		return nil, fmt.Errorf("生成支付链接失败: %w", err)
	}

	return &PaymentResponse{
		PaymentNo:  generatePaymentNo("AL"),
		PayURL:     payURL.String(),
		ExpireTime: time.Now().Add(30 * time.Minute),
	}, nil
}

// payApp APP支付
func (a *AlipayChannel) payApp(ctx context.Context, req *PaymentRequest) (*PaymentResponse, error) {
	param := alipay.TradeAppPay{}
	param.Subject = req.Subject
	param.OutTradeNo = req.OrderNo
	param.TotalAmount = fmt.Sprintf("%.2f", ToYuan(req.Amount))
	param.TimeoutExpress = "30m"
	param.ProductCode = "QUICK_MSECURITY_PAY"

	// 生成签名后的参数字符串
	// 客户端使用这个参数调起支付宝
	payParam, err := a.client.TradeAppPay(param)
	if err != nil {
		return nil, fmt.Errorf("生成支付参数失败: %w", err)
	}

	return &PaymentResponse{
		PaymentNo:  generatePaymentNo("AL"),
		PayData:    payParam.String(),
		ExpireTime: time.Now().Add(30 * time.Minute),
	}, nil
}

// ParseNotify 解析支付回调
func (a *AlipayChannel) ParseNotify(ctx context.Context, notifyData map[string]string) (*NotifyResult, error) {
	// 验证签名
	verified := a.VerifySign(notifyData)
	if !verified {
		return nil, fmt.Errorf("签名验证失败")
	}

	// 解析回调类型
	notifyType := notifyData["notify_type"]
	
	var result *NotifyResult
	
	if notifyType == "trade_status_sync" {
		// 交易状态通知
		tradeStatus := notifyData["trade_status"]
		result = &NotifyResult{
			OrderNo:   notifyData["out_trade_no"],
			TradeNo:   notifyData["trade_no"],
			Amount:    ToFen(parseFloat(notifyData["receipt_amount"])),
			PaidTime:  parseTime(notifyData["gmt_payment"]),
			RawData:   notifyData["notify_data"],
		}

		// 转换交易状态
		switch tradeStatus {
		case "TRADE_SUCCESS", "TRADE_FINISHED":
			result.Status = PaymentStatusPaid
		case "WAIT_BUYER_PAY":
			result.Status = PaymentStatusPending
		case "TRADE_CLOSED":
			result.Status = PaymentStatusClosed
		}
	}

	return result, nil
}

// Refund 申请退款
func (a *AlipayChannel) Refund(ctx context.Context, req *RefundRequest) (*RefundResponse, error) {
	param := alipay.TradeRefund{}
	param.TradeNo = req.TradeNo
	param.RefundAmount = fmt.Sprintf("%.2f", ToYuan(req.RefundAmount))
	param.OutRequestNo = req.RefundNo
	param.RefundReason = req.Reason

	resp, err := a.client.TradeRefund(param)
	if err != nil {
		return nil, fmt.Errorf("退款失败: %w", err)
	}

	if !resp.Response.Success {
		return nil, fmt.Errorf("退款失败: %s", resp.Response.SubMsg)
	}

	return &RefundResponse{
		RefundNo:     req.RefundNo,
		RefundAmount: req.RefundAmount,
		Status:       RefundStatusSuccess,
		RefundTime:   time.Now(),
	}, nil
}

// ParseRefundNotify 解析退款回调
func (a *AlipayChannel) ParseRefundNotify(ctx context.Context, notifyData map[string]string) (*RefundNotifyResult, error) {
	// 验证签名
	if !a.VerifySign(notifyData) {
		return nil, fmt.Errorf("签名验证失败")
	}

	notifyDataJSON := notifyData["notify_data"]
	var data map[string]interface{}
	json.Unmarshal([]byte(notifyDataJSON), &data)

	return &RefundNotifyResult{
		RefundNo:     data["out_request_no"].(string),
		TradeNo:      data["trade_no"].(string),
		RefundAmount: ToFen(data["refund_amount"].(float64)),
		Status:       RefundStatusSuccess,
		RefundTime:   time.Now(),
		RawData:      notifyDataJSON,
	}, nil
}

// QueryPayment 查询支付状态
func (a *AlipayChannel) QueryPayment(ctx context.Context, req *QueryRequest) (*QueryResponse, error) {
	param := alipay.TradeQuery{}
	param.TradeNo = req.TradeNo
	param.OutTradeNo = req.OrderNo

	resp, err := a.client.TradeQuery(param)
	if err != nil {
		return nil, fmt.Errorf("查询失败: %w", err)
	}

	result := &QueryResponse{
		TradeNo: resp.Response.TradeNo,
		Amount:  ToFen(parseFloat(resp.Response.TotalAmount)),
	}

	// 转换状态
	switch resp.Response.TradeStatus {
	case "TRADE_SUCCESS", "TRADE_FINISHED":
		result.Status = PaymentStatusPaid
		result.PaidTime = parseTime(resp.Response.SendPayDate)
	case "WAIT_BUYER_PAY":
		result.Status = PaymentStatusPending
	case "TRADE_CLOSED":
		result.Status = PaymentStatusClosed
	}

	return result, nil
}

// QueryRefund 查询退款状态
func (a *AlipayChannel) QueryRefund(ctx context.Context, refundNo string) (*RefundResponse, error) {
	param := alipay.TradeFastpayRefundQuery{}
	param.OutRequestNo = refundNo
	param.TradeNo = ""

	resp, err := a.client.TradeFastpayRefundQuery(param)
	if err != nil {
		return nil, fmt.Errorf("查询失败: %w", err)
	}

	return &RefundResponse{
		RefundNo:     refundNo,
		RefundAmount: ToFen(parseFloat(resp.Response.RefundAmount)),
		Status:       RefundStatusSuccess,
	}, nil
}

// ClosePayment 关闭交易
func (a *AlipayChannel) ClosePayment(ctx context.Context, orderNo string) error {
	param := alipay.TradeClose{}
	param.OutTradeNo = orderNo

	resp, err := a.client.TradeClose(param)
	if err != nil {
		return fmt.Errorf("关闭交易失败: %w", err)
	}

	if !resp.Response.Success {
		return fmt.Errorf("关闭交易失败: %s", resp.Response.SubMsg)
	}

	return nil
}

// BuildSign 构建签名（示例）
func (a *AlipayChannel) BuildSign(params map[string]string) string {
	// 按照支付宝签名规则构建签名
	// 1. 移除 sign、sign_type
	// 2. 按字典序排列参数
	// 3. 使用 & 连接
	// 4. 使用 RSA2 签名
	
	delete(params, "sign")
	delete(params, "sign_type")
	
	// 排序
	keys := make([]string, 0, len(params))
	for k := range params {
		keys = append(keys, k)
	}
	sort.Strings(keys)
	
	// 拼接
	var signData strings.Builder
	for _, k := range keys {
		if signData.Len() > 0 {
			signData.WriteByte('&')
		}
		signData.WriteString(k)
		signData.WriteByte('=')
		signData.WriteString(params[k])
	}
	
	// 使用私钥签名
	// 实际使用 crypto/rsa 包
	return ""
}

	keys := make([]string, 0, len(params))
```

```go
支付服务

职责：
1. 统一支付入口，屏蔽渠道差异
2. 支付流程编排
3. 回调处理
4. 交易记录管理

// PaymentService 支付服务
type PaymentService struct {
	repo          *repository.PaymentRepository
	channelMgr    *channel.ChannelManager
	notifyService *NotifyService
}

// NewPaymentService 创建支付服务
func NewPaymentService(
	repo *repository.PaymentRepository,
	channelMgr *channel.ChannelManager,
) *PaymentService {
	return &PaymentService{
		repo:       repo,
		channelMgr: channelMgr,
	}
}

// Pay 发起支付
func (s *PaymentService) Pay(ctx context.Context, req *pb.PayRequest) (*pb.PayResponse, error) {
	// ========== 第1步：参数验证 ==========
	if req.Amount <= 0 {
		return &pb.PayResponse{
			Code:    400,
			Message: "金额必须大于0",
			Success: false,
		}, nil
	}

	// ========== 第2步：检查重复订单 ==========
	existing, _ := s.repo.FindByOrderNo(req.OrderNo)
	if existing != nil && existing.Status != model.PaymentStatusClosed {
		return &pb.PayResponse{
			Code:    409,
			Message: "订单已存在，请勿重复提交",
			Success: false,
		}, nil
	}

	// ========== 第3步：创建支付记录 ==========
	payment := &model.Payment{
		OrderNo:     req.OrderNo,
		Channel:     int(req.Channel),
		Method:      int(req.Method),
		Amount:      req.Amount,
		Currency:    req.Currency,
		Status:      model.PaymentStatusPending,
		NotifyURL:   req.NotifyUrl,
		ReturnURL:   req.ReturnUrl,
		ClientIP:    req.ClientIp,
		UserID:      req.UserId,
		Subject:     req.Subject,
		ExpireTime:  time.Now().Add(30 * time.Minute),
	}

	if err := s.repo.Create(payment); err != nil {
		return nil, fmt.Errorf("创建支付记录失败: %w", err)
	}

	// ========== 第4步：调用支付渠道 ==========
	channelInstance := s.channelMgr.GetChannel(channel.PaymentChannel(req.Channel))
	if channelInstance == nil {
		return &pb.PayResponse{
			Code:    400,
			Message: "不支持的支付渠道",
			Success: false,
		}, nil
	}

	// 构建渠道请求
	channelReq := &channel.PaymentRequest{
		OrderNo:   req.OrderNo,
		Subject:   req.Subject,
		Amount:    req.Amount,
		Currency:  req.Currency,
		Method:    channel.PayMethodType(req.Method),
		NotifyURL: req.NotifyUrl,
		ReturnURL: req.ReturnUrl,
		ClientIP:  req.ClientIp,
		UserID:    req.UserId,
	}

	// 调用渠道支付
	channelResp, err := channelInstance.Pay(ctx, channelReq)
	if err != nil {
		// 更新支付状态为失败
		payment.Status = model.PaymentStatusClosed
		payment.Remark = err.Error()
		s.repo.Update(payment)

		return &pb.PayResponse{
			Code:    500,
			Message: fmt.Sprintf("发起支付失败: %v", err),
			Success: false,
		}, nil
	}

	// ========== 第5步：更新支付记录 ==========
	payment.PaymentNo = channelResp.PaymentNo
	payment.TradeNo = channelResp.TradeNo
	payment.TradeData = channelResp.PayData
	payment.PayURL = channelResp.PayURL
	payment.ExpireTime = channelResp.ExpireTime
	s.repo.Update(payment)

	// ========== 第6步：返回响应 ==========
	return &pb.PayResponse{
		Code:         0,
		Message:      "支付发起成功",
		Success:      true,
		PaymentNo:    payment.PaymentNo,
		TradeNo:      channelResp.TradeNo,
		PayUrl:       channelResp.PayURL,
		PayData:      channelResp.PayData,
		ExpireTime:   channelResp.ExpireTime.Unix(),
	}, nil
}

// HandleNotify 处理支付回调
func (s *PaymentService) HandleNotify(ctx context.Context, channelType int, notifyData map[string]string) error {
	// 获取支付渠道
	channelInstance := s.channelMgr.GetChannel(channel.PaymentChannel(channelType))
	if channelInstance == nil {
		return fmt.Errorf("不支持的支付渠道")
	}

	// 解析回调
	notifyResult, err := channelInstance.ParseNotify(ctx, notifyData)
	if err != nil {
		return fmt.Errorf("解析回调失败: %w", err)
	}

	// 更新支付状态
	payment, err := s.repo.FindByOrderNo(notifyResult.OrderNo)
	if err != nil || payment == nil {
		return fmt.Errorf("订单不存在")
	}

	// 状态检查（幂等性）
	if payment.Status == model.PaymentStatusPaid {
		return nil // 已处理，直接返回
	}

	// 更新状态
	payment.TradeNo = notifyResult.TradeNo
	payment.PaidAmount = notifyResult.Amount
	payment.PaidTime = &notifyResult.PaidTime
	payment.Status = model.PaymentStatus(notifyResult.Status)
	payment.RawNotifyData = notifyResult.RawData

	if err := s.repo.Update(payment); err != nil {
		return fmt.Errorf("更新支付状态失败: %w", err)
	}

	// 发送通知（通知订单服务等）
	// 这里使用事件驱动，实际可用消息队列
	s.notifyService.NotifyPaymentSuccess(ctx, payment)

	return nil
}

// Query 查询支付状态
func (s *PaymentService) Query(ctx context.Context, req *pb.QueryRequest) (*pb.QueryResponse, error) {
	payment, err := s.repo.FindByOrderNo(req.OrderNo)
	if err != nil || payment == nil {
		return &pb.QueryResponse{
			Code:    404,
			Message: "订单不存在",
			Success: false,
		}, nil
	}

	var paidAt int64
	if payment.PaidTime != nil {
		paidAt = payment.PaidTime.Unix()
	}

	return &pb.QueryResponse{
		Code:       0,
		Message:    "查询成功",
		Success:    true,
		PaymentNo:  payment.PaymentNo,
		TradeNo:    payment.TradeNo,
		Status:     int32(payment.Status),
		Amount:     payment.Amount,
		PaidAt:     paidAt,
	}, nil
}

// Refund 申请退款
func (s *PaymentService) Refund(ctx context.Context, req *pb.RefundRequest) (*pb.RefundResponse, error) {
	// 查询原支付记录
	payment, err := s.repo.FindByOrderNo(req.OrderNo)
	if err != nil || payment == nil {
		return &pb.RefundResponse{
			Code:    404,
			Message: "原订单不存在",
			Success: false,
		}, nil
	}

	// 检查支付状态
	if payment.Status != model.PaymentStatusPaid {
		return &pb.RefundResponse{
			Code:    400,
			Message: "只有已支付的订单可以退款",
			Success: false,
		}, nil
	}

	// 检查退款金额
	if req.RefundAmount > payment.PaidAmount {
		return &pb.RefundResponse{
			Code:    400,
			Message: "退款金额不能超过支付金额",
			Success: false,
		}, nil
	}

	// 创建退款记录
	refund := &model.Refund{
		OrderNo:     req.OrderNo,
		RefundNo:   req.RefundNo,
		PaymentID:  payment.ID,
		Amount:      req.RefundAmount,
		Reason:      req.Reason,
		Status:      model.RefundStatusPending,
	}
	s.repo.CreateRefund(refund)

	// 调用渠道退款
	channelInstance := s.channelMgr.GetChannel(channel.PaymentChannel(payment.Channel))
	channelResp, err := channelInstance.Refund(ctx, &channel.RefundRequest{
		OrderNo:      req.OrderNo,
		TradeNo:      payment.TradeNo,
		RefundNo:     req.RefundNo,
		RefundAmount: req.RefundAmount,
		Reason:       req.Reason,
	})

	if err != nil {
		refund.Status = model.RefundStatusFailed
		refund.Remark = err.Error()
		s.repo.UpdateRefund(refund)

		return &pb.RefundResponse{
			Code:    500,
			Message: fmt.Sprintf("退款失败: %v", err),
			Success: false,
		}, nil
	}

	// 更新退款记录
	refund.Status = model.RefundStatus(channelResp.Status)
	refund.TradeRefundNo = channelResp.RefundTime.Format("20060102150405")
	refund.ProcessedAt = &channelResp.RefundTime
	s.repo.UpdateRefund(refund)

	// 如果全额退款，更新支付状态
	if req.RefundAmount == payment.PaidAmount {
		payment.Status = model.PaymentStatusRefunded
		s.repo.Update(payment)
	} else {
		payment.Status = model.PaymentStatusRefunding
		s.repo.Update(payment)
	}

	return &pb.RefundResponse{
		Code:         0,
		Message:      "退款申请成功",
		Success:      true,
		RefundNo:     req.RefundNo,
		RefundAmount: req.RefundAmount,
		Status:       int32(channelResp.Status),
	}, nil
}

// ClosePayment 关闭交易
func (s *PaymentService) ClosePayment(ctx context.Context, req *pb.ClosePaymentRequest) (*pb.ClosePaymentResponse, error) {
	payment, err := s.repo.FindByOrderNo(req.OrderNo)
	if err != nil || payment == nil {
		return &pb.ClosePaymentResponse{
			Code:    404,
			Message: "订单不存在",
			Success: false,
		}, nil
	}

	// 只有待支付订单可以关闭
	if payment.Status != model.PaymentStatusPending {
		return &pb.ClosePaymentResponse{
			Code:    400,
			Message: "只有待支付订单可以关闭",
			Success: false,
		}, nil
	}

	// 调用渠道关闭
	channelInstance := s.channelMgr.GetChannel(channel.PaymentChannel(payment.Channel))
	if err := channelInstance.ClosePayment(ctx, req.OrderNo); err != nil {
		return &pb.ClosePaymentResponse{
			Code:    500,
			Message: fmt.Sprintf("关闭失败: %v", err),
			Success: false,
		}, nil
	}

	// 更新状态
	payment.Status = model.PaymentStatusClosed
	s.repo.Update(payment)

	return &pb.ClosePaymentResponse{
		Code:    0,
		Message: "关闭成功",
		Success: true,
	}, nil
}
```
