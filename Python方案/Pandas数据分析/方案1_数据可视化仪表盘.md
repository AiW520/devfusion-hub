### 1.1 项目概述
本项目是一个功能完整的**数据可视化仪表盘**，支持上传 CSV/Excel 数据文件，自动进行数据清洗和预处理，提供丰富的可视化图表（折线图、柱状图、饼图、散点图等），支持拖拽式报表设计和数据导出。

### 1.2 核心技术亮点
- **Pandas 数据处理**：数据清洗、转换、聚合
- **ECharts 可视化**：丰富的图表类型
- **拖拽式报表**：可视化配置
- **数据导出**：Excel/PDF 报告
- **缓存优化**：常用查询结果缓存

### 1.3 适用场景

- 运营数据监控
- 财务报表

# backend/services/data_processor.py
"""
数据处理服务

使用 Pandas 进行数据处理和分析
"""
import pandas as pd
import numpy as np
from typing import Dict, List, Any, Optional, Tuple
from datetime import datetime
import json


class DataProcessor:
    """
    数据处理器
    
    功能：
    1. 文件读取（CSV、Excel）
    2. 数据清洗
    3. 数据转换
    4. 统计分析
    5. 数据聚合
    """
    
    def __init__(self, df: pd.DataFrame = None):
        """初始化"""
        self.df = df
        self.original_df = df.copy() if df is not None else None
        self.metadata = self._generate_metadata()
    
    @classmethod
    def from_file(cls, file_path: str, **kwargs) -> 'DataProcessor':
        """
        从文件创建处理器
        
        Args:
            file_path: 文件路径
            **kwargs: pandas read 参数
            
        Returns:
            DataProcessor 实例
        """
        if file_path.endswith('.csv'):
            df = pd.read_csv(file_path, **kwargs)
        elif file_path.endswith(('.xlsx', '.xls')):
            df = pd.read_excel(file_path, **kwargs)
        else:
            raise ValueError(f"不支持的文件格式: {file_path}")
        
        return cls(df)
    
    @classmethod
    def from_dict(cls, data: Dict, **kwargs) -> 'DataProcessor':
        """从字典创建"""
        df = pd.DataFrame(data)
        return cls(df)
    
    def _generate_metadata(self) -> Dict:
        """生成数据元信息"""
        if self.df is None:
            return {}
        
        metadata = {
            'row_count': len(self.df),
            'column_count': len(self.df.columns),
            'columns': [],
            'dtypes': {},
            'memory_usage': self.df.memory_usage(deep=True).sum()
        }
        
        for col in self.df.columns:
            col_info = {
                'name': col,
                'dtype': str(self.df[col].dtype),
                'null_count': int(self.df[col].isnull().sum()),
                'unique_count': int(self.df[col].nunique()),
                'is_numeric': pd.api.types.is_numeric_dtype(self.df[col])
            }
            
            # 数值类型统计
            if col_info['is_numeric']:
                col_info['min'] = float(self.df[col].min()) if not pd.isna(self.df[col].min()) else None
                col_info['max'] = float(self.df[col].max()) if not pd.isna(self.df[col].max()) else None
                col_info['mean'] = float(self.df[col].mean()) if not pd.isna(self.df[col].mean()) else None
                col_info['median'] = float(self.df[col].median()) if not pd.isna(self.df[col].median()) else None
            
            metadata['columns'].append(col_info)
            metadata['dtypes'][col] = col_info['dtype']
        
        return metadata
    
    # ============ 数据清洗 ============
    
    def handle_missing(self, strategy: str = 'drop', fill_value: Any = None) -> 'DataProcessor':
        """
        处理缺失值
        
        Args:
            strategy: 处理策略
                - 'drop': 删除包含缺失值的行
                - 'fill': 用指定值填充
                - 'forward': 前向填充
                - 'backward': 后向填充
                - 'mean': 数值列用均值填充
                - 'median': 数值列用中位数填充
                - 'mode': 用众数填充
            fill_value: 填充值（strategy='fill' 时使用）
        """
        df = self.df.copy()
        
        if strategy == 'drop':
            df = df.dropna()
        elif strategy == 'fill':
            df = df.fillna(fill_value)
        elif strategy == 'forward':
            df = df.fillna(method='ffill')
        elif strategy == 'backward':
            df = df.fillna(method='bfill')
        elif strategy == 'mean':
            numeric_cols = df.select_dtypes(include=[np.number]).columns
            df[numeric_cols] = df[numeric_cols].fillna(df[numeric_cols].mean())
        elif strategy == 'median':
            numeric_cols = df.select_dtypes(include=[np.number]).columns
            df[numeric_cols] = df[numeric_cols].fillna(df[numeric_cols].median())
        elif strategy == 'mode':
            for col in df.columns:
                if df[col].isnull().any():
                    df[col] = df[col].fillna(df[col].mode()[0] if len(df[col].mode()) > 0 else None)
        
        self.df = df
        self.metadata = self._generate_metadata()
        return self
    
    def remove_duplicates(self, subset: List[str] = None, keep: str = 'first') -> 'DataProcessor':
        """
        删除重复行
        
        Args:
            subset: 用于判断重复的列
            keep: 保留策略 ('first', 'last', False)
        """
        self.df = self.df.drop_duplicates(subset=subset, keep=keep)
        self.metadata = self._generate_metadata()
        return self
    
    def remove_outliers(self, column: str, method: str = 'iqr', threshold: float = 1.5) -> 'DataProcessor':
        """
        移除异常值
        
        Args:
            column: 列名
            method: 方法
                - 'iqr': 四分位距法
                - 'zscore': Z-score 法
            threshold: 阈值
        """
        df = self.df.copy()
        
        if method == 'iqr':
            Q1 = df[column].quantile(0.25)
            Q3 = df[column].quantile(0.75)
            IQR = Q3 - Q1
            lower = Q1 - threshold * IQR
            upper = Q3 + threshold * IQR
            df = df[(df[column] >= lower) & (df[column] <= upper)]
        elif method == 'zscore':
            z_scores = np.abs((df[column] - df[column].mean()) / df[column].std())
            df = df[z_scores <= threshold]
        
        self.df = df
        self.metadata = self._generate_metadata()
        return self
    
    def convert_dtype(self, column: str, dtype: str) -> 'DataProcessor':
        """转换数据类型"""
        self.df[column] = self.df[column].astype(dtype)
        self.metadata = self._generate_metadata()
        return self
    
    # ============ 数据转换 ============
    
    def rename_columns(self, mapping: Dict[str, str]) -> 'DataProcessor':
        """重命名列"""
        self.df = self.df.rename(columns=mapping)
        return self
    
    def select_columns(self, columns: List[str]) -> 'DataProcessor':
        """选择列"""
        self.df = self.df[columns]
        return self
    
    def filter_rows(self, condition: str) -> 'DataProcessor':
        """
        过滤行
        
        Args:
            condition: 过滤条件表达式
        """
        self.df = self.df.query(condition)
        return self
    
    def add_column(self, name: str, value: Any, expression: str = None) -> 'DataProcessor':
        """
        添加列
        
        Args:
            name: 列名
            value: 固定值
            expression: 计算表达式
        """
        if expression:
            self.df[name] = self.df.eval(expression)
        else:
            self.df[name] = value
        return self
    
    def group_aggregate(
        self,
        group_by: List[str],
        aggregations: Dict[str, List[str]]
    ) -> pd.DataFrame:
        """
        分组聚合
        
        Args:
            group_by: 分组列
            aggregations: 聚合配置
                {
                    'sales': ['sum', 'mean', 'count'],
                    'profit': ['sum', 'max']
                }
                
        Returns:
            聚合后的 DataFrame
        """
        agg_dict = {}
        for col, funcs in aggregations.items():
            agg_dict[col] = funcs
        
        result = self.df.groupby(group_by).agg(agg_dict).reset_index()
        
        # 扁平化列名
        result.columns = ['_'.join(col).strip('_') if isinstance(col, tuple) else col 
                         for col in result.columns]
        
        return result
    
    def pivot_table(
        self,
        index: str,
        columns: str,
        values: str,
        aggfunc: str = 'sum'
    ) -> pd.DataFrame:
        """透视表"""
        return pd.pivot_table(
            self.df,
            index=index,
            columns=columns,
            values=values,
            aggfunc=aggfunc
        ).reset_index()
    
    def time_series_resample(self, date_column: str, freq: str) -> 'DataProcessor':
        """
        时间序列重采样
        
        Args:
            date_column: 日期列名
            freq: 频率 ('D', 'W', 'M', 'Q', 'Y')
        """
        self.df[date_column] = pd.to_datetime(self.df[date_column])
        self.df = self.df.set_index(date_column)
        self.df = self.df.resample(freq).sum().reset_index()
        return self
    
    # ============ 统计分析 ============
    
    def descriptive_stats(self, column: str = None) -> Dict:
        """
        描述性统计
        
        Returns:
            统计指标字典
        """
        if column:
            data = self.df[column]
        else:
            data = self.df
        
        if pd.api.types.is_numeric_dtype(data):
            return {
                'count': int(data.count()),
                'mean': float(data.mean()),
                'std': float(data.std()),
                'min': float(data.min()),
                'q25': float(data.quantile(0.25)),
                'q50': float(data.quantile(0.50)),
                'q75': float(data.quantile(0.75)),
                'max': float(data.max()),
                'skewness': float(data.skew()),
                'kurtosis': float(data.kurtosis())
            }
        else:
            return {
                'count': int(data.count()),
                'unique': int(data.nunique()),
                'mode': data.mode().tolist(),
                'value_counts': data.value_counts().head(10).to_dict()
            }
    
    def correlation_matrix(self, method: str = 'pearson') -> pd.DataFrame:
        """相关系数矩阵"""
        numeric_df = self.df.select_dtypes(include=[np.number])
        return numeric_df.corr(method=method)
    
    def distribution(self, column: str, bins: int = 20) -> Dict:
        """分布统计"""
        data = self.df[column]
        
        hist, bin_edges = np.histogram(data.dropna(), bins=bins)
        
        return {
            'histogram': [int(x) for x in hist],
            'bin_edges': [float(x) for x in bin_edges],
            'bins_count': bins
        }
    
    # ============ 导出 ============
    
    def to_dict(self, orient: str = 'records') -> List[Dict]:
        """导出为字典列表"""
        return self.df.to_dict(orient=orient)
    
    def to_csv(self, path: str, **kwargs):
        """导出为 CSV"""
        self.df.to_csv(path, index=False, **kwargs)
    
    def to_excel(self, path: str, **kwargs):
        """导出为 Excel"""
        self.df.to_excel(path, index=False, **kwargs)
    
    def to_json(self, path: str = None, **kwargs):
        """导出为 JSON"""
        return self.df.to_json(path, orient='records', force_ascii=False, **kwargs)


class ChartDataGenerator:
    """
    图表数据生成器
    
    根据图表类型生成 ECharts 格式的数据
    """
    
    def __init__(self, df: pd.DataFrame):
        self.df = df
    
    def line_chart(
        self,
        x_column: str,
        y_columns: List[str],
        title: str = None
    ) -> Dict:
        """折线图数据"""
        series = []
        for col in y_columns:
            series.append({
                'name': col,
                'type': 'line',
                'data': self.df[[x_column, col]].values.tolist()
            })
        
        return {
            'title': {'text': title or '折线图'},
            'xAxis': {
                'type': 'category',
                'data': self.df[x_column].tolist()
            },
            'yAxis': {'type': 'value'},
            'series': series,
            'tooltip': {'trigger': 'axis'}
        }
    
    def bar_chart(
        self,
        x_column: str,
        y_column: str,
        title: str = None
    ) -> Dict:
        """柱状图数据"""
        return {
            'title': {'text': title or '柱状图'},
            'tooltip': {'trigger': 'axis'},
            'xAxis': {
                'type': 'category',
                'data': self.df[x_column].tolist(),
                'axisLabel': {'rotate': 45}
            },
            'yAxis': {'type': 'value'},
            'series': [{
                'name': y_column,
                'type': 'bar',
                'data': self.df[y_column].tolist(),
                'itemStyle': {
                    'color': '#5470c6'
                }
            }]
        }
    
    def pie_chart(
        self,
        name_column: str,
        value_column: str,
        title: str = None
    ) -> Dict:
        """饼图数据"""
        data = self.df[[name_column, value_column]].values.tolist()
        
        return {
            'title': {'text': title or '饼图'},
            'tooltip': {'trigger': 'item'},
            'series': [{
                'name': value_column,
                'type': 'pie',
                'radius': ['40%', '70%'],
                'avoidLabelOverlap': False,
                'itemStyle': {
                    'borderRadius': 10,
                    'borderColor': '#fff',
                    'borderWidth': 2
                },
                'label': {
                    'show': True,
                    'formatter': '{b}: {c} ({d}%)'
                },
                'data': [
                    {'name': row[0], 'value': row[1]}
                    for row in data
                ]
            }]
        }
    
    def scatter_chart(
        self,
        x_column: str,
        y_column: str,
        size_column: str = None,
        title: str = None
    ) -> Dict:
        """散点图数据"""
        data = []
        for _, row in self.df.iterrows():
            point = [row[x_column], row[y_column]]
            if size_column:
                point.append(row[size_column])
            data.append(point)
        
        series = [{
            'type': 'scatter',
            'symbolSize': 20 if not size_column else lambda x: x[2] / 10,
            'data': data
        }]
        
        return {
            'title': {'text': title or '散点图'},
            'tooltip': {'trigger': 'item'},
            'xAxis': {'type': 'value', 'name': x_column},
            'yAxis': {'type': 'value', 'name': y_column},
            'series': series
        }
    
    def dashboard_summary(self) -> Dict:
        """仪表盘汇总数据"""
        summary = {
            'total_rows': len(self.df),
            'total_columns': len(self.df.columns),
            'numeric_columns': [],
            'categorical_columns': [],
            'key_metrics': {}
        }
        
        for col in self.df.columns:
            if pd.api.types.is_numeric_dtype(self.df[col]):
                summary['numeric_columns'].append(col)
                summary['key_metrics'][col] = {
                    'min': float(self.df[col].min()),
                    'max': float(self.df[col].max()),
                    'mean': float(self.df[col].mean()),
                    'sum': float(self.df[col].sum())
                }
            else:
                summary['categorical_columns'].append(col)
        
        return summary

# backend/routers/data.py
"""
数据处理 API
"""
from fastapi import APIRouter, UploadFile, File, HTTPException, Form
from fastapi.responses import StreamingResponse
from typing import Optional, List, Dict
import pandas as pd
import io
import json

from services.data_processor import DataProcessor, ChartDataGenerator
from models.schema import (
    ChartRequest, ChartResponse, DataSummaryResponse,
    AggregationRequest, AggregationResponse
)

router = APIRouter(prefix="/api/data", tags=["数据处理"])

# 存储当前处理的数据（生产环境用 Redis）
_data_store: Dict[str, DataProcessor] = {}


@router.post("/upload")
async def upload_file(
    file: UploadFile = File(...),
    session_id: str = Form(...)
) -> Dict:
    """
    上传数据文件
    
    支持 CSV 和 Excel 格式
    """
    if not file.filename.endswith(('.csv', '.xlsx', '.xls')):
        raise HTTPException(status_code=400, detail="只支持 CSV 和 Excel 文件")
    
    try:
        contents = await file.read()
        
        if file.filename.endswith('.csv'):
            df = pd.read_csv(io.BytesIO(contents))
        else:
            df = pd.read_excel(io.BytesIO(contents))
        
        processor = DataProcessor(df)
        
        # 保存到存储
        _data_store[session_id] = processor
        
        return {
            'session_id': session_id,
            'filename': file.filename,
            'metadata': processor.metadata
        }
    
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))


@router.post("/clean")
async def clean_data(
    session_id: str,
    operations: List[Dict]
) -> DataSummaryResponse:
    """
    数据清洗
    
    operations = [
        {'type': 'dropna', 'columns': ['col1']},
        {'type': 'fillna', 'column': 'col2', 'value': 0},
        {'type': 'remove_duplicates'},
        {'type': 'rename', 'mapping': {'old': 'new'}}
    ]
    """
    if session_id not in _data_store:
        raise HTTPException(status_code=404, detail="数据不存在")
    
    processor = _data_store[session_id]
    
    for op in operations:
        op_type = op.get('type')
        
        if op_type == 'dropna':
            processor.df = processor.df.dropna(subset=op.get('columns'))
        elif op_type == 'fillna':
            processor.df[op['column']] = processor.df[op['column']].fillna(op['value'])
        elif op_type == 'remove_duplicates':
            processor.df = processor.df.drop_duplicates()
        elif op_type == 'rename':
            processor.df = processor.df.rename(columns=op['mapping'])
        elif op_type == 'convert_dtype':
            processor.df[op['column']] = processor.df[op['column']].astype(op['dtype'])
    
    processor.metadata = processor._generate_metadata()
    
    return DataSummaryResponse(
        metadata=processor.metadata,
        preview=processor.df.head(10).to_dict(orient='records')
    )


@router.post("/aggregate")
async def aggregate_data(
    request: AggregationRequest,
    session_id: str
) -> AggregationResponse:
    """
    数据聚合
    """
    if session_id not in _data_store:
        raise HTTPException(status_code=404, detail="数据不存在")
    
    processor = _data_store[session_id]
    
    # 执行聚合
    result = processor.group_aggregate(
        group_by=request.group_by,
        aggregations=request.aggregations
    )
    
    return AggregationResponse(
        columns=result.columns.tolist(),
        data=result.to_dict(orient='records')
    )


@router.post("/chart")
async def generate_chart(
    request: ChartRequest,
    session_id: str
) -> ChartResponse:
    """
    生成图表数据
    """
    if session_id not in _data_store:
        raise HTTPException(status_code=404, detail="数据不存在")
    
    processor = _data_store[session_id]
    generator = ChartDataGenerator(processor.df)
    
    # 根据图表类型生成数据
    chart_data = None
    
    if request.chart_type == 'line':
        chart_data = generator.line_chart(
            x_column=request.x_column,
            y_columns=request.y_columns,
            title=request.title
        )
    elif request.chart_type == 'bar':
        chart_data = generator.bar_chart(
            x_column=request.x_column,
            y_column=request.y_columns[0] if request.y_columns else '',
            title=request.title
        )
    elif request.chart_type == 'pie':
        chart_data = generator.pie_chart(
            name_column=request.x_column,
            value_column=request.y_columns[0] if request.y_columns else '',
            title=request.title
        )
    elif request.chart_type == 'scatter':
        chart_data = generator.scatter_chart(
            x_column=request.x_column,
            y_column=request.y_columns[0] if request.y_columns else '',
            title=request.title
        )
    
    return ChartResponse(
        chart_type=request.chart_type,
        data=chart_data,
        metadata=processor.metadata
    )


@router.get("/summary/{session_id}")
async def get_summary(session_id: str) -> DataSummaryResponse:
    """获取数据摘要"""
    if session_id not in _data_store:
        raise HTTPException(status_code=404, detail="数据不存在")
    
    processor = _data_store[session_id]
    
    return DataSummaryResponse(
        metadata=processor.metadata,
        preview=processor.df.head(10).to_dict(orient='records')
    )


@router.get("/export/{session_id}")
async def export_data(
    session_id: str,
    format: str = 'csv'
):
    """导出数据"""
    if session_id not in _data_store:
        raise HTTPException(status_code=404, detail="数据不存在")
    
    processor = _data_store[session_id]
    
    if format == 'csv':
        output = io.StringIO()
        processor.df.to_csv(output, index=False)
        output.seek(0)
        
        return StreamingResponse(
            iter([output.getvalue()]),
            media_type="text/csv",
            headers={"Content-Disposition": f"attachment; filename=data.csv"}
        )
    elif format == 'json':
        return processor.to_json(orient='records')

// src/components/charts/ECharts.tsx
import React, { useEffect, useRef } from 'react'
import * as echarts from 'echarts'

interface ChartProps {
  option: any
  style?: React.CSSProperties
  onClick?: (params: any) => void
}

export const ECharts: React.FC<ChartProps> = ({ option, style, onClick }) => {
  const chartRef = useRef<HTMLDivElement>(null)
  const chartInstance = useRef<echarts.ECharts>()

  useEffect(() => {
    if (!chartRef.current) return

    // 初始化图表
    chartInstance.current = echarts.init(chartRef.current)

    // 绑定点击事件
    if (onClick) {
      chartInstance.current.on('click', onClick)
    }

    return () => {
      chartInstance.current?.dispose()
    }
  }, [])

  // 更新数据
  useEffect(() => {
    if (chartInstance.current && option) {
      chartInstance.current.setOption(option, { notMerge: true })
    }
  }, [option])

  return <div ref={chartRef} style={style || { width: '100%', height: '400px' }} />
}

**参考答案：**

2. **Z-score**：|z| > 3 视为异常
3. **可视化**：箱线图、散点图
4. **处理**：删除、替换、保留
