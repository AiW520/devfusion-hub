```python
# routers/content.py
from fastapi import APIRouter, HTTPException, Query
from typing import List, Optional
from datetime import datetime

router = APIRouter(prefix="/api/contents", tags=["内容管理"])

@router.post("/", response_model=Content)
async def create_content(content: ContentCreate):
    """创建内容"""
    content.created_at = datetime.utcnow()
    content.updated_at = datetime.utcnow()
    
    result = await db.contents.insert_one(content.dict(by_alias=True))
    content.id = str(result.inserted_id)
    
    return content


@router.get("/")
async def list_contents(
    status: Optional[str] = None,
    category_id: Optional[str] = None,
    tag: Optional[str] = None,
    author_id: Optional[str] = None,
    page: int = 1,
    page_size: int = 20,
    sort: str = "-created_at"
):
    """内容列表（支持过滤、分页）"""
    query = {}
    
    if status:
        query["status"] = status
    if category_id:
        query["category_id"] = category_id
    if tag:
        query["tags"] = tag
    if author_id:
        query["author_id"] = author_id
    
    # 排序
    sort_field = sort.lstrip("-")
    sort_direction = -1 if sort.startswith("-") else 1
    
    # 分页查询
    skip = (page - 1) * page_size
    
    cursor = db.contents.find(query) \
        .sort(sort_field, sort_direction) \
        .skip(skip) \
        .limit(page_size)
    
    contents = await cursor.to_list(length=page_size)
    total = await db.contents.count_documents(query)
    
    # 转换 ObjectId
    for content in contents:
        content["_id"] = str(content["_id"])
    
    return {
        "items": contents,
        "total": total,
        "page": page,
        "page_size": page_size,
        "pages": (total + page_size - 1) // page_size
    }


@router.get("/{content_id}")
async def get_content(content_id: str):
    """获取内容详情"""
    content = await db.contents.find_one({"_id": content_id})
    
    if not content:
        raise HTTPException(status_code=404, detail="内容不存在")
    
    # 增加浏览数（使用 Redis 缓存批量更新）
    await redis.incr(f"content:views:{content_id}")
    
    content["_id"] = str(content["_id"])
    return content


@router.patch("/{content_id}")
async def update_content(content_id: str, updates: dict):
    """更新内容"""
    updates["updated_at"] = datetime.utcnow()
    
    result = await db.contents.update_one(
        {"_id": content_id},
        {"$set": updates}
    )
    
    if result.matched_count == 0:
        raise HTTPException(status_code=404, detail="内容不存在")
    
    return {"status": "updated"}


@router.post("/{content_id}/publish")
async def publish_content(content_id: str):
    """发布内容"""
    result = await db.contents.update_one(
        {"_id": content_id},
        {
            "$set": {
                "status": "published",
                "published_at": datetime.utcnow(),
                "updated_at": datetime.utcnow()
            }
        }
    )
    
    if result.matched_count == 0:
        raise HTTPException(status_code=404, detail="内容不存在")
    
    return {"status": "published"}
```

from pydantic import BaseModel, Field
from typing import Optional, List, Dict, Any
from datetime import datetime
from enum import Enum


class LogLevel(str, Enum):
    DEBUG = "DEBUG"
    INFO = "INFO"
    WARNING = "WARNING"
    ERROR = "ERROR"
    CRITICAL = "CRITICAL"


class LogEntry(BaseModel):
    """日志条目"""
    id: Optional[str] = Field(None, alias="_id")
    
    # 时间
    timestamp: datetime = Field(default_factory=datetime.utcnow)
    date: str = Field(...)  # YYYY-MM-DD 格式，便于分区
    
    # 日志级别
    level: LogLevel = LogLevel.INFO
    
    # 来源
    service: str
    host: Optional[str] = None
    process_id: Optional[int] = None
    
    # 内容
    message: str
    logger: Optional[str] = None
    thread: Optional[str] = None
    
    # 堆栈
    stack_trace: Optional[str] = None
    
    # 自定义字段
    extra: Dict[str, Any] = Field(default_factory=dict)
    
    # 解析后的结构化数据
    parsed: Optional[Dict[str, Any]] = None
    
    class Config:
        use_enum_values = True


class AlertRule(BaseModel):
    """告警规则"""
    id: Optional[str] = Field(None, alias="_id")
    
    name: str
    description: Optional[str] = None
    
    # 条件
    condition: Dict[str, Any]  # MongoDB 查询条件
    threshold: Optional[int] = None  # 触发阈值
    time_window: int = 300  # 时间窗口（秒）
    
    # 动作
    action_type: str  # "webhook", "email", "slack"
    action_config: Dict[str, Any]

**实时日志流**：
```python
# routers/logs.py
from fastapi import APIRouter, HTTPException, Query
from fastapi.responses import StreamingResponse
from typing import Optional
import asyncio
import json

router = APIRouter(prefix="/api/logs", tags=["日志"])


@router.post("/")
async def ingest_log(log: LogEntry):
    """接收日志"""
    # 设置日期字段
    log.date = log.timestamp.strftime("%Y-%m-%d")
    
    # 插入 MongoDB
    result = await db.logs.insert_one(log.dict(by_alias=True))
    
    # 推送到 Redis 列表（实时流）
    await redis.lpush("logs:stream", log.json())
    await redis.ltrim("logs:stream", 0, 9999)  # 保留最近 10000 条
    
    return {"id": str(result.inserted_id)}


@router.get("/stream")
async def stream_logs(
    service: Optional[str] = None,
    level: Optional[str] = None,
    limit: int = Query(100, le=1000)
):
    """实时日志流 (Server-Sent Events)"""
    async def event_generator():
        # 使用 Redis BRPOP 实现阻塞读取
        while True:
            # 从 Redis 获取日志
            log = await redis.brpop("logs:stream", timeout=30)
            
            if log:
                _, log_data = log
                log_entry = json.loads(log_data)
                
                # 过滤
                if service and log_entry.get("service") != service:
                    continue
                if level and log_entry.get("level") != level:
                    continue
                
                yield f"data: {log_data}\n\n"
    
    return StreamingResponse(
        event_generator(),
        media_type="text/event-stream",
        headers={
            "Cache-Control": "no-cache",
            "Connection": "keep-alive",
        }
    )


@router.get("/search")
async def search_logs(
    query: str,
    start_time: Optional[datetime] = None,
    end_time: Optional[datetime] = None,
    service: Optional[str] = None,
    level: Optional[str] = None,
    page: int = 1,
    page_size: int = 50
):
    """搜索日志"""
    # 构建查询
    filter_query = {"$text": {"$search": query}}  # 需要创建全文索引
    
    if start_time or end_time:
        filter_query["timestamp"] = {}
        if start_time:
            filter_query["timestamp"]["$gte"] = start_time
        if end_time:
            filter_query["timestamp"]["$lte"] = end_time
    
    if service:
        filter_query["service"] = service
    if level:
        filter_query["level"] = level
    
    # 查询
    skip = (page - 1) * page_size
    
    cursor = db.logs.find(filter_query) \
        .sort("timestamp", -1) \
        .skip(skip) \
        .limit(page_size)
    
    logs = await cursor.to_list(length=page_size)
    total = await db.logs.count_documents(filter_query)
    
    for log in logs:
        log["_id"] = str(log["_id"])
    
    return {
        "items": logs,
        "total": total,
        "page": page,
        "page_size": page_size
    }


@router.get("/aggregate")
async def aggregate_logs(
    group_by: str = "service",  # "service", "level", "hour"
    start_time: Optional[datetime] = None,
    end_time: Optional[datetime] = None,
    interval: str = "1h"  # "1m", "5m", "1h", "1d"
):
    """聚合统计"""
    match_stage = {}
    
    if start_time or end_time:
        match_stage["timestamp"] = {}
        if start_time:
            match_stage["timestamp"]["$gte"] = start_time
        if end_time:
            match_stage["timestamp"]["$lte"] = end_time
    
    # MongoDB 聚合管道
    pipeline = [
        {"$match": match_stage},
        {
            "$group": {
                "_id": f"${group_by}",
                "count": {"$sum": 1},
                "errors": {
                    "$sum": {"$cond": [{"$eq": ["$level", "ERROR"]}, 1, 0]}
                }
            }
        },
        {"$sort": {"count": -1}},
        {"$limit": 20}
    ]
    
    result = await db.logs.aggregate(pipeline).to_list(length=20)
    
    return {
        "group_by": group_by,
        "intervals": result
    }
```

### 问题 1: MongoDB vs PostgreSQL 的选择？

**参考答案**：
- MongoDB：灵活 schema、大量写入、日志系统
- PostgreSQL：强一致性、复杂查询、事务支持
