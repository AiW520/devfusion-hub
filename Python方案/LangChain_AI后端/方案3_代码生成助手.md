## 📋 项目介绍

### 1.1 项目概述
本项目是一个**智能代码生成助手**，支持根据自然语言描述生成代码、代码解释、代码审查、代码翻译（Python↔JavaScript↔Java）等功能。基于 LangChain 的 Agent 架构，可以调用多种工具完成复杂任务。

### 1.2 核心技术亮点
- **LangChain Agent**：自主决策的工具调用
- **Code Interpreter**：沙箱代码执行环境
- **多语言支持**：Python、JavaScript、Java、TypeScript
- **代码高亮**：语法分析和格式化
- **版本对比**：新旧代码 diff 展示

### 1.3 适用场景
- AI 编程助手
- 代码审查和优化
- 技术文档自动生成
- LeetCode 刷题辅助
- 代码翻译和移植

```python
"""
数据模型 - 代码助手 API 的数据结构
"""
from pydantic import BaseModel, Field
from typing import List, Optional, Dict, Any
from enum import Enum
from datetime import datetime


class ProgrammingLanguage(str, Enum):
    """编程语言枚举"""
    PYTHON = "python"
    JAVASCRIPT = "javascript"
    TYPESCRIPT = "typescript"
    JAVA = "java"
    GO = "go"
    RUST = "rust"
    UNKNOWN = "unknown"


class CodeGenerationRequest(BaseModel):
    """代码生成请求"""
    description: str = Field(..., min_length=10, max_length=2000)
    language: ProgrammingLanguage = Field(default=ProgrammingLanguage.PYTHON)
    framework: Optional[str] = Field(None, description="框架，如 'fastapi', 'react'")
    constraints: Optional[List[str]] = Field(default_factory=list, description="约束条件")
    examples: Optional[List[str]] = Field(default_factory=list, description="示例代码")
    
    class Config:
        json_schema_extra = {
            "example": {
                "description": "实现一个快速排序算法，要求时间复杂度 O(n log n)",
                "language": "python",
                "constraints": ["使用递归", "原地排序"],
                "examples": ["示例输入: [3,1,4,1,5]"]
            }
        }


class CodeTranslationRequest(BaseModel):
    """代码翻译请求"""
    source_code: str = Field(..., min_length=1)
    source_language: ProgrammingLanguage
    target_language: ProgrammingLanguage
    preserve_comments: bool = True


class CodeExplanationRequest(BaseModel):
    """代码解释请求"""
    code: str = Field(..., min_length=1)
    language: Optional[ProgrammingLanguage] = None
    detail_level: str = Field(default="medium", pattern="^(simple|medium|detailed)$")


class CodeReviewRequest(BaseModel):
    """代码审查请求"""
    code: str = Field(..., min_length=10)
    language: Optional[ProgrammingLanguage] = None
    focus_areas: Optional[List[str]] = Field(
        default_factory=lambda: ["correctness", "performance", "readability"]
    )


class CodeExecutionRequest(BaseModel):
    """代码执行请求"""
    code: str = Field(..., min_length=1)
    language: ProgrammingLanguage
    input_data: Optional[str] = Field(None, description="标准输入")
    timeout: int = Field(default=5, ge=1, le=30, description="超时秒数")


class CodeGenerationResponse(BaseModel):
    """代码生成响应"""
    code: str
    language: ProgrammingLanguage
    explanation: str
    complexity: Dict[str, str]  # {"time": "O(n log n)", "space": "O(n)"}
    test_cases: List[Dict[str, Any]]


class CodeExecutionResponse(BaseModel):
    """代码执行响应"""
    success: bool
    output: str
    error: Optional[str] = None
    execution_time: float  # 毫秒
    memory_usage: Optional[str] = None


class CodeReviewResponse(BaseModel):
    """代码审查响应"""
    score: int = Field(ge=1, le=10)
    issues: List[Dict[str, Any]]
    suggestions: List[str]
    summary: str
```

### 2.4 代码生成服务 code_generator.py
```python
"""
代码生成服务 - 基于 LangChain 的代码生成

核心功能：
1. 自然语言到代码的转换
2. 多轮对话优化代码
3. 代码解释和文档生成
"""
from typing import Dict, List, Optional, Any
from langchain.chat_models import ChatOpenAI
from langchain.prompts import PromptTemplate
from langchain.output_parsers import PydanticOutputParser
from langchain.schema import HumanMessage, SystemMessage
import os


class CodeGenerator:
    """
    代码生成器
    
    基于 LangChain + GPT-4 的代码生成
    """
    
    def __init__(self, model: str = "gpt-4"):
        """
        初始化代码生成器
        
        Args:
            model: 使用的模型（gpt-4 生成代码效果更好）
        """
        self.llm = ChatOpenAI(
            model=model,
            temperature=0.3,  # 代码生成用较低温度
            api_key=os.getenv("OPENAI_API_KEY"),
            base_url=os.getenv("OPENAI_API_BASE", "https://api.openai.com/v1")
        )
        
        # 系统提示词 - 扮演资深工程师
        self.system_prompt = """你是一个资深的{language}工程师，擅长编写高质量、可维护的代码。

【工作原则】
1. 代码必须是可运行的，不要留下 TODO
2. 遵循语言的编码规范和最佳实践
3. 添加必要的注释说明
4. 考虑边界情况和错误处理
5. 保持代码简洁但不过度优化

【代码风格】
- Python: 遵循 PEP 8，使用类型提示
- JavaScript: 遵循 ESLint 标准，使用 ES6+ 语法
- Java: 遵循 Oracle 编码规范

【输出格式】
请按以下格式返回：
```code
[生成的代码]
```

同时请简要说明：
1. 代码的核心逻辑
2. 时间/空间复杂度
3. 可能的改进方向
"""
    
    def generate(
        self,
        description: str,
        language: str = "python",
        framework: Optional[str] = None,
        constraints: Optional[List[str]] = None,
        context: Optional[str] = None
    ) -> Dict[str, Any]:
        """
        生成代码
        
        Args:
            description: 功能描述
            language: 目标语言
            framework: 使用的框架
            constraints: 约束条件
            context: 额外上下文
            
        Returns:
            {"code": str, "explanation": str, "complexity": dict}
        """
        # 构建提示词
        prompt = self._build_prompt(
            description=description,
            language=language,
            framework=framework,
            constraints=constraints,
            context=context
        )
        
        # 调用 LLM
        response = self.llm.invoke([
            SystemMessage(content=self.system_prompt.format(language=language)),
            HumanMessage(content=prompt)
        ])
        
        # 解析结果
        return self._parse_response(response.content, language)
    
    def _build_prompt(
        self,
        description: str,
        language: str,
        framework: Optional[str],
        constraints: Optional[List[str]],
        context: Optional[str]
    ) -> str:
        """构建生成提示词"""
        prompt = f"""【任务描述】
{description}

【目标语言】
{language}"""
        
        if framework:
            prompt += f"\n\n【使用框架】\n{framework}"
        
        if constraints:
            prompt += f"\n\n【约束条件】\n" + "\n".join(f"- {c}" for c in constraints)
        
        if context:
            prompt += f"\n\n【额外上下文】\n{context}"
        
        prompt += "\n\n请生成代码并给出简要说明。"
        
        return prompt
    
    def _parse_response(self, response: str, language: str) -> Dict[str, Any]:
        """
        解析 LLM 响应
        
        从输出中提取代码和说明
        """
        # 提取代码块
        code_start = response.find("```") + 3
        code_end = response.find("```", code_start)
        
        if code_start > 2 and code_end > code_start:
            code = response[code_start:code_end].strip()
            # 移除语言标识（如 ```python）
            if code.startswith(language):
                code = code[len(language):].strip()
        else:
            code = response.strip()
        
        # 提取说明（代码块之后的部分）
        explanation = response[code_end + 3:].strip() if code_end > 0 else ""
        
        # 简单分析复杂度（实际应该用 AST 分析）
        complexity = self._analyze_complexity(code, language)
        
        return {
            "code": code,
            "explanation": explanation or "代码已生成",
            "complexity": complexity,
            "language": language
        }
    
    def _analyze_complexity(self, code: str, language: str) -> Dict[str, str]:
        """
        简单分析代码复杂度
        
        注意：这是基于规则的简单分析，精确分析需要 AST
        """
        # 常见算法复杂度模式
        patterns = {
            "O(1)": ["return", "if", "else", "="],
            "O(log n)": ["binary_search", "while.*<.*:", "while.*>.*:"],
            "O(n)": ["for.*in", "while", "map", "for i in range"],
            "O(n log n)": ["sorted", "sort", "merge", "quick"],
            "O(n²)": ["for.*for", "嵌套循环"],
        }
        
        time_complexity = "O(n)"  # 默认
        space_complexity = "O(1)"  # 默认
        
        # 简单的关键词匹配
        code_lower = code.lower()
        
        if "quick" in code_lower and "sort" in code_lower:
            time_complexity = "O(n log n)"
        elif "nested" in code_lower or code.count("for") >= 2:
            time_complexity = "O(n²)"
        elif "binary" in code_lower or "search" in code_lower:
            time_complexity = "O(log n)"
        elif "return" in code_lower and code.count("\n") < 5:
            time_complexity = "O(1)"
        
        # 检查空间复杂度
        if "list" in code_lower or "[]" in code:
            space_complexity = "O(n)"
        
        return {
            "time": time_complexity,
            "space": space_complexity
        }
    
    def generate_tests(
        self,
        code: str,
        language: str,
        test_cases: Optional[List[Dict]] = None
    ) -> str:
        """
        生成测试代码
        
        Args:
            code: 要测试的代码
            language: 语言
            test_cases: 预设测试用例
            
        Returns:
            测试代码
        """
        prompt = f"""请为以下代码生成测试用例：

```{language}
{code}
```

要求：
1. 使用标准的测试框架（Python: pytest, JavaScript: Jest, Java: JUnit）
2. 覆盖正常情况和边界情况
3. 包含断言和预期结果

请只返回测试代码，不要包含其他说明。"""
        
        response = self.llm.invoke([
            HumanMessage(content=prompt)
        ])
        
        return response.content


class CodeTranslator:
    """
    代码翻译器
    
    将代码从一种语言翻译到另一种语言
    """
    
    def __init__(self):
        self.llm = ChatOpenAI(
            model="gpt-4",
            temperature=0.2,  # 翻译用更低温度
            api_key=os.getenv("OPENAI_API_KEY")
        )
    
    def translate(
        self,
        code: str,
        source_lang: str,
        target_lang: str,
        preserve_comments: bool = True
    ) -> Dict[str, Any]:
        """
        翻译代码
        
        Args:
            code: 源代码
            source_lang: 源语言
            target_lang: 目标语言
            preserve_comments: 是否保留注释
            
        Returns:
            {"code": str, "notes": list}
        """
        notes = []
        
        # 翻译提示词
        prompt = f"""请将以下 {source_lang} 代码翻译为 {target_lang}。

{'保留所有注释' if preserve_comments else '移除注释，只保留核心代码'}

```{source_lang}
{code}
```

翻译要求：
1. 保持相同的逻辑和功能
2. 遵循目标语言的编码规范
3. 如果有语言特有的实现方式，需要适配
4. 标注需要特别注意的地方

请返回翻译后的代码。"""
        
        response = self.llm.invoke([HumanMessage(content=prompt)])
        
        # 提取代码
        code_start = response.content.find("```") + 3
        code_end = response.content.find("```", code_start)
        
        if code_start > 2 and code_end > code_start:
            translated_code = response.content[code_start:code_end].strip()
            if translated_code.startswith(target_lang):
                translated_code = translated_code[len(target_lang):].strip()
        else:
            translated_code = response.content.strip()
        
        # 提取注意事项
        notes_text = response.content[code_end + 3:].strip()
        if notes_text:
            notes = [n.strip() for n in notes_text.split("\n") if n.strip()]
        
        return {
            "code": translated_code,
            "notes": notes,
            "source_language": source_lang,
            "target_language": target_lang
        }


class CodeExplainer:
    """
    代码解释器
    
    解释代码的功能和实现细节
    """
    
    # 语言映射
    LANGUAGE_NAMES = {
        "python": "Python",
        "javascript": "JavaScript",
        "typescript": "TypeScript",
        "java": "Java",
        "go": "Go",
        "rust": "Rust"
    }
    
    def __init__(self):
        self.llm = ChatOpenAI(
            model="gpt-4",
            temperature=0.3,
            api_key=os.getenv("OPENAI_API_KEY")
        )
    
    def explain(
        self,
        code: str,
        language: Optional[str] = None,
        detail_level: str = "medium"
    ) -> Dict[str, Any]:
        """
        解释代码
        
        Args:
            code: 要解释的代码
            language: 语言（可选，自动检测）
            detail_level: 详细程度 (simple/medium/detailed)
            
        Returns:
            {"summary": str, "breakdown": list, "flow": str}
        """
        lang_name = self.LANGUAGE_NAMES.get(language, "该语言") if language else "这"
        
        # 不同详细程度的提示词
        prompts = {
            "simple": f"""请简单解释以下 {lang_name} 代码的功能：

```{language or ""}
{code}
```

用 2-3 句话概括。""",
            
            "medium": f"""请详细解释以下 {lang_name} 代码：

```{language or ""}
{code}
```

请按以下格式回答：
1. 功能概述（一句话）
2. 核心逻辑说明
3. 主要函数/方法的作用
4. 数据流""",
            
            "detailed": f"""请全面分析以下 {lang_name} 代码：

```{language or ""}
{code}
```

请按以下格式回答：
1. 功能概述
2. 代码结构分析
3. 每个函数的详细说明
4. 数据流和控制流
5. 潜在问题和改进建议
6. 复杂度分析"""
        }
        
        prompt = prompts.get(detail_level, prompts["medium"])
        response = self.llm.invoke([HumanMessage(content=prompt)])
        
        return self._parse_explanation(response.content, detail_level)
    
    def _parse_explanation(
        self,
        content: str,
        detail_level: str
    ) -> Dict[str, Any]:
        """解析解释结果"""
        # 按行分割
        lines = content.strip().split("\n")
        
        # 提取各个部分
        result = {
            "summary": "",
            "breakdown": [],
            "flow": "",
            "raw": content
        }
        
        # 简单解析（实际可以用更复杂的方法）
        if detail_level == "simple":
            result["summary"] = content.strip()
        else:
            current_section = ""
            current_content = []
            
            for line in lines:
                # 检测章节标题
                if line.strip().startswith(("1.", "2.", "3.", "4.", "5.", "6.")):
                    # 保存上一个章节
                    if current_section:
                        content_text = "\n".join(current_content).strip()
                        if current_section in ["功能概述", "功能概述（一句话）"]:
                            result["summary"] = content_text
                        else:
                            result["breakdown"].append({
                                "title": current_section,
                                "content": content_text
                            })
                    
                    current_section = line.strip()
                    current_content = []
                elif line.strip().startswith("-"):
                    current_content.append(line)
                else:
                    current_content.append(line)
            
            # 保存最后一个章节
            if current_content:
                content_text = "\n".join(current_content).strip()
                if current_section in ["功能概述", "功能概述（一句话）"]:
                    result["summary"] = content_text
                else:
                    result["breakdown"].append({
                        "title": current_section,
                        "content": content_text
                    })
        
        return result

### 2.5 代码执行器 code_executor.py
```python
"""
代码执行器 - 安全的代码运行环境

注意：这是一个简单的执行器，生产环境需要：
1. 沙箱隔离（Docker/Kubernetes）
2. 资源限制（CPU/内存/时间）
3. 安全审计
"""
from typing import Dict, Any, Optional
import subprocess
import tempfile
import os
import time
from datetime import datetime


class CodeExecutor:
    """
    代码执行器
    
    支持多种语言的代码执行
    注意：这是演示版本，生产环境需要完整的沙箱隔离！
    """
    
    # 语言到运行命令的映射
    LANGUAGE_COMMANDS = {
        "python": {
            "extension": ".py",
            "command": ["python3", "{file}"],
            "version_cmd": ["python3", "--version"]
        },
        "javascript": {
            "extension": ".js",
            "command": ["node", "{file}"],
            "version_cmd": ["node", "--version"]
        },
        "java": {
            "extension": ".java",
            "command": ["java", "{file}"],
            "version_cmd": ["java", "-version"]
        }
    }
    
    def __init__(self, timeout: int = 5, max_output_length: int = 10000):
        """
        初始化执行器
        
        Args:
            timeout: 执行超时（秒）
            max_output_length: 最大输出长度
        """
        self.timeout = timeout
        self.max_output_length = max_output_length
    
    def execute(
        self,
        code: str,
        language: str,
        input_data: Optional[str] = None
    ) -> Dict[str, Any]:
        """
        执行代码
        
        Args:
            code: 要执行的代码
            language: 语言
            input_data: 标准输入
            
        Returns:
            {"success": bool, "output": str, "error": str, "execution_time": float}
        """
        lang_config = self.LANGUAGE_COMMANDS.get(language)
        
        if not lang_config:
            return {
                "success": False,
                "output": "",
                "error": f"不支持的语言: {language}",
                "execution_time": 0
            }
        
        # 创建临时文件
        with tempfile.NamedTemporaryFile(
            mode='w',
            suffix=lang_config["extension"],
            delete=False,
            encoding='utf-8'
        ) as f:
            f.write(code)
            file_path = f.name
        
        try:
            # 构建命令
            command = [
                cmd.replace("{file}", file_path)
                for cmd in lang_config["command"]
            ]
            
            # 执行
            start_time = time.time()
            
            # 如果有输入，通过 stdin 传入
            input_bytes = input_data.encode() if input_data else None
            
            result = subprocess.run(
                command,
                capture_output=True,
                timeout=self.timeout,
                input=input_bytes,
                cwd=tempfile.gettempdir()
            )
            
            execution_time = (time.time() - start_time) * 1000  # 毫秒
            
            # 解析输出
            stdout = result.stdout.decode('utf-8', errors='replace')
            stderr = result.stderr.decode('utf-8', errors='replace')
            
            # 截断过长输出
            if len(stdout) > self.max_output_length:
                stdout = stdout[:self.max_output_length] + "\n... (output truncated)"
            
            return {
                "success": result.returncode == 0,
                "output": stdout,
                "error": stderr if stderr else None,
                "execution_time": execution_time,
                "return_code": result.returncode
            }
        
        except subprocess.TimeoutExpired:
            return {
                "success": False,
                "output": "",
                "error": f"执行超时（超过 {self.timeout} 秒）",
                "execution_time": self.timeout * 1000
            }
        
        except Exception as e:
            return {
                "success": False,
                "output": "",
                "error": f"执行错误: {str(e)}",
                "execution_time": 0
            }
        
        finally:
            # 清理临时文件
            if os.path.exists(file_path):
                os.remove(file_path)
    
    def execute_python(self, code: str, test_input: str = "") -> Dict[str, Any]:
        """专门执行 Python 代码（支持更复杂的场景）"""
        return self.execute(code, "python", test_input)
    
    def execute_javascript(self, code: str, test_input: str = "") -> Dict[str, Any]:
        """专门执行 JavaScript 代码"""
        return self.execute(code, "javascript", test_input)


class PythonSandbox:
    """

    # 允许的模块

    ) -> Dict[str, Any]:

```python
"""
代码审查服务 - AI 驱动的代码审查

功能：
1. 代码质量评分
2. 问题检测
3. 改进建议
"""
from typing import Dict, List, Any
from langchain.chat_models import ChatOpenAI
from langchain.schema import HumanMessage, SystemMessage
import os


class CodeReviewer:
    """
    代码审查器
    
    基于 GPT-4 的代码审查
    """
    
    def __init__(self):
        self.llm = ChatOpenAI(
            model="gpt-4",
            temperature=0.3,
            api_key=os.getenv("OPENAI_API_KEY")
        )
        
        self.review_prompt = """你是一个资深的代码审查工程师，负责审查代码并提供改进建议。

【审查维度】
1. 正确性 (Correctness)：逻辑是否正确，边界情况是否处理
2. 性能 (Performance)：时间和空间复杂度是否合理
3. 可读性 (Readability)：命名、注释、代码结构
4. 安全性 (Security)：潜在的安全漏洞
5. 可维护性 (Maintainability)：代码结构、模块化程度

【输出格式】
请按以下 JSON 格式返回：
```json
{{
    "score": 8,
    "summary": "总体评价...",
    "issues": [
        {{
            "severity": "high/medium/low",
            "category": "correctness/performance/readability",
            "location": "具体位置",
            "description": "问题描述",
            "suggestion": "修改建议"
        }}
    ],
    "suggestions": ["改进建议1", "改进建议2"]
}}
```

【审查的代码】
```{language}
{code}
```"""
    
    def review(
        self,
        code: str,
        language: str = "python",
        focus_areas: List[str] = None
    ) -> Dict[str, Any]:
        """
        审查代码
        
        Args:
            code: 要审查的代码
            language: 编程语言
            focus_areas: 重点审查领域
            
        Returns:
            {"score": int, "issues": list, "suggestions": list, "summary": str}
        """
        # 构建提示词
        focus_text = ""
        if focus_areas:
            focus_text = f"\n\n【重点审查】\n" + "\n".join(f"- {area}" for area in focus_areas)
        
        prompt = self.review_prompt.format(
            language=language,
            code=code
        ) + focus_text
        
        # 调用 LLM
        response = self.llm.invoke([HumanMessage(content=prompt)])
        
        # 解析结果
        return self._parse_review(response.content)
    
    def _parse_review(self, content: str) -> Dict[str, Any]:
        """解析审查结果"""
        # 尝试提取 JSON
        json_start = content.find("```json")
        if json_start == -1:
            json_start = content.find("```")
        
        if json_start != -1:
            json_start += 7
            json_end = content.find("```", json_start)
            if json_end != -1:
                json_str = content[json_start:json_end].strip()
                try:
                    import json
                    return json.loads(json_str)
                except:
                    pass
        
        # 如果无法解析 JSON，返回原始内容
        return {
            "score": 7,
            "summary": content,
            "issues": [],
            "suggestions": []
        }
    
    def suggest_improvements(
        self,
        code: str,
        language: str = "python"
    ) -> str:
        """
        获取代码改进建议
        """
        prompt = f"""请为以下 {language} 代码提供具体的改进建议，包括：
1. 可以优化的部分
2. 更好的实现方式
3. 代码风格改进

```{language}
{code}
```

请只返回改进建议，不要返回修改后的完整代码。"""
        
        response = self.llm.invoke([HumanMessage(content=prompt)])
        return response.content

      console.error('解释失败:', error)

      setLoading(false)
