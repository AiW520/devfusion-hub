## 📋 项目介绍

### 1.1 项目概述
本项目是一个功能完整的**在线教育平台**，包含课程管理、视频点播、在线测验、学习进度追踪、作业批改、讨论区等核心功能，支持多种角色（学生、教师、管理员）的使用。

### 1.2 核心技术亮点
- **视频处理**：Django + FFmpeg 视频转码
- **在线测验**：题库管理、随机组卷、自动判题
- **学习追踪**：进度记录、数据分析
- **作业系统**：在线提交、批改评分
- **讨论区**：课程讨论、答疑解惑

### 1.3 适用场景
- 在线教育平台
- 企业培训系统
- MOOC 慕课平台
- 知识付费平台

class QuizSerializer(serializers.ModelSerializer):
    """测验"""
    questions = serializers.SerializerMethodField()

    class Meta:
        model = Quiz
        fields = ['id', 'lesson', 'title', 'description',
                  'question_count', 'time_limit', 'passing_score',
                  'total_score', 'questions']

    def get_questions(self, obj):
        """获取题目（根据配置决定是否返回答案）"""
        questions = obj.lesson.question_set.all()
        
        # 随机抽取
        if obj.is_shuffle:
            import random
            questions = random.sample(list(questions), min(obj.question_count, len(questions)))
        
        # 是否返回答案（用户提交后才能看）
        context = self.context
        if context.get('show_answer'):
            return QuestionSerializer(questions, many=True).data
        else:
            # 不返回答案字段
            data = QuestionSerializer(questions, many=True).data
            for q in data:
                q.pop('answer', None)
            return data

from rest_framework import viewsets, status

class CourseViewSet(viewsets.ReadOnlyModelViewSet):
    """课程浏览"""
    queryset = Course.objects.filter(status='published')
    lookup_field = 'slug'
    
    def get_serializer_class(self):
        if self.action == 'retrieve':
            return CourseDetailSerializer
        return CourseListSerializer
    
    def get_permissions(self):
        if self.action in ['retrieve', 'list', 'chapters']:
            return [AllowAny()]
        return [IsAuthenticated()]
    
    def get_queryset(self):
        queryset = super().get_queryset()
        queryset = queryset.select_related('teacher', 'category')
        
        # 过滤
        category = self.request.query_params.get('category')
        if category:
            queryset = queryset.filter(category_id=category)
        
        level = self.request.query_params.get('level')
        if level:
            queryset = queryset.filter(level=level)
        
        # 搜索
        search = self.request.query_params.get('search')
        if search:
            queryset = queryset.filter(
                Q(title__icontains=search) |
                Q(description__icontains=search)
            )
        
        # 排序
        ordering = self.request.query_params.get('ordering', '-created_at')
        if ordering == 'popular':
            ordering = '-enrolled_count'
        elif ordering == 'rating':
            ordering = '-rating'
        elif ordering == 'newest':
            ordering = '-created_at'
        
        queryset = queryset.order_by(ordering)
        
        return queryset
    
    @action(detail=True, methods=['get'])
    def chapters(self, request, slug=None):
        """获取课程章节"""
        course = self.get_object()
        chapters = course.chapters.prefetch_related('lessons')
        
        # 如果用户已登录，标记已完成的课时
        user = request.user
        completed_lessons = set()
        if user.is_authenticated:
            enrollment = Enrollment.objects.filter(user=user, course=course).first()
            if enrollment:
                completed = LessonProgress.objects.filter(
                    enrollment=enrollment,
                    is_completed=True
                ).values_list('lesson_id', flat=True)
                completed_lessons = set(completed)
        
        from .serializers import ChapterSerializer
        chapters_data = ChapterSerializer(chapters, many=True).data
        
        # 添加完成状态
        for chapter in chapters_data:
            for lesson in chapter['lessons']:
                lesson['is_completed'] = lesson['id'] in completed_lessons
        
        return Response(chapters_data)


class EnrollmentViewSet(viewsets.ModelViewSet):
    """学习记录"""
    serializer_class = EnrollmentSerializer
    permission_classes = [IsAuthenticated]
    
    def get_queryset(self):
        return Enrollment.objects.filter(user=self.request.user)
    
    @action(detail=True, methods=['post'])
    def enroll(self, request, pk=None):
        """报名课程"""
        course = Course.objects.get(pk=pk)
        
        # 检查是否已报名
        if Enrollment.objects.filter(user=request.user, course=course).exists():
            return Response(
                {'error': '您已报名该课程'},
                status=status.HTTP_400_BAD_REQUEST
            )
        
        enrollment = Enrollment.objects.create(
            user=request.user,
            course=course
        )
        
        # 更新课程报名人数
        Course.objects.filter(pk=course.pk).update(
            enrolled_count=F('enrolled_count') + 1
        )
        
        serializer = self.get_serializer(enrollment)
        return Response(serializer.data, status=status.HTTP_201_CREATED)
    
    @action(detail=True, methods=['post'])
    def complete_lesson(self, request, pk=None):
        """完成课时"""
        enrollment = self.get_object()
        lesson_id = request.data.get('lesson_id')
        watch_time = request.data.get('watch_time', 0)
        
        try:
            lesson = Lesson.objects.get(id=lesson_id)
        except Lesson.DoesNotExist:
            return Response(
                {'error': '课时不存在'},
                status=status.HTTP_404_NOT_FOUND
            )
        
        # 更新或创建进度
        progress, created = LessonProgress.objects.update_or_create(
            user=request.user,
            lesson=lesson,
            enrollment=enrollment,
            defaults={
                'watched_duration': watch_time,
                'progress_percent': min(100, watch_time * 100 // max(lesson.video_duration or 1, 1)),
                'is_completed': True,
                'completed_at': timezone.now()
            }
        )
        
        # 更新课程进度
        self._update_course_progress(enrollment)
        
        return Response({
            'status': 'success',
            'progress': progress.progress_percent
        })
    
    def _update_course_progress(self, enrollment):
        """更新课程进度"""
        total_lessons = enrollment.course.lesson_count
        completed = enrollment.lesson_progress.filter(
            is_completed=True
        ).count()
        
        progress = int(completed * 100 / max(total_lessons, 1))
        
        enrollment.completed_lessons = completed
        enrollment.progress = progress
        enrollment.last_study_at = timezone.now()
        enrollment.save(update_fields=[
            'completed_lessons', 'progress', 'last_study_at', 'updated_at'
        ])
        
        # 如果 100%，更新状态
        if progress >= 100:
            enrollment.status = 'completed'
            enrollment.save(update_fields=['status'])

class QuizViewSet(viewsets.ModelViewSet):
    """测验"""
    queryset = Quiz.objects.all()
    serializer_class = QuizSerializer
    
    def get_permissions(self):
        if self.action in ['retrieve', 'list']:
            return [AllowAny()]
        return [IsAuthenticated()]
    
    @action(detail=True, methods=['post'])
    def start(self, request, pk=None):
        """开始测验"""
        quiz = self.get_object()
        enrollment = Enrollment.objects.filter(
            user=request.user,
            course=quiz.lesson.chapter.course
        ).first()
        
        if not enrollment:
            return Response(
                {'error': '请先报名课程'},
                status=status.HTTP_400_BAD_REQUEST
            )
        
        # 检查尝试次数
        attempts = QuizAttempt.objects.filter(
            user=request.user, quiz=quiz
        ).count()
        
        if quiz.attempts_allowed > 0 and attempts >= quiz.attempts_allowed:
            return Response(
                {'error': '已达到最大尝试次数'},
                status=status.HTTP_400_BAD_REQUEST
            )
        
        # 创建测验记录
        attempt = QuizAttempt.objects.create(
            user=request.user,
            quiz=quiz,
            enrollment=enrollment,
            status='in_progress'
        )
        
        # 返回题目
        serializer = QuizSerializer(
            quiz,
            context={'show_answer': False}
        )
        
        return Response({
            'attempt_id': attempt.id,
            'quiz': serializer.data,
            'time_limit': quiz.time_limit,
            'started_at': attempt.started_at
        })
    
    @action(detail=True, methods=['post'])
    def submit(self, request, pk=None):
        """提交测验"""
        quiz = self.get_object()
        attempt_id = request.data.get('attempt_id')
        answers = request.data.get('answers', {})
        
        try:
            attempt = QuizAttempt.objects.get(
                id=attempt_id,
                user=request.user,
                quiz=quiz
            )
        except QuizAttempt.DoesNotExist:
            return Response(
                {'error': '测验记录不存在'},
                status=status.HTTP_404_NOT_FOUND
            )
        
        if attempt.status != 'in_progress':
            return Response(
                {'error': '测验已提交'},
                status=status.HTTP_400_BAD_REQUEST
            )
        
        # 评分
        score, total = self._calculate_score(quiz, answers)
        is_passed = score >= quiz.passing_score
        
        # 更新记录
        attempt.answers = answers
        attempt.score = score
        attempt.is_passed = is_passed
        attempt.status = 'completed'
        attempt.submitted_at = timezone.now()
        attempt.time_spent = (
            attempt.submitted_at - attempt.started_at
        ).total_seconds()
        attempt.save()
        
        return Response({
            'score': score,
            'total': total,
            'is_passed': is_passed,
            'passing_score': quiz.passing_score
        })
    
    def _calculate_score(self, quiz, answers):
        """计算分数"""
        score = 0
        total = 0
        
        for question in quiz.lesson.question_set.all():
            total += 1
            user_answer = answers.get(str(question.id))
            
            if user_answer == question.answer:
                score += quiz.total_score / quiz.question_count
        
        return int(score), total
